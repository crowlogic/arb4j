\section*{Cram{\'e}r's Pathwise Inversion Formula for the Orthogonal Random
Measure}

\subsection*{The Spectral Representation}

For a centered stationary Gaussian process $\xi (t)$ on $\mathbb{R}$, there
exists an orthogonal random measure $\zeta (\lambda)$ such that
\begin{equation}
  \xi (t) = \int_{- \infty}^{\infty} e^{it \lambda}  \hspace{0.17em} d \zeta
  (\lambda)
\end{equation}

\subsection*{Recovery of the Spectral Distribution Function}

The spectral distribution function $F (\lambda)$ is recovered from the
autocovariance $r (t) =\mathbb{E} [\xi (t + \tau) \xi (t)]$ by
\begin{equation}
  F (\lambda_2) - F (\lambda_1) = \frac{1}{2 \pi} \lim_{T \to \infty}  \int_{-
  T}^T \frac{e^{- i \lambda_2 t} - e^{- i \lambda_1 t}}{- it} r (t) 
  \hspace{0.17em} dt
\end{equation}

\subsection*{Recovery of the Orthogonal Random Measure Increment}

The increment of the orthogonal random measure over the interval $(\lambda_1,
\lambda_2]$ is obtained pathwise from the sample trajectory $\xi (t)$ by the
quadratic mean limit
\begin{equation}
  \zeta (\lambda_2) - \zeta (\lambda_1) = \frac{1}{2 \pi}  \text{l.i.m.}_{T
  \to \infty} \int_{- T}^T \frac{e^{- i \lambda_2 t} - e^{- i \lambda_1 t}}{-
  it} \xi (t)  \hspace{0.17em} dt
\end{equation}

\subsection*{The Inverse Fourier Transform Identity}

The kernel is the inverse Fourier transform of the indicator function of the
interval $(\lambda_1, \lambda_2]$:
\begin{equation}
  \textbf{1}_{(\lambda_1, \lambda_2]} (\lambda) = \frac{1}{2 \pi}  \int_{-
  \infty}^{\infty} \frac{e^{- i \lambda_2 t} - e^{- i \lambda_1 t}}{- it}
  e^{it \lambda}  \hspace{0.17em} dt
\end{equation}

\subsection*{Real Part of the Orthogonal Random Measure}

Setting the lower bound to $\lambda_1 = 0$, the real part $u (\lambda)$ of
$\zeta (\lambda)$ is
\begin{equation}
  u (\lambda) = \frac{1}{\pi}  \text{l.i.m.}_{T \to \infty} \int_{- T}^T
  \frac{\sin (\lambda t)}{t} \xi (t)  \hspace{0.17em} dt
\end{equation}

\subsection*{Imaginary Part of the Orthogonal Random Measure}

The imaginary part $v (\lambda)$ of $\zeta (\lambda)$ is
\begin{equation}
  v (\lambda) = \frac{1}{\pi}  \text{l.i.m.}_{T \to \infty} \int_{- T}^T
  \frac{1 - \cos (\lambda t)}{t} \xi (t)  \hspace{0.17em} dt
\end{equation}

\subsection*{The Complete Orthogonal Random Measure}

The full orthogonal random measure is
\begin{equation}
  \zeta (\lambda) = u (\lambda) + iv (\lambda)
\end{equation}
