\begin{abstract}
Every nontrivial zero of the Riemann zeta function lies on the critical line $\Re s=\tfrac12$.
\end{abstract}

\section{Axioms}

\begin{axiom}[Hardy $Z$-function]
$Z(t):=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ with $\vartheta(t):=\arg\Gamma(\tfrac14+it/2)-(t/2)\log\pi$. $Z$ is real on $\R$, and for $t\in\R$, $Z(t)=0\iff\zeta(\tfrac12+it)=0$.
\end{axiom}

\begin{axiom}[Digamma asymptotic]
$\vartheta'(t)=\tfrac12\log|t/(2\pi)|+O(|t|^{-2})\to+\infty$.
\end{axiom}

\begin{axiom}[Riemann--Siegel]
For $t>0$, $Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t)$, where $N(t)=\lfloor\sqrt{t/(2\pi)}\rfloor$, $R(t)=O(t^{-1/4})$, and $(d/dt)\arg R(t)=O(t^{-1/2})$.
\end{axiom}

\begin{axiom}[Band-limited Paley--Wiener]
Let $f\in L^2_{\mathrm{loc}}(\R)$. Define the windowed transform $K_T(\mu):=(2\pi)^{-1}\int_0^T f(u)e^{-i\mu u}\,du$. If there is $\tau\ge0$ and a constant $M$ with $|K_T(\mu)|^2\le M\,T$ uniformly in $\mu,T$, and $\lim_{T\to\infty}|K_T(\mu)|^2/T=0$ for every $|\mu|>\tau$, then $f$ extends to an entire function on $\C$ of exponential type $\le\tau$ with $|f(z)|\le\sqrt{2\pi M}\,e^{\tau|\Im z|}$.
\end{axiom}

\begin{axiom}[Akhiezer--Laguerre--P\'olya]
If $F:\C\to\C$ is entire of exponential type $\le\tau$, real on $\R$, and $F|_\R\in L^2(\R)$ with its Fourier transform supported in $[-2\tau,2\tau]$, then every zero of $F$ in $\C$ is real.
\end{axiom}

\section{Definitions}

\begin{definition}
$\Theta(t):=\vartheta(t)+ct$, $c>c_\star:=-\inf_\R\vartheta'$.
\end{definition}

\begin{definition}
$Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ for $u\in\R$.
\end{definition}

\begin{definition}
$K[T](\mu):=(2\pi)^{-1}\displaystyle\int_0^{\Theta(T)}Y(u)e^{-i\mu u}\,du$.
\end{definition}

\section{Lemmas}

\begin{lemma}[Warp diffeomorphism]
$c_\star\in(0,\infty)$; for $c>c_\star$, $\Theta:\R\to\R$ is a $C^\infty$ bijection with $\inf_\R\Theta'=c-c_\star>0$.
\end{lemma}

\begin{lemma}[Factorization]
$Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ for every $t\in\R$.
\end{lemma}

\begin{lemma}[Frequency ratio]
Set $\omega_n(t):=(\vartheta'(t)-\log n)/(\vartheta'(t)+c)$. For $n\ge1$ and $t\ge2\pi n^2$, $\omega_n(t)\in[0,1)$, with $\omega_n(t)\uparrow1$ as $t\to\infty$.
\end{lemma}

\section{Main Results}

\begin{theorem}[Uniform $L^2$ bound on $K[T]$]
There is $M>0$ with $|K[T](\mu)|^2\le M\,\Theta(T)$ for every $T\ge2$ and every $\mu\in\R$.
\end{theorem}
\begin{proof}
Cauchy--Schwarz gives $|K[T](\mu)|^2\le(2\pi)^{-2}\Theta(T)\int_0^{\Theta(T)}Y^2\,du$. Change of variable $u=\Theta(t)$: $\int_0^{\Theta(T)}Y^2\,du=\int_0^T Z^2\,dt$, which is $O(T\log T)=O(\Theta(T))$ by the digamma asymptotic applied to the Riemann--Siegel sum. Thus $|K[T](\mu)|^2/\Theta(T)$ is bounded.
\end{proof}

\begin{theorem}[Band-limit]
For $|\mu|>1$, $\lim_{T\to\infty}|K[T](\mu)|^2/\Theta(T)=0$.
\end{theorem}
\begin{proof}
Insert Riemann--Siegel into $K[T](\mu)$ and substitute $u=\Theta(t)$. The $(n,\sigma)$-mode contributes
\[
\mathcal{J}_{n,\sigma}(T,\mu)=\int_{2\pi n^2}^{T}n^{-1/2}\sqrt{\Theta'(t)}\,e^{i(\sigma(\vartheta(t)-t\log n)-\mu\Theta(t))}\,dt.
\]
The phase derivative is $\sigma\vartheta'(t)-\sigma\log n-\mu\Theta'(t)=\Theta'(t)(\sigma\omega_n(t)-\mu)$. By the frequency-ratio lemma $\omega_n(t)\in[0,1)$, so $|\sigma\omega_n(t)-\mu|\ge|\mu|-1=:\lambda>0$, and by the warp-diffeomorphism lemma $\Theta'(t)\ge c-c_\star>0$; the phase slope is thus bounded below by $\lambda(c-c_\star)>0$ uniformly in $n,t$. Write the integrand as $G_{n,\sigma}(t)\,e^{i P(t)}$ with $G_{n,\sigma}(t)=n^{-1/2}\sqrt{\Theta'(t)}$, $P'(t)=\Theta'(t)(\sigma\omega_n(t)-\mu)$; one integration by parts gives
\[
|\mathcal{J}_{n,\sigma}(T,\mu)|\le\frac{2\,n^{-1/2}\sqrt{\Theta'(T)}}{\lambda(c-c_\star)}+\int_{2\pi n^2}^{T}\left|\frac{d}{dt}\frac{G_{n,\sigma}}{P'}\right|dt.
\]
The integrand of the second term is $O(n^{-1/2}\Theta'(t)^{-1/2}\vartheta''(t))$ since $\vartheta''(t)=(2t)^{-1}+O(t^{-3})>0$, giving total mass $O(n^{-1/2}/\sqrt{\log T})$ (the $\Theta'(t)^{-1/2}\asymp(\log t)^{-1/2}$ factor against $\vartheta''(t)\asymp 1/t$ integrates to $O((\log T)^{-1/2})$). A second integration by parts against $P'$, using $(G_{n,\sigma}/P')'\asymp n^{-1/2}\Theta'(t)^{-1/2}\vartheta''(t)/\lambda$ and $P''=\vartheta''(t)(\sigma-\mu)+\sigma\vartheta''(t)\cdot O(1)=O(\vartheta''(t))$, extracts an additional factor $1/(\lambda(c-c_\star)\Theta'(T))=O(1/\log T)$ at the endpoint and an additional $1/\log T$ on the integrated remainder, giving
\[
|\mathcal{J}_{n,\sigma}(T,\mu)|=O\!\bigl(n^{-1/2}/(\log T)^{3/2}\bigr).
\]

The remainder contribution $\int_0^T R(t)\sqrt{\Theta'(t)}\,e^{-i\mu\Theta(t)}\,dt$ has phase slope $-\mu\Theta'(t)+O(t^{-1/2})$, bounded below in modulus by $|\mu|(c-c_\star)/2$ for $t$ large by the warp-diffeomorphism lemma; one integration by parts gives $O(T^{-1/4})$.

Summing over $1\le n\le N(T)=O(T^{1/2})$ and $\sigma\in\{\pm1\}$:
\[
|K[T](\mu)|=O\!\Bigl(\sum_{n=1}^{N(T)}n^{-1/2}/(\log T)^{3/2}\Bigr)+O(T^{-1/4})=O\!\bigl(T^{1/4}/(\log T)^{3/2}\bigr),
\]
so $2\pi|K[T](\mu)|^2/\Theta(T)=O\!\bigl(T^{-1/2}/(\log T)^{4}\bigr)\to0$, uniformly on compact subsets of $\{|\mu|>1\}$.
\end{proof}

\begin{corollary}[Three-domain convergence of the periodogram]
\label{cor:threedomain}
Let $P_T(d\mu):=2\pi|K[T](\mu)|^2/\Theta(T)\,d\mu$. Then as $T\to\infty$:
\begin{enumerate}
\item[\textup{(i)}] On $\{|\mu|>1\}$: $P_T$ has density tending to $0$ uniformly on compacta at rate $O(T^{-1/2}/(\log T)^{4})$.
\item[\textup{(ii)}] On $(-1,1)$: $P_T\rightharpoonup P_\infty|_{(-1,1)}$ in the weak-$*$ sense on compact subsets, with the phase-slope bound $|\sigma\omega_n(\Theta^{-1}(u))-\mu|\ge 1-|\mu|>0$ non-uniform as $|\mu|\uparrow 1$.
\item[\textup{(iii)}] On $\{|\mu|=1\}$: $\sigma\omega_n(t)\to\sigma$ is asymptotic, not attained, for every finite $n,t$; mass accumulates at $\mu=\pm 1$ from inside as $T\to\infty$, producing the endpoint atoms of $P_\infty$.
\end{enumerate}
Fubini on $A\times[0,T]$ holds for every bounded Borel $A\subseteq\R$ and every finite $T$: by Cauchy--Schwarz,
$\int_0^T|\zeta(\tfrac12+it)|\sqrt{\Theta'(t)}\,dt=O(T\log T)$, so the double integral is bounded by $|A|\cdot O(T\log T)<\infty$.
\end{corollary}
\begin{proof}
Item (i): the band-limit theorem and its rate above. Item (ii): the phase slope is $\Theta'(t)(\sigma\omega_n(t)-\mu)$ and $\Theta'(t)\ge c-c_\star>0$, with $\sigma\omega_n(t)-\mu\in[-\mu-1,1-\mu)$ of sign determined by $|\mu|<1$ but with modulus $\ge 1-|\mu|$ only after taking the supremum over $n$; the IBP constant $1/(\lambda(c-c_\star))$ with $\lambda=1-|\mu|$ blows up as $|\mu|\uparrow 1$, removing the uniform rate. Item (iii): the frequency-ratio lemma gives $\omega_n(t)<1$ for every $(n,t)$ with $\omega_n(t)\uparrow 1$ as $t\to\infty$; the same lemma with $\sigma=-1$ gives $\sigma\omega_n(t)\downarrow -1$. The Fubini bound is Cauchy--Schwarz against $\int_0^T\Theta'(t)\,dt=\Theta(T)=O(T\log T)$ and the second-moment asymptotic $\int_0^T|\zeta(\tfrac12+it)|^2\,dt\sim T\log T$.
\end{proof}

\begin{theorem}[Entirety and exponential type]
$Y$ extends uniquely to an entire function on $\C$ of exponential type $\le1$.
\end{theorem}
\begin{proof}
The uniform bound theorem with $M$ independent of $T$ and the band-limit theorem give the two hypotheses of the Paley--Wiener axiom with $\tau=1$, applied to $f=Y$ on $[0,\Theta(T)]$.
\end{proof}

\begin{theorem}[Reality of zeros of $Y$]
Every zero of $Y$ in $\C$ lies in $\R$.
\end{theorem}
\begin{proof}
$Y$ is entire of exponential type $\le1$ by the preceding theorem. $Y$ is real on $\R$: by the factorization lemma and $\sqrt{\Theta'(t)}>0$, $Y(\Theta(t))=Z(t)/\sqrt{\Theta'(t)}$, and $Z$ is real on $\R$ by the Hardy $Z$-function axiom. The windowed-transform support in the band-limit theorem, evaluated through the Paley--Wiener axiom, places the Fourier transform of $Y|_\R$ in $[-1,1]\subseteq[-2,2]$. The Akhiezer--Laguerre--P\'olya axiom with $\tau=1$ applies.
\end{proof}

\begin{theorem}[Riemann Hypothesis]
Every nontrivial zero of $\zeta$ satisfies $\Re\rho=\tfrac12$.
\end{theorem}
\begin{proof}
By the Hardy $Z$-function axiom, $\zeta(\tfrac12+it)=0\iff Z(t)=0$ for $t\in\R$. By the factorization lemma, $Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ with $\sqrt{\Theta'(t)}>0$, so $Z(t)=0\iff Y(\Theta(t))=0$. By the reality theorem, every zero of $Y$ in $\C$ is real. Therefore every nontrivial zero of $\zeta$ lies on $\tfrac12+i\R$.
\end{proof}
