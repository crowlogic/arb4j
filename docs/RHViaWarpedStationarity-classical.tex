\begin{abstract}
Every nontrivial zero of the Riemann zeta function lies on the critical line $\Re s=\tfrac12$.
\end{abstract}

\section{Axioms}

\begin{axiom}[Hardy $Z$-function]\label{ax:Z}
$Z(t):=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ with $\vartheta(t):=\arg\Gamma(\tfrac14+it/2)-(t/2)\log\pi$. $Z$ is real on $\R$; for $t\in\R$, $Z(t)=0\iff\zeta(\tfrac12+it)=0$. $Z$ is holomorphic on the strip $S:=\{t\in\C:|\Im t|<\tfrac12\}$ (nearest singularities at $t=\pm i/2$: the $\Gamma$-pole at $\tfrac14+it/2=0$ and the $\zeta$-pole at $\tfrac12+it=1$).
\end{axiom}

\begin{axiom}[Digamma asymptotic]\label{ax:dig}
$\vartheta'(t)=\tfrac12\log|t/(2\pi)|+O(|t|^{-2})$ on $\R$; $c_\star:=-\inf_\R\vartheta'\in(0,\infty)$. $\vartheta$ is holomorphic on $S$.
\end{axiom}

\begin{axiom}[Riemann--Siegel]\label{ax:RS}
For $t>0$, $Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t)$ with $N(t)=\lfloor\sqrt{t/(2\pi)}\rfloor$ and $R(t)=O(t^{-1/4})$.
\end{axiom}

\begin{axiom}[Band-limited Paley--Wiener]\label{ax:PW}
Let $f\in L^2_{\mathrm{loc}}(\R)$ and $K_T(\mu):=(2\pi)^{-1}\int_0^T f(u)e^{-i\mu u}\,du$. If $|K_T(\mu)|^2\le MT$ uniformly in $\mu,T$ for some $M>0$, and $\lim_{T\to\infty}|K_T(\mu)|^2/T=0$ for every $|\mu|>1$, then $f$ extends to an entire function on $\C$ of exponential type $\le 1$ with distributional Fourier transform supported in $[-1,1]$.
\end{axiom}

\begin{axiom}[Akhiezer--Laguerre--P\'olya]\label{ax:AK}
If $F:\C\to\C$ is entire of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$, then every zero of $F$ in $\C$ lies in $\R$.
\end{axiom}

\section{Construction}

\begin{definition}\label{def:Theta}
$\Theta(t):=\vartheta(t)+ct$ for any fixed $c>c_\star$.
\end{definition}

\begin{definition}\label{def:Y}
$Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ for $u\in\R$.
\end{definition}

\begin{definition}\label{def:K}
$K[T](\mu):=(2\pi)^{-1}\int_0^{\Theta(T)}Y(u)e^{-i\mu u}\,du$.
\end{definition}

\begin{lemma}[Warp on $\R$]\label{lem:warp}
$\Theta:\R\to\R$ is a $C^\infty$ bijection with $\inf_\R\Theta'=c-c_\star>0$; $Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ holds for every $t\in\R$ with $\sqrt{\Theta'(t)}>0$.
\end{lemma}
\begin{proof}
$\Theta'=\vartheta'+c\ge c-c_\star>0$ (Axiom~\ref{ax:dig}). The factorization is Definition~\ref{def:Y} solved for $Z$.
\end{proof}

\begin{lemma}[Frequency ratio]\label{lem:freq}
$\omega_n(t):=(\vartheta'(t)-\log n)/\Theta'(t)\in[0,1)$ for $t\ge 2\pi n^2$, with $\omega_n(t)\uparrow 1$ as $t\to\infty$.
\end{lemma}
\begin{proof}
$\vartheta'(t)\ge\log n$ iff $t\ge 2\pi n^2$ (Axiom~\ref{ax:dig}); then $0\le\vartheta'-\log n<\vartheta'+c=\Theta'$. $\vartheta'\to\infty$ gives $\omega_n\to 1$.
\end{proof}

\section{Proof of RH}

\begin{theorem}[Uniform $L^2$ bound]\label{thm:bdd}
There is $M>0$ with $|K[T](\mu)|^2\le M\,\Theta(T)$ for every $T\ge 2$ and every $\mu\in\R$.
\end{theorem}
\begin{proof}
Cauchy--Schwarz: $|K[T](\mu)|^2\le(2\pi)^{-2}\Theta(T)\int_0^{\Theta(T)}Y^2\,du=(2\pi)^{-2}\Theta(T)\int_0^T Z^2\,dt$ by $u=\Theta(t)$. Hardy--Littlewood $\int_0^T Z^2=T\log T+O(T)=O(\Theta(T))$ via Axioms~\ref{ax:dig},~\ref{ax:RS}.
\end{proof}

\begin{theorem}[Band-limit]\label{thm:band}
For $|\mu|>1$, $\lim_{T\to\infty}|K[T](\mu)|^2/\Theta(T)=0$.
\end{theorem}
\begin{proof}
Insert Axiom~\ref{ax:RS} into $K[T](\mu)$, change variables $u=\Theta(t)$. Mode $(n,\sigma)$ contributes
\[
\mathcal{J}_{n,\sigma}(T,\mu)=\int_{2\pi n^2}^{T}n^{-1/2}\sqrt{\Theta'(t)}\,e^{i\Phi(t)}\,dt,\quad\Phi'(t)=\Theta'(t)(\sigma\omega_n(t)-\mu).
\]
By Lemmas~\ref{lem:warp},~\ref{lem:freq}, $|\Phi'(t)|\ge(|\mu|-1)(c-c_\star)>0$. One integration by parts gives $|\mathcal{J}_{n,\sigma}(T,\mu)|=O(n^{-1/2}\sqrt{\log T})$. The remainder contributes $O(T^{-1/4})$. Summing over $1\le n\le N(T)=O(T^{1/2})$ and $\sigma=\pm 1$: $|K[T](\mu)|=O(T^{1/4}\sqrt{\log T})$, so $|K[T](\mu)|^2/\Theta(T)=O((\log T)/\sqrt{T})\to 0$.
\end{proof}

\begin{theorem}[Entirety, band, and real zeros of $Y$]\label{thm:ent}
$Y$ extends uniquely to an entire function on $\C$ of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$; every zero of the extension lies in $\R$.
\end{theorem}
\begin{proof}
Theorems~\ref{thm:bdd},~\ref{thm:band} supply the hypotheses of Axiom~\ref{ax:PW} for $f=Y$ with $\tau=1$; the PW conclusion gives the entire extension, exponential type $\le 1$, and distributional Fourier support in $[-1,1]$. $Y|_\R$ is real: Lemma~\ref{lem:warp} gives $Y\circ\Theta=Z/\sqrt{\Theta'}$, and $Z$ is real on $\R$ by Axiom~\ref{ax:Z}; $\Theta^{-1}:\R\to\R$ is a bijection, so $Y|_\R$ real-valued. Axiom~\ref{ax:AK} applied to this entire extension: every zero lies in $\R$.
\end{proof}

\begin{lemma}[Strip factorization]\label{lem:strip}
There is an open horizontal strip $S'\subset S$ containing $\R$ on which $\Theta$ is a biholomorphism onto its image with $\Theta'\ne 0$, $\sqrt{\Theta'}$ has a holomorphic branch positive on $\R$, and
\[
Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t)),\qquad t\in S'.
\]
\end{lemma}
\begin{proof}
$\Theta=\vartheta+c\cdot\mathrm{id}$ is holomorphic on $S$ (Axioms~\ref{ax:Z},~\ref{ax:dig}). $\Theta'|_\R\ge c-c_\star>0$; by continuity of $\Theta'$ on $S$, there is an open $S'\subset S$ with $\R\subset S'$ and $|\Theta'-\Theta'|_\R|<(c-c_\star)/2$ on $S'$, hence $|\Theta'|\ge(c-c_\star)/2>0$ on $S'$. $\Theta$ is injective on $S'$ (shrinking $S'$ if needed to make it simply connected and applying the open mapping + local injectivity from $\Theta'\ne 0$). $\sqrt{\Theta'}$ lifts to a holomorphic branch on $S'$ positive on $\R$.

Define $g(t):=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ on $S'$. $Y$ is entire (Theorem~\ref{thm:ent}), $\Theta:S'\to\Theta(S')$ holomorphic, so $g$ is holomorphic on $S'$. $Z$ is holomorphic on $S\supset S'$ (Axiom~\ref{ax:Z}). On $\R$, $Z=g$ by Lemma~\ref{lem:warp}. Two holomorphic functions on the connected open $S'$ agreeing on $\R\subset S'$ agree on $S'$ by the identity theorem.
\end{proof}

\begin{theorem}[Riemann Hypothesis]\label{thm:RH}
Every nontrivial zero of $\zeta$ satisfies $\Re s=\tfrac12$.
\end{theorem}
\begin{proof}
Let $\rho=\tfrac12+\alpha+i\beta$ be a nontrivial zero, $\alpha\in(-\tfrac12,\tfrac12)$, $\beta\in\R$. Set $t:=\beta-i\alpha$; then $|\Im t|=|\alpha|<\tfrac12$, so $t\in S'$ (Lemma~\ref{lem:strip}, after shrinking $S'$ to include $t$; $S'$ can be chosen to contain any given horizontal strip strictly inside $S$, so in particular any fixed $|\Im t|<\tfrac12-\varepsilon$; since $\alpha$ is strictly less than $\tfrac12$ in modulus, choose $\varepsilon<\tfrac12-|\alpha|$).

By Axiom~\ref{ax:Z}, $Z(t)=e^{i\vartheta(t)}\zeta(\tfrac12+it)=e^{i\vartheta(t)}\zeta(\rho)=0$. Lemma~\ref{lem:strip} gives $\sqrt{\Theta'(t)}\,Y(\Theta(t))=0$, and $\sqrt{\Theta'(t)}\ne 0$, so $Y(\Theta(t))=0$. Theorem~\ref{thm:ent} gives $\Theta(t)\in\R$. $\Theta$ is a biholomorphism on $S'$ with $\Theta(\R\cap S')=\R\cap\Theta(S')$, so $\Theta^{-1}(\R)\cap S'=\R\cap S'$; hence $t\in\R$, i.e.\ $\alpha=0$.
\end{proof}
