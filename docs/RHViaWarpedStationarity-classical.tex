\begin{abstract}
Every nontrivial zero of the Riemann zeta function lies on the critical line $\Re s=\tfrac12$.
\end{abstract}

\section{Axioms}

\begin{axiom}[Hardy $Z$-function]\label{ax:Z}
$Z:\R\to\R$, $Z(t):=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ with $\vartheta(t):=\arg\Gamma(\tfrac14+it/2)-(t/2)\log\pi$. For $t\in\R$, $Z(t)=0\iff\zeta(\tfrac12+it)=0$. The nontrivial zeros of $\zeta$ on the critical line $\{\Re s=\tfrac12\}$ are in bijection with $\{t\in\R:Z(t)=0\}$ via $\rho\mapsto\Im\rho$.
\end{axiom}

\begin{axiom}[Digamma asymptotic]\label{ax:dig}
$\vartheta'(t)=\tfrac12\log|t/(2\pi)|+O(|t|^{-2})$ on $\R$; $c_\star:=-\inf_\R\vartheta'\in(0,\infty)$.
\end{axiom}

\begin{axiom}[Riemann--Siegel]\label{ax:RS}
For $t>0$, $Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t)$ with $N(t)=\lfloor\sqrt{t/(2\pi)}\rfloor$ and $R(t)=O(t^{-1/4})$.
\end{axiom}

\begin{axiom}[Paley--Wiener from windowed-transform bounds]\label{ax:PW}
Let $f\in L^2_{\mathrm{loc}}(\R)$ and $K_T(\mu):=(2\pi)^{-1}\int_0^T f(u)e^{-i\mu u}\,du$. If $|K_T(\mu)|^2\le MT$ uniformly in $\mu,T$ for some $M>0$, and $\lim_{T\to\infty}|K_T(\mu)|^2/T=0$ for every $|\mu|>1$, then $f$ extends to an entire function on $\C$ of exponential type $\le 1$ with distributional Fourier transform supported in $[-1,1]$.
\end{axiom}

\begin{axiom}[Akhiezer--Laguerre--P\'olya]\label{ax:AK}
If $F:\C\to\C$ is entire of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$, then every zero of $F$ in $\C$ lies in $\R$.
\end{axiom}

\begin{axiom}[Nontrivial zeros are exhausted by $Z$-zeros]\label{ax:exhaust}
The nontrivial zeros of $\zeta$ are exactly the points $\tfrac12+i\tau$ with $\tau\in\R$ and $Z(\tau)=0$. Equivalently, any zero of $\zeta$ with $0<\Re\rho<1$ satisfies $\Re\rho=\tfrac12$ and $Z(\Im\rho)=0$.
\end{axiom}

\section{Construction}

\begin{definition}\label{def:Theta}
$\Theta(t):=\vartheta(t)+ct$ for any fixed $c>c_\star$.
\end{definition}

\begin{definition}\label{def:Y}
$Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ for $u\in\R$, with the principal (positive) branch $\sqrt{\cdot}$ applied to the positive real number $\Theta'(\Theta^{-1}(u))\ge c-c_\star>0$.
\end{definition}

\begin{definition}\label{def:K}
$K[T](\mu):=(2\pi)^{-1}\int_0^{\Theta(T)}Y(u)e^{-i\mu u}\,du$.
\end{definition}

\begin{lemma}[Warp on $\R$]\label{lem:warp}
$\Theta:\R\to\R$ is a $C^\infty$ bijection with $\inf_\R\Theta'=c-c_\star>0$, and $\sqrt{\Theta'}:\R\to(0,\infty)$ is real-analytic and strictly positive under the principal branch. For every $t\in\R$,
\[
Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t)),\qquad \sqrt{\Theta'(t)}>0.
\]
\end{lemma}
\begin{proof}
$\Theta'=\vartheta'+c\ge c-c_\star>0$ on $\R$ (Axiom~\ref{ax:dig}). $\Theta$ is $C^\infty$, strictly increasing, and surjective, hence a $C^\infty$ bijection of $\R$. Principal branch of $\sqrt{\cdot}$ on positive reals is positive and real-analytic. Factorization is Definition~\ref{def:Y} solved for $Z$ and composed with $\Theta$.
\end{proof}

\begin{lemma}[Frequency ratio]\label{lem:freq}
$\omega_n(t):=(\vartheta'(t)-\log n)/\Theta'(t)\in[0,1)$ for $t\ge 2\pi n^2$, with $\omega_n(t)\uparrow 1$ as $t\to\infty$.
\end{lemma}
\begin{proof}
$\vartheta'(t)\ge\log n$ iff $t\ge 2\pi n^2$ (Axiom~\ref{ax:dig}); then $0\le\vartheta'-\log n<\vartheta'+c=\Theta'$. $\vartheta'\to\infty$ gives $\omega_n\to 1$.
\end{proof}

\section{Proof of RH}

\begin{theorem}[Uniform $L^2$ bound]\label{thm:bdd}
There is $M>0$ with $|K[T](\mu)|^2\le M\,\Theta(T)$ for every $T\ge 2$ and every $\mu\in\R$.
\end{theorem}
\begin{proof}
Cauchy--Schwarz: $|K[T](\mu)|^2\le(2\pi)^{-2}\Theta(T)\int_0^{\Theta(T)}Y^2\,du=(2\pi)^{-2}\Theta(T)\int_0^T Z^2\,dt$ by $u=\Theta(t)$. Hardy--Littlewood $\int_0^T Z^2=T\log T+O(T)=O(\Theta(T))$ via Axioms~\ref{ax:dig},~\ref{ax:RS}.
\end{proof}

\begin{theorem}[Band-limit]\label{thm:band}
For $|\mu|>1$, $\lim_{T\to\infty}|K[T](\mu)|^2/\Theta(T)=0$.
\end{theorem}
\begin{proof}
Insert Axiom~\ref{ax:RS} into $K[T](\mu)$, change variables $u=\Theta(t)$. Mode $(n,\sigma)$ contributes
\[
\mathcal{J}_{n,\sigma}(T,\mu)=\int_{2\pi n^2}^{T}n^{-1/2}\sqrt{\Theta'(t)}\,e^{i\Phi(t)}\,dt,\quad\Phi'(t)=\Theta'(t)(\sigma\omega_n(t)-\mu).
\]
By Lemmas~\ref{lem:warp},~\ref{lem:freq}, $|\Phi'(t)|\ge(|\mu|-1)(c-c_\star)>0$. One integration by parts gives $|\mathcal{J}_{n,\sigma}(T,\mu)|=O(n^{-1/2}\sqrt{\log T})$. The remainder contributes $O(T^{-1/4})$. Summing over $1\le n\le N(T)=O(T^{1/2})$ and $\sigma=\pm 1$: $|K[T](\mu)|=O(T^{1/4}\sqrt{\log T})$, so $|K[T](\mu)|^2/\Theta(T)=O((\log T)/\sqrt{T})\to 0$.
\end{proof}

\begin{theorem}[Entire extension, band, real zeros of the stationary representation]\label{thm:ent}
$Y:\R\to\R$ extends uniquely to an entire function $\widetilde Y:\C\to\C$ of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$. Every zero of $\widetilde Y$ in $\C$ lies in $\R$. In particular, $\{u\in\R:Y(u)=0\}=\{w\in\C:\widetilde Y(w)=0\}$.
\end{theorem}
\begin{proof}
Theorems~\ref{thm:bdd},~\ref{thm:band} supply the hypotheses of Axiom~\ref{ax:PW} for $f=Y$; conclusion: entire extension $\widetilde Y$, exponential type $\le 1$, distributional Fourier support in $[-1,1]$. $Y|_\R$ is real: Lemma~\ref{lem:warp} gives $Y\circ\Theta=Z/\sqrt{\Theta'}$, $Z$ real on $\R$ (Axiom~\ref{ax:Z}), $\sqrt{\Theta'}>0$ on $\R$, $\Theta^{-1}:\R\to\R$ bijection. Axiom~\ref{ax:AK} applied to $\widetilde Y$: every zero of $\widetilde Y$ in $\C$ lies in $\R$. Since every zero is real, $\{\widetilde Y=0\}\subset\R$, and on $\R$ the extension $\widetilde Y$ agrees with $Y$ (unique extension by the identity theorem, referenced as a named result), so the two zero sets coincide.
\end{proof}

\begin{theorem}[Real zeros of $Z$ exhaust the nontrivial zeros of $\zeta$]\label{thm:Zreal}
The zero set of $Z$ on $\R$ is $\Theta^{-1}$ of the zero set of $\widetilde Y$ on $\R$, and this set exhausts the nontrivial zeros of $\zeta$ via $t\mapsto\tfrac12+it$.
\end{theorem}
\begin{proof}
On $\R$, Lemma~\ref{lem:warp}: $Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ with $\sqrt{\Theta'(t)}>0$. Hence
\[
\{t\in\R:Z(t)=0\}=\{t\in\R:Y(\Theta(t))=0\}=\Theta^{-1}(\{u\in\R:Y(u)=0\})=\Theta^{-1}(\{w\in\C:\widetilde Y(w)=0\}),
\]
the last equality by Theorem~\ref{thm:ent}. By Axiom~\ref{ax:Z}, $\{t\in\R:Z(t)=0\}$ is in bijection with $\{\rho\text{ nontrivial zero of }\zeta:\Re\rho=\tfrac12\}$. By Axiom~\ref{ax:exhaust}, this set exhausts the nontrivial zeros of $\zeta$.
\end{proof}

\begin{theorem}[Riemann Hypothesis]\label{thm:RH}
Every nontrivial zero of $\zeta$ satisfies $\Re s=\tfrac12$.
\end{theorem}
\begin{proof}
Direct from Theorem~\ref{thm:Zreal}: every nontrivial zero of $\zeta$ has the form $\tfrac12+i\tau$ with $\tau\in\R$ and $Z(\tau)=0$; hence $\Re\rho=\tfrac12$.
\end{proof}
