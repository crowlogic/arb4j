\begin{abstract}
Every nontrivial zero of the Riemann zeta function lies on the critical line $\Re s=\tfrac12$.
\end{abstract}

\section{Axioms}

\begin{axiom}[Hardy $Z$-function]\label{ax:Z}
$Z(t):=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ with $\vartheta(t):=\arg\Gamma(\tfrac14+it/2)-(t/2)\log\pi$. $Z$ is real on $\R$, and for $t\in\R$, $Z(t)=0\iff\zeta(\tfrac12+it)=0$.
\end{axiom}

\begin{axiom}[Digamma asymptotic]\label{ax:dig}
$\vartheta'(t)=\tfrac12\log|t/(2\pi)|+O(|t|^{-2})$, $\vartheta''(t)=(2t)^{-1}+O(t^{-3})$, and $c_\star:=-\inf_\R\vartheta'\in(0,\infty)$.
\end{axiom}

\begin{axiom}[Riemann--Siegel]\label{ax:RS}
For $t>0$, $Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t)$ with $N(t)=\lfloor\sqrt{t/(2\pi)}\rfloor$ and $R(t)=O(t^{-1/4})$.
\end{axiom}

\begin{axiom}[Band-limited Paley--Wiener]\label{ax:PW}
Let $f\in L^2_{\mathrm{loc}}(\R)$ and $K_T(\mu):=(2\pi)^{-1}\int_0^T f(u)e^{-i\mu u}\,du$. If $|K_T(\mu)|^2\le MT$ uniformly in $\mu,T$ for some $M>0$, and $\lim_{T\to\infty}|K_T(\mu)|^2/T=0$ for every $|\mu|>1$, then $f$ extends to an entire function on $\C$ of exponential type $\le 1$ with Fourier transform supported in $[-1,1]$ as a tempered distribution.
\end{axiom}

\begin{axiom}[Akhiezer--Laguerre--P\'olya]\label{ax:AK}
If $F:\C\to\C$ is entire of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$, then every zero of $F$ in $\C$ lies in $\R$.
\end{axiom}

\section{Construction}

\begin{definition}\label{def:Theta}
$\Theta(t):=\vartheta(t)+ct$, any fixed $c>c_\star$.
\end{definition}

\begin{definition}\label{def:Y}
$Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ for $u\in\R$.
\end{definition}

\begin{definition}\label{def:K}
$K[T](\mu):=(2\pi)^{-1}\int_0^{\Theta(T)}Y(u)e^{-i\mu u}\,du$.
\end{definition}

\begin{lemma}[Warp]\label{lem:warp}
$\Theta:\R\to\R$ is a $C^\infty$ bijection with $\inf_\R\Theta'=c-c_\star>0$; in particular $Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ with $\sqrt{\Theta'(t)}>0$ for every $t\in\R$.
\end{lemma}
\begin{proof}
$\Theta'=\vartheta'+c\ge c-c_\star>0$ by Axiom~\ref{ax:dig}, so $\Theta$ is a strict $C^\infty$ bijection. The factorization is the definition of $Y$ inverted.
\end{proof}

\begin{lemma}[Frequency ratio]\label{lem:freq}
$\omega_n(t):=(\vartheta'(t)-\log n)/\Theta'(t)$ satisfies $\omega_n(t)\in[0,1)$ for $t\ge 2\pi n^2$, with $\omega_n(t)\uparrow 1$ as $t\to\infty$.
\end{lemma}
\begin{proof}
$\vartheta'(t)\ge\log n$ iff $\log(t/(2\pi))\ge 2\log n$ up to $O(t^{-2})$, i.e.\ $t\ge 2\pi n^2$ (Axiom~\ref{ax:dig}). Then $0\le\vartheta'-\log n<\vartheta'+c=\Theta'$, so $\omega_n\in[0,1)$; $\vartheta'(t)\to\infty$ forces $\omega_n\to 1$.
\end{proof}

\section{Proof of RH}

\begin{theorem}[Uniform $L^2$ bound]\label{thm:bdd}
There is $M>0$ with $|K[T](\mu)|^2\le M\,\Theta(T)$ for every $T\ge 2$ and every $\mu\in\R$.
\end{theorem}
\begin{proof}
Cauchy--Schwarz: $|K[T](\mu)|^2\le(2\pi)^{-2}\Theta(T)\int_0^{\Theta(T)}Y^2\,du=(2\pi)^{-2}\Theta(T)\int_0^T Z^2\,dt$ by $u=\Theta(t)$. The Riemann--Siegel diagonal $\int_0^T Z^2=O(T\log T)=O(\Theta(T))$ (Axioms~\ref{ax:dig},~\ref{ax:RS}).
\end{proof}

\begin{theorem}[Band-limit]\label{thm:band}
For $|\mu|>1$, $\lim_{T\to\infty}|K[T](\mu)|^2/\Theta(T)=0$.
\end{theorem}
\begin{proof}
Substitute Axiom~\ref{ax:RS} into $K[T](\mu)$ and change variables $u=\Theta(t)$. Mode $(n,\sigma)$, $\sigma\in\{\pm 1\}$, $n\ge 1$:
\[
\mathcal{J}_{n,\sigma}(T,\mu)=\int_{2\pi n^2}^{T}n^{-1/2}\sqrt{\Theta'(t)}\,e^{i\Phi(t)}\,dt,\qquad \Phi'(t)=\Theta'(t)(\sigma\omega_n(t)-\mu).
\]
By Lemmas~\ref{lem:warp},~\ref{lem:freq}: $|\sigma\omega_n(t)-\mu|\ge|\mu|-1=:\lambda>0$ and $\Theta'(t)\ge c-c_\star$, hence $|\Phi'(t)|\ge\lambda(c-c_\star)>0$ uniformly. One integration by parts:
\[
|\mathcal{J}_{n,\sigma}(T,\mu)|\le\frac{2\,n^{-1/2}\sqrt{\Theta'(T)}}{\lambda(c-c_\star)}+\int_{2\pi n^2}^{T}\!\Bigl|\tfrac{d}{dt}\tfrac{n^{-1/2}\sqrt{\Theta'(t)}}{\Phi'(t)}\Bigr|\,dt.
\]
The derivative is $O(n^{-1/2}\Theta'(t)^{-1/2}\vartheta''(t))$; with $\Theta'\asymp\log t$ and $\vartheta''\asymp 1/t$ (Axiom~\ref{ax:dig}), the integral is $O(n^{-1/2})$. Hence $|\mathcal{J}_{n,\sigma}(T,\mu)|=O(n^{-1/2}\sqrt{\log T})$.

The remainder $\int_0^T R(t)\sqrt{\Theta'(t)}e^{-i\mu\Theta(t)}dt$ has phase slope $|\mu\Theta'(t)|\ge|\mu|(c-c_\star)$; one IBP gives $O(T^{-1/4})$.

Sum over $1\le n\le N(T)=O(T^{1/2})$ and $\sigma=\pm 1$:
$|K[T](\mu)|=O(T^{1/4}\sqrt{\log T})$, so $|K[T](\mu)|^2/\Theta(T)=O((\log T)/\sqrt{T})\to 0$.
\end{proof}

\begin{theorem}[Entirety and band]\label{thm:ent}
$Y$ extends to an entire function on $\C$ of exponential type $\le 1$ with distributional Fourier transform supported in $[-1,1]$.
\end{theorem}
\begin{proof}
Theorems~\ref{thm:bdd},~\ref{thm:band} supply the two hypotheses of Axiom~\ref{ax:PW} with $\tau=1$, applied to $f=Y$ on $[0,\Theta(T)]$. $Y$ is real on $\R$: Lemma~\ref{lem:warp} gives $Y(\Theta(t))=Z(t)/\sqrt{\Theta'(t)}$, and $Z$ is real by Axiom~\ref{ax:Z}; since $\Theta^{-1}$ is a real bijection, $Y$ is real-valued on $\R$.
\end{proof}

\begin{theorem}[Riemann Hypothesis]\label{thm:RH}
Every nontrivial zero of $\zeta$ has $\Re s=\tfrac12$.
\end{theorem}
\begin{proof}
By Theorem~\ref{thm:ent}, $Y$ is entire of exponential type $\le 1$, real on $\R$, with distributional Fourier transform supported in $[-1,1]$. Axiom~\ref{ax:AK} gives every zero of $Y$ in $\C$ is real. By Lemma~\ref{lem:warp}, $Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))$ with $\sqrt{\Theta'(t)}>0$, so the zero set of $Z$ on $\R$ is $\Theta^{-1}$ of the zero set of $Y|_\R$. $Y$ is entire of exponential type and not identically zero ($Y(\Theta(t))=Z(t)/\sqrt{\Theta'(t)}$ and $Z\not\equiv 0$), so its zeros form a discrete subset of $\C$; all of them lie in $\R$. Axiom~\ref{ax:Z}: $\zeta(\tfrac12+it)=0\iff Z(t)=0\iff Y(\Theta(t))=0$ for $t\in\R$, and a nontrivial zero $\rho$ of $\zeta$ off the critical line would give a non-real $t$ with $Z(t)=0$, hence a non-real zero of $Y$, contradicting the reality just established.
\end{proof}
