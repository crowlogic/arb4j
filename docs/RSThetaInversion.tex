\title{Lagrange Inversion Formula for the Riemann-Siegel Theta Function}
\maketitle

\begin{theorem}[Lagrange Inversion for $\theta(t)$]
Let $\theta(t) = \sum_{k=0}^{\infty} a_{2k+1} t^{2k+1}$ with $a_1 = -\frac{1}{2}\log\pi \neq 0$. Define $\phi(t) = \frac{t}{\theta(t)} = \sum_{j=0}^{\infty} \phi_{2j} t^{2j}$ with $\phi_0 = a_1^{-1}$. The compositional inverse $t(w)$ satisfying $\theta(t(w)) = w$ is
\[
t(w) = \sum_{n=1}^{\infty} c_n w^n
\]
where
\[
c_n = \frac{1}{n(n-1)!} \frac{d^{n-1}}{dt^{n-1}} (\phi(t))^n \bigg|_{t=0}.
\]
\end{theorem}

\begin{proof}
Since $\theta(0) = 0$ and $\theta'(0) = a_1 \neq 0$, the function $\theta$ admits a local analytic inverse. By the Lagrange-BÃ¼rmann theorem,
\[
c_n = \frac{1}{n(n-1)!} \frac{d^{n-1}}{dt^{n-1}} (\phi(t))^n \bigg|_{t=0}.
\]

Since $\phi(t) = \sum_{j=0}^{\infty} \phi_{2j} t^{2j}$ is even, the power $(\phi(t))^n$ has Taylor expansion
\[
(\phi(t))^n = \sum_{\ell=0}^{\infty} \left(\sum_{\substack{\kappa \in (\mathbb{Z}_{\geq 0})^{\mathbb{Z}_{\geq 0}} \\ \sum_{i=0}^{\infty} \kappa(i) = n \\ \sum_{i=0}^{\infty} 2i \kappa(i) = 2\ell}} \frac{n!}{\prod_{i=0}^{\infty} \kappa(i)!} \prod_{i=0}^{\infty} \phi_{2i}^{\kappa(i)}\right) t^{2\ell}
\]
where the inner sum is over all functions $\kappa \colon \mathbb{Z}_{\geq 0} \to \mathbb{Z}_{\geq 0}$ with finite support satisfying the indicated constraints.

The derivative evaluation gives
\[
\frac{d^{n-1}}{dt^{n-1}} (\phi(t))^n \bigg|_{t=0} = (n-1)! \sum_{\substack{\kappa \in (\mathbb{Z}_{\geq 0})^{\mathbb{Z}_{\geq 0}} \\ \sum_{i=0}^{\infty} \kappa(i) = n \\ \sum_{i=0}^{\infty} 2i \kappa(i) = n-1}} \frac{n!}{\prod_{i=0}^{\infty} \kappa(i)!} \prod_{i=0}^{\infty} \phi_{2i}^{\kappa(i)}
\]
when $n$ is odd, and equals zero when $n$ is even.

Therefore
\[
c_n = \frac{1}{n} \sum_{\substack{\kappa \in (\mathbb{Z}_{\geq 0})^{\mathbb{Z}_{\geq 0}} \\ \sum_{i=0}^{\infty} \kappa(i) = n \\ \sum_{i=0}^{\infty} 2i \kappa(i) = n-1}} \frac{n!}{\prod_{i=0}^{\infty} \kappa(i)!} \prod_{i=0}^{\infty} \phi_{2i}^{\kappa(i)}
\]
for odd $n$, and $c_n = 0$ for even $n$.
\end{proof}
