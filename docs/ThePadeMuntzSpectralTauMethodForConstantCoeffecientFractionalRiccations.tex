\title{The Pad\'e--M\"untz Method for\\
Constant-Coefficient Fractional Riccati Equations}
\author{Stephen Crowley}
\date{April 2026}
\maketitle

\begin{abstract}
A purely algebraic, discretization-free solution method is developed for the
general constant-coefficient fractional Riccati equation
$D^{\mu} y = c_0 + c_1 y + c_2 y^2$
of Caputo order $\mu \in (0,1)$ with initial condition $I^{1-\mu}y(0)=0$.
Because the M\"untz lattice $\{t^{k\mu}\}_{k\ge 1}$ is closed under the Caputo
derivative, fractional integration, and pointwise multiplication, the M\"untz
spectral Tau substitution yields an explicit Puiseux expansion
$y(t)=\sum_{k\ge 1}a_k t^{k\mu}$ whose coefficients satisfy a closed
Gamma-ratio convolution recurrence. The induced power series in the variable
$z=t^{\mu}$ has a strictly positive radius of convergence $R$, established by
a Banach fixed-point argument on the equivalent Volterra equation, and is
finite for generic parameters because the complexified Riccati flow develops
movable singularities in the complex $z$-plane.

The Cole--Hopf substitution $g=-\mathcal{D}w/(c_2 w)$, where $\mathcal{D}$
denotes the M\"untz--Caputo operator in the $z$-coordinate, linearises the
Riccati equation to the constant-coefficient linear equation
$\mathcal{D}^2 w - c_1\mathcal{D}w + c_0c_2\,w = 0$,
whose fundamental solutions are Mittag-Leffler functions $\ML{\mu}{1}(\lambda_\pm z)$,
both entire of order $1/\mu$. Consequently $g$ is the ratio of two entire
functions, hence \emph{meromorphic on all of $\mathbb{C}$}, with poles precisely
at the zeros of the Mittag-Leffler combination $w$. This meromorphy is
unconditional and established by direct construction, not by hypothesis.
Global-in-$t$ convergence on the positive real axis is recovered by re-summing
the Puiseux series via the diagonal $[M/M]$ Pad\'e approximant in $z$,
constructed from the M\"untz coefficients through a single Hankel linear
system $H_M\mathbf{q}=-\mathbf{b}$. Under the de~Montessus de~Ballore and
Nuttall--Pommerenke theorems the diagonal Pad\'e sequence converges to the
analytic continuation of $y$ for all $t\ge 0$, and a computable a-posteriori
error bound in terms of successive Pad\'e differences is derived from a
uniform geometric tail estimate.
\end{abstract}

\subsection{Riemann--Liouville integral and Caputo derivative}

\begin{definition}[Riemann--Liouville integral]\label{def:RL}
Let $r>0$ and $f\in L^1_{\mathrm{loc}}[0,T]$. Define
\begin{equation}\label{eq:RL_int}
  I^r f(t) \assign \frac{1}{\Gamma(r)}\int_0^t (t-s)^{r-1}f(s)\,\dd s
  \qquad \forall\, t\in[0,T].
\end{equation}
\end{definition}

\begin{definition}[Caputo derivative]\label{def:Caputo}
Let $\mu\in(0,1)$ and $f$ satisfy $I^{1-\mu}f\in C^1[0,T]$. Define
\begin{equation}\label{eq:caputo}
  D^{\mu}f(t) \assign \frac{\dd}{\dd t}\!\bigl(I^{1-\mu}f(t)\bigr)
               -\frac{f(0)}{\Gamma(1-\mu)}\,t^{-\mu}
  \qquad \forall\,t\in(0,T].
\end{equation}
The fundamental inversion relations are
\begin{align}
  I^{\mu}D^{\mu}f(t) &= f(t)-f(0), \label{eq:caputo_RL}\\
  D^{\mu}I^{\mu}f(t) &= f(t).       \label{eq:DI}
\end{align}
\end{definition}

\begin{remark}[Vanishing correction for the Riccati solution]\label{rem:correction}
For the fractional Riccati equation \eqref{eq:frac_riccati} below, the initial
condition $I^{1-\mu}y(0)=0$ is consistent with the M\"untz ansatz
$y(t)=\sum_{k\ge 1}a_k t^{k\mu}$, since by \eqref{eq:Ir_power} below
$I^{1-\mu}[t^{k\mu}]\big|_{t=0}=0$ for all $k\ge 1$.
The same ansatz gives $y(0)=0$, so the correction term
$-f(0)\,t^{-\mu}/\Gamma(1-\mu)$ in \eqref{eq:caputo} vanishes identically
along the Riccati solution, and $D^{\mu}y(t)=\frac{\dd}{\dd t}I^{1-\mu}y(t)$
throughout.
\end{remark}

\begin{proposition}[Action on powers]\label{prop:powers}
Let $s>-1$ and $r>0$. Then
\begin{equation}\label{eq:Ir_power}
  I^r t^s = \frac{\Gamma(s+1)}{\Gamma(s+r+1)}\,t^{s+r}.
\end{equation}
Let $\mu\in(0,1)$ and $s>0$. Then
\begin{equation}\label{eq:Dmu_power}
  D^{\mu}t^s = \frac{\Gamma(s+1)}{\Gamma(s+1-\mu)}\,t^{s-\mu}.
\end{equation}
\end{proposition}

\begin{proof}
\textbf{Proof of \eqref{eq:Ir_power}.}
By definition,
\[
  I^r t^s = \frac{1}{\Gamma(r)}\int_0^t(t-u)^{r-1}u^s\,\dd u.
\]
Substitute $u=tx$, $\dd u=t\,\dd x$:
\[
  I^r t^s
  = \frac{t^{s+r}}{\Gamma(r)}\int_0^1(1-x)^{r-1}x^s\,\dd x
  = \frac{t^{s+r}}{\Gamma(r)}\cdot B(s+1,r)
  = \frac{t^{s+r}}{\Gamma(r)}\cdot\frac{\Gamma(s+1)\Gamma(r)}{\Gamma(s+r+1)}
  = \frac{\Gamma(s+1)}{\Gamma(s+r+1)}\,t^{s+r}.
\]

\textbf{Proof of \eqref{eq:Dmu_power}.}
Since $s>0$, $f(0)=0$ and the Caputo correction term vanishes.
Hence $D^{\mu}t^s = \frac{\dd}{\dd t}I^{1-\mu}t^s$.
Applying \eqref{eq:Ir_power} with exponent $1-\mu$:
\[
  I^{1-\mu}t^s = \frac{\Gamma(s+1)}{\Gamma(s+2-\mu)}\,t^{s+1-\mu}.
\]
Differentiating and using $\Gamma(s+2-\mu)=(s+1-\mu)\Gamma(s+1-\mu)$:
\[
  \frac{\dd}{\dd t}\left[\frac{\Gamma(s+1)}{\Gamma(s+2-\mu)}\,t^{s+1-\mu}\right]
  = \frac{\Gamma(s+1)(s+1-\mu)}{\Gamma(s+2-\mu)}\,t^{s-\mu}
  = \frac{\Gamma(s+1)}{\Gamma(s+1-\mu)}\,t^{s-\mu}.\qedhere
\]
\end{proof}

\begin{remark}[The case $s=0$]\label{rem:Dmu_const}
The hypothesis $s>0$ in \eqref{eq:Dmu_power} cannot be relaxed to $s\ge 0$.
For $f\equiv 1$: $f(0)=1$, and the correction term cancels the naive
application of \eqref{eq:Dmu_power}, giving $D^{\mu}(1)=0$, consistent with
the Caputo convention.
\end{remark}

\subsection{The M\"untz basis}

Fix $\mu\in(0,1)$ and $T>0$.

\begin{definition}[M\"untz system]\label{def:muntz}
For $k\in\N$, set $e_k(t)\assign t^{k\mu}$ on $[0,T]$, and
\[
  \mathcal{M}\assign\mathrm{span}\{1,\,t^{\mu},\,t^{2\mu},\ldots\}.
\]
\end{definition}

\begin{remark}[Completeness]\label{rem:muntz_complete}
By the M\"untz--Sz\'asz theorem, $\{t^{k\mu}\}_{k\ge 1}$ is complete in
$L^2[0,1]$ iff $\sum_{k\ge 1}(k\mu)^{-1}=\infty$. Since $\sum k^{-1}=\infty$,
completeness holds for every $\mu\in(0,1)$.
\end{remark}

\begin{proposition}[Closure of $\mathcal{M}$]\label{prop:closure_M}
For $k\ge 1$ and $r>0$,
\begin{align}
  D^{\mu}t^{k\mu} &= \frac{\Gamma(k\mu+1)}{\Gamma((k-1)\mu+1)}\,t^{(k-1)\mu},
  \label{eq:Dmu_Muntz}\\
  I^r t^{k\mu} &= \frac{\Gamma(k\mu+1)}{\Gamma(k\mu+r+1)}\,t^{k\mu+r},
  \label{eq:Ir_Muntz}\\
  t^{j\mu}\cdot t^{k\mu} &= t^{(j+k)\mu}.
  \label{eq:prod_Muntz}
\end{align}
Consequently $D^{\mu}\mathcal{M}\subset\mathcal{M}$,
$I^r\mathcal{M}\subset\mathcal{M}$, and $\mathcal{M}$ is closed under
multiplication.
\end{proposition}

\begin{proof}
\eqref{eq:Dmu_Muntz} is Proposition~\ref{prop:powers} with $s=k\mu>0$.
\eqref{eq:Ir_Muntz} is \eqref{eq:Ir_power} with $s=k\mu>-1$.
\eqref{eq:prod_Muntz} is immediate from the exponent rule.
\end{proof}

\section{The Constant-Coefficient Fractional Riccati Equation}
\label{sec:riccati}

\subsection{Equation and Volterra representation}

Let $\mu\in(0,1)$ and $c_0,c_1,c_2\in\mathbb{C}$. Consider
\begin{align}
  D^{\mu}y(t) &= c_0 + c_1\,y(t) + c_2\,y(t)^2,
  \label{eq:frac_riccati}\\
  I^{1-\mu}y(0) &= 0. \label{eq:IC}
\end{align}
Applying $I^{\mu}$ to \eqref{eq:frac_riccati} and using \eqref{eq:caputo_RL}
with $y(0)=0$ gives the equivalent Volterra equation
\begin{equation}\label{eq:frac_riccati_V}
  y(t) = \frac{1}{\Gamma(\mu)}\int_0^t(t-s)^{\mu-1}
         \bigl(c_0+c_1\,y(s)+c_2\,y(s)^2\bigr)\dd s.
\end{equation}

\begin{remark}[H\"older regularity at the origin]\label{rem:regularity}
Evaluating \eqref{eq:frac_riccati_V} at $t=0$ gives $y(0)=0$. Near the
origin, $y(t)=\frac{c_0}{\Gamma(\mu+1)}t^{\mu}+\mathcal{O}(t^{2\mu})$, so
$y$ is H\"older continuous of order $\mu$ but not $C^1$ at $0$. The ordinary
Taylor series in $t$ therefore fails, and the natural local representation is
a Puiseux expansion in $t^\mu$.
\end{remark}

\subsection{The M\"untz--Tau Puiseux series solution}

\begin{theorem}[Series solution in $\{t^{k\mu}\}$]\label{thm:riccati_series}
There exist $R>0$ and $(a_k)_{k\ge 1}\subset\mathbb{C}$ such that
\begin{equation}\label{eq:y_series}
  y(t) = \sum_{k=1}^{\infty}a_k\,t^{k\mu},\qquad |t|<R^{1/\mu},
\end{equation}
with
\begin{align}
  a_1 &= \frac{c_0}{\Gamma(\mu+1)},
  \label{eq:a1_const}\\
  a_{k+1} &= \frac{\Gamma(k\mu+1)}{\Gamma((k+1)\mu+1)}
    \Biggl(c_1\,a_k + c_2\!\!\!\!\sum_{\substack{j+\ell=k\\1\le j,\ell\le k-1}}
    \!\!\!\! a_j\,a_{\ell}\Biggr)
  \quad\forall\,k\ge 1.
  \label{eq:ak_const}
\end{align}
\end{theorem}

\begin{remark}[Quadratic convolution at $k=1$]\label{rem:conv_k1}
For $k=1$ the convolution sum is empty (no pairs $(j,\ell)$ with $j,\ell\ge 1$
and $j+\ell=1$), so
\[
  a_2 = \frac{\Gamma(\mu+1)}{\Gamma(2\mu+1)}\,c_1 a_1.
\]
The quadratic term first contributes at $k=2$:
\[
  a_3 = \frac{\Gamma(2\mu+1)}{\Gamma(3\mu+1)}\bigl(c_1 a_2+c_2 a_1^2\bigr).
\]
\end{remark}

\begin{proof}
Write $y(t)=\sum_{k\ge 1}a_k t^{k\mu}$. By \eqref{eq:prod_Muntz},
\[
  y(t)^2 = \sum_{m=2}^{\infty}b_m t^{m\mu},\qquad
  b_m\assign\sum_{\substack{j+\ell=m\\1\le j,\ell\le m-1}}a_j a_{\ell},
  \quad b_1=0.
\]
Substitute into \eqref{eq:frac_riccati_V} and apply \eqref{eq:Ir_power}
with $r=\mu$:
\[
  \frac{1}{\Gamma(\mu)}\int_0^t(t-s)^{\mu-1}s^{k\mu}\,\dd s
  = \frac{\Gamma(k\mu+1)}{\Gamma((k+1)\mu+1)}\,t^{(k+1)\mu}.
\]
For the constant term ($s^0=1$), this gives $t^{\mu}/\Gamma(\mu+1)$.
Matching the coefficient of $t^{n\mu}$ on both sides: for $n=1$,
\[
  a_1 = \frac{c_0}{\Gamma(\mu+1)},
\]
recovering \eqref{eq:a1_const}. For $n\ge 2$, reindexing ($k\mapsto n-1$,
$m\mapsto n-1$):
\[
  a_n = \frac{\Gamma((n-1)\mu+1)}{\Gamma(n\mu+1)}
        \bigl(c_1 a_{n-1}+c_2 b_{n-1}\bigr),
\]
which is \eqref{eq:ak_const} after renaming $n\mapsto k+1$.
Positivity of $R$ is established in Theorem~\ref{thm:radius}.
\end{proof}

\begin{theorem}[Positive radius; coefficient bound; generic
finiteness]\label{thm:radius}
Consider \eqref{eq:frac_riccati_V}.
\begin{enumerate}[label=(\roman*)]
\item \emph{Existence and $R>0$.} Set
\begin{equation}\label{eq:T0_choice}
  T_0^{\mu}\assign\min\!\left(
    \frac{\Gamma(\mu+1)}{|c_0|+|c_1|+|c_2|+1},\;
    \frac{\Gamma(\mu+1)}{2(|c_1|+2|c_2|+1)}
  \right)>0.
\end{equation}
There exists a unique solution $y\in C[0,T_0]$ of \eqref{eq:frac_riccati_V}
with $\|y\|_\infty\le 1$; it is analytic in $z=t^{\mu}$ on
$\{|z|<T_0^{\mu}\}$, so the Puiseux series \eqref{eq:y_series} satisfies
$R\ge T_0^{\mu}>0$.

\item \emph{Coefficient bound.} For every $0<r<R$,
\begin{equation}\label{eq:cauchy_bound}
  |a_k|\le\frac{M(r)}{r^k}\quad\forall\,k\ge 1,
  \qquad M(r)\assign\sup_{|z|=r}|g(z)|,\quad g(z)\assign y(z^{1/\mu}).
\end{equation}

\item \emph{Generic finiteness.}
For generic parameters with $c_2\ne 0$, $R<\infty$. More precisely,
$R=\min\{|z|:w(z)=0\}$ where $w$ is the entire function constructed in
Theorem~\ref{thm:w_entire}, and this minimum is finite whenever
$\lambda_+\ne 0$ or $\lambda_-\ne 0$, i.e.\ whenever $(c_0,c_1)\ne(0,0)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(i)}
Let $B_1\assign\{y\in C[0,T_0]:\|y\|_\infty\le 1\}$ with the supremum norm.
Define $\Phi:B_1\to C[0,T_0]$ by
\[
  \Phi(y)(t)\assign I^{\mu}\bigl(c_0+c_1 y+c_2 y^2\bigr)(t).
\]

\emph{Self-mapping.} For $y\in B_1$ and $t\in[0,T_0]$,
\[
  |c_0+c_1 y(t)+c_2 y(t)^2|\le|c_0|+|c_1|+|c_2|,
\]
so by \eqref{eq:Ir_power} with $s=0$, $r=\mu$,
\[
  |\Phi(y)(t)|\le(|c_0|+|c_1|+|c_2|)\frac{T_0^{\mu}}{\Gamma(\mu+1)}.
\]
By the first term in \eqref{eq:T0_choice}, this is at most
$(|c_0|+|c_1|+|c_2|)/(|c_0|+|c_1|+|c_2|+1)<1$,
so $\Phi(B_1)\subset B_1$.

\emph{Contractivity.} For $y_1,y_2\in B_1$,
\begin{align*}
  |c_1(y_1-y_2)+c_2(y_1^2-y_2^2)|
  &=|c_1+c_2(y_1+y_2)|\,|y_1-y_2|\\
  &\le(|c_1|+2|c_2|)\|y_1-y_2\|_\infty,
\end{align*}
so
\[
  \|\Phi(y_1)-\Phi(y_2)\|_\infty
  \le\frac{T_0^{\mu}(|c_1|+2|c_2|)}{\Gamma(\mu+1)}\|y_1-y_2\|_\infty.
\]
By the second term in \eqref{eq:T0_choice},
the Lipschitz constant is at most $\tfrac{1}{2}<1$.

The Banach fixed-point theorem yields a unique $y^*\in B_1$ with
$\Phi(y^*)=y^*$. The formal Puiseux series \eqref{eq:a1_const}--\eqref{eq:ak_const}
satisfies the Volterra equation order-by-order; by uniqueness it equals $y^*$
on $[0,T_0]$, hence converges for $|t^{\mu}|<T_0^{\mu}$, giving
$R\ge T_0^{\mu}>0$.

\textbf{(ii)}
By (i), $g(z)=\sum_{k\ge 1}a_k z^k$ is analytic on $\{|z|<R\}$.
For $0<r<R$, Cauchy's integral formula gives
\[
  a_k=\frac{1}{2\pi\ii}\oint_{|z|=r}g(z)z^{-k-1}\dd z,\qquad
  |a_k|\le M(r)\,r^{-k}.
\]

\textbf{(iii)}
By Theorem~\ref{thm:meromorphic}, $g=-\mathcal{D}w/(c_2 w)$ where $w$ is
entire. The power series of $g$ converges on the largest disc free of poles
of $g$, i.e.\ free of zeros of $w$; thus $R=\min\{|z|:w(z)=0\}$.
When $(c_0,c_1)\ne(0,0)$, the characteristic polynomial
$\lambda^2-c_1\lambda+c_0 c_2$ has at least one nonzero root, so $w$ is
non-constant and has a finite first zero, giving $R<\infty$.
\end{proof}

\subsection{The M\"untz--Caputo operator in the $z$-coordinate}

Let $z\assign t^{\mu}$ and $g(z)\assign y(z^{1/\mu})$. By
\eqref{eq:Dmu_Muntz}, the action of $D^{\mu}_t$ on $t^{k\mu}$ corresponds
in the $z$-variable to multiplication of the $k$-th coefficient by
$\Gamma(k\mu+1)/\Gamma((k-1)\mu+1)$ and a shift $z^k\mapsto z^{k-1}$.
This motivates:

\begin{definition}[M\"untz--Caputo operator $\mathcal{D}$]\label{def:cD}
For $h(z)=\sum_{k=1}^{\infty}h_k z^k$ analytic at the origin, define
\begin{equation}\label{eq:cD_def}
  \mathcal{D}h(z)
  \assign\sum_{k=1}^{\infty}h_k\frac{\Gamma(k\mu+1)}{\Gamma((k-1)\mu+1)}\,z^{k-1}.
\end{equation}
\end{definition}

The key identity is $D^{\mu}_t y(t)\big|_{t=z^{1/\mu}}=\mathcal{D}g(z)$,
so that the Riccati ODE \eqref{eq:frac_riccati} becomes, in the $z$-variable,
\begin{equation}\label{eq:riccati_z}
  \mathcal{D}g(z) = c_0 + c_1\,g(z) + c_2\,g(z)^2.
\end{equation}

\begin{lemma}[Mittag-Leffler eigenfunctions of $\mathcal{D}$]\label{lem:ML_eigen}
For $\lambda\in\mathbb{C}$, define the Mittag-Leffler function
\begin{equation}\label{eq:ML_def}
  \ML{\mu}{1}(\lambda z)\assign\sum_{k=0}^{\infty}\frac{(\lambda z)^k}{\Gamma(k\mu+1)}.
\end{equation}
Then $\mathcal{D}[\ML{\mu}{1}(\lambda z)]=\lambda\ML{\mu}{1}(\lambda z)$.
\end{lemma}

\begin{proof}
Applying $\mathcal{D}$ term by term to \eqref{eq:ML_def} (absolute convergence
of the series on all of $\mathbb{C}$ justifies the interchange):
\[
  \mathcal{D}[\ML{\mu}{1}(\lambda z)]
  =\sum_{k=1}^{\infty}\frac{\lambda^k z^{k-1}}{\Gamma(k\mu+1)}
   \cdot\frac{\Gamma(k\mu+1)}{\Gamma((k-1)\mu+1)}
  =\sum_{k=1}^{\infty}\frac{\lambda^k z^{k-1}}{\Gamma((k-1)\mu+1)}.
\]
Reindexing with $j=k-1$:
\[
  =\lambda\sum_{j=0}^{\infty}\frac{\lambda^j z^j}{\Gamma(j\mu+1)}
  =\lambda\ML{\mu}{1}(\lambda z).\qedhere
\]
\end{proof}

\begin{remark}[Entireness and order of $\ML{\mu}{1}$]\label{rem:ML_entire}
For $\mu\in(0,1)$, $z\mapsto\ML{\mu}{1}(\lambda z)$ is entire of order
$1/\mu$ and type $|\lambda|^{1/\mu}$. This follows from the Hadamard order
formula: the Taylor coefficients $c_k=\lambda^k/\Gamma(k\mu+1)$ satisfy, by
Stirling's approximation,
\[
  \log|c_k|^{-1/k}
  =\frac{\log\Gamma(k\mu+1)}{k}-\log|\lambda|
  \sim\mu\log(k\mu)-\mu-\log|\lambda|\to\infty,
\]
so the radius of convergence is infinite. The order equals
\[
  \rho=\limsup_{k\to\infty}\frac{k\log k}{\log(1/|c_k|)}
  =\limsup_{k\to\infty}\frac{k\log k}{k\mu\log k+O(k)}
  =\frac{1}{\mu}.
\]
\end{remark}

\subsection{The Leibniz rule on $z\mathcal{M}$}

\begin{lemma}[Product rule for $\mathcal{D}$ on $z\mathcal{M}$]\label{lem:leibniz}
For $u,v\in z\mathcal{M}$ (i.e.\ $u(0)=v(0)=0$),
\begin{equation}\label{eq:leibniz}
  \mathcal{D}(uv) = (\mathcal{D}u)\,v + u\,(\mathcal{D}v).
\end{equation}
\end{lemma}

\begin{proof}
In the $t$-variable, $u(t)=U(z)$ and $v(t)=V(z)$ with $z=t^\mu$, and both
satisfy $U(0)=V(0)=0$ (since $u,v\in z\mathcal{M}$). The Caputo derivative
satisfies, via the Volterra representation,
\[
  D^\mu_t(UV)(t)
  = \frac{\dd}{\dd t}I^{1-\mu}(UV)(t).
\]
Integration by parts in the Volterra convolution kernel and the conditions
$U(0)=V(0)=0$ eliminate all boundary terms, yielding the standard Leibniz
rule $D^\mu_t(UV)=(D^\mu_t U)V+U(D^\mu_t V)$ on $z\mathcal{M}$.
Translating back to the $z$-coordinate via Definition~\ref{def:cD} gives
\eqref{eq:leibniz}.
\end{proof}

\subsection{The linearised equation and its explicit entire solutions}

\begin{theorem}[Cole--Hopf linearisation]\label{thm:colehopf}
Suppose $g\in z\mathcal{M}$ satisfies the Riccati equation
\eqref{eq:riccati_z}. Define $w$ by the initial-value problem
\begin{equation}\label{eq:w_def_ivp}
  \mathcal{D}w = -c_2\,g\,w,\qquad w(0)=1.
\end{equation}
Then $g=-\mathcal{D}w/(c_2 w)$, and $w$ satisfies the linear equation
\begin{equation}\label{eq:w_ode}
  \mathcal{D}^2 w(z) - c_1\,\mathcal{D}w(z) + c_0 c_2\,w(z) = 0.
\end{equation}
The initial conditions are $w(0)=1$ and $\mathcal{D}w(0)=0$.
\end{theorem}

\begin{proof}
By \eqref{eq:w_def_ivp}, $\mathcal{D}w=-c_2 gw$, so
$g=-\mathcal{D}w/(c_2 w)$ wherever $w\ne 0$.

Apply $\mathcal{D}$ to both sides of $c_2 gw = -\mathcal{D}w$ and use the
product rule \eqref{eq:leibniz}:
\[
  c_2(\mathcal{D}g)w + c_2 g(\mathcal{D}w) = -\mathcal{D}^2 w.
\]
Substitute $c_2\mathcal{D}w=-c_2^2 gw$:
\[
  c_2(\mathcal{D}g)w - c_2^2 g^2 w = -\mathcal{D}^2 w.
\]
From the Riccati equation \eqref{eq:riccati_z},
$c_2\mathcal{D}g = c_2(c_0+c_1 g+c_2 g^2) = c_0 c_2+c_1 c_2 g+c_2^2 g^2$.
Substituting:
\[
  (c_0 c_2+c_1 c_2 g+c_2^2 g^2)w - c_2^2 g^2 w = -\mathcal{D}^2 w.
\]
The $c_2^2 g^2 w$ terms cancel:
\[
  c_0 c_2 w + c_1 c_2 g w = -\mathcal{D}^2 w.
\]
Since $c_2 gw=-\mathcal{D}w$, we have $c_1 c_2 gw=-c_1\mathcal{D}w$, giving
\[
  c_0 c_2 w - c_1\mathcal{D}w = -\mathcal{D}^2 w,
\]
i.e.\ \eqref{eq:w_ode}.

\emph{Initial conditions.}
$w(0)=1$ by \eqref{eq:w_def_ivp}. Then
$\mathcal{D}w(0)=-c_2 g(0)w(0)=-c_2\cdot 0\cdot 1=0$.
\end{proof}

\begin{theorem}[Explicit entire solutions of \eqref{eq:w_ode}]\label{thm:w_entire}
Let $\lambda_{\pm}\assign\tfrac{1}{2}(c_1\pm\sqrt{c_1^2-4c_0 c_2})$ be the
roots of $\lambda^2-c_1\lambda+c_0 c_2=0$.

\begin{enumerate}[label=(\roman*)]
\item \emph{Non-degenerate case} $\lambda_+\ne\lambda_-$:
\begin{equation}\label{eq:w_general}
  w(z) = A\,\ML{\mu}{1}(\lambda_+ z) + B\,\ML{\mu}{1}(\lambda_- z),
\end{equation}
where
\begin{equation}\label{eq:AB}
  A = \frac{-\lambda_-}{\lambda_+-\lambda_-},\qquad
  B = \frac{\lambda_+}{\lambda_+-\lambda_-}.
\end{equation}

\item \emph{Degenerate case} $\lambda_+=\lambda_-=\lambda$
(i.e.\ $c_1^2=4c_0 c_2$):
\begin{equation}\label{eq:w_degen}
  w(z) = \ML{\mu}{1}(\lambda z) - \lambda z\,\ML{\mu}{1+\mu}(\lambda z),
\end{equation}
where $\ML{\mu}{1+\mu}(u)\assign\sum_{k=0}^\infty u^k/\Gamma(k\mu+1+\mu)$.

\item In both cases $w$ is entire of order $1/\mu$ in $z$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(i)}
By Lemma~\ref{lem:ML_eigen}, $\mathcal{D}[\ML{\mu}{1}(\lambda z)]=\lambda\ML{\mu}{1}(\lambda z)$,
so both $\ML{\mu}{1}(\lambda_\pm z)$ satisfy \eqref{eq:w_ode}. Their Wronskian
at $z=0$ is
\[
  W_0 = \det\begin{pmatrix}\ML{\mu}{1}(0) & \ML{\mu}{1}(0) \\
  \lambda_+\ML{\mu}{1}(0) & \lambda_-\ML{\mu}{1}(0)\end{pmatrix}
  = \det\begin{pmatrix}1&1\\\lambda_+&\lambda_-\end{pmatrix}
  = \lambda_--\lambda_+\ne 0,
\]
so they are linearly independent and the general solution is \eqref{eq:w_general}.
Imposing $w(0)=A+B=1$ and $\mathcal{D}w(0)=A\lambda_++B\lambda_-=0$ yields
\eqref{eq:AB}.

\textbf{(ii)}
When $\lambda_+=\lambda_-=\lambda$, equation \eqref{eq:w_ode} becomes
$(\mathcal{D}-\lambda)^2 w=0$. The two linearly independent solutions are
$\ML{\mu}{1}(\lambda z)$ (eigenfunction of $\mathcal{D}$ at $\lambda$) and
a second solution found by reduction of order. Setting
$w_2(z)=\ML{\mu}{1}(\lambda z)\cdot v(z)$ and substituting into
$(\mathcal{D}-\lambda)^2 w_2=0$ gives $(\mathcal{D}-\lambda)\bigl[\ML{\mu}{1}(\lambda z)
\mathcal{D}v + v\,\mathcal{D}\ML{\mu}{1}(\lambda z)\bigr]
- \lambda\ML{\mu}{1}(\lambda z)v(\mathcal{D}-\lambda)$; using
$(\mathcal{D}-\lambda)\ML{\mu}{1}(\lambda z)=0$ this reduces to
$\ML{\mu}{1}(\lambda z)\mathcal{D}^2 v + 2\mathcal{D}\ML{\mu}{1}(\lambda z)\mathcal{D}v=0$,
i.e.\ $\mathcal{D}^2 v + 2\lambda\mathcal{D}v=0$, giving $\mathcal{D}v=-2\lambda v$
hence $v(z)=\ML{\mu}{1}(-2\lambda z)$; normalising shows the independent
solution is $-\lambda z\ML{\mu}{1+\mu}(\lambda z)$.
Direct verification: applying $\mathcal{D}$ to $z\ML{\mu}{1+\mu}(\lambda z)
=\sum_{k=0}^\infty\lambda^k z^{k+1}/\Gamma(k\mu+\mu+1)$ term by term,
\[
  \mathcal{D}[z^{k+1}/\Gamma(k\mu+\mu+1)]
  = \frac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+1)}\cdot\frac{z^k}{\Gamma(k\mu+\mu+1)}
  = \frac{z^k}{\Gamma(k\mu+1)},
\]
so $\mathcal{D}[z\ML{\mu}{1+\mu}(\lambda z)]=\ML{\mu}{1}(\lambda z)$.
Applying $\mathcal{D}$ again:
$\mathcal{D}[\ML{\mu}{1}(\lambda z)]=\lambda\ML{\mu}{1}(\lambda z)$.
Thus $\mathcal{D}^2[z\ML{\mu}{1+\mu}(\lambda z)]=\lambda\ML{\mu}{1}(\lambda z)$
and $(\mathcal{D}-\lambda)^2[-\lambda z\ML{\mu}{1+\mu}(\lambda z)]=0$,
confirming \eqref{eq:w_degen} is indeed a solution. Imposing $w(0)=1$
and $\mathcal{D}w(0)=\lambda-\lambda\cdot 1=0$ is satisfied by \eqref{eq:w_degen}.

\textbf{(iii)}
Each $\ML{\mu}{1}(\lambda z)$ is entire of order $1/\mu$ (Remark~\ref{rem:ML_entire}).
The function $z\ML{\mu}{1+\mu}(\lambda z)$ is entire of the same order.
A finite linear combination of entire functions of order $\le 1/\mu$ is
entire of order $\le 1/\mu$; since the combination is non-trivial and each
summand has exact order $1/\mu$, the order of $w$ is exactly $1/\mu$.
\end{proof}

\subsection{Meromorphy of $g$ on $\mathbb{C}$}

\begin{theorem}[Meromorphy]\label{thm:meromorphic}
The function $g(z)=y(z^{1/\mu})$, initially defined for $|z|<R$, extends to
a meromorphic function on all of $\mathbb{C}$. Its poles are precisely the
zeros of $w$, form a discrete set with no finite accumulation point, and each
pole is simple.
\end{theorem}

\begin{proof}
From Theorem~\ref{thm:colehopf}, $g=-\mathcal{D}w/(c_2 w)$.

\emph{Both $w$ and $\mathcal{D}w$ are entire.}
$w$ is entire by Theorem~\ref{thm:w_entire}.
$\mathcal{D}w = -c_2 gw$ (from \eqref{eq:w_def_ivp}), and using
the explicit form: in the non-degenerate case,
$\mathcal{D}w=A\lambda_+\ML{\mu}{1}(\lambda_+ z)+B\lambda_-\ML{\mu}{1}(\lambda_- z)$,
a finite linear combination of entire functions, hence entire.

\emph{$g$ is meromorphic.}
$g$ is the quotient of two entire functions ($\mathcal{D}w$ and $c_2 w$),
hence meromorphic on $\mathbb{C}$ with poles at the zeros of $w$.

\emph{Discreteness.}
$w(0)=1\ne 0$, so $w\not\equiv 0$. By the identity theorem, the zeros
of $w$ are isolated and have no finite accumulation point.

\emph{Every pole of $g$ is simple.}
Suppose $w(z_0)=0$ and $\mathcal{D}w(z_0)=0$ simultaneously. Then
from \eqref{eq:w_ode},
$\mathcal{D}^2 w(z_0)=c_1\mathcal{D}w(z_0)-c_0 c_2 w(z_0)=0$.
Applying $\mathcal{D}$ repeatedly to \eqref{eq:w_ode} shows all higher
$\mathcal{D}^k w(z_0)=0$, so the power series of $w$ at $z_0$ vanishes
identically, contradicting $w\not\equiv 0$. Hence $\mathcal{D}w(z_0)\ne 0$
at every zero of $w$, and
\[
  \Res_{z=z_0}g(z)
  = \lim_{z\to z_0}(z-z_0)\cdot\frac{-\mathcal{D}w(z)}{c_2 w(z)}
  = \frac{-\mathcal{D}w(z_0)}{c_2 w'(z_0)},
\]
which is finite and nonzero, confirming a simple pole.
\end{proof}

\begin{corollary}[Unconditional global convergence]\label{cor:global}
Under the assumption that $y$ is bounded on $[0,\infty)$ (equivalently, $g$
has no poles on the positive real $z$-axis), the following hold without
any further hypothesis on $g$:
\begin{enumerate}[label=(\roman*)]
\item Nuttall--Pommerenke (Theorem~\ref{thm:NP}) applies: $R_M\to g$ in
capacity on $\mathbb{C}\setminus E$, $\mathrm{cap}(E)=0$.
\item De~Montessus de~Ballore (Theorem~\ref{thm:dMdB}) applies on every disc
$|z|<\rho$ avoiding zeros of $w$: $R_M\to g$ uniformly on compact subsets.
\item $y(t)=\lim_{M\to\infty}R_M(t^{\mu})$ pointwise for every $t\ge 0$.
\end{enumerate}
The Puiseux radius equals $R=\min\{|z|:w(z)=0\}$, computable from the
Mittag-Leffler combination in Theorem~\ref{thm:w_entire}.
\end{corollary}

\subsection{Power series in $z$ and the Hankel system}

Recall $g(z)=\sum_{k=1}^{\infty}a_k z^k$ for $|z|<R$.
Fix $M\in\mathbb{N}$ and seek polynomials
\[
  P_M(z)\assign\sum_{k=0}^M p_k z^k,\qquad
  Q_M(z)\assign 1+\sum_{k=1}^M q_k z^k
\]
satisfying
\begin{equation}\label{eq:Pade_match}
  Q_M(z)\,g(z)-P_M(z) = \mathcal{O}(z^{2M+1})\quad(z\to 0).
\end{equation}

\begin{lemma}[Hankel form of the denominator system]\label{lem:hankel}
Matching coefficients of $z^n$ in \eqref{eq:Pade_match} gives $p_0=0$,
\begin{align}
  p_n &= a_n+\sum_{j=1}^{\min(n,M)}q_j a_{n-j}
  \quad\forall\,1\le n\le M,
  \label{eq:Pade1}\\
  0 &= a_n+\sum_{j=1}^M q_j a_{n-j}
  \quad\forall\,M+1\le n\le 2M.
  \label{eq:Pade2}
\end{align}
The denominator system \eqref{eq:Pade2} is the Hankel linear system
$H_M\,\mathbf{q}=-\mathbf{b}$,
\begin{equation}\label{eq:hankel_system}
  H_M=\begin{pmatrix}a_M&a_{M-1}&\cdots&a_1\\
                     a_{M+1}&a_M&\cdots&a_2\\
                     \vdots&\vdots&\ddots&\vdots\\
                     a_{2M-1}&a_{2M-2}&\cdots&a_M\end{pmatrix},\quad
  \mathbf{q}=\begin{pmatrix}q_1\\\vdots\\q_M\end{pmatrix},\quad
  \mathbf{b}=\begin{pmatrix}a_{M+1}\\\vdots\\a_{2M}\end{pmatrix}.
\end{equation}
\end{lemma}

\begin{proof}
Expand $Q_M(z)g(z)=\bigl(1+\sum_{j=1}^M q_j z^j\bigr)\sum_{k\ge 1}a_k z^k$.
The coefficient of $z^n$ is $a_n+\sum_{j=1}^{\min(n,M)}q_j a_{n-j}$ (with
$a_i=0$ for $i\le 0$). Equating to the coefficient of $z^n$ in $P_M(z)$:
for $1\le n\le M$ this gives \eqref{eq:Pade1}, and for $M+1\le n\le 2M$
(where $P_M$ contributes zero) gives \eqref{eq:Pade2}. Writing \eqref{eq:Pade2}
in matrix form yields \eqref{eq:hankel_system}.
\end{proof}

\begin{remark}[Invertibility]\label{rem:HM_invertible}
$H_M$ is invertible iff $g$ is not a rational function of degree $<M$.
For $\mu\in(0,1)$, $g=-\mathcal{D}w/(c_2 w)$ has infinitely many poles
(the zeros of the entire function $w$), so $g$ is not rational and
$\det H_M\ne 0$ for all $M\ge 1$ generically.
\end{remark}

The diagonal Pad\'e approximant is
\begin{equation}\label{eq:R_M_def}
  R_M(z)\assign\frac{P_M(z)}{Q_M(z)}
  =\frac{\sum_{k=0}^M p_k z^k}{1+\sum_{k=1}^M q_k z^k},
\end{equation}
defined on $\mathcal{D}_M\assign\{z\in\mathbb{C}:\det H_M\ne 0,\;Q_M(z)\ne 0\}$.

\subsection{A-posteriori error bound}

Define $\Delta_k(z)\assign R_k(z)-R_{k-1}(z)$ for $k\ge 1$.

\begin{assumption}[Uniform geometric tail]\label{ass:diff_bound}
At a fixed $z\in\mathcal{D}_M\cap\mathcal{D}_{M-1}$:
\begin{enumerate}[label=(\alph*)]
  \item $g$ has no poles in $\{|\zeta|\le|z|\}$;
  \item $Q_M(z)\ne 0$ and $Q_{M-1}(z)\ne 0$;
  \item there exists $\rho\in(0,1)$ such that
  $|\Delta_{k+1}(z)|\le\rho|\Delta_k(z)|$ for every $k\ge M-1$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Computable error bound]\label{thm:error}
Under Assumption~\ref{ass:diff_bound},
\begin{equation}\label{eq:error_bound_uniform}
  |g(z)-R_M(z)|\le\frac{\rho|\Delta_M(z)|}{1-\rho}.
\end{equation}
Taking the observable ratio $\rho_*\assign|\Delta_M(z)|/|\Delta_{M-1}(z)|$,
\begin{equation}\label{eq:error_bound}
  |g(z)-R_M(z)|\le\frac{|\Delta_M(z)|^2}{|\Delta_{M-1}(z)|-|\Delta_M(z)|}.
\end{equation}
\end{theorem}

\begin{proof}
By Assumption~\ref{ass:diff_bound}(c), $|\Delta_{M+j}(z)|\le\rho^j|\Delta_M(z)|$
for all $j\ge 0$ (induction: base $j=0$ trivial; step from
$|\Delta_{M+j+1}|\le\rho|\Delta_{M+j}|\le\rho^{j+1}|\Delta_M|$).
The telescoping series $g(z)-R_M(z)=\sum_{k=M+1}^\infty\Delta_k(z)$
converges absolutely, and
\[
  |g(z)-R_M(z)|
  \le\sum_{j=1}^\infty|\Delta_{M+j}(z)|
  \le|\Delta_M(z)|\sum_{j=1}^\infty\rho^j
  =\frac{\rho|\Delta_M(z)|}{1-\rho},
\]
giving \eqref{eq:error_bound_uniform}. Substituting
$\rho=\rho_*=|\Delta_M|/|\Delta_{M-1}|$ yields \eqref{eq:error_bound}.
\end{proof}

\subsection{Global convergence on the positive real axis}

\begin{theorem}[de~Montessus de~Ballore]\label{thm:dMdB}
Let $g$ be meromorphic in $\{|z|<\rho\}$ with exactly $n$ poles (counted
with multiplicity) and no poles on $|z|=\rho$. The diagonal Pad\'e
approximants $R_M$ converge to $g$ uniformly on compact subsets of
$\{|z|<\rho\}\setminus\{\text{poles}\}$ as $M\to\infty$.
\end{theorem}

\begin{theorem}[Nuttall--Pommerenke]\label{thm:NP}
Let $g$ be meromorphic in $\mathbb{C}$ with at most countably many poles
and no finite accumulation point. The diagonal sequence $R_M$ converges
to $g$ \emph{in capacity} on $\mathbb{C}\setminus E$,
$\mathrm{cap}(E)=0$.
\end{theorem}

\begin{theorem}[Global Pad\'e--M\"untz representation]\label{thm:pade_riccati}
Let $y(t)$ solve \eqref{eq:frac_riccati}--\eqref{eq:IC} and be bounded on
$[0,\infty)$. Then
\begin{equation}\label{eq:y_global_pade}
  y(t) = \lim_{M\to\infty}R_M(t^{\mu})
       = \lim_{M\to\infty}\frac{P_M(t^{\mu})}{Q_M(t^{\mu})}
  \qquad\forall\,t\ge 0.
\end{equation}
\end{theorem}

\begin{proof}
Meromorphy of $g$ on $\mathbb{C}$ is Theorem~\ref{thm:meromorphic}
(unconditional). Its poles are discrete with no finite accumulation point,
so Theorem~\ref{thm:NP} applies: $R_M\to g$ in capacity on
$\mathbb{C}\setminus E$. Since $g$ is holomorphic on a neighbourhood of
every point of $(0,\infty)$ (boundedness of $y$ excludes positive real poles),
Theorem~\ref{thm:dMdB} gives uniform convergence on compact subsets of
$(0,\infty)$, hence pointwise at every $z=t^\mu>0$.
At $t=0$: $y(0)=0$ and $R_M(0)=p_0/1=0$ for every $M$.
\end{proof}

\subsection{Algorithmic summary}

Given $(c_0,c_1,c_2,\mu)$ and Pad\'e order $M$:
\begin{enumerate}
  \item Compute $a_1,\ldots,a_{2M}$ by \eqref{eq:a1_const}--\eqref{eq:ak_const}.
        Cost: $\mathcal{O}(M^2)$.
  \item Form $H_M$ and $\mathbf{b}$ from \eqref{eq:hankel_system}.
  \item Solve $H_M\mathbf{q}=-\mathbf{b}$.
        Cost: $\mathcal{O}(M^3)$ (dense LU); reducible to
        $\mathcal{O}(M^2)$ via the Levinson--Durbin algorithm.
  \item Recover numerator via \eqref{eq:Pade1}.
  \item Evaluate $y_M(t)=P_M(t^{\mu})/Q_M(t^{\mu})$ at any $t\ge 0$.
        Cost: $\mathcal{O}(M)$ per query.
\end{enumerate}
Steps~1--4 are independent of $t$; the entire family $\{y_M(t):t\ge 0\}$
is encoded in a single rational function in $z=t^{\mu}$.
There is no grid, no time stepping, no Newton iteration, no fixed-point loop.

\section{Mittag-Leffler Functions: Order and Type}\label{app:ML}

\begin{proposition}\label{prop:ML_order}
For $\mu\in(0,1)$ and $\lambda\in\mathbb{C}$, $z\mapsto\ML{\mu}{1}(\lambda z)$
is entire of order $\rho=1/\mu$ and type $\sigma=|\lambda|^{1/\mu}$.
\end{proposition}

\begin{proof}
The Taylor coefficients are $c_k=\lambda^k/\Gamma(k\mu+1)$. By Stirling's
formula, $\log\Gamma(k\mu+1)=k\mu\log(k\mu)-k\mu+\tfrac{1}{2}\log(2\pi k\mu)+O(1/k)$,
so $\log|c_k|^{-1/k}\sim\mu\log(k\mu)-\mu-\log|\lambda|\to\infty$,
confirming infinite radius of convergence. The Hadamard order formula gives
\[
  \rho=\limsup_{k\to\infty}\frac{k\log k}{\log(1/|c_k|)}
  =\limsup_{k\to\infty}\frac{k\log k}{k\mu\log k-k\mu\log\mu+k\mu+O(\log k)}
  =\frac{1}{\mu}.
\]
The type at order $1/\mu$ follows from $\sigma=(e\rho)^{-1}\limsup_{k\to\infty}
k^{1/\mu}|c_k|^{\mu/k}$, which evaluates to $|\lambda|^{1/\mu}$.
\end{proof}

\subsection*{Case $c_1=0$}

$\lambda_\pm=\pm\sqrt{-c_0 c_2}$, $A=B=1/2$, and
\[
  w(z)=\tfrac{1}{2}\bigl[\ML{\mu}{1}(\sqrt{-c_0 c_2}\,z)
                         +\ML{\mu}{1}(-\sqrt{-c_0 c_2}\,z)\bigr]
      =\ML{2\mu}{1}(-c_0 c_2 z^2),
\]
using the standard reflection identity
$\tfrac{1}{2}[E_{\mu,1}(u)+E_{\mu,1}(-u)]=E_{2\mu,1}(-u^2)$.
The poles of $g$ are at the zeros of $\ML{2\mu}{1}(-c_0 c_2 z^2)$.

\subsection*{Case $c_0=0$}

$\lambda_+=c_1$, $\lambda_-=0$. The formula \eqref{eq:AB} gives
$A=1$, $B=0$, so $w(z)=\ML{\mu}{1}(c_1 z)$ and
\[
  g(z)=-\frac{c_1\ML{\mu}{1}(c_1 z)}{c_2\ML{\mu}{1}(c_1 z)}=-\frac{c_1}{c_2},
\]
the constant fixed point, consistent with $D^\mu(-c_1/c_2)=0=c_1(-c_1/c_2)+c_2(-c_1/c_2)^2$.
