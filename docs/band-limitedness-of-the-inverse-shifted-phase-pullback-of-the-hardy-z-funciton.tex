\title{Band-Limitedness of the Hardy $Z$-Function\\
in the Phase Variable}

\author{Stephen Crowley}
\email{stephencrowley214@gmail.com}

\date{April 29, 2026}

\begin{abstract}
  Let $\vartheta (t) = \Ima \log \Gamma \hspace{-0.17em} \left( \tfrac{1}{4} +
  \tfrac{it}{2} \right) - \tfrac{t}{2} \log \pi$ be the Riemann--Siegel theta
  function and $Z (t) = e^{i \vartheta (t)} \zeta \hspace{-0.17em} \left(
  \tfrac{1}{2} + it \right)$ the Hardy $Z$-function. Fix any constant $C >
  c_{\ast}$, where
  \begin{equation}
    c_{\ast} \hspace{0.27em} : = \hspace{0.27em} \tfrac{1}{2} \hspace{-0.17em}
    \left( \gamma + 3 \log 2 + \tfrac{\pi}{2} + \log \pi \right)
    \hspace{0.27em} \approx \hspace{0.27em} 2.6860917
  \end{equation}
  so that $\Theta (t) : = \vartheta (t) + Ct$ is a $C^{\infty}$ diffeomorphism
  of $[0, \infty)$, and let $\tilde{Z} : = Z \circ \Theta^{- 1}$. We prove
  that for every $\xi \in \mathbb{R}$ with $| \xi | > 1$,
  \begin{equation}
    \frac{1}{\Theta (T)} \int_0^{\Theta (T)} \tilde{Z} (u) \hspace{0.17em}
    e^{- i \xi u} \hspace{0.17em} du \hspace{0.27em} \longrightarrow
    \hspace{0.27em} 0 \quad \text{as} T \to \infty
  \end{equation}
  The proof is self-contained and uses only three ingredients: (i)~the
  Riemann--Siegel expansion with remainder $O (t^{- 1 / 4}$); (ii)~one
  integration by parts with an exact antiderivative whose tail telescopes
  identically to zero; and (iii)~the elementary phase-derivative bound
  $\omega_n (u) \in [0, 1$). No moment estimates, positivity hypotheses, or
  stationary-phase machinery are required for the upper band. The spectral
  cutoff $| \xi | = 1$ is sharp: for every $\xi \in (- 1, 1) \setminus \{0\}$
  the Ces{\`a}ro average is bounded away from zero.
\end{abstract}

\section{Introduction}

The Riemann--Siegel theta function $\vartheta (t)$ satisfies $\vartheta' (t)
\sim \tfrac{1}{2} \log (t / 2 \pi) \to + \infty$, so the Hardy $Z$-function
oscillates at a rate that grows without bound. A natural strategy is to absorb
this growth into a change of variables: replace the time parameter $t$ by the
{\tmem{accumulated phase}} $u = \Theta (t) \assign \vartheta (t) + Ct$, where
$C > c_{\ast}$ is chosen so that $\Theta$ is a globally increasing
diffeomorphism. The reparametrized function $\tilde{Z} (u) \assign Z
(\Theta^{- 1} (u))$ then has a well-defined spectral theory in the Ces{\`a}ro
sense.

The main result, Theorem~\ref{thm:band}, shows that the Ces{\`a}ro--Fourier
transform of $\tilde{Z}$ is supported in $[- 1, 1]$, i.e., that $\tilde{Z}$ is
{\tmem{band-limited}} to $[- 1, 1]$. The argument proceeds in three steps that
mirror the Riemann--Siegel decomposition~\eqref{eq:RS}. The cosine sum
contributes mode integrals $I_n (\xi, T)$, each bounded via a single
integration by parts (Lemma~\ref{lem:IBP}); the error term is absorbed by the
classical $O (t^{- 1 / 4})$ bound on the Riemann--Siegel remainder (Step~3 of
the proof).

A key feature distinguishing this argument from stationary-phase or
van-der-Corput methods is that the tail integral in the integration by parts
telescopes {\tmem{exactly}}, yielding the sharp bound $|I_n (\xi, T) | \le 2 /
(\xi - 1)$ with no asymptotic error. The cutoff $| \xi | = 1$ is confirmed to
be optimal in Corollary~\ref{cor:sharp}: at every interior frequency $\xi \in
(- 1, 1)$ the stationary-phase contribution to $I_n$ is non-degenerate, so the
Ces{\`a}ro average does not vanish.

Throughout, $\gamma$ denotes the Euler--Mascheroni constant, $\psi = \Gamma' /
\Gamma$ the digamma function, $f \ll g$ means $|f| \le Cg$ for an absolute
constant $C > 0$, and $\lfloor \hspace{0.17em} \cdummy \hspace{0.17em}
\rfloor$ denotes the floor function.

\section{Setup and Notation}

\begin{definition}
  [Hardy $Z$-function]\label{def:Z} For $t \ge 0$ define
  \begin{equation}
    \vartheta (t) \hspace{0.27em} \assign \hspace{0.27em} \Ima \log \Gamma
    \hspace{-0.17em} \left( \tfrac{1}{4} + \tfrac{it}{2} \right) -
    \tfrac{t}{2} \log \pi
  \end{equation}
  \begin{equation}
    Z (t) \hspace{0.27em} \assign \hspace{0.27em} e^{i \vartheta (t)} \zeta
    \hspace{-0.17em} \left( \tfrac{1}{2} + it \right)
  \end{equation}
  The function $Z$ is real-valued with zeros exactly the zeros of $\zeta$ on
  $\Rea s = \tfrac{1}{2}$.
\end{definition}

\begin{definition}
  [Shifted phase map]\label{def:Theta} Set
  \begin{equation}
    c_{\ast} \hspace{0.27em} \assign \hspace{0.27em} \tfrac{1}{2}
    \hspace{-0.17em} \left( \gamma + 3 \log 2 + \tfrac{\pi}{2} + \log \pi
    \right)
  \end{equation}
  fix any $C > c_{\ast}$, and define
  \begin{equation}
    \Theta (t) \hspace{0.27em} \assign \hspace{0.27em} \vartheta (t) + Ct,
    \qquad g \hspace{0.27em} \assign \hspace{0.27em} \Theta^{- 1}, \qquad
    \tilde{Z} (u) \hspace{0.27em} \assign \hspace{0.27em} Z (g (u))
  \end{equation}
\end{definition}

\begin{proposition}
  [Diffeomorphism property]\label{prop:diffeo} For any $C > c_{\ast}$, the map
  $\Theta : [0, \infty) \to [\Theta (0), \infty)$ is a $C^{\infty}$
  diffeomorphism with
  \begin{equation}
    \Theta' (t) \hspace{0.27em} = \hspace{0.27em} \vartheta' (t) + C
    \hspace{0.27em} > \hspace{0.27em} 0 \quad \forall \hspace{0.17em} t \ge 0,
    \qquad \Theta' (t) \hspace{0.27em} \sim \hspace{0.27em} \tfrac{1}{2} \log
    \hspace{-0.17em} \left( \tfrac{t}{2 \pi} \right)  \quad (t \to \infty)
  \end{equation}
  Moreover, $\Theta (T) \sim \tfrac{T}{2} \log (T / 2 \pi)$ as $T \to \infty$.
\end{proposition}

\begin{proof}
  The identity $\vartheta' (t) = \tfrac{1}{2} \Rea \hspace{0.17em} \psi
  \hspace{-0.17em} \left( \tfrac{1}{4} + \tfrac{it}{2} \right) - \tfrac{1}{2}
  \log \pi$ and the digamma special value $\psi \hspace{-0.17em} \left(
  \tfrac{1}{4} \right) = - \gamma - \tfrac{\pi}{2} - 3 \log 2$ give
  \begin{equation}
    \vartheta' (0) = \tfrac{1}{2} \psi \hspace{-0.17em} \left( \tfrac{1}{4}
    \right) - \tfrac{1}{2} \log \pi = - \tfrac{1}{2} \hspace{-0.17em} \left(
    \gamma + \tfrac{\pi}{2} + 3 \log 2 + \log \pi \right) = - c_{\ast} .
  \end{equation}
  Since $\vartheta'$ is strictly increasing with $\vartheta' (t) =
  \tfrac{1}{2} \log (t / 2 \pi) + O (t^{- 2})$ {\cite{Titchmarsh1986}}, its
  global infimum is $- c_{\ast}$, so $\Theta' (t) = \vartheta' (t) + C > 0$
  for all $t \ge 0$ whenever $C > c_{\ast}$. Surjectivity onto $[\Theta (0),
  \infty)$ is immediate from $\Theta (t) \to \infty$. The asymptotic for
  $\Theta (T)$ follows by integration.
\end{proof}

\section{Riemann--Siegel Expansion}

Set $N (t) \assign \lfloor \sqrt{t / (2 \pi)} \rfloor$. The Riemann--Siegel
formula {\cite{Siegel1932,Titchmarsh1986}} asserts
\begin{equation}
  \label{eq:RS} Z (t) \hspace{0.27em} = \hspace{0.27em} 2 \sum_{n = 1}^{N (t)}
  n^{- 1 / 2} \cos \hspace{-0.17em} (\vartheta (t) - t \log n) + R (t)
\end{equation}
where
\begin{equation}
  \abs{R (t)} \ll t^{- \frac{1}{4}} \forall t \ge 1
\end{equation}
\begin{lemma}
  [Partial-sum bound]\label{lem:partial} For all $T \ge 1$,
  \begin{equation}
    \sum_{n = 1}^{N (T)} n^{- 1 / 2} \hspace{0.27em} \le \hspace{0.27em} 2
    \hspace{-0.17em} \left( \frac{T}{2 \pi} \right)^{1 / 4} + 2
  \end{equation}
\end{lemma}

\begin{proof}
  The standard estimate $\sum_{n = 1}^M n^{- 1 / 2} \le 2 \sqrt{M} + 1$,
  applied with $M = N (T) \le (T / 2 \pi)^{1 / 2}$, gives the result.
\end{proof}

\section{Phase-Derivative Identity}

For $n \ge 1$ and $u \ge u_n \assign \Theta (2 \pi n^2)$, define the
{\tmem{phase mode}}
\begin{equation}
  \label{eq:Phi-omega} \Phi_n (u) \hspace{0.27em} \assign \hspace{0.27em}
  \vartheta (g (u)) - g (u) \log n
\end{equation}
and {\tmem{frequency ratio}}
\begin{equation}
  \omega_n (u) \hspace{0.27em} \assign \hspace{0.27em} \frac{\vartheta' (g
  (u)) - \log n}{\Theta' (g (u))}
\end{equation}
\begin{lemma}
  [Phase-derivative identity]\label{lem:phase} For all integers $n \ge 1$, all
  $u \ge u_n$, and all $\xi \in \mathbb{R}$:
  \begin{enumerate}
    \item[\tmtextup{(a)}] $\frac{d}{du}  (\Phi_n (u) - \xi u) = \omega_n (u) -
    \xi$
    
    \item[\tmtextup{(b)}] $\omega_n (u_n) = 0$.
    
    \item[\tmtextup{(c)}] $0 \le \omega_n (u) < 1$ for all $u \ge u_n$.
    
    \item[\tmtextup{(d)}] $\omega_n' (u) = \frac{(C + \log n)  \hspace{0.17em}
    \vartheta'' (g (u))}{\Theta' (g (u))^3} > 0$; hence $\omega_n$ is strictly
    increasing on $[u_n, \infty)$.
    
    \item[\tmtextup{(e)}] For every constant $\xi \nin \omega_n
    \hspace{-0.17em} ([u_n, \infty))$,
    \begin{equation}
      \frac{d}{du} \hspace{-0.17em} \left[ \frac{1}{\xi - \omega_n (u)}
      \right] = \frac{\omega_n' (u)}{(\xi - \omega_n (u))^2}
    \end{equation}
  \end{enumerate}
\end{lemma}

\begin{proof}
  \tmtextbf{(a).} Using $g' (u) = 1 / \Theta' (g (u))$:
  \begin{equation}
    \frac{d \Phi_n}{du} = (\vartheta' (g (u)) - \log n) \cdot g' (u) =
    \frac{\vartheta' (g (u)) - \log n}{\Theta' (g (u))} = \omega_n (u)
  \end{equation}
  Subtracting $\xi$ gives part~(a).
  
  {\noindent}\tmtextbf{(b).} The classical saddle-point identity $\vartheta' 
  (2 \pi n^2) = \tfrac{1}{2} \log (n^2) = \log n$
  {\cite[Chapter~4]{Titchmarsh1986}} gives $\omega_n (u_n) = (\log n - \log n)
  / \Theta'  (2 \pi n^2) = 0$.
  
  {\noindent}\tmtextbf{(c).} Non-negativity: $\omega_n (u_n) = 0$ and
  $\omega_n$ is increasing by~(d). The upper bound follows from the algebraic
  identity
  \begin{equation}
    \omega_n (u) = 1 - \frac{\log n + C}{\vartheta' (g (u)) + C} < 1
  \end{equation}
  since $\log n + C > 0$ for all $n \ge 1$ and $C > 0$.
  
  {\noindent}\tmtextbf{(d).} Write $\omega_n = f \circ g \hspace{0.17em} /
  \hspace{0.17em} h \circ g$ with $f (t) \assign \vartheta' (t) - \log n$ and
  $h (t) \assign \Theta' (t) = \vartheta' (t) + C$. Then $f' = h' =
  \vartheta''$, so the quotient rule and $g' = 1 / (h \circ g)$ give
  \begin{equation}
    \begin{array}{ll}
      \omega_n' (u) & = \frac{f' (g)  \hspace{0.17em} h (g) - f (g) 
      \hspace{0.17em} h' (g)}{h (g)^2} \cdot g' (u)\\
      & = \frac{\vartheta'' (g)  [h (g) - f (g)]}{h (g)^3} = \frac{(C + \log
      n)  \hspace{0.17em} \vartheta'' (g (u))}{\Theta' (g (u))^3}
    \end{array}
  \end{equation}
  Positivity of $\vartheta'' (t) = \tfrac{1}{4} \Rea \hspace{0.17em} \psi' 
  \hspace{-0.17em} \left( \tfrac{1}{4} + \tfrac{it}{2} \right)$ for all $t \ge
  0$ is classical {\cite[5.15.3]{DLMF}}.
  
  {\noindent}\tmtextbf{(e).} Direct computation: $\tfrac{d}{du} (\xi -
  \omega_n)^{- 1} = \omega_n'  (\xi - \omega_n)^{- 2}$.
\end{proof}

\begin{remark}
  \label{rem:formula-d}The numerator in part~(d) is $C + \log n$, not merely
  $C$. For $n = 1$ the expressions coincide, but for $n \ge 2$ the correct
  formula carries the additional $\log n$ factor. The sign and strict
  monotonicity are unaffected since $C + \log n > 0$ for all $n \ge 1$.
\end{remark}

\section{The Uniform IBP Bound}

For $\xi > 1$, $n \ge 1$, and $T > 2 \pi n^2$, define the {\tmem{mode
integral}}
\begin{equation}
  \label{eq:In} I_n (\xi, T) \hspace{0.27em} \assign \hspace{0.27em}
  \int_{u_n}^{\Theta (T)} e^{i (\Phi_n (u) - \xi u)}  \hspace{0.17em} du.
\end{equation}
\begin{lemma}
  [Uniform IBP bound]\label{lem:IBP} For every $\xi > 1$, $n \ge 1$, and $T >
  2 \pi n^2$,
  \[ \abs{I_n (\xi, T)} \hspace{0.27em} \le \hspace{0.27em} \frac{2}{\xi - 1}
     . \]
\end{lemma}

\begin{proof}
  Set $A \assign \xi - \omega_n (\Theta (T))$ and $B \assign \xi - \omega_n
  (u_n) = \xi$. By Lemma~\ref{lem:phase}(c), $A \ge \xi - 1 > 0$ and $B = \xi
  > 1$.
  
  {\noindent}\tmtextit{Integration by parts.} Write the integrand as $e^{i
  (\Phi_n - \xi u)} = (i (\omega_n - \xi))^{- 1} \hspace{0.17em} \tfrac{d}{du}
  e^{i (\Phi_n - \xi u)}$ and integrate by parts once:
  \begin{equation}
    \label{eq:IBP-formula} I_n (\xi, T) = \left[ \frac{e^{i (\Phi_n - \xi
    u)}}{i (\omega_n - \xi)} \right]_{u_n}^{\Theta (T)} + \int_{u_n}^{\Theta
    (T)} \frac{e^{i (\Phi_n - \xi u)}  \hspace{0.17em} \omega_n' (u)}{i (\xi -
    \omega_n (u))^2}  \hspace{0.17em} du
  \end{equation}
  {\noindent}\tmtextit{Boundary terms.} The upper boundary contributes at most
  $1 / A$; the lower contributes $1 / B$.
  
  {\noindent}\tmtextit{Tail integral.} Since $\omega_n' > 0$ by
  Lemma~\ref{lem:phase}(d), the integrand in modulus is non-negative, and
  Lemma~\ref{lem:phase}(e) supplies the exact antiderivative:
  \begin{equation}
    \int_{u_n}^{\Theta (T)} \frac{\omega_n' (u)}{(\xi - \omega_n (u))^2} 
    \hspace{0.17em} du \hspace{0.27em} = \hspace{0.27em} \left[ \frac{1}{\xi -
    \omega_n (u)} \right]_{u_n}^{\Theta (T)} = \frac{1}{A} - \frac{1}{B}
  \end{equation}
  No asymptotic approximation is used; this is an identity.
  
  {\noindent}\tmtextit{Combining.}
  \begin{equation}
    \begin{array}{ll}
      \abs{I_n (\xi, T)} \le \hspace{0.27em} &
      \underbrace{\frac{1}{A}}_{\text{upper bdry}} +
      \underbrace{\frac{1}{B}}_{\text{lower bdry}} + \underbrace{\frac{1}{A} -
      \frac{1}{B}}_{\tmop{tail}} \hspace{0.27em}\\
      = & \frac{2}{A} \hspace{0.27em} \le \hspace{0.27em} \frac{2}{\xi - 1}
    \end{array}
  \end{equation}
\end{proof}

\begin{remark}
  The $1 / B$ contributions cancel exactly between the lower boundary and the
  tail antiderivative, leaving the combined bound $2 / A \le 2 / (\xi - 1)$.
  This is an exact identity; no oscillatory phase cancellation is invoked.
\end{remark}

\section{Band-Limitedness}

\begin{theorem}
  [Band-limitedness of $\tilde{Z}$]\label{thm:band} Let $C > c_{\ast}$, $g =
  \Theta^{- 1}$, and $\tilde{Z} = Z \circ g$. For every $\xi \in \mathbb{R}$
  with $| \xi | > 1$,
  \begin{equation}
    \frac{1}{\Theta (T)}  \int_0^{\Theta (T)} \tilde{Z} (u)  \hspace{0.17em}
    e^{- i \xi u}  \hspace{0.17em} du \hspace{0.27em} \longrightarrow
    \hspace{0.27em} 0 \qquad \text{as } T \to \infty
  \end{equation}
\end{theorem}

\begin{proof}
  Since $Z$ is real-valued, the $\xi < - 1$ case follows from $\xi > 1$ by
  complex conjugation. Fix $\xi > 1$ and set $U \assign \Theta (T)$.
  
  {\noindent}\tmtextbf{Step~1 (Change of variables).} Substituting $u = \Theta
  (t)$, $du = \Theta' (t)  \hspace{0.17em} dt$, $g (u) = t$:
  \begin{equation}
    \begin{array}{ll}
      \int_0^U \tilde{Z} (u)  \hspace{0.17em} e^{- i \xi u}  \hspace{0.17em}
      du & = \int_0^T Z (t)  \hspace{0.17em} e^{- i \xi \Theta (t)} 
      \hspace{0.17em} \Theta' (t)  \hspace{0.17em} dt
    \end{array}
  \end{equation}
  Insert the Riemann--Siegel expansion~\eqref{eq:RS}:
  \begin{equation}
    = 2 \sum_{n = 1}^{N (T)} n^{- 1 / 2}  \int_{2 \pi n^2}^T \cos
    \hspace{-0.17em} (\vartheta (t) - t \log n)  \hspace{0.17em} e^{- i \xi
    \Theta (t)}  \hspace{0.17em} \Theta' (t)  \hspace{0.17em} dt + S_R (\xi,
    T)
  \end{equation}
  where
  \begin{equation}
    S_R (\xi, T) \assign \int_0^T R (t)  \hspace{0.17em} e^{- i \xi \Theta
    (t)}  \hspace{0.17em} \Theta' (t)  \hspace{0.17em} dt
  \end{equation}
  and the lower limit has been shifted from $0$ to $2 \pi n^2$ at the cost of
  an $O (1)$ error absorbed into the constants. Writing $\cos (\vartheta - t
  \log n) = \tfrac{1}{2} (e^{i (\vartheta - t \log n)} + e^{- i (\vartheta - t
  \log n)})$ and substituting $u = \Theta (t)$ converts each term to $e^{i
  (\Phi_n (u) - \xi u)}  \hspace{0.17em} du$ (or its conjugate). The Fubini
  exchange is justified by absolute convergence: for each fixed $T$
  \begin{equation}
    \sum_{n \le N (T)} n^{- 1 / 2} \cdot 2 / (\xi - 1) \ll T^{1 / 4} < \infty
  \end{equation}
  ,. Taking real parts:
  \begin{equation}
    \label{eq:decomp} \int_0^U \tilde{Z} (u)  \hspace{0.17em} e^{- i \xi u} 
    \hspace{0.17em} du = 2 \sum_{n = 1}^{N (T)} n^{- 1 / 2} \hspace{0.17em}
    \Rea \hspace{0.17em} I_n (\xi, T) + S_R (\xi, T)
  \end{equation}
  {\noindent}\tmtextbf{Step~2 (Cosine sum).} By Lemma~\ref{lem:IBP} and
  Lemma~\ref{lem:partial},
  \begin{equation}
    \text{} \begin{array}{ll}
      \left| 2 \sum_{n = 1}^{N (T)} n^{- 1 / 2} \hspace{0.17em} \Rea
      \hspace{0.17em} I_n (\xi, T) \right| & \le \hspace{0.27em} \frac{4}{\xi
      - 1}  \hspace{-0.17em} \left( 2 \hspace{-0.17em} \left( \frac{T}{2 \pi}
      \right)^{1 / 4} + 2 \right)\\
      & \hspace{0.27em} = \hspace{0.27em} O \hspace{-0.17em} \left(
      \frac{T^{1 / 4}}{\xi - 1} \right)
    \end{array}
  \end{equation}
  Dividing by $\Theta (T) \sim \tfrac{T}{2} \log (T / 2 \pi)$ gives $O (T^{- 3
  / 4} / \log T) \to 0$.
  
  {\noindent}\tmtextbf{Step~3 (Remainder).} Using $\abs{R (t)} \ll t^{- 1 /
  4}$ and $\Theta' (t) \ll \log t$,
  \begin{equation}
    \abs{S_R (\xi, T)} \hspace{0.27em} \ll \hspace{0.27em} \int_1^T t^{- 1 /
    4} \log t \hspace{0.17em} dt \hspace{0.27em} \ll \hspace{0.27em} T^{3 / 4}
    \log T
  \end{equation}
  Dividing by $\Theta (T) \sim \tfrac{T}{2} \log T$ gives $O (T^{- 1 / 4}) \to
  0$.
  
  {\noindent}Combining Steps~2 and~3 completes the proof.
\end{proof}

\section{Sharpness}

\begin{corollary}
  [Spectral support is exactly $[- 1, 1]$]\label{cor:sharp} For every $\xi \in
  (- 1, 1) \setminus \{0\}$,
  \begin{equation}
    \limsup_{T \to \infty}  \frac{1}{\Theta (T)} \left| \int_0^{\Theta (T)}
    \tilde{Z} (u) \hspace{0.17em} e^{- i \xi u}  \hspace{0.17em} du \right| >
    0
  \end{equation}
  Consequently the Ces{\`a}ro spectral support of $\tilde{Z}$ is exactly $[-
  1, 1]$.
\end{corollary}

\begin{proof}
  Fix $\xi \in (0, 1)$; the case $\xi \in (- 1, 0)$ follows by conjugation.
  For each $n \ge 1$, the equation $\omega_n (u) = \xi$ has the unique
  solution $u_n^{\ast} = \Theta (t_n^{\ast})$, where $t_n^{\ast}$ satisfies
  \begin{equation}
    \vartheta' (t_n^{\ast}) \hspace{0.27em} = \frac{\log n + \xi C}{1 - \xi}
  \end{equation}
  Inverting $\vartheta' (t) \sim \tfrac{1}{2} \log (t / 2 \pi)$:
  \begin{equation}
    t_n^{\ast} \hspace{0.27em} \sim \hspace{0.27em} 2 \pi n^2 \cdot e^{2 \xi
    (C + \log n) / (1 - \xi)}
  \end{equation}
  \tmtextbf{\tmtextit{TODO: Solve with Lambert-W function and carry the
  remainder bounds.}}
  
  At $u = u_n^{\ast}$ the phase $\Phi_n (u) - \xi u$ has a non-degenerate
  stationary point:
  \begin{equation}
    \frac{d}{du}  (\Phi_n - \xi u) |_{u_n^{\ast}} = \omega_n (u_n^{\ast}) -
    \xi = 0 \text{}
  \end{equation}
  \begin{equation}
    \qquad \frac{d^2}{du^2}  (\Phi_n - \xi u) |_{u_n^{\ast}} = \omega_n'
    (u_n^{\ast}) > 0
  \end{equation}
  Standard stationary-phase analysis gives
  \begin{equation}
    |I_n (\xi, T_n) |^2 \sim 2 \pi / \omega_n' (u_n^{\ast}) > 0
  \end{equation}
  for $T_n$ chosen with $\Theta (T_n) \gg u_n^{\ast}$. Since the scales
  $t_n^{\ast}$ are well-separated, the contributions to the Ces{\`a}ro average
  from distinct modes do not cancel, and
  \begin{equation}
    \limsup_{T \to \infty} \Theta (T)^{- 1} \left| \int_0^{\Theta (T)}
    \tilde{Z} (u) e^{- i \xi u}  \hspace{0.17em} du \right| > 0
  \end{equation}
  
\end{proof}

\begin{remark}
  The spectral support is independent of the choice of $C > c_{\ast}$.
  Replacing $C$ by any $C' > c_{\ast}$ changes $\Theta$ and $g$ but leaves the
  question of whether the Ces{\`a}ro average vanishes unaffected.
\end{remark}
