\begin{abstract}
Let $\zeta$ denote the Riemann zeta function and $Z(t)=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ the Hardy $Z$-function, with $\vartheta$ the Riemann--Siegel theta function. For each nontrivial zero $\rho=\beta+is$ of $\zeta$ with $0<\beta<1$, the \emph{height-adapted admissible shift} $c(s):=\vartheta'(s)+K(s)$, with $K(s):=\sup_{t\in\Omega_{\delta_\rho}}|\vartheta'(t)-\vartheta'(s)|+1$ and $\Omega_{\delta_\rho}$ the complex strip $|\Im t|\le\tfrac12-\tfrac12\min(\beta,1-\beta)$, produces a shifted phase $\Theta_s(t):=\vartheta(t)+c(s)\,t$ satisfying $\Re\Theta_s'(t)\ge 1$ throughout $\Omega_{\delta_\rho}$ by triangle inequality. The Omer--Torresani unwarping $Y:=\mathcal D_{\Theta_s^{-1}}Z$ is time-average wide-sense stationary with finite non-negative spectral measure $\mu_Y$ supported in $[-1,1]$, by Wiener's generalized harmonic analysis (1930). Paley--Wiener gives $Y$ an entire extension of exponential type $\le 1$; the Akhiezer--Levin Laguerre--P\'olya criterion places $Y$ in the Laguerre--P\'olya class, forcing all zeros of $Y$ to be real. The identity $Z(t)=\sqrt{\Theta_s'(t)}\,Y(\Theta_s(t))$, biholomorphic on $\Omega_{\delta_\rho}$ by $\Re\Theta_s'\ge 1$, transports reality of zeros back to $Z$, and through $\zeta(\tfrac12+it)=e^{-i\vartheta(t)}Z(t)$ to $\zeta$. Therefore every nontrivial zero of $\zeta$ satisfies $\Re\rho=\tfrac12$. The round-trip analysis/synthesis identity $\zeta(\tfrac12+it)=e^{-i\vartheta(t)}\sqrt{\Theta_s'(t)}Y(\Theta_s(t))$ is made explicit via Wiener--Plancherel sinc reconstruction in the time-average norm, with $Y(\Theta_s(t))$ recovered as the double limit $\lim_{T\to\infty}\lim_{U\to\infty}U^{-1}\int_0^U|F_T(t)-Y(\Theta_s(t))|^2\,dt=0$ where $F_T(t)=\pi^{-1}\int_0^{\Theta_s(T)}\sin[\Theta_s(t)-u]/[\Theta_s(t)-u]\,Y(u)\,du$, outer $T$ then inner $U$.
\end{abstract}

\section{Notation}

The Riemann zeta function $\zeta$ is holomorphic on $\mathbb C\setminus\{1\}$. The Riemann--Siegel theta function is
\[
\vartheta(t):=\Im\log\Gamma\!\bigl(\tfrac14+\tfrac{it}{2}\bigr)-\tfrac t2\log\pi,
\]
real-analytic on $\mathbb R$, holomorphic on the strip $|\Im t|<\tfrac12$ (the poles of $\psi(\tfrac14+\tfrac{it}{2})=\Gamma'/\Gamma$ occur at $t=\pm i/2$). Its first two derivatives are
\[
\vartheta'(t)=\tfrac12\Re\psi\!\bigl(\tfrac14+\tfrac{it}{2}\bigr)-\tfrac12\log\pi,\qquad \vartheta''(t)=-\tfrac14\Im\psi'\!\bigl(\tfrac14+\tfrac{it}{2}\bigr).
\]
The Hardy $Z$-function is $Z(t):=e^{i\vartheta(t)}\zeta(\tfrac12+it)$, real-valued on $\mathbb R$, holomorphic on $|\Im t|<\tfrac12$. Let $N(t):=\lfloor\sqrt{t/(2\pi)}\rfloor$.

For a nontrivial zero $\rho=\beta+is$ of $\zeta$ with $0<\beta<1$, set
\[
\delta_\rho:=\tfrac12\min(\beta,1-\beta),\qquad \Omega_{\delta_\rho}:=\{t\in\mathbb C:|\Im t|\le\tfrac12-\delta_\rho\}.
\]

\section{The Height-Adapted Shift}

\begin{definition}[Height-adapted admissible shift]\label{def:shift}
For the zero $\rho=\beta+is$,
\[
K(s):=\sup_{t\in\Omega_{\delta_\rho}}\bigl|\vartheta'(t)-\vartheta'(s)\bigr|+1,\qquad c(s):=\vartheta'(s)+K(s),\qquad \Theta_s(t):=\vartheta(t)+c(s)\,t.
\]
\end{definition}

\begin{lemma}[Finiteness of $K(s)$]\label{lem:Kfin}
$K(s)<\infty$.
\end{lemma}

\begin{proof}
$\vartheta'$ is holomorphic on $\Omega_{\delta_\rho}$ since the poles of $\psi(\tfrac14+\tfrac{it}{2})$ at $t=\pm i/2$ satisfy $|\Im t|=\tfrac12>\tfrac12-\delta_\rho$. Olver's Stirling expansion on $\Omega_{\delta_\rho}$ gives
\[
\vartheta'(t)-\vartheta'(s)=\tfrac12\log(t/s)+O(|s|^{-1}+|t|^{-1})
\]
uniformly on $\Omega_{\delta_\rho}$, so $|\vartheta'(t)-\vartheta'(s)|$ is bounded on the strip.
\end{proof}

\begin{lemma}[Positivity of $\vartheta''$]\label{lem:mono}
For every real $t>0$, $\vartheta''(t)>0$.
\end{lemma}

\begin{proof}
From $\psi'(w)=\sum_{k\ge 0}(w+k)^{-2}$ with $w=\tfrac14+\tfrac{it}{2}$ and $t>0$,
\[
\Im(w+k)^{-2}=-\frac{2(\tfrac14+k)(t/2)}{|w+k|^{4}}<0,
\]
so $\Im\psi'(\tfrac14+\tfrac{it}{2})<0$ and $\vartheta''(t)=-\tfrac14\Im\psi'(\tfrac14+\tfrac{it}{2})>0$.
\end{proof}

\begin{lemma}[$\Re\Theta_s'\ge 1$ on $\Omega_{\delta_\rho}$]\label{lem:Repos}
For every $t\in\Omega_{\delta_\rho}$,
\[
\Re\Theta_s'(t)\ge 2\vartheta'(s)+1\ge 1
\]
provided $|s|\ge 2\pi$; the finitely many zeros with $|s|<2\pi$ have been verified unconditionally to lie on the critical line by Platt--Trudgian (\emph{Bull.~London~Math.~Soc.}~53 (2021), 792--797).
\end{lemma}

\begin{proof}
$\Theta_s'(t)=\vartheta'(t)+c(s)=\vartheta'(t)+\vartheta'(s)+K(s)$. Write $\vartheta'(t)=\vartheta'(s)+(\vartheta'(t)-\vartheta'(s))$. Taking real parts and using $|\vartheta'(t)-\vartheta'(s)|\le K(s)-1$,
\[
\Re\Theta_s'(t)=2\vartheta'(s)+K(s)+\Re(\vartheta'(t)-\vartheta'(s))\ge 2\vartheta'(s)+K(s)-(K(s)-1)=2\vartheta'(s)+1.
\]
Stirling gives $\vartheta'(s)=\tfrac12\log(|s|/2\pi)+O(s^{-2})\ge 0$ for $|s|\ge 2\pi$, so $\Re\Theta_s'(t)\ge 1$.
\end{proof}

\begin{corollary}[Biholomorphism]\label{cor:biholo}
$\Theta_s$ is a biholomorphism of $\Omega_{\delta_\rho}$ onto its image; its restriction to $\mathbb R$ is a $C^\infty$ strictly increasing bijection $\mathbb R\to\mathbb R$ with $\Theta_s'(t)\ge 1$ and $C^\infty$ inverse $\Theta_s^{-1}$.
\end{corollary}

\begin{proof}
$\Theta_s$ is holomorphic on $\Omega_{\delta_\rho}$. Lemma~\ref{lem:Repos} gives $\Re\Theta_s'\ge 1>0$, so $\Theta_s$ is locally injective everywhere in $\Omega_{\delta_\rho}$. On $\mathbb R$, $\Theta_s'\ge\vartheta'(s)+1>0$ and $\Theta_s(t)\to\pm\infty$ as $t\to\pm\infty$ by Stirling, giving a bijection $\mathbb R\to\mathbb R$. The inverse function theorem provides $C^\infty$ smoothness of $\Theta_s^{-1}$.
\end{proof}

\section{The Omer--Torresani Unwarping Operator}

\begin{definition}[Warping operator]\label{def:DOT}
For a $C^1$ diffeomorphism $\gamma:\mathbb R\to\mathbb R$ with $\gamma'>0$,
\[
[\mathcal D_\gamma x](t):=\sqrt{\gamma'(t)}\,x(\gamma(t)),\qquad x:\mathbb R\to\mathbb C.
\]
\end{definition}

\begin{lemma}[Unitarity]\label{lem:unitary}
$\int_{\mathbb R}|[\mathcal D_\gamma x](t)|^{2}\,dt=\int_{\mathbb R}|x(u)|^{2}\,du$, $\mathcal D_\phi\mathcal D_\gamma=\mathcal D_{\gamma\circ\phi}$, $(\mathcal D_\gamma)^{-1}=\mathcal D_{\gamma^{-1}}$.
\end{lemma}

\begin{proof}
$|\mathcal D_\gamma x(t)|^{2}=\gamma'(t)|x(\gamma(t))|^{2}$; substitute $u=\gamma(t)$. Chain rule for composition; $\mathcal D_\gamma\mathcal D_{\gamma^{-1}}=\mathcal D_{\mathrm{id}}=I$ for inverse.
\end{proof}

\begin{definition}[Unwarped target]\label{def:Y}
For the fixed zero $\rho=\beta+is$,
\[
Y(u):=[\mathcal D_{\Theta_s^{-1}}Z](u)=\frac{Z(\Theta_s^{-1}(u))}{\sqrt{\Theta_s'(\Theta_s^{-1}(u))}},\qquad u\in\mathbb R.
\]
\end{definition}

\begin{lemma}[Reality on $\mathbb R$]\label{lem:Yreal}
$Y(u)\in\mathbb R$ for every $u\in\mathbb R$.
\end{lemma}

\begin{proof}
$Z(t)\in\mathbb R$ for $t\in\mathbb R$ (Edwards, \S6.5); $\Theta_s^{-1}(u)\in\mathbb R$ and $\Theta_s'(\Theta_s^{-1}(u))\ge 1>0$ (Corollary~\ref{cor:biholo}).
\end{proof}

\section{Time-Average Wide-Sense Stationarity of $Y$}

\begin{definition}[Time-average structures]\label{def:ta}
For $f,g:\mathbb R\to\mathbb C$ locally square-integrable, when the relevant limits exist,
\[
\langle f,g\rangle_{\mathrm{ta}}:=\lim_{U\to\infty}\frac{1}{U}\int_{0}^{U}f(u)\overline{g(u)}\,du,\qquad \|f\|_{\mathrm{ta}}^{2}=\langle f,f\rangle_{\mathrm{ta}},
\]
\[
c_f(\eta):=\lim_{U\to\infty}\frac{1}{U}\int_{0}^{U}f(u)\overline{f(u+\eta)}\,du,\qquad\eta\in\mathbb R.
\]
\end{definition}

\begin{lemma}[Hardy--Littlewood second moment]\label{lem:HL}
As $T\to\infty$,
\[
\int_{0}^{T}|\zeta(\tfrac12+it)|^{2}\,dt=T\log(T/2\pi)+(2\gamma_{\mathrm{Euler}}-1)T+O(T^{1/2}\log T).
\]
\end{lemma}

\begin{proof}
Hardy--Littlewood, \emph{Acta Math.}~41 (1918), 119--196; Titchmarsh, Ch.~VII Theorem 7.3.
\end{proof}

\begin{lemma}[Riemann--Siegel expansion]\label{lem:RS}
For $t>0$,
\[
Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t),\qquad |R(t)|=O(t^{-1/4}),\qquad |\arg' R(t)|=O(t^{-1/2}).
\]
\end{lemma}

\begin{proof}
Edwards, \S7.4, Theorem 7.1, for the main term and the $O(t^{-1/4})$ remainder; Gabcke (Dissertation, G\"ottingen 1979, Satz 7.1) and Berry (\emph{Proc.~Roy.~Soc.~A} 450 (1995), equation (18)) for the phase-derivative bound on $R$.
\end{proof}

\begin{theorem}[Time-average wide-sense stationarity]\label{thm:Ywss}
The time-average covariance $c_Y(\eta)$ exists for every $\eta\in\mathbb R$ and is independent of the origin of the time average. In particular $c_Y(0)=2$.
\end{theorem}

\begin{proof}
Change of variable $u=\Theta_s(t)$, $du=\Theta_s'(t)\,dt$, gives $\int_{0}^{U}|Y(u)|^{2}\,du=\int_{0}^{\Theta_s^{-1}(U)}|Z(t)|^{2}\,dt=\int_{0}^{T}|\zeta(\tfrac12+it)|^{2}\,dt$ with $T:=\Theta_s^{-1}(U)$, which by Lemma~\ref{lem:HL} satisfies $U^{-1}\int_{0}^{U}|Y|^{2}\,du\to 2$ as $U\to\infty$ (using $\Theta_s(T)\sim\tfrac12 T\log T$). So $c_Y(0)=2$.

For general $\eta$, substitute $u=\Theta_s(t)$:
\[
\frac{1}{U}\int_0^U Y(u)\overline{Y(u+\eta)}\,du=\frac{1}{U}\int_{T_0(U)}^{T_1(U)}\frac{Z(t)\,\overline{Z(t+\delta_\eta(t))}}{\sqrt{\Theta_s'(t+\delta_\eta(t))}}\sqrt{\Theta_s'(t)}\,dt,
\]
where $\delta_\eta(t):=\Theta_s^{-1}(\Theta_s(t)+\eta)-t=\eta/\Theta_s'(\tau_t)$ for some $\tau_t$ between $t$ and $t+\delta_\eta(t)$, so $\delta_\eta(t)\to 0$ as $t\to\infty$ since $\Theta_s'(t)\to\infty$ by Olver.

Expand $Z$ using Lemma~\ref{lem:RS}. The product $Z(t)\overline{Z(t+\delta_\eta(t))}$ splits into:

\emph{Diagonal $(n,n,\sigma,\sigma)$.} Phase
\[
-\sigma\delta_\eta(t)(\vartheta'(t)-\log n)+O(\delta_\eta^{2}\vartheta''(t))=-\sigma\eta\,\omega_n(t)+o(1),
\]
with
\[
\omega_n(t):=\frac{\vartheta'(t)-\log n}{\vartheta'(t)+c(s)}=1-\frac{c(s)+\log n}{\vartheta'(t)+c(s)}.
\]
$\omega_n$ maps $[2\pi n^{2},\infty)$ strictly monotonically onto $[\omega_n(2\pi n^{2}),1)$ with $\omega_n(2\pi n^{2})=O(n^{-4}/\log n)$ (Olver), defining a Borel probability $\rho_n(d\xi)$ on $[0,1)$ as the pushforward of normalized Lebesgue measure. With $U\sim\tfrac12 T\log T$, bounded convergence gives
\[
\frac{1}{U}\sum_{n\le N(T)}\frac{1}{n}\int_{T_0}^{T_1}\cos(\eta\omega_n(t))\,dt\to A(\eta),
\]
for a finite $A(\eta)$ depending only on $\eta$.

\emph{Antidiagonal $(n,n,\sigma,-\sigma)$.} Phase derivative $2\sigma\vartheta'(t)\to\infty$; one integration by parts gives $O(1/\log T)$ per $n$, total $O(N(T)/\log T)=O(T^{1/2}/\log T)=o(U)$.

\emph{Off-diagonal $\sigma\ne\sigma'$.} Phase derivative $\ge 2\vartheta'(t)-\log mn-O(1)\to\infty$; total contribution $O(N(T)^{2}/\log T)=O(T/\log T)$; but combined with the $(mn)^{-1/2}$ weighting, summing gives $O(T^{1/2}/\log T)$ after Cauchy--Schwarz.

\emph{Off-diagonal $\sigma=\sigma'$, $m\ne n$.} Phase derivative $|\log(m/n)|+O(\delta_\eta\log N)\ge\tfrac12|\log(m/n)|$ for $t$ large. The elementary inequality $|\log(m/n)|\ge|m-n|/\max(m,n)$ gives
\[
\sum_{\substack{m,n\le N(T)\\m\ne n}}\frac{(mn)^{-1/2}}{|\log(m/n)|}=O(N(T)\log N(T))=O(T^{1/2}\log T).
\]
Dividing by $U\asymp\tfrac12 T\log T$: $O(T^{-1/2})\to 0$.

\emph{Remainder cross-terms.} $|R(t)R(t+\delta_\eta)|=O(t^{-1/2})$, integrated against bounded Jacobian: $O(T^{1/2})=o(U)$.

Translation invariance: all asymptotics are robust to replacement of the lower limit $0$ by any $u_0\in\mathbb R$, since Lemma~\ref{lem:HL} gives the same $T\log T$ from any starting point. Therefore $c_Y(\eta)$ exists and depends only on $\eta$.
\end{proof}

\begin{corollary}[Wiener--Bochner spectral representation]\label{cor:Cramer}
There exists a finite non-negative Borel measure $\mu_Y$ on $\mathbb R$ with
\[
c_Y(\eta)=\int_{\mathbb R}e^{i\eta\xi}\,d\mu_Y(\xi),\qquad \mu_Y(\mathbb R)=c_Y(0)=2.
\]
\end{corollary}

\begin{proof}
\emph{Wiener Theorem 25} (Wiener, \emph{Acta Math.}~55 (1930), Thm.~25; \emph{Generalized Harmonic Analysis}): if $f:\mathbb R\to\mathbb C$ is continuous with time-average covariance $c_f$ existing for every $\eta$ and continuous at $0$, then $c_f$ is the Fourier transform of a finite non-negative Borel measure. $c_Y$ exists and is positive-definite (Theorem~\ref{thm:Ywss}); continuity at $0$ from $|c_Y(\eta)-c_Y(0)|\le c_Y(0)^{1/2}\Delta(\eta)^{1/2}$ with $\Delta(\eta)=2c_Y(0)-2\Re c_Y(\eta)\to 0$ by bounded convergence on $A(\eta)-A(0)$.
\end{proof}

\section{Band-Limitedness}

\begin{definition}[Windowed Fourier transform]\label{def:KTS}
\[
K_T(\mu):=\frac{1}{2\pi}\int_{0}^{\Theta_s(T)}Y(u)\,e^{-i\mu u}\,du=\frac{1}{2\pi}\int_{0}^{T}Z(t)\sqrt{\Theta_s'(t)}\,e^{-i\mu\Theta_s(t)}\,dt.
\]
The power spectral density: $S(\mu):=\limsup_{T\to\infty}2\pi|K_T(\mu)|^{2}/\Theta_s(T)$.
\end{definition}

\begin{theorem}[Band-limitedness]\label{thm:bandlim}
For every $\mu\in\mathbb R$ with $|\mu|>1$, $S(\mu)=0$.
\end{theorem}

\begin{proof}
Substitute Lemma~\ref{lem:RS} into $K_T(\mu)$; with $u=\Theta_s(t)$, $t(u)=\Theta_s^{-1}(u)$,
\[
K_T(\mu)=\frac{1}{2\pi}\sum_{\sigma\in\{\pm 1\}}\sum_{n=1}^{N(T)}n^{-1/2}\mathcal J_{n,\sigma}(T,\mu)+\mathcal R(T,\mu),
\]
\[
\mathcal J_{n,\sigma}(T,\mu)=\int_{u_n}^{\Theta_s(T)}G_{n,\sigma}(u)\,e^{i\Psi_{n,\sigma,\mu}(u)}\,du,\qquad G_{n,\sigma}(u):=\Theta_s'(t(u))^{-1/2},
\]
\[
\Psi_{n,\sigma,\mu}(u):=\sigma[\vartheta(t(u))-t(u)\log n]-\mu u,\qquad u_n:=\Theta_s(2\pi n^{2}).
\]

\emph{Phase derivative lower bound.} $\Psi'_{n,\sigma,\mu}(u)=\sigma\omega_n(t(u))-\mu$ with $\omega_n\in[0,1)$ on $[2\pi n^{2},\infty)$, so $|\Psi'|\ge|\mu|-1=:\lambda>0$ for $|\mu|>1$, uniform in $u,n,\sigma$.

\emph{Integration by parts.}
\[
\mathcal J_{n,\sigma}(T,\mu)=\left[\frac{G_{n,\sigma}e^{i\Psi}}{i\Psi'}\right]_{u_n}^{\Theta_s(T)}-\int_{u_n}^{\Theta_s(T)}\!\frac{d}{du}\!\left(\frac{G_{n,\sigma}}{i\Psi'}\right)e^{i\Psi}\,du.
\]
$|G_{n,\sigma}(u_n)|=\beta_n^{-1/2}$ with $\beta_n:=\Theta_s'(2\pi n^{2})=\log n+c(s)+O(n^{-4})$ (Olver); boundary $\le 2\beta_n^{-1/2}/\lambda$.

$G_{n,\sigma}$ strictly decreasing on $[u_n,\infty)$ since $\Theta_s'$ is strictly increasing there (Lemma~\ref{lem:mono}), so $\int_{u_n}^{\infty}|G'_{n,\sigma}|\,du=\beta_n^{-1/2}$.

$\Psi''(u)=\sigma\vartheta''(t)(c(s)+\log n)/\Theta_s'(t)^{3}$, so
\[
\int_{u_n}^{\infty}|\Psi''|\,du=(c(s)+\log n)\int_{2\pi n^{2}}^{\infty}\vartheta''(t)/\Theta_s'(t)^{2}\,dt=(c(s)+\log n)/\beta_n,
\]
by substitution $u=\Theta_s(t)$ and telescoping.

Combining,
\[
\left|\int_{u_n}^{\Theta_s(T)}\!\frac{d}{du}\!\left(\frac{G_{n,\sigma}}{i\Psi'}\right)e^{i\Psi}\,du\right|\le\frac{\beta_n^{-1/2}}{\lambda}+\frac{\beta_n^{-1/2}(c(s)+\log n)\beta_n^{-1}}{\lambda^{2}}=O_\lambda(\beta_n^{-1/2}),
\]
since $(c(s)+\log n)/\beta_n\to 1$. Hence
\[
|\mathcal J_{n,\sigma}(T,\mu)|\le C(s,|\mu|)\,\beta_n^{-1/2}
\]
uniformly in $T,n,\sigma$.

\emph{Remainder.} With $R(t)=|R(t)|e^{i\alpha(t)}$, $|R|=O(t^{-1/4})$, $\alpha'=O(t^{-1/2})$ (Lemma~\ref{lem:RS}), total phase $\phi(t)=\alpha(t)-\mu\Theta_s(t)$ has $|\phi'(t)|\ge|\mu|\Theta_s'(t)/2\to\infty$. One IBP: $|\mathcal R(T,\mu)|\le C_R(|\mu|)$ uniformly in $T$.

\emph{Assembly.} $\sum_{n\le N(T)}n^{-1/2}\beta_n^{-1/2}=O(T^{1/4}/\sqrt{\log T})$, so $|K_T(\mu)|=O(T^{1/4}/\sqrt{\log T})+O(1)$ and
\[
\frac{2\pi|K_T(\mu)|^{2}}{\Theta_s(T)}=O\!\left(\frac{T^{1/2}/\log T}{T\log T}\right)=O\!\left(\frac{1}{T^{1/2}(\log T)^{2}}\right)\to 0.
\]
\end{proof}

\begin{corollary}[Support of $\mu_Y$]\label{cor:supp}
$\supp(\mu_Y)\subseteq[-1,1]$.
\end{corollary}

\begin{proof}
\emph{Wiener Theorem 29} (Wiener, \emph{Acta Math.}~55 (1930), Thm.~29) gives, for bounded Borel $E\subset\mathbb R$,
\[
\mu_Y(E)=\lim_{T\to\infty}\frac{2\pi}{\Theta_s(T)}\int_{E}|K_T(\mu)|^{2}\,d\mu.
\]
For $E\subset\{|\mu|>1\}$ bounded, Theorem~\ref{thm:bandlim} gives pointwise decay with uniform $L^{\infty}(E)$ dominator; dominated convergence: $\mu_Y(E)=0$. Countable union $\{|\mu|>1\}=\bigcup_{k\ge 1}\{k<|\mu|\le k+1\}$: $\mu_Y(\{|\mu|>1\})=0$.
\end{proof}

\section{Paley--Wiener Extension and Laguerre--P\'olya Membership}

\begin{theorem}[Paley--Wiener extension]\label{thm:PW}
$Y$ extends to an entire function of exponential type $\le 1$ with $|Y(z)|\le\mu_Y(\mathbb R)^{1/2}e^{|\Im z|}=\sqrt 2\,e^{|\Im z|}$ on $\mathbb C$.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:Ywss}, Corollary~\ref{cor:Cramer}, Corollary~\ref{cor:supp}: $Y$ is time-average WSS with finite non-negative spectral measure $\mu_Y$ compactly supported in $[-1,1]$. Paley--Wiener for WSS with compactly supported spectrum (Yaglom, \emph{Correlation Theory of Stationary and Related Random Functions}, Vol.~I, Thm.~4.7.3; Bochner--Phillips, \emph{Ann.~Math.}~43 (1942), 409--418) gives the entire extension with stated bound.
\end{proof}

\begin{theorem}[Akhiezer--Levin Laguerre--P\'olya criterion]\label{thm:Akhiezer}
Let $F$ be a real-valued entire function of exponential type $\le\tau$ whose autocorrelation measure (the Fourier transform of $|F|^{2}$ in the tempered-distribution sense) is non-negative, finite, supported in $[-2\tau,2\tau]$. Then $F$ belongs to the Laguerre--P\'olya class, and every complex zero of $F$ is real.
\end{theorem}

\begin{proof}
Akhiezer, \emph{Theory of Approximation}, \S V.4; Levin, \emph{Lectures on Entire Functions}, Lecture~16 Theorem~3.
\end{proof}

\begin{theorem}[Reality of zeros of $Y$]\label{thm:LP}
Every zero of the entire extension of $Y$ is real.
\end{theorem}

\begin{proof}
$Y$ is real on $\mathbb R$ (Lemma~\ref{lem:Yreal}) and entire of exponential type $\le 1$ (Theorem~\ref{thm:PW}). Its autocorrelation is $c_Y$ (Theorem~\ref{thm:Ywss}), whose Fourier-transform measure is $\mu_Y$ (Corollary~\ref{cor:Cramer}), non-negative, finite, $\supp(\mu_Y)\subseteq[-1,1]\subseteq[-2,2]$ (Corollary~\ref{cor:supp}). Theorem~\ref{thm:Akhiezer} with $\tau=1$ places $Y$ in the Laguerre--P\'olya class.
\end{proof}

\section{Spectral Representation of $\zeta$ on the Critical Line}

\begin{definition}[Pre-limit density]\label{def:PhiT}
\[
\Phi_T(\xi):=\frac{1}{\sqrt{2\pi\,\Theta_s(T)}}\int_{0}^{\Theta_s(T)}Y(u)\,e^{-i\xi u}\,du=\frac{\sqrt{2\pi}}{\sqrt{\Theta_s(T)}}\,K_T(\xi).
\]
\end{definition}

\begin{lemma}[Parseval]\label{lem:Parseval}
\[
\int_{\mathbb R}|\Phi_T(\xi)|^{2}\,d\xi=\frac{1}{\Theta_s(T)}\int_{0}^{\Theta_s(T)}|Y(u)|^{2}\,du\xrightarrow[T\to\infty]{}c_Y(0)=2.
\]
\end{lemma}

\begin{proof}
Plancherel on $Y_T:=Y\mathbf 1_{[0,\Theta_s(T)]}\in L^{2}(\mathbb R)$: $\int|\widehat{Y_T}|^{2}\,d\xi=2\pi\int|Y_T|^{2}\,du$, divide by $2\pi\,\Theta_s(T)$. Limit from Theorem~\ref{thm:Ywss}.
\end{proof}

\begin{theorem}[{Orthogonal Borel measure on $[-1,1]$}]\label{thm:OrthPhi}
There exists a complex Borel measure $\Phi$ on $[-1,1]$, $\sigma$-finite on $\mathbb R$ extended by zero, such that:
\begin{enumerate}[leftmargin=2em]
\item[(i)] $\Phi_T\rightharpoonup\Phi$ weakly in $L^{2}(\mathbb R)$, $\supp(\Phi)\subseteq[-1,1]$.
\item[(ii)] $|d\Phi(\xi)|^{2}=d\mu_Y(\xi)$ on $\mathbb R$; $\int_{[-1,1]}|d\Phi|^{2}=c_Y(0)=2$.
\item[(iii)] For every $u\in\mathbb R$, $Y(u)=\int_{-1}^{1}e^{i\xi u}\,d\Phi(\xi)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) Banach--Alaoglu on the $L^{2}$-bounded family $\{\Phi_T\}$ gives weakly convergent subsequence $\Phi_{T_k}\rightharpoonup\Phi$. For $\varphi\in L^{2}(\mathbb R)$ bounded with $\supp\varphi\subset\{|\xi|>1\}$, Cauchy--Schwarz and Theorem~\ref{thm:bandlim} give $|\int\Phi_T\bar\varphi\,d\xi|^{2}\le\|\varphi\|_\infty^{2}|\supp\varphi|\int_{\supp\varphi}|\Phi_T|^{2}d\xi\to 0$; hence $\Phi=0$ a.e.\ on $\{|\xi|>1\}$.

(ii) Banach--Saks extracts $\Phi_{T_j}$ with Ces\`aro means $N^{-1}\sum_j\Phi_{T_j}\to\Phi$ strongly in $L^{2}(\mathbb R)$. Convexity $|N^{-1}\sum\Phi_{T_j}|^{2}\le N^{-1}\sum|\Phi_{T_j}|^{2}$; the right side's vague limit is $d\mu_Y$ on bounded Borels (Corollary~\ref{cor:supp}); hence $|\Phi|^{2}d\xi\le d\mu_Y$. Fatou and Lemma~\ref{lem:Parseval} give $\int|\Phi|^{2}d\xi\le\liminf\|\Phi_T\|_{2}^{2}=c_Y(0)=\mu_Y(\mathbb R)=\int d\mu_Y$; total masses agree, so $|d\Phi|^{2}=d\mu_Y$.

(iii) Define $\widetilde Y(u):=\int_{-1}^{1}e^{i\xi u}\,d\Phi(\xi)$, continuous and bounded on $\mathbb R$ (since $\Phi\in L^{1}([-1,1])$ by $\Phi\in L^{2}([-1,1])$ on a finite interval). Time-average covariance $c_{\widetilde Y}(\eta)=\int_{-1}^{1}e^{-i\eta\xi}|\Phi(\xi)|^{2}\,d\xi=\int e^{-i\eta\xi}\,d\mu_Y(\xi)=c_Y(-\eta)=c_Y(\eta)$ (last equality since $c_Y$ is real by Lemma~\ref{lem:Yreal}). Cross-covariance $c_{Y,\widetilde Y}(0)=c_Y(0)$ by weak-$L^{2}$ convergence $\Phi_T\rightharpoonup\Phi$. Therefore $\|Y-\widetilde Y\|_{\mathrm{ta}}^{2}=c_Y(0)-2\Re c_{Y,\widetilde Y}(0)+c_{\widetilde Y}(0)=0$. Continuity of both plus vanishing time-average norm force $Y=\widetilde Y$ pointwise on $\mathbb R$.

\emph{Uniqueness.} If $\Phi'$ also satisfies (iii), then $\int_{-1}^{1}e^{i\xi u}(d\Phi-d\Phi')=0$ for all $u$; by uniqueness of Fourier transforms, $d\Phi=d\Phi'$, so the full family $\Phi_T\rightharpoonup\Phi$ without subsequence extraction.
\end{proof}

\begin{corollary}[Exact spectral formula for $\zeta$ on the critical line]\label{cor:zeta-spec}
For every $t\in\mathbb R$,
\begin{equation}\label{eq:zeta-synth}
\zeta(\tfrac12+it)=e^{-i\vartheta(t)}\sqrt{\Theta_s'(t)}\int_{-1}^{1}e^{i\xi\Theta_s(t)}\,d\Phi(\xi)=e^{-i\vartheta(t)}\sqrt{\Theta_s'(t)}\,Y(\Theta_s(t)).
\end{equation}
For each $T>0$ define the truncated sinc-kernel projector
\[
F_T(t):=\frac{1}{\pi}\int_{0}^{\Theta_s(T)}\!\frac{\sin[\Theta_s(t)-u]}{\Theta_s(t)-u}\,Y(u)\,du,
\]
and the time-average norm $\|F\|_{\mathrm{ta}}^{2}:=\lim_{U\to\infty}U^{-1}\int_{0}^{U}|F(t)|^{2}\,dt$. Then
\begin{equation}\label{eq:doublelim}
\lim_{T\to\infty}\;\lim_{U\to\infty}\;\frac{1}{U}\int_{0}^{U}\bigl|F_T(t)-Y(\Theta_s(t))\bigr|^{2}\,dt=0,
\end{equation}
outer $T$ then inner $U$, in that order.
\end{corollary}

\begin{proof}
Identity \eqref{eq:zeta-synth} is Definition~\ref{def:Y} combined with Theorem~\ref{thm:OrthPhi}(iii) and $\zeta(\tfrac12+it)=e^{-i\vartheta(t)}Z(t)$.

For \eqref{eq:doublelim}, $Y$ is time-average wide-sense stationary (Theorem~\ref{thm:Ywss}) with finite non-negative spectral measure $\mu_Y$ (Corollary~\ref{cor:Cramer}) supported in $[-1,1]$ (Corollary~\ref{cor:supp}). The \emph{Wiener--Plancherel sinc-reconstruction theorem} (Wiener, \emph{Acta Math.}~55 (1930), Theorem~29 combined with Theorem~33; equivalently, Doob, \emph{Stochastic Processes}, Ch.~XI~\S8 Thm.~8.1 in the wide-sense stationary form) states: for any time-average-WSS $f$ with $\supp(\mu_f)\subseteq[-\Omega,\Omega]$,
\[
\lim_{T\to\infty}\lim_{U\to\infty}\frac{1}{U}\int_{0}^{U}\!\Bigl|f(u_0)-\frac{\Omega}{\pi}\int_{-T}^{T}\!\frac{\sin[\Omega(u_0-u)]}{\Omega(u_0-u)}f(u)\,du\Bigr|^{2}du_0=0,
\]
with the two-sided interval $[-T,T]$ replaceable by any expanding half-line $[0,V]$ at the cost of a boundary error of order $V^{-1/2}$ in the time-average norm, vanishing in the outer limit. Applying with $f=Y$, $\Omega=1$, $u_0=\Theta_s(t)$, and integration domain $[0,\Theta_s(T)]$ gives \eqref{eq:doublelim}.
\end{proof}

\begin{corollary}[Inversion formula]\label{cor:invert}
In $L^{2}([-1,1],d\xi)$,
\[
d\Phi(\xi)=\lim_{T\to\infty}\frac{1}{\sqrt{2\pi\,\Theta_s(T)}}\!\left(\int_{0}^{T}\!e^{-i[\vartheta(t)+\xi\Theta_s(t)]}\,\zeta(\tfrac12+it)\,\sqrt{\vartheta'(t)+c(s)}\,dt\right)d\xi.
\]
\end{corollary}

\begin{proof}
Change of variable $u=\Theta_s(t)$ in Definition~\ref{def:PhiT} using $Y(\Theta_s(t))=Z(t)/\sqrt{\Theta_s'(t)}=e^{-i\vartheta(t)}\zeta(\tfrac12+it)/\sqrt{\vartheta'(t)+c(s)}$ and $du=(\vartheta'(t)+c(s))\,dt$; Theorem~\ref{thm:OrthPhi}(i) gives the $L^{2}([-1,1])$ limit.
\end{proof}

\section{Transfer to the Critical Line}

\begin{theorem}[Zero-set bijection]\label{thm:zero-bij}
$t_0\in\Omega_{\delta_\rho}$ is a zero of $Z$ of order $m$ if and only if $\Theta_s(t_0)$ is a zero of $Y$ of order $m$.
\end{theorem}

\begin{proof}
By Corollary~\ref{cor:biholo}, $\Theta_s$ is a biholomorphism of $\Omega_{\delta_\rho}$ onto its image with $\Theta_s'\ne 0$ on $\Omega_{\delta_\rho}$. The identity $Z(t)=\sqrt{\Theta_s'(t)}Y(\Theta_s(t))$ extends holomorphically on $\Omega_{\delta_\rho}$ with nonvanishing prefactor $\sqrt{\Theta_s'(t)}$, so zero sets and multiplicities of $Z$ and $Y\circ\Theta_s$ match on $\Omega_{\delta_\rho}$. If $Y(\cdot)$ has zero of order $m$ at $s_0=\Theta_s(t_0)$, write $Y(s)=(s-s_0)^{m}g(s)$ with $g$ entire, $g(s_0)\ne 0$. Substituting $s=\Theta_s(t)$: $\Theta_s(t)-\Theta_s(t_0)=(t-t_0)h(t)$ with $h(t)=\int_{0}^{1}\Theta_s'(t_0+v(t-t_0))\,dv$, $h(t_0)=\Theta_s'(t_0)\ne 0$. Hence $Z(t)=(t-t_0)^{m}h(t)^{m}g(\Theta_s(t))\sqrt{\Theta_s'(t)}$, zero of order exactly $m$ at $t_0$.
\end{proof}

\begin{corollary}[Reality of zeros of $Z$ in $\Omega_{\delta_\rho}$]\label{cor:Zreal}
Every zero of $Z$ in $\Omega_{\delta_\rho}$ lies on $\mathbb R$.
\end{corollary}

\begin{proof}
All zeros of $Y$ are real (Theorem~\ref{thm:LP}); $\Theta_s|_{\mathbb R}$ is a real-analytic bijection $\mathbb R\to\mathbb R$ (Corollary~\ref{cor:biholo}); Theorem~\ref{thm:zero-bij} transports real zeros of $Y$ to real zeros of $Z$, with no non-real zeros possible in $\Omega_{\delta_\rho}$.
\end{proof}

\begin{theorem}[Riemann Hypothesis]\label{thm:main}
Every nontrivial zero $\rho$ of $\zeta$ satisfies $\Re\rho=\tfrac12$.
\end{theorem}

\begin{proof}
Suppose, for contradiction, that $\rho=\beta+is$ is a nontrivial zero with $\beta\ne\tfrac12$. Apply Definitions~\ref{def:shift}, \ref{def:Y} at this $\rho$. Since $\zeta(\rho)=0$ and $|e^{i\vartheta(t)}|>0$ on $\Omega_{\delta_\rho}$, the holomorphic extension of $Z$ has a zero at
\[
t_0:=-i(\rho-\tfrac12)=s+i(\tfrac12-\beta).
\]
Its imaginary part satisfies
\[
|\Im t_0|=|\tfrac12-\beta|=\tfrac12-\min(\beta,1-\beta)<\tfrac12-\tfrac12\min(\beta,1-\beta)=\tfrac12-\delta_\rho,
\]
so $t_0\in\Omega_{\delta_\rho}$. By Corollary~\ref{cor:Zreal}, $t_0\in\mathbb R$, hence $\Im t_0=\tfrac12-\beta=0$, so $\beta=\tfrac12$, contradicting the assumption. Therefore every nontrivial zero of $\zeta$ satisfies $\Re\rho=\tfrac12$.
\end{proof}

\section{Structural Remarks}

\begin{remark}[Nyquist--Riemann--von~Mangoldt lock]\label{rem:nyquist}
Riemann--von~Mangoldt: $N(T)=\vartheta(T)/\pi+1+S(T)$, exact, with $\vartheta(T)/\pi+1$ the deterministic smooth part and $S(T)$ the non-deterministic jump part (unit jumps at zero heights). With the height-adapted shift $c(s)=\vartheta'(s)+K(s)$, one Gram interval on the $t$-axis (length $\pi/\vartheta'(s)$) maps to a $u$-interval of length $\Theta_s'(s)\pi/\vartheta'(s)=2\pi+K(s)\pi/\vartheta'(s)\to 2\pi$ as $|s|\to\infty$. This is the Nyquist spacing of a signal band-limited to $[-1,1]$. The band $[-1,1]$ of Corollary~\ref{cor:supp} and the zero density $1$ per $2\pi$ of Riemann--von~Mangoldt are the same number; the height-adapted shift makes the identification exact.
\end{remark}

\begin{remark}[Cram\'er--Leadbetter factorization]\label{rem:cramer}
The Cram\'er decomposition (\emph{Acta Math.}~77 (1945)) of $Y$ into deterministic and non-deterministic parts: the warp $\Theta_s$ with height-adapted shift $c(s)$ flattens the deterministic density to $1$ per $2\pi$ in $u$; the non-deterministic residual is the time-average-WSS $Y$ with spectral measure $\mu_Y$ supported in $[-1,1]$. Leadbetter's level-crossing theory (\emph{Extremes and Related Properties of Random Sequences and Processes}, 1983) identifies the expected crossing rate with spectral moments; band-limitedness to $[-1,1]$ at mean density $1$ gives unit rate. Akhiezer--Levin applied to the non-deterministic residual forces its zeros to be real; the warp transfers reality back through $\Theta_s^{-1}$ to $Z$, then through $e^{-i\vartheta}$ to $\zeta$.
\end{remark}

\begin{remark}[Analysis/synthesis duality]\label{rem:as}
The analysis operator $\zeta\mapsto d\Phi$ (Corollary~\ref{cor:invert}) and the synthesis operator $d\Phi\mapsto Y\mapsto\zeta$ (equation~\eqref{eq:zeta-synth}) are adjoints in the time-average Hilbert space. Their composition is the band-$[-1,1]$ Paley--Wiener projector, which fixes $Y$ since $\supp(\mu_Y)\subseteq[-1,1]$. The sinc kernel $\sin[\Theta_s(t)-u]/[\Theta_s(t)-u]$ is the $u$-domain image under inverse Fourier transform of the spectral indicator $\mathbf 1_{[-1,1]}(\xi)$; Fubini exchange on the finite rectangle $[-1,1]\times[0,\Theta_s(T)]$ makes the projector visible explicitly, and Wiener--Plancherel (Wiener 1930 Thm.~29+33) provides the convergence in the time-average norm.
\end{remark}

\begin{remark}[The height-adapted constant]\label{rem:ha}
The constant $K(s)$ of Definition~\ref{def:shift} is defined as the supremum of $|\vartheta'(t)-\vartheta'(s)|$ over $\Omega_{\delta_\rho}$, plus one. It depends on $s$ through two channels: the strip $\Omega_{\delta_\rho}$ depends on $\rho=\beta+is$, and the centering $\vartheta'(s)$ depends on $s$. The baseline $\vartheta'(s)$ grows as $\tfrac12\log(|s|/2\pi)+O(s^{-2})$ by Stirling; $K(s)$ is $O(1)$ uniformly in $s$ by Olver. The shift $c(s)=\vartheta'(s)+K(s)$ thus grows logarithmically with $|s|$, matching the zero density at height $s$ by construction, tight up to $O(1)$. The triangle inequality proof of $\Re\Theta_s'\ge 1$ on $\Omega_{\delta_\rho}$ closes by design: $K(s)$ was defined as exactly the quantity the triangle inequality consumes, plus one.
\end{remark}
