\begin{abstract}
Let $\zeta$ denote the Riemann zeta function and $Z(t)=e^{i\vartheta(t)}\zeta(\tfrac12+it)$
the Hardy $Z$-function, with $\vartheta$ the Riemann--Siegel theta function.
Fix a shift $c>c_*:=-\inf_{t\in\mathbb R}\vartheta'(t)$ and set $\Theta(t):=\vartheta(t)+ct$,
a $C^\infty$ bijection $\mathbb R\to\mathbb R$ with $\Theta'>0$.
Let $Y:\mathbb R\to\mathbb R$ be the Omer--Torresani unwarping
$Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$.
The time-average covariance $c_Y(\eta)$ exists for every $\eta\in\mathbb R$, is
positive-definite and continuous, and Bochner's theorem yields a finite non-negative
spectral measure $\mu_Y$ with $\mu_Y(\mathbb R)=2$.
A single integration by parts on the Riemann--Siegel mode expansion, in which the
$\vartheta''$ contribution cancels algebraically, gives
$\operatorname{supp}\mu_Y\subseteq[-1,1]$.
The Paley--Wiener theorem for functions admitting a compactly supported Bochner
spectral measure extends $Y$ to an entire function of exponential type $\le 1$
with $|Y(z)|\le\sqrt{2}\,e^{|\Im z|}$; the Akhiezer--Laguerre--P\'olya criterion
then places $Y$ in the Laguerre--P\'olya class, so every complex zero of $Y$ is real.
For a nontrivial zero $\rho=\beta+i\gamma$ of $\zeta$ with $\beta\in(0,\tfrac12]$
(using $\rho\mapsto 1-\bar\rho$), parametrize the vertical segment by
$t(s):=\gamma+is$, $s\in\mathbb C$, and set $F(s):=Z(t(s))$ and $G(s):=\sqrt{\Theta'(t(s))}\,Y(\Theta(t(s)))$,
which are holomorphic on a connected open set $N\subset\mathbb C$ containing both
the real segment $[0,\tfrac12-\beta]$ and an interval of the imaginary axis
$\{ir:|r|<\varepsilon\}$.
On the imaginary axis, $t(ir)=\gamma-r\in\mathbb R$, and Lemma~\ref{lem:Zfact}
gives $F(ir)=Z(\gamma-r)=\sqrt{\Theta'(\gamma-r)}\,Y(\Theta(\gamma-r))=G(ir)$.
The 1D identity theorem on $N$ yields $F\equiv G$, and evaluating at
$s=\tfrac12-\beta$ gives $Z(t_\rho)=\sqrt{\Theta'(t_\rho)}\,Y(\Theta(t_\rho))$
where $t_\rho:=\gamma+i(\tfrac12-\beta)$.
Since $\tfrac12+it_\rho=\rho$, $Z(t_\rho)=0$; with $\sqrt{\Theta'(t_\rho)}\ne 0$,
$Y(\Theta(t_\rho))=0$; reality of zeros of $Y$ gives $\Theta(t_\rho)\in\mathbb R$;
a direct computation of $\Im\Theta(t_\rho)=\int_0^{\tfrac12-\beta}\Re\Theta'(\gamma+is)\,ds$,
combined with $\Re\Theta'(\gamma+is)>0$ along the segment
(choice of $c$ large enough), forces $\tfrac12-\beta=0$.
\end{abstract}

\section{The Shifted Phase and the Unwarped Function}

\begin{lemma}[Lower bound of $\vartheta'$ on $\mathbb R$]\label{lem:cstar}
$c_*:=-\inf_{t\in\mathbb R}\vartheta'(t)$ is finite.
\end{lemma}

\begin{proof}
$\vartheta'(t)=\tfrac12\Re\psi(\tfrac14+\tfrac{it}{2})-\tfrac12\log\pi$ is continuous
on $\mathbb R$ and, by the classical digamma asymptotic applied to
$\psi(\tfrac14+\tfrac{it}{2})$, satisfies $\vartheta'(t)=\tfrac12\log|t/(2\pi)|+O(|t|^{-2})$
as $|t|\to\infty$, hence $\vartheta'(t)\to+\infty$.
A continuous function on $\mathbb R$ tending to $+\infty$ at infinity is bounded below.
\end{proof}

\begin{definition}[Admissible shift and shifted phase]\label{def:Theta}
Fix $c>c_*$ and set $\Theta(t):=\vartheta(t)+ct$.
\end{definition}

\begin{lemma}[Properties of $\Theta$]\label{lem:Theta}
For $c>c_*$, $\Theta'(t)=\vartheta'(t)+c>0$ for every $t\in\mathbb R$, and
$\Theta:\mathbb R\to\mathbb R$ is a $C^\infty$ bijection with $C^\infty$ inverse.
\end{lemma}

\begin{proof}
$c>c_*=-\inf\vartheta'$ gives $\vartheta'+c>0$ on $\mathbb R$; positivity and
$\Theta(\pm\infty)=\pm\infty$ give bijectivity; smoothness of the inverse by the
inverse function theorem.
\end{proof}

\begin{definition}[Omer--Torresani warping]\label{def:DOT}
For a $C^1$ diffeomorphism $\gamma:\mathbb R\to\mathbb R$ with $\gamma'>0$,
\[
[\mathcal D_\gamma x](u):=\sqrt{\gamma'(u)}\,x(\gamma(u)),\qquad x:\mathbb R\to\mathbb C.
\]
\end{definition}

\begin{lemma}[Unitarity]\label{lem:unitary}
$\mathcal D_\gamma:L^2(\mathbb R)\to L^2(\mathbb R)$ is unitary with
$\mathcal D_\gamma^{-1}=\mathcal D_{\gamma^{-1}}$.
\end{lemma}

\begin{proof}
Change of variable $u=\gamma(t)$, $du=\gamma'(t)dt$.
\end{proof}

\begin{definition}[Unwarped $Y$]\label{def:Y}
$Y:\mathbb R\to\mathbb R$, $Y(u):=Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}=[\mathcal D_{\Theta^{-1}}Z](u)$.
\end{definition}

\begin{lemma}[Pointwise factorization identity]\label{lem:Zfact}
For every $u\in\mathbb R$, with $t:=\Theta^{-1}(u)$,
\[
Z(t)=\sqrt{\Theta'(t)}\,Y(\Theta(t))=\sqrt{\Theta'(t)}\,Y(u).
\]
\end{lemma}

\begin{proof}
From Definition~\ref{def:Y}, $Y(\Theta(t))=Z(t)/\sqrt{\Theta'(t)}$; multiply both
sides by $\sqrt{\Theta'(t)}$.
\end{proof}

\section{Time-Average WSS of $Y$}

\begin{definition}[Time-average covariance]\label{def:covg}
$c_f(\eta):=\lim_{U\to\infty}U^{-1}\int_0^U f(u)\overline{f(u+\eta)}\,du$
when the limit exists.
\end{definition}

\begin{lemma}[Hardy--Littlewood second moment]\label{lem:HL}
$\int_0^T|\zeta(\tfrac12+it)|^2dt=T\log(T/2\pi)+(2\gamma-1)T+O(T^{1/2}\log T)$.
\end{lemma}

\begin{proof}
Titchmarsh, Ch.~VII, Thm.~7.3.
\end{proof}

\begin{lemma}[Riemann--Siegel]\label{lem:RS}
For $t>0$,
\[
Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\vartheta(t)-t\log n)+R(t),
\qquad N(t)=\lfloor\sqrt{t/(2\pi)}\rfloor,\quad R(t)=O(t^{-1/4}),
\]
and $(d/dt)\arg R(t)=O(t^{-1/2})$.
\end{lemma}

\begin{proof}
Edwards \S7.4 Thm.~7.1; Gabcke Satz~7.1; Berry eq.~18.
\end{proof}

\begin{lemma}[Instantaneous $u$-frequency ratio]\label{lem:omegabound}
For every $n\ge 1$ and every $t\ge 2\pi n^2$,
\[
\omega_n(t):=\frac{\vartheta'(t)-\log n}{\vartheta'(t)+c}\in[0,1).
\]
\end{lemma}

\begin{proof}
\emph{Sub-claim (A): stationary-phase threshold.}
$\vartheta'(t)\ge\log n$ for every $t\ge 2\pi n^2$.
The mode $n^{-1/2}\cos(\vartheta(t)-t\log n)$ has stationary point at $t=2\pi n^2$
where $\vartheta'(t)=\log n$ (Edwards, \S7.4, preceding Thm.~7.1).
Verification: $\vartheta''(t)>0$ on $(0,\infty)$ since
$\vartheta''(t)=-\tfrac14\Im\psi'(\tfrac14+\tfrac{it}{2})$ and
$\Im(s+k)^{-2}=-2(\tfrac14+k)(t/2)|s+k|^{-4}<0$ for $t>0$, $k\ge 0$, so
$\vartheta'$ is strictly increasing on $(0,\infty)$.
The digamma asymptotic at $t=2\pi n^2$ gives
$\vartheta'(2\pi n^2)=\log n+O(n^{-2})$; monotonicity propagates
$\vartheta'(t)\ge\log n$ to $[2\pi n^2,\infty)$.

\emph{Sub-claim (B): admissible-shift positivity.}
$\vartheta'(t)+c>0$ on $\mathbb R$ by $c>c_*$ (Lemma~\ref{lem:Theta}).

\emph{Combination.}
(A) gives numerator $\ge 0$; (B) gives denominator $>0$; hence $\omega_n(t)\ge 0$.
Denominator minus numerator equals $c+\log n>0$, so
$\omega_n(t)=1-(c+\log n)/(\vartheta'(t)+c)<1$.
\end{proof}

\begin{theorem}[Time-average WSS of $Y$]\label{thm:Ywss}
$c_Y(\eta)$ exists for every $\eta\in\mathbb R$ and depends only on $\eta$.
\end{theorem}

\begin{proof}
$U^{-1}\int_0^U|Y|^2du\to 2$ by
$\int_0^U|Y|^2du=\int_0^{\Theta^{-1}(U)}|Z|^2dt\sim T\log T\sim 2U$
via Lemma~\ref{lem:HL} and $U=\Theta(T)\sim\tfrac T2\log T$.

Substitute $u=\Theta(t)$ in $c_Y(\eta)$; set
$\delta_\eta(t):=\Theta^{-1}(\Theta(t)+\eta)-t=\eta/\Theta'(\tau_t)\to 0$.
Insert Riemann--Siegel (Lemma~\ref{lem:RS}); the product
$Z(t)\overline{Z(t+\delta_\eta(t))}$ splits into diagonal $(m=n,\sigma=\sigma')$,
antidiagonal $(m=n,\sigma=-\sigma')$, off-diagonal $(m\ne n)$, and remainder
cross-terms.

\emph{Off-diagonal $\sigma\ne\sigma'$.}
$|d\Phi/dt|\ge 2\vartheta'(t)-O(1)\to\infty$;
single IBP gives $O((\log T)^{-1})\sum_{m,n}(mn)^{-1/2}=o(U)$.

\emph{Off-diagonal $\sigma=\sigma'$, $m\ne n$.}
$|d\Phi/dt|\ge\tfrac12|\log(m/n)|$;
$\sum_{m\ne n\le N(T)}(mn)^{-1/2}/|\log(m/n)|=O(T^{1/2}\log T)=o(U)$.

\emph{Remainder.}
$|R|=O(t^{-1/4})$ cross-terms Cauchy--Schwarz to $O(T^{1/2})=o(U)$.

\emph{Diagonal.}
$(n,n,\sigma,\sigma)$ contributes, after phase expansion
$\delta_\eta(t)[\vartheta'(t)-\log n]\to\eta\omega_n(t)$
with $\omega_n(t)\in[0,1)$ for $t\ge 2\pi n^2$ by Lemma~\ref{lem:omegabound}:
\[
\frac{1}{U}\sum_{n\le N(T)}\frac{1}{n}\int_{T_0}^{T_1}\cos(\eta\omega_n(t))\,dt
\longrightarrow A(\eta),
\]
finite limit depending only on $\eta$, by bounded convergence against the
push-forward $\rho_n=(\omega_n)_*(\mathrm{Leb})$.

\emph{Antidiagonal.}
$O(N(T)/\log T)=o(U)$.

\emph{Translation invariance.}
Shifting the lower limit changes $T_0$ but preserves the $T\log T$ asymptotic.
\end{proof}

\begin{corollary}[Bochner spectral measure]\label{cor:Cramer}
There is a finite non-negative Borel measure $\mu_Y$ on $\mathbb R$ with
$c_Y(\eta)=\int e^{i\eta\xi}d\mu_Y(\xi)$ and $\mu_Y(\mathbb R)=2$.
\end{corollary}

\begin{proof}
$c_Y$ is positive-definite (Ces\`aro limit of positive-definite kernels) and continuous
at $0$ ($|c_Y(\eta)-c_Y(0)|^2\le c_Y(0)\Delta(\eta)\to 0$ by the explicit diagonal series).
Bochner's theorem applies.
\end{proof}

\section{Band-Limit via Single Integration by Parts}

\begin{definition}[Windowed Fourier, PSD]\label{def:KT}
$K_T(\mu):=(2\pi)^{-1}\int_0^T Z(t)\sqrt{\Theta'(t)}e^{-i\mu\Theta(t)}dt$;
$S(\mu):=\limsup_T 2\pi|K_T(\mu)|^2/\Theta(T)$.
\end{definition}

\begin{theorem}[Band-limit]\label{thm:bandlim}
For every $c>c_*$ and every $|\mu|>1$: $S(\mu)=0$.
\end{theorem}

\begin{proof}
Insert Lemma~\ref{lem:RS} into $K_T$ and substitute $u=\Theta(t)$.
Mode $(n,\sigma)$ of the main sum contributes
\[
\mathcal J_{n,\sigma}(T,\mu)
=\int_{u_n}^{\Theta(T)} G_{n,\sigma}(u)\,e^{i\Psi_{n,\sigma,\mu}(u)}\,du,
\qquad G_{n,\sigma}(u)=\Theta'(t(u))^{-1/2},
\]
where $u_n=\Theta(2\pi n^2)$, $t(u)=\Theta^{-1}(u)$, and the phase is
\[
\Psi_{n,\sigma,\mu}(u)
=\sigma\vartheta(t(u))-\sigma t(u)\log n-\mu u
=(\sigma-\mu)\vartheta(t(u))-\sigma t(u)\log n-\mu c\,t(u).
\]

\emph{Phase derivative and instantaneous $u$-frequency.}
Differentiating in $u$ via $dt/du=1/\Theta'(t)$,
\[
\Psi'_{n,\sigma,\mu}(u)
=\frac{(\sigma-\mu)\vartheta'(t)-\sigma\log n-\mu c}{\Theta'(t)}
=\sigma\omega_n(t(u))-\mu.
\]
By Lemma~\ref{lem:omegabound}, $|\sigma\omega_n|\le|\omega_n|<1$, so for $|\mu|>1$:
\[
|\Psi'_{n,\sigma,\mu}(u)|\ge|\mu|-1=:\lambda>0.
\]

\emph{Exact cancellation in the IBP ratio.}
Set $x:=\Theta'(t(u))=\vartheta'(t(u))+c$.
Then $G_{n,\sigma}(u)=x^{-1/2}$ and
\[
\frac{d\Psi}{du}=\frac{(\sigma-\mu)x-\sigma(c+\log n)}{x},
\qquad
Q_{n,\sigma,\mu}(u):=\frac{G_{n,\sigma}(u)}{d\Psi/du}=\frac{x^{1/2}}{(\sigma-\mu)x-\sigma(c+\log n)}.
\]
$Q_{n,\sigma,\mu}$ is a rational function of $x$ alone; no $\vartheta''(t)$ enters,
and no bound on $\vartheta''$ is required.
As $x\to\infty$, $Q_{n,\sigma,\mu}(u)=O(x^{-1/2})$ and
$(dQ_{n,\sigma,\mu}/dx)=O(x^{-3/2})$.

\emph{IBP bound.}
With $\beta_n:=\Theta'(2\pi n^2)=\log n+c+O(n^{-4})$:
\[
\int_{u_n}^\infty|G'_{n,\sigma}|\,du=\beta_n^{-1/2},
\qquad
\int_{u_n}^\infty|\Psi''_{n,\sigma,\mu}|\,du=\frac{c+\log n}{\beta_n},
\]
and a single IBP gives
\[
|\mathcal J_{n,\sigma}(T,\mu)|\le\frac{C(c,|\mu|)}{\lambda}\,\beta_n^{-1/2}
\le\frac{C}{\lambda(\log n)^{1/2}}.
\]

\emph{Remainder contribution.}
The Riemann--Siegel remainder contributes
$\int_{T_0}^T R(t)\sqrt{\Theta'(t)}\,e^{-i\mu\Theta(t)}\,dt$;
the phase slope $\mu\Theta'(t)+\arg R'(t)$ is bounded below in absolute value
by $|\mu|c/2$ for large $t$ (Lemma~\ref{lem:RS} on $\arg R'$), and IBP yields
$O(T^{-1/4})$ uniformly.

\emph{Summing over modes.}
\[
|K_T(\mu)|
\le C\sum_{n=1}^{N(T)}\frac{n^{-1/2}}{\lambda(\log n)^{1/2}}+C
=O\!\left(\frac{T^{1/4}}{\sqrt{\log T}}\right).
\]
Since $\Theta(T)\sim\tfrac T2\log T$,
\[
\frac{2\pi|K_T(\mu)|^2}{\Theta(T)}
=O\!\left(\frac{1}{T^{1/2}(\log T)^2}\right)\to 0.
\qedhere
\]
\end{proof}

\begin{corollary}[Support of $\mu_Y$]\label{cor:supp}
$\operatorname{supp}(\mu_Y)\subseteq[-1,1]$.
\end{corollary}

\begin{proof}
Wiener (Acta Math.~55 (1930), Thm.~29):
$\mu_Y(E)=\lim_T(2\pi/\Theta(T))\int_E|K_T(\mu)|^2d\mu$ for bounded Borel $E$.
Theorem~\ref{thm:bandlim} gives pointwise vanishing on $E\subseteq\{|\mu|>1\}$;
dominated convergence gives $\mu_Y(E)=0$; countable additivity gives
$\mu_Y(\{|\mu|>1\})=0$.
\end{proof}

\section{Entire Extension and Reality of Zeros of $Y$}

\begin{theorem}[Paley--Wiener extension]\label{thm:PW}
$Y$ extends to an entire function of exponential type $\le 1$ with
$|Y(z)|\le\sqrt{2}\,e^{|\Im z|}$.
\end{theorem}

\begin{proof}
\emph{Theorem (Paley--Wiener for time-average WSS functions; Yaglom, Vol.~I,
Thm.~4.7.3; Bochner--Phillips, Ann.~Math.~43 (1942), 409--418).}
Let $f:\mathbb R\to\mathbb C$ be locally square-integrable and admit a time-average
covariance
\[
c_f(\eta):=\lim_{U\to\infty}\frac1U\int_0^U f(u)\overline{f(u+\eta)}\,du
\]
existing for every $\eta\in\mathbb R$, positive-definite, and continuous at $0$.
Let $\mu_f$ be the Bochner spectral measure, i.e.\
$c_f(\eta)=\int_{\mathbb R}e^{i\eta\xi}\,d\mu_f(\xi)$, and assume
$\operatorname{supp}\mu_f\subseteq[-\tau,\tau]$.
Then $f$ extends to an entire function on $\mathbb C$ of exponential type $\le\tau$
with
\[
|f(z)|\le\mu_f(\mathbb R)^{1/2}e^{\tau|\Im z|}\qquad(z\in\mathbb C).
\]

\emph{Application.}
$Y$ satisfies the hypotheses with $\tau=1$, $\mu_f=\mu_Y$, $\mu_Y(\mathbb R)=2$
(Theorem~\ref{thm:Ywss}, Corollary~\ref{cor:Cramer}, Corollary~\ref{cor:supp}),
giving $|Y(z)|\le\sqrt{2}\,e^{|\Im z|}$ and exponential type $\le 1$.
\end{proof}

\begin{theorem}[Akhiezer--Laguerre--P\'olya]\label{thm:Akhiezer}
If $F$ is a real-valued entire function of exponential type $\le\tau$ whose
time-average autocorrelation measure is finite, non-negative, and supported in
$[-2\tau,2\tau]$, then $F$ belongs to the Laguerre--P\'olya class $\mathcal{LP}$
and every complex zero of $F$ is real.
\end{theorem}

\begin{proof}
Akhiezer, \emph{Theory of Approximation}, \S V.4;
Levin, \emph{Lectures on Entire Functions}, Lecture~16, Thm.~3.
\end{proof}

\begin{theorem}[Reality of zeros of $Y$]\label{thm:Yzeros}
Every complex zero of the entire extension of $Y$ is real.
\end{theorem}

\begin{proof}
$Y$ is real on $\mathbb R$, entire of exponential type $\le 1$ (Theorem~\ref{thm:PW}),
with autocorrelation $\mu_Y\ge 0$, finite, supported in $[-1,1]\subseteq[-2,2]$
(Corollary~\ref{cor:supp}).
Apply Theorem~\ref{thm:Akhiezer} with $\tau=1$.
\end{proof}

\section{The Riemann Hypothesis}

\begin{theorem}[Riemann hypothesis]\label{thm:main}
Every nontrivial zero $\rho$ of $\zeta$ satisfies $\Re\rho=\tfrac12$.
\end{theorem}

\begin{proof}
Let $\rho=\beta+i\gamma$ be a nontrivial zero with $\beta\in(0,1)$,
$\gamma\in\mathbb R$.

\emph{Reduction to $\beta\le\tfrac12$.}
The functional equation $\xi(s)=\xi(1-s)$ with reality of $\xi$ on
$\mathbb R\cap\{\Re s=\tfrac12\}$ gives the symmetry $\rho\mapsto 1-\bar\rho$, so
it suffices to prove $\beta=\tfrac12$ assuming $\beta\in(0,\tfrac12]$.

\emph{One-parameter complexification.}
Define $t:\mathbb C\to\mathbb C$ by $t(s):=\gamma+is$; then $t(ir)=\gamma-r\in\mathbb R$
for every $r\in\mathbb R$, and $t(\tfrac12-\beta)=\gamma+i(\tfrac12-\beta)=:t_\rho$.
Let $m_\rho:=-\inf_{s\in[0,\tfrac12-\beta]}\Re\vartheta'(\gamma+is)$, which is finite
since $\vartheta'$ is holomorphic on the open set $\{|\Im t|<\tfrac12\}$ (poles of
$\psi(\tfrac14+\tfrac{it}{2})$ lie at $\Im t=\pm(2k+\tfrac12)$, $k\ge 0$) and the
segment $\{\gamma+is:s\in[0,\tfrac12-\beta]\}$ is compactly contained in it.
Fix $c>\max(c_*,m_\rho)$.

\emph{Holomorphy of $Z$ on a strip.}
$Z(t)=e^{i\vartheta(t)}\zeta(\tfrac12+it)$ is holomorphic on
$\Omega:=\{t\in\mathbb C:|\Im t|<\tfrac12\}$: $\vartheta$ is holomorphic on $\Omega$
(the $\psi$-poles lie at $\Im t=\pm(2k+\tfrac12)$, $k\ge 0$), and
$\tfrac12+it\in\mathbb C\setminus\{1\}$ for every $t\in\Omega$ since $t=-i/2$
lies on $\partial\Omega$ not in $\Omega$, so $\zeta(\tfrac12+it)$ is holomorphic on
$\Omega$.

\emph{The two holomorphic functions on a common domain.}
Let $N:=t^{-1}(\Omega)=\{s\in\mathbb C:|\Re s|<\tfrac12\}$, a connected open
vertical strip containing both the real interval
$[0,\tfrac12-\beta]\subset(-\tfrac12,\tfrac12)$ and the full imaginary axis
$i\mathbb R$.
Define
\[
F(s):=Z(t(s)),
\qquad
G(s):=\sqrt{\Theta'(t(s))}\,Y(\Theta(t(s))),
\]
where $\sqrt{\cdot}$ is the principal branch.
$F$ is holomorphic on $N$ by holomorphy of $Z$ on $\Omega$.
For $s\in N$, $\Theta'(t(s))=\vartheta'(t(s))+c$ is holomorphic, and on the real
segment $[0,\tfrac12-\beta]\subset N$ one has $\Re\Theta'(\gamma+is)\ge c-m_\rho>0$
by the choice of $c$, so $\Theta'(t(s))$ avoids $(-\infty,0]$ in a connected open
neighborhood of that segment inside $N$.
The entire extension of $Y$ (Theorem~\ref{thm:PW}) makes $Y\circ\Theta\circ t$
holomorphic on all of $N$.
Let $N'\subset N$ be the connected component of
$\{s\in N:\Theta'(t(s))\notin(-\infty,0]\}$ containing the real segment
$[0,\tfrac12-\beta]$; then $F,G$ are both holomorphic on $N'$, and
$i\mathbb R\cap N'$ is nonempty (since $s=0\in N'$ and $N'$ is open, it contains
an imaginary interval $\{ir:|r|<\varepsilon\}$ for some $\varepsilon>0$).

\emph{Coincidence on the imaginary segment.}
For $s=ir\in N'$, $t(ir)=\gamma-r\in\mathbb R$, so
$F(ir)=Z(\gamma-r)$, and Lemma~\ref{lem:Zfact} at the real argument $\gamma-r$
gives $Z(\gamma-r)=\sqrt{\Theta'(\gamma-r)}\,Y(\Theta(\gamma-r))=G(ir)$.
Hence $F=G$ on the open imaginary interval $\{ir:|r|<\varepsilon\}\subset N'$.

\emph{1D identity theorem.}
$N'$ is connected and open in $\mathbb C$, $F-G$ is holomorphic on $N'$, and the
zero set of $F-G$ contains an open interval of $i\mathbb R$, which has accumulation
points in $N'$.
Therefore $F\equiv G$ on $N'$; at $s_0:=\tfrac12-\beta\in N'$,
\[
Z(t_\rho)=F(s_0)=G(s_0)=\sqrt{\Theta'(t_\rho)}\,Y(\Theta(t_\rho)).
\]
$\tfrac12+it_\rho=\tfrac12+i\gamma-(\tfrac12-\beta)=\beta+i\gamma=\rho$ and
$t_\rho\in\Omega$, so $Z(t_\rho)=e^{i\vartheta(t_\rho)}\zeta(\rho)=0$.
The principal-branch factor $\sqrt{\Theta'(t_\rho)}$ is nonzero since
$\Theta'(t_\rho)\notin(-\infty,0]$; hence $Y(\Theta(t_\rho))=0$.

\emph{Reality of $\Theta(t_\rho)$.}
$Y$ is entire (Theorem~\ref{thm:PW}) with every complex zero real
(Theorem~\ref{thm:Yzeros}), so $\Theta(t_\rho)\in\mathbb R$.

\emph{Conclusion.}
With $\Theta'(t(s))$ holomorphic on the real segment $s\in[0,\tfrac12-\beta]$
and $\Re\Theta'(\gamma+is)\ge c-m_\rho>0$ there,
\[
\Im\Theta(t_\rho)=\Im\Theta(\gamma+i(\tfrac12-\beta))
=\int_0^{\tfrac12-\beta}\Re\Theta'(\gamma+is)\,ds
\ge (\tfrac12-\beta)(c-m_\rho).
\]
The left side is $0$ since $\Theta(t_\rho)\in\mathbb R$, and the right side is
$\ge 0$ with equality iff $\tfrac12-\beta=0$.
Hence $\beta=\tfrac12$.
\end{proof}
