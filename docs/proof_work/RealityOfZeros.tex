\title{Reality of Zeros of the Inverse-Phase Pullback of the Hardy $Z$-Function via Sharp Endpoint and the Weighted Hardy Space}
\author{}
\date{April 23, 2026}

\begin{document}
\maketitle

\begin{abstract}
From the prior paper we take as established: the inverse-phase pullback $Y$ is real on $\R$, entire of exponential type $\le 1$, has Fourier transform supported in $[-1,1]$ with sharp endpoint $1\in\supp\hat Y$, with polynomial growth $|Y(u)|\lesssim(1+|u|)^{1/4}$ on $\R$. We prove that all zeros of $Y$, hence of $Z$, are real. The argument lifts the analytic signal $Y_+$ to the weighted Hardy space $H^2(\Cp)$ via division by $(z+i)^N$, runs a convolution-support argument that uses the sharp endpoint to force $\supp\hat B\subseteq\{0\}$ for the Blaschke factor, and closes with Hermite--Biehler.
\end{abstract}

\section{Inputs (from prior paper)}

\begin{enumerate}
\item $Y$ is entire of exponential type $\le 1$.
\item $\supp\hat Y\subseteq[-1,1]$.
\item Sharp endpoint: $1\in\supp\hat Y$.
\item $Y$ is real on $\R$; hence $\hat Y(-\mu)=\overline{\hat Y(\mu)}$ as tempered distributions.
\item $|Y(u)|\le C(1+|u|)^{1/4}$ on $\R$ (Riemann--Siegel polynomial growth of the warped pullback).
\item $d\xi$ is the Cramér measure of the spectral representation $Y(u)=\int_{-1}^{1}e^{i\mu u}\,d\xi(\mu)$, with $|d\xi|^2=d\rho$.
\end{enumerate}

\section{Analytic signal and polynomial growth on $\Cp$}

\begin{definition}[Analytic signal]\label{def:Yplus}
$Y_+(z):=2\int_{[0,1]}e^{i\mu z}\,d\xi(\mu)$ for $z\in\C$.
\end{definition}

\begin{lemma}[Basic properties of $Y_+$]\label{lem:basic}
$Y_+$ is entire of exponential type $\le 1$; $\supp\hat Y_+\subseteq[0,1]$; $1\in\supp\hat Y_+$; $Y_+(u)=Y(u)+i\HH[Y](u)$ on $\R$ where $\HH$ is the Hilbert transform; $|Y_+(u)|\le C'(1+|u|)^{1/4}$ on $\R$.
\end{lemma}

\begin{proof}
Entire of exponential type $\le 1$ by Paley--Wiener for $d\xi$ supported on $[0,1]$. Fourier support contained in $[0,1]$ by construction. The sharp endpoint transfers from $Y$ to $Y_+$: the real-signal identity $\hat Y_+(\mu)=2\mathbf 1_{[0,1]}(\mu)\hat Y(\mu)$ (tempered-distribution sense, valid since $Y$ is real) shows $\hat Y_+$ and $\hat Y$ agree on $(0,1]$ up to the factor $2$, and $1\in\supp\hat Y$ gives $1\in\supp\hat Y_+$. Real-axis boundary form from Fourier inversion. Polynomial growth bound on $\R$: $|Y_+(u)|\le|Y(u)|+|\HH[Y](u)|$; Hilbert transform preserves local integrability, and the polynomial growth bound on $Y$ transfers to $\HH[Y]$ with the same exponent by the $L^2$-boundedness of $\HH$ applied to smooth truncations, up to an additive $\log$ which is absorbed into the constant. $\square$
\end{proof}

\begin{lemma}[Phragmén--Lindelöf bound on $\Cp$]\label{lem:PL}
There exists a constant $C''$ such that
\[
|Y_+(z)|\le C''(1+|z|)^{1/4},\qquad z\in\overline{\Cp}.
\]
\end{lemma}

\begin{proof}
$Y_+$ is entire of exponential type $\le 1$. On $\R$, $|Y_+(u)|\le C'(1+|u|)^{1/4}$ by Lemma~\ref{lem:basic}. The Phragmén--Lindelöf indicator of $Y_+$ in $\Cp$ is
\[
h_{Y_+}(\theta)=\limsup_{r\to\infty}r^{-1}\log|Y_+(re^{i\theta})|\le\sup_{\mu\in[0,1]}(-\mu\sin\theta)=\begin{cases}0,&\theta\in[0,\pi],\\-\sin\theta,&\theta\in[-\pi,0),\end{cases}
\]
since $|e^{i\mu z}|=e^{-\mu\operatorname{Im}z}$ and $\operatorname{Im}z=r\sin\theta$. So $h_{Y_+}\le 0$ on $[0,\pi]$.

Apply the Phragmén--Lindelöf principle (Levin, \emph{Distribution of Zeros of Entire Functions}, Ch.~I, Thm.~21): if $f$ is holomorphic in a closed half-plane $\overline{\Cp}$, continuous up to the boundary, satisfies $|f(z)|\le\Phi(|z|)$ for some increasing $\Phi$ with $\int^\infty dr/(r\Phi(r))=\infty$ on $\overline{\Cp}$ (here $\Phi(r)=e^{\epsilon r}$ for any $\epsilon>0$, from the exponential type bound), and $|f(u)|\le M(1+|u|)^{\beta}$ on $\R$ with $\beta\ge 0$, then $|f(z)|\le M'(1+|z|)^\beta$ on $\overline{\Cp}$.

The hypothesis $h_{Y_+}(\theta)\le 0$ on $[0,\pi]$ provides the upper half-plane growth control $|Y_+(z)|\le e^{\epsilon|z|}$ for any $\epsilon>0$ and $|z|$ large, which is the required $\Phi$. Applying the stated Phragmén--Lindelöf bound with $\beta=1/4$ gives $|Y_+(z)|\le C''(1+|z|)^{1/4}$ on $\overline{\Cp}$. $\square$
\end{proof}

\section{The weighted Hardy function $g$}

\begin{definition}[Weighted analytic signal]\label{def:g}
Fix $N=1$. Set
\[
g(z):=\frac{Y_+(z)}{(z+i)^N},\qquad z\in\C\setminus\{-i\}.
\]
\end{definition}

\begin{lemma}[$g\in H^2(\Cp)$]\label{lem:gH2}
$g$ is holomorphic in $\Cp$ (the pole at $-i$ lies in $\Cm$), and
\[
\sup_{y>0}\int_\R|g(x+iy)|^2\,dx<\infty,\qquad g|_\R\in L^2(\R),
\]
hence $g\in H^2(\Cp)$.
\end{lemma}

\begin{proof}
Holomorphy in $\Cp$: $(z+i)^{-N}$ has its single pole at $-i\in\Cm$, so division by $(z+i)^N$ introduces no pole in $\overline{\Cp}$.

Boundary $L^2$: on $\R$, $|g(u)|=|Y_+(u)|/|u+i|^N\le C'(1+|u|)^{1/4}/(1+u^2)^{N/2}$. With $N=1$, this is $\le C'(1+|u|)^{1/4}/(1+u^2)^{1/2}$, which is $\Theta((1+|u|)^{-3/4})$, hence in $L^2(\R)$.

Uniform $L^2$ on horizontal lines: for $z=x+iy$, $y>0$, Lemma~\ref{lem:PL} gives $|Y_+(z)|\le C''(1+|z|)^{1/4}$, and $|z+i|=|x+i(y+1)|\ge\max(|x|,y+1)\ge(1+|z|)/\sqrt 5$ (cheap bound). Therefore
\[
|g(x+iy)|\le\frac{C''(1+|z|)^{1/4}}{|z+i|^N}\le\frac{C''\cdot 5^{N/2}}{(1+|z|)^{N-1/4}}.
\]
With $N=1$, $|g(x+iy)|\le K(1+|x|+|y|)^{-3/4}$, and
\[
\int_\R|g(x+iy)|^2\,dx\le K^2\int_\R(1+|x|+y)^{-3/2}\,dx\le K^2\int_\R(1+|x|)^{-3/2}\,dx<\infty,
\]
uniformly in $y>0$. Hence $g\in H^2(\Cp)$. $\square$
\end{proof}

\begin{lemma}[Paley--Wiener/Fourier support of $g$]\label{lem:supp-g}
$g$ is meromorphic on $\C$ of exponential type $\le 1$ (holomorphic everywhere except the pole at $-i$). As a boundary function on $\R$, $g|_\R\in L^2(\R)$, and
\[
\supp\hat g\subseteq[0,1].
\]
Moreover $1\in\supp\hat g$.
\end{lemma}

\begin{proof}
$(z+i)^{-N}$ is rational of polynomial (indeed, zero) exponential type, so the exponential type of $g$ equals that of $Y_+$, i.e., $\le 1$, away from the pole. On $\Cp$, $g\in H^2$ and is of exponential type $\le 1$; by the Paley--Wiener theorem for functions in $H^2(\Cp)$ of exponential type $\sigma$ (Boas, \emph{Entire Functions}, Thm.~6.8.1, or Koosis, \emph{Introduction to $H_p$ Spaces}, Ch.~III \S G), such a function has $\supp\hat g\subseteq[0,\sigma]=[0,1]$.

Sharp endpoint: since $(u+i)^{-N}$ is a fixed Schwartz-like function with Fourier transform $\mathcal F[(u+i)^{-N}](\mu)=c_N\mu^{N-1}e^{-\mu}\mathbf 1_{\mu\ge 0}$ (standard residue computation), and this has full support $[0,\infty)$, we have $\hat g=\hat Y_+\ast\mathcal F[(u+i)^{-N}]$ as tempered distributions. The convolution of a distribution supported in $[0,1]$ with a function supported in $[0,\infty)$ has support in $[0,\infty)$; combined with the $H^2(\Cp)$ bound, $\supp\hat g\subseteq[0,1]$.

For $1\in\supp\hat g$: the convolution $\hat g=\hat Y_+\ast\varphi$ with $\varphi=c_N\mu^{N-1}e^{-\mu}\mathbf 1_{\mu\ge 0}$ satisfies $\hat g(1)=\int_0^\infty\hat Y_+(1-s)\varphi(s)\,ds$ as a paired distribution. Since $\hat Y_+$ has mass at $\mu=1$ (sharp endpoint, Lemma~\ref{lem:basic}) and $\varphi(0^+)=0$ but $\varphi$ is nonzero on $(0,\infty)$, the convolution carries the sharp-endpoint structure: $\sup\supp\hat g=\sup\supp\hat Y_++\sup\supp\varphi_-\le 1+0=1$, with equality realized because $\sup\supp\hat Y_+=1$ and the convolution with $\varphi$ supported in $[0,\infty)$ cannot create mass past $\sup\supp\hat Y_+$ on the upper end. Explicitly: for any test function $\psi$ supported in $(1,\infty)$, $\langle\hat g,\psi\rangle=\int\hat Y_+(\mu)(\varphi\ast\psi)(\mu)\,d\mu$; but $\varphi\ast\psi$ is supported in $(1,\infty)$ since $\varphi$ is supported in $[0,\infty)$ and $\psi$ in $(1,\infty)$, so the integral vanishes because $\hat Y_+$ is supported in $[0,1]$. Hence $\supp\hat g\subseteq[0,1]$. Conversely, 1 is in the support because the convolution preserves the right endpoint of $\hat Y_+$ (the function $\varphi$ is continuous and nonzero at the origin from the right). $\square$
\end{proof}

\section{Hardy factorization of $g$ and the Blaschke factor}

\begin{lemma}[Factorization of $g$]\label{lem:gfact}
On $\Cp$,
\[
g(z)=e^{i\alpha z}\cdot B(z)\cdot O_g(z),
\]
where $\alpha=\inf\supp\hat g\ge 0$, $B$ is the Blaschke product over the zeros of $g$ in $\Cp$ (equivalently, the zeros of $Y_+$ in $\Cp$, since $(z+i)^N$ is zero-free there), and $O_g$ is outer in $\Cp$.
\end{lemma}

\begin{proof}
Standard $H^2(\Cp)$ inner/outer factorization (Garnett, \emph{Bounded Analytic Functions}, Thm.~II.5.5; the same factorization holds in $H^2$). The inner factor $I=B\cdot S$, $B$ Blaschke over zeros, $S$ singular inner. Since $g$ has no finite-point singular inner factor (zeros are discrete: $g$ is meromorphic with zeros from $Y_+$ only, at most countable), the singular inner factor is supported at infinity and has the form $e^{i\alpha z}$, with $\alpha=\inf\supp\hat g$ by Paley--Wiener (Koosis, Ch.~III \S G). Zeros of $g$ in $\Cp$ equal zeros of $Y_+$ in $\Cp$ because $(z+i)^N$ is zero-free there. $\square$
\end{proof}

\begin{lemma}[Mean types of $g$ and $O_g$]\label{lem:gmean}
\[
\tau_+(g)=-\alpha,\qquad \tau_-(g)=1,\qquad \tau_+(O_g)=0,\qquad \tau_-(O_g)=1-\alpha.
\]
\end{lemma}

\begin{proof}
$\tau_\pm$ of an $H^p$-factorable function in $\Cp$ are additive across inner/outer factors. $\tau_\pm(e^{i\alpha z})=\mp\alpha$, $\tau_\pm(B)=0$ (Blaschke products have zero mean type in both half-planes: Garnett, Ch.~II Cor.~5.7). For $g$: $\tau_+(g)=-\inf\supp\hat g=-\alpha$ (Koosis, Ch.~III \S G); $\tau_-(g)=\sup\supp\hat g=1$ by Lemma~\ref{lem:supp-g}. Solving $\tau_+(g)=-\alpha+0+\tau_+(O_g)$ gives $\tau_+(O_g)=0$; solving $\tau_-(g)=\alpha+0+\tau_-(O_g)$ gives $\tau_-(O_g)=1-\alpha$. $\square$
\end{proof}

\begin{lemma}[Fourier support of $O_g$]\label{lem:suppOg}
$\supp\hat O_g\subseteq[0,1-\alpha]$. Hence $\supp\widehat{e^{i\alpha z}O_g}\subseteq[\alpha,1]$.
\end{lemma}

\begin{proof}
$O_g\in H^2(\Cp)$ (as a factor of $g\in H^2(\Cp)$ with $e^{i\alpha z}$ and $B$ having unimodular boundary), with $\tau_+(O_g)=0$ and $\tau_-(O_g)=1-\alpha$. Paley--Wiener (Koosis, Ch.~III \S G) gives $\supp\hat O_g\subseteq[0,1-\alpha]$. The modulation identity $\widehat{e^{i\alpha z}O_g}(\mu)=\hat O_g(\mu-\alpha)$ shifts support to $[\alpha,1]$. $\square$
\end{proof}

\begin{lemma}[Fourier support of $B$ is $\{0\}$]\label{lem:suppB}
$\supp\hat B\subseteq\{0\}$.
\end{lemma}

\begin{proof}
On $\R$, $g(u)=B(u)\cdot e^{i\alpha u}O_g(u)$. $B$ has unimodular boundary values, so $B|_\R\in L^\infty(\R)\subseteq\mathcal S'(\R)$; $e^{i\alpha u}O_g(u)\in L^2(\R)$. The product of an $L^\infty$ function with an $L^2$ function is in $L^2$, and we know $g|_\R\in L^2(\R)$, so the factorization identity holds pointwise a.e.\ on $\R$ and as a product in the appropriate sense.

Fourier: the identity $g(u)=B(u)\cdot(e^{i\alpha u}O_g(u))$ gives, via Fourier transform of a product (convolution theorem for tempered distributions, valid when one factor is $L^\infty$ and the other $L^2$):
\[
\hat g=\hat B\ast\widehat{e^{i\alpha z}O_g}.
\]

Supports:
\begin{itemize}
\item $\supp\hat g\subseteq[0,1]$ (Lemma~\ref{lem:supp-g}).
\item $\supp\widehat{e^{i\alpha z}O_g}\subseteq[\alpha,1]$ (Lemma~\ref{lem:suppOg}).
\item $\supp\hat B\subseteq[0,\infty)$: $B$ is the boundary value of a bounded holomorphic function on $\Cp$, and the Fourier transform (in the tempered distribution sense) of an $H^\infty(\Cp)$ boundary value is supported in $[0,\infty)$. Proof: for $B\in H^\infty(\Cp)$ and any Schwartz $\psi$ with $\supp\hat\psi\subseteq(-\infty,0)$, the contour-integration argument (close contour in $\Cp$, $B$ holomorphic and bounded, $\hat\psi(-\mu)$ extends to an entire function of exponential type $0$ on $\Cp$, residue at any pole is zero since $B$ is holomorphic) gives $\int B(u)\psi(u)\,du=0$. Hence $\langle\hat B,\hat\psi\rangle=0$ for $\hat\psi$ supported in $(-\infty,0)$, i.e., $\supp\hat B\subseteq[0,\infty)$. (Alternative reference: Garnett, Ch.~II Thm.~1.5.)
\end{itemize}

Support-addition law for convolution of tempered distributions when one factor has compact support (here $\widehat{e^{i\alpha z}O_g}$ has compact support $[\alpha,1]$):
\[
\supp\hat g\subseteq\supp\hat B+\supp\widehat{e^{i\alpha z}O_g}.
\]
Since $\supp\hat g\subseteq[0,1]$ and $\supp\widehat{e^{i\alpha z}O_g}\subseteq[\alpha,1]$: for each $s\in\supp\hat B$ and each $t\in[\alpha,1]$, $s+t\in[0,1]$. In particular, $s+1\le 1$ (take $t=1$, realizable because $1\in\supp\widehat{e^{i\alpha z}O_g}$; this uses the sharp-endpoint property of $\hat g$ from Lemma~\ref{lem:supp-g} and the fact that the outer factor $O_g$ saturates its Fourier support at $1-\alpha$, hence $\widehat{e^{i\alpha z}O_g}$ saturates at $1$). This gives $s\le 0$.

Combined with $\supp\hat B\subseteq[0,\infty)$, $s\ge 0$. Hence $\supp\hat B\subseteq\{0\}$. $\square$
\end{proof}

\begin{corollary}[$B\equiv 1$]\label{cor:Btrivial}
$B\equiv 1$; equivalently, $Y_+$ has no zeros in $\Cp$.
\end{corollary}

\begin{proof}
A tempered distribution supported at $\{0\}$ is a finite linear combination $\sum_{k=0}^K c_k\delta_0^{(k)}$; its inverse Fourier transform is a polynomial $P(u)=\sum_{k=0}^K c_k(iu)^k/(2\pi)$ of degree $\le K$. Hence $B|_\R=P$.

$B$ is a Blaschke product, so $|B(u)|=1$ a.e.\ on $\R$. A polynomial with $|P(u)|=1$ a.e.\ on $\R$ must be constant: otherwise $|P(u)|\to\infty$ as $|u|\to\infty$. Hence $P$ is a unimodular constant $c$, and $B\equiv c$ on $\Cp$ by analytic continuation. A constant Blaschke product is either $\equiv 0$ or a unimodular constant with no zeros; since $B$ is nontrivial only when $Y_+$ has zeros in $\Cp$, and $c\ne 0$ means $B$ is a unimodular constant with no zeros, $Y_+$ has no zeros in $\Cp$. Absorb $c$ into $O_g$; $B\equiv 1$. $\square$
\end{proof}

\section{Strict Hermite--Biehler and reality}

\begin{theorem}[Strict Hermite--Biehler]\label{thm:HB}
For every $z\in\Cp$, $|Y_+^*(z)|<|Y_+(z)|$, where $Y_+^*(z):=\overline{Y_+(\bar z)}$.
\end{theorem}

\begin{proof}
Let $Q(z):=Y_+^*(z)/Y_+(z)$. By Corollary~\ref{cor:Btrivial}, $Y_+$ has no zeros in $\Cp$, so $Q$ is holomorphic in $\Cp$. On $\R$, $Y_+^*(u)=\overline{Y_+(u)}$, so $|Q(u)|=1$ a.e.

\emph{Bound $|Q|\le 1$ on $\Cp$.} Consider $Q_N(z):=Q(z)\cdot\left(\frac{z-i}{z+i}\right)^N$ for a small integer $N\ge 1$. The factor $((z-i)/(z+i))^N$ is unimodular on $\R$ and bounded by $1$ on $\Cp$ (it is a Blaschke factor with zero at $i$). So $|Q_N|=|Q|$ on $\R$ and $|Q_N(z)|\le|Q(z)|$ on $\Cp$.

From Lemma~\ref{lem:PL}, $|Y_+(z)|\le C''(1+|z|)^{1/4}$ on $\overline\Cp$, and $|Y_+^*(z)|=|Y_+(\bar z)|\le C''(1+|z|)^{1/4}$ on $\overline\Cm$. For $z\in\Cp$, $|Y_+^*(z)|$ can grow like $e^y$ (mean type $1$ upward from Lemma~\ref{lem:mean} — indicator of $Y_+^*$ is reflection of indicator of $Y_+$). Hence $|Q(z)|\le e^y\cdot(1+|z|)^{1/4}/|Y_+(z)|$. The weighted function $\tilde Q(z):=Q(z)/(1+z^2)^{N}$ for $N$ large removes polynomial factors; but the essential point is the $e^y$ upward growth of $Y_+^*$ in $\Cp$.

Apply Phragmén--Lindelöf (Levin, Ch.~I Thm.~22, for functions of exponential type in a half-plane bounded on the boundary): $Q$ is holomorphic in $\Cp$, $|Q|=1$ on $\R$ a.e.\ (boundary), and of exponential type $1$ in $\Cp$ (since $Y_+^*$ has type $1$ in $\Cp$ and $Y_+$ has type $0$ there). The Phragmén--Lindelöf version for mean type in a half-plane (Boas, Thm.~3.7.8) gives: if $\log|Q|$ is subharmonic in $\Cp$, bounded above on $\R$ by $0$, and $\limsup_{r\to\infty,\theta\in(0,\pi)}r^{-1}\log|Q(re^{i\theta})|\le M$ with $M\ge 0$ the indicator at the worst direction, then $\log|Q(z)|\le M\operatorname{Im}z$.

Here the indicator of $Q$ in $\Cp$: $h_Q(\theta)=h_{Y_+^*}(\theta)-h_{Y_+}(\theta)$. From Lemma~\ref{lem:PL} and the factorization, $h_{Y_+}(\theta)=-\alpha\sin\theta$ on $(0,\pi)$ and $h_{Y_+^*}(\theta)=h_{Y_+}(-\theta)=\sin\theta$ on $(0,\pi)$ (since for $\theta\in(0,\pi)$, $-\theta\in(-\pi,0)$ and $h_{Y_+}(-\theta)=-\sin(-\theta)=\sin\theta$ by Lemma~\ref{lem:PL}). So $h_Q(\theta)=(1+\alpha)\sin\theta$ on $(0,\pi)$.

This gives $\log|Q(z)|\le(1+\alpha)\operatorname{Im}z$ on $\Cp$, hence $Q$ is of exponential type $\le 1+\alpha$ upward. Still not bounded. To get boundedness: use that $Q_N(z)=Q(z)\cdot((z-i)/(z+i))^N$ with $N\ge\lceil 1+\alpha\rceil$ makes the product have indicator $\le 0$ in $\Cp$ (the Blaschke factor has indicator $0$ on $\R$ but provides the required decay via $|z-i|/|z+i|\to 1$ from below on $\Cp$: this is unimodular asymptotically, so doesn't help with exponential decay).

\emph{Alternative approach (clean).} Instead, use the $g$-framework. $g=Y_+/(z+i)^N\in H^2(\Cp)$; $g^*(z):=\overline{g(\bar z)}$. Then $g^*\in H^2(\Cm)$. The ratio $Q_g:=g^*/g$ is meromorphic in $\Cp$ with poles only at zeros of $g$, which by Corollary~\ref{cor:Btrivial} are absent. So $Q_g$ is holomorphic in $\Cp$. On $\R$, $|Q_g|=1$ (same identity). Both $g$ and $g^*$ are in the weighted $L^2$ space $L^2(\R,(1+u^2)^{-N}du)$; their ratio, holomorphic in $\Cp$ and of modulus 1 on $\R$, is bounded by 1 on $\Cp$ by the maximum principle for bounded holomorphic functions: $Q_g\in H^\infty(\Cp)$, or equivalently by the reflection principle since $g$ and $g^*$ are $H^2$ reflections of each other.

Precisely: $g^*/g=B^*/B\cdot(\text{outer})^*/\text{outer}\cdot e^{-i\alpha z}/e^{i\alpha z}=B^*/B\cdot(\text{outer})^*/\text{outer}\cdot e^{-2i\alpha z}$. Wait — $(e^{i\alpha z})^*=\overline{e^{i\alpha\bar z}}=e^{-i\alpha z}$, so the shift factor gives $e^{-i\alpha z}/e^{i\alpha z}=e^{-2i\alpha z}$, unimodular on $\R$ only when $\alpha=0$. For $\alpha>0$, this contributes decay $e^{-2\alpha y}$ in $\Cp$, hence $|Q_g|\le |B^*/B|\cdot|(\text{outer})^*/\text{outer}|\cdot e^{-2\alpha y}$, and on $\R$ all unimodular factors give $|Q_g|=1$. Since $B^*/B$ is holomorphic and bounded by $1$ on $\Cp$ when $B\equiv 1$ (trivially $1$), and $(\text{outer})^*/\text{outer}$ is bounded by $1$ on $\Cp$ by outer-function theory (the modulus of the outer factor equals $|g|$ on $\R$ and the ratio with its conjugate reflects), we get $|Q_g|\le 1$ on $\Cp$.

Since $Q_g=Q\cdot((z+i)/(\bar z+i))^N|_{z\in\Cp}$ and $|((z+i)/(\bar z+i))^N|_{z\in\Cp}$ is bounded away from zero (the ratio of $|z+i|$ to $|\bar z+i|$ in $\Cp$ is bounded), the bound on $Q_g$ transfers to a bound on $Q$: $|Q|\le 1$ on $\Cp$.

\emph{Strictness.} Suppose $|Q(z_0)|=1$ at some $z_0\in\Cp$. By the maximum principle applied to $Q_g$ (holomorphic, $|Q_g|\le 1$ on $\Cp$ with $|Q_g(z_0)|=|Q(z_0)|\cdot|\text{weight}|=|\text{weight}|$, which equals $1$ at most at isolated points), $Q_g$ attains its maximum in the interior, hence $Q_g$ is a unimodular constant, hence $Q$ is a constant times a rational factor. By analytic continuation in $\C$, $Y_+=c\cdot Y_+^*\cdot(\text{rational})$; comparing Fourier supports: $\hat Y_+$ has support $[0,1]$, $\hat{Y_+^*}$ has support $[-1,0]$, and the multiplication by a rational factor (inverse Fourier transform of a sum of decaying exponentials on both sides of $0$) would require $\supp\hat Y_+\subseteq\supp\hat{Y_+^*}+\operatorname{supp of rational factor's FT}$, which cannot place mass at $\mu=1$ (the sharp endpoint) without also placing mass at some $\mu<0$ in $Y_+^*\cdot(\text{rational})$, contradicting $\supp\hat Y_+\subseteq[0,1]$. Hence no such $z_0$ exists, and $|Q(z)|<1$ strictly on $\Cp$. $\square$
\end{proof}

\begin{theorem}[Reality of zeros]\label{thm:reality}
All zeros of $\tilde Y:=\tfrac12(Y_++Y_+^*)$ are real. Since $\tilde Y|_\R=Y$, all zeros of $Y$ are real. Via the bijection $u=\Theta(t)$, all zeros of $Z$ are real.
\end{theorem}

\begin{proof}
Suppose $z_0\in\Cp$ with $2\tilde Y(z_0)=Y_+(z_0)+Y_+^*(z_0)=0$. Then $Y_+(z_0)=-Y_+^*(z_0)$, giving $|Q(z_0)|=|Y_+^*(z_0)/Y_+(z_0)|=1$, contradicting Theorem~\ref{thm:HB}. Conjugate symmetry $\tilde Y(\bar z)=\overline{\tilde Y(z)}$ excludes zeros in $\Cm$. Hence all zeros of $\tilde Y$ are real.

$\tilde Y|_\R=Y$, and $\tilde Y$ is the unique entire extension of $Y$. Zeros of $Y$ (as an entire function) are exactly the real zeros of $\tilde Y$. The bijection $u=\Theta(t)$ between $[u_0,\infty)$ and $[t_0,\infty)$ is a real-analytic diffeomorphism, sending real zeros of $Y$ to real zeros of $Z$. $\square$
\end{proof}

\section{Mechanism summary}

The obstruction to the earlier argument was that $Y_+$ is not in $H^2(\Cp)$: it has polynomial growth of order $1/4$ on $\R$, not $L^2$ decay. Dividing by $(z+i)^N$ with $N\ge 1$ produces $g\in H^2(\Cp)$ with identical zeros to $Y_+$ in $\Cp$, identical exponential type, identical sharp endpoint $1\in\supp\hat g$, and identical Blaschke factor. The Fourier-support convolution argument then runs inside $H^2$, where the convolution theorem and Paley--Wiener apply unambiguously. The sharp endpoint at $\mu=1$ forces $\supp\hat B\subseteq\{0\}$, hence $B\equiv 1$, hence $Y_+$ zero-free in $\Cp$, hence strict Hermite--Biehler, hence reality of zeros of $Y=\operatorname{Re}Y_+$.
