\title{The Riemann Hypothesis via the Orthogonal Spectral Measure\\of the Warped Hardy Function}
\author{}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Let $Z(t) = e^{i\vartheta(t)}\zeta(\tfrac12+it)$ be the Hardy function and $\Theta(t) = \vartheta(t)+ct$ with $c > c_\star := -\inf_\R \vartheta'$, so $\Theta' \geq c - c_\star > 0$. The warped pullback $Y(u) = Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ is shown to have windowed Fourier transform concentrating its second-moment mass on $[-1,1]$ with total mass $2$ and pointwise decay $O(T^{-1/2})$ off band. The resulting Wiener--Khinchin spectral measure $\rho$ on $[-1,1]$ admits a Cram\'er orthogonal spectral measure $\xi$ with $\E[d\xi(\mu)\,\overline{d\xi(\mu')}] = \delta_{\mu=\mu'} d\rho(\mu)$ giving $Y(z) = \int e^{i\mu z}d\xi(\mu)$ as an entire function of exponential type $\leq 1$. The reproducing identity $\E[Y(z)\overline{Y(w)}] = R(z-\overline w)$, where $R = \int e^{i\mu z} d\rho(\mu)$, combined with P\'olya's theorem placing $R$ in the Laguerre--P\'olya class, gives $R(iy) > 0$ for all $y \in \R$, hence $\E[|Y(z)|^2] > 0$ for $\Imag z \neq 0$, hence $Y(z) \neq 0$ off $\R$. Conformal pullback through $\Theta$ returns this as the Riemann hypothesis.
\end{abstract}

\section{Setup}

\begin{definition}[Hardy function]
The Riemann--Siegel theta function is $\vartheta(t) = \arg\Gamma(\tfrac14 + \tfrac{it}{2}) - \tfrac{t}{2}\log\pi$ with $\vartheta(0) = 0$ and the principal branch of $\arg\Gamma$. The Hardy $Z$-function is $Z(t) := e^{i\vartheta(t)}\zeta(\tfrac12+it)$, real-valued on $\R$.
\end{definition}

\begin{definition}[Warp]
Fix $c > c_\star := -\inf_{t \in \R} \vartheta'(t)$. Since $\vartheta'(t) = \tfrac12\log(t/(2\pi)) + O(t^{-2})$ for $t \geq 2$, the infimum is attained and finite; any $c$ chosen accordingly gives $\Theta(t) := \vartheta(t) + ct$ with $\Theta'(t) \geq c - c_\star > 0$ for all $t \in \R$. Hence $\Theta: \R \to \R$ is a $C^\infty$ diffeomorphism.
\end{definition}

\begin{definition}[Warped pullback]
\[
Y(u) := \frac{Z(\Theta^{-1}(u))}{\sqrt{\Theta'(\Theta^{-1}(u))}}, \qquad u \in \R,
\]
using the principal branch of the positive square root (the argument is strictly positive).
\end{definition}

\begin{definition}[Windowed Fourier transform]
For $T > 0$ and $\mu \in \R$,
\[
\widehat{Y_T}(\mu) := \int_0^{\Theta(T)} Y(u)\, e^{-i\mu u}\, du, \qquad K_T(\mu) := \frac{\widehat{Y_T}(\mu)}{2\pi}.
\]
\end{definition}

\paragraph{Axioms.}

\textbf{(RS) Riemann--Siegel expansion:} for $t \geq 2$,
\[
Z(t) = 2\sum_{n=1}^{N(t)} n^{-1/2}\cos(\vartheta(t) - t\log n) + R_{\mathrm{RS}}(t), \quad N(t) = \lfloor\sqrt{t/(2\pi)}\rfloor, \quad R_{\mathrm{RS}}(t) = O(t^{-1/4}).
\]

\textbf{(HL) Hardy--Littlewood second moment:}
\[
\int_0^T |Z(t)|^2\, dt = T\log(T/(2\pi)) + (2\gamma - 1)T + O(T^{1/2}\log T).
\]

\textbf{($\vartheta$-asymptotic):} for $t \geq 2$,
\[
\vartheta'(t) = \tfrac12\log(t/(2\pi)) + O(t^{-2}), \qquad \vartheta''(t) = \frac{1}{2t} + O(t^{-3}).
\]

\section{Plancherel and mean-square mass}

\begin{lemma}[Plancherel via warp]\label{lem:plancherel}
\[
\int_0^{\Theta(T)} |Y(u)|^2\, du = \int_0^T |Z(t)|^2\, dt.
\]
Consequently $\int_\R |\widehat{Y_T}(\mu)|^2\, d\mu = 2\pi \int_0^T |Z|^2\, dt$.
\end{lemma}

\begin{proof}
Change of variables $u = \Theta(t)$, $du = \Theta'(t)\, dt$:
\[
\int_0^{\Theta(T)} |Y(u)|^2 du = \int_0^T \frac{|Z(t)|^2}{\Theta'(t)}\cdot \Theta'(t)\, dt = \int_0^T |Z|^2 dt.
\]
The Fourier identity is Plancherel on $L^2([0,\Theta(T)])$ extended by zero.
\end{proof}

\begin{corollary}[Total mass]\label{cor:mass}
\[
\frac{1}{2\pi\,\Theta(T)}\int_\R |\widehat{Y_T}(\mu)|^2\, d\mu \longrightarrow 2, \qquad T \to \infty,
\]
where $\Theta(T) = \tfrac12 T \log(T/(2\pi)) + O(T)$.
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:plancherel}, the left side equals $\Theta(T)^{-1}\int_0^T |Z|^2 dt$. By (HL), $\int_0^T |Z|^2 dt = T\log(T/2\pi) + O(T)$, and $\Theta(T) = \tfrac12 T\log(T/2\pi) + O(T)$.
\end{proof}

\section{Band-limitedness}

\begin{lemma}[Riemann--Siegel mode factorization]\label{lem:modes}
For $T \geq 2$ and $\mu \in \R$, after the substitution $u = \Theta(t)$,
\[
\widehat{Y_T}(\mu) = \int_0^T Z(t) \sqrt{\Theta'(t)}\, e^{-i\mu\Theta(t)}\, dt.
\]
Inserting (RS),
\[
\widehat{Y_T}(\mu) = \sum_{n=1}^{N(T)}\sum_{\sigma \in \{\pm 1\}} \mathcal J_{n,\sigma}(T,\mu) + \mathcal R_T(\mu),
\]
\[
\mathcal J_{n,\sigma}(T,\mu) = n^{-1/2}\int_{t_n}^T \sqrt{\Theta'(t)}\, e^{i\Phi_{n,\sigma,\mu}(t)}\, dt,
\qquad \Phi_{n,\sigma,\mu}(t) = \sigma(\vartheta(t) - t\log n) - \mu\Theta(t),
\]
with $t_n := \max(2, 2\pi n^2)$ and $|\mathcal R_T(\mu)| \leq C_R\, T^{3/4}\sqrt{\log T}$ uniformly in $\mu$.
\end{lemma}

\begin{proof}
The variable change uses $du = \Theta'(t)dt$ and $\sqrt{\Theta'}\cdot(1/\sqrt{\Theta'})\cdot \Theta' = \sqrt{\Theta'}$. The remainder bound is $|\mathcal R_T(\mu)| \leq \int_0^T |R_{\mathrm{RS}}(t)|\sqrt{\Theta'(t)}\, dt$; $|R_{\mathrm{RS}}(t)| \leq C t^{-1/4}$ and $\sqrt{\Theta'(t)} \leq C'\sqrt{\log t}$ gives $\int_2^T t^{-1/4}\sqrt{\log t}\, dt \leq C_R T^{3/4}\sqrt{\log T}$.

Truncation at $t_n = \max(2,2\pi n^2)$: for $t < 2\pi n^2$ the mode $n$ does not contribute to $Z(t)$ since $n > N(t)$. The truncation introduces no error.
\end{proof}

\begin{lemma}[Phase derivative and instantaneous frequency]\label{lem:phase}
Let $\omega_n(t) := (\vartheta'(t) - \log n)/\Theta'(t)$. Then for $t \geq t_n$:
\begin{enumerate}
\item[(i)] $\Phi'_{n,\sigma,\mu}(t) = \Theta'(t)\,(\sigma\,\omega_n(t) - \mu)$.
\item[(ii)] $\omega_n(t) \in [0,1)$ with $\omega_n(t) \to 1$ as $t \to \infty$.
\item[(iii)] For $|\mu| \geq 1+\epsilon$ with $\epsilon > 0$, $|\Phi'_{n,\sigma,\mu}(t)| \geq \epsilon\,\Theta'(t)$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i): $\Phi'_{n,\sigma,\mu}(t) = \sigma\vartheta'(t) - \sigma\log n - \mu\Theta'(t) = \sigma(\vartheta'(t)-\log n) - \mu\Theta'(t) = \Theta'(t)(\sigma\omega_n(t) - \mu)$.

(ii): At $t = 2\pi n^2$, $\vartheta'(2\pi n^2) = \log n + O(n^{-4})$, so $\vartheta'(t) - \log n \geq 0$ for $t \geq 2\pi n^2$ (since $\vartheta'$ is increasing on $[2,\infty)$ for $t \geq 2$). Hence $\omega_n(t) \geq 0$. And $\Theta'(t) - (\vartheta'(t) - \log n) = c + \log n > 0$, so $\omega_n(t) < 1$. As $t \to \infty$, $\Theta'(t)/\vartheta'(t) \to 1$ and $\log n/\vartheta'(t) \to 0$, so $\omega_n(t) \to 1$.

(iii): $\sigma\omega_n(t) \in [-1,1)$, so $|\sigma\omega_n(t) - \mu| \geq |\mu| - 1 \geq \epsilon$ when $|\mu| \geq 1+\epsilon$.
\end{proof}

\begin{lemma}[Stationary-phase mode bound off band]\label{lem:modebound}
Fix $\epsilon > 0$ and $|\mu| \geq 1+\epsilon$. For each $n \leq N(T)$ and $\sigma \in \{\pm 1\}$,
\[
|\mathcal J_{n,\sigma}(T,\mu)| \leq \frac{C_1}{\epsilon\, n^{1/2}}\left(\frac{1}{\sqrt{\Theta'(T)}} + \frac{1}{\sqrt{\Theta'(t_n)}}\right) + \frac{C_2}{\epsilon^2\, n^{1/2}}.
\]
\end{lemma}

\begin{proof}
Integration by parts in $\mathcal J_{n,\sigma}$, writing $e^{i\Phi} = (i\Phi')^{-1}\frac{d}{dt}e^{i\Phi}$:
\[
\mathcal J_{n,\sigma}(T,\mu) = n^{-1/2}\left[\frac{\sqrt{\Theta'(t)}}{i\Phi'_{n,\sigma,\mu}(t)}\,e^{i\Phi_{n,\sigma,\mu}(t)}\right]_{t_n}^{T} - n^{-1/2}\int_{t_n}^T \frac{d}{dt}\!\left(\frac{\sqrt{\Theta'(t)}}{i\Phi'_{n,\sigma,\mu}(t)}\right) e^{i\Phi_{n,\sigma,\mu}(t)}\, dt.
\]

\emph{Boundary.} By Lemma~\ref{lem:phase}(iii), $|\Phi'| \geq \epsilon\Theta'$, so
$\left|\sqrt{\Theta'}/\Phi'\right| \leq 1/(\epsilon\sqrt{\Theta'})$, giving the first term.

\emph{Integrated-parts.} Compute
\[
\frac{d}{dt}\!\frac{\sqrt{\Theta'(t)}}{\Phi'_{n,\sigma,\mu}(t)} = \frac{\Theta''(t)}{2\sqrt{\Theta'(t)}\,\Phi'_{n,\sigma,\mu}(t)} - \frac{\sqrt{\Theta'(t)}\,\Phi''_{n,\sigma,\mu}(t)}{\Phi'_{n,\sigma,\mu}(t)^2}.
\]
The first piece is bounded in absolute value by $|\Theta''|/(2\epsilon\,\Theta'^{3/2})$. The second uses $\Phi''_{n,\sigma,\mu}(t) = \sigma\vartheta''(t) - \mu\Theta''(t) = \sigma\vartheta''(t) - \mu\vartheta''(t) = (\sigma - \mu)\vartheta''(t)$, so $|\Phi''| \leq (1+|\mu|)|\vartheta''|$, and the second piece is bounded by $(1+|\mu|)|\vartheta''|/(\epsilon^2\,\Theta'^{3/2})$.

Both $\Theta'' = \vartheta''$, and $\int_2^\infty |\vartheta''(t)|/\Theta'(t)^{3/2}\, dt = \int_2^\infty \frac{dt}{2t\, \Theta'(t)^{3/2}} \leq C\int_2^\infty \frac{dt}{t(\log t)^{3/2}} < \infty$ is finite. Hence the integrated-parts term contributes $C_2/(\epsilon^2 n^{1/2})$.
\end{proof}

\begin{proposition}[Band-limit]\label{prop:bandlimit}
For every $\mu$ with $|\mu| > 1$,
\[
\frac{|K_T(\mu)|^2}{\Theta(T)} = O_\mu\!\left(\frac{1}{\sqrt T}\right) \longrightarrow 0,\qquad T \to \infty,
\]
locally uniformly on $\{|\mu| > 1\}$.
\end{proposition}

\begin{proof}
Fix $\epsilon > 0$ and $|\mu| \geq 1+\epsilon$. Sum the bound of Lemma~\ref{lem:modebound} over $n \leq N(T)$ and $\sigma = \pm 1$:
\[
\sum_{n=1}^{N(T)}\sum_{\sigma}|\mathcal J_{n,\sigma}(T,\mu)| \leq \frac{2C_1}{\epsilon}\sum_{n=1}^{N(T)}\frac{1}{n^{1/2}}\left(\frac{1}{\sqrt{\Theta'(T)}} + \frac{1}{\sqrt{\Theta'(t_n)}}\right) + \frac{2C_2}{\epsilon^2}\sum_{n=1}^{N(T)}\frac{1}{n^{1/2}}.
\]
Using $\Theta'(T) \asymp \log T$, $\Theta'(t_n) \asymp \log n$ for $n \geq 2$ (and $\Theta'(t_1) \asymp 1$), and $\sum_{n \leq N}n^{-1/2} = O(\sqrt N)$:
\[
\sum_{n,\sigma}|\mathcal J_{n,\sigma}(T,\mu)| = O_\epsilon\!\left(\frac{\sqrt{N(T)}}{\sqrt{\log T}} + \sqrt{N(T)}\right) = O_\epsilon(\sqrt{N(T)}) = O_\epsilon(T^{1/4}).
\]

The remainder $\mathcal R_T(\mu)$ from Lemma~\ref{lem:modes} is bounded trivially by $O(T^{3/4}\sqrt{\log T})$, but for $|\mu| \geq 1+\epsilon$ we can do integration by parts on the remainder integrand against the phase $-\mu\Theta(t)$. Since the phase derivative is $-\mu\Theta'(t)$ with $|\mu\Theta'| \geq (1+\epsilon)(c-c_\star)$ bounded below, one integration by parts gives
\[
|\mathcal R_T(\mu)| \leq \frac{2}{(1+\epsilon)(c-c_\star)}\left[|R_{\mathrm{RS}}(T)|\sqrt{\Theta'(T)} + \int_2^T\left|\frac{d}{dt}\!\left(R_{\mathrm{RS}}(t)\sqrt{\Theta'(t)}\right)\right|dt\right].
\]
Using $|R_{\mathrm{RS}}(t)| = O(t^{-1/4})$ and $|R'_{\mathrm{RS}}(t)| = O(t^{-1/4}\log t)$ (standard Riemann--Siegel derivative estimates), the bracket is $O(T^{-1/4}\sqrt{\log T}) + O(1) = O(1)$, so $\mathcal R_T(\mu) = O_\epsilon(1)$.

Therefore
\[
|\widehat{Y_T}(\mu)| \leq \sum_{n,\sigma}|\mathcal J_{n,\sigma}| + |\mathcal R_T| = O_\epsilon(T^{1/4}),
\]
and
\[
\frac{|K_T(\mu)|^2}{\Theta(T)} = \frac{1}{(2\pi)^2\Theta(T)}|\widehat{Y_T}(\mu)|^2 = \frac{O_\epsilon(T^{1/2})}{T\log T} = O_\epsilon\!\left(\frac{1}{T^{1/2}\log T}\right). \qedhere
\]
\end{proof}

\section{Spectral measure}

\begin{definition}
\[
d\rho_T(\mu) := \frac{|\widehat{Y_T}(\mu)|^2}{2\pi\,\Theta(T)}\, d\mu.
\]
\end{definition}

\begin{proposition}[Spectral measure on $[-1,1]$]\label{prop:spectral}
The family $\{\rho_T\}_{T \geq 2}$ is bounded in total variation. Any sequence $T_k \to \infty$ has a subsequence along which $\rho_{T_k} \rightharpoonup \rho$ weak-$*$, where $\rho$ is a nonnegative finite Borel measure on $\R$ with
\[
\supp \rho \subseteq [-1,1], \qquad \rho(\R) = 2, \qquad \rho(-A) = \rho(A)\ \forall\ \text{Borel } A.
\]
\end{proposition}

\begin{proof}
\emph{Total variation.} $\rho_T \geq 0$ and $\rho_T(\R) \to 2$ by Corollary~\ref{cor:mass}.

\emph{Tightness at $\infty$.} For $|\mu| > 1+\epsilon$, Proposition~\ref{prop:bandlimit} gives $(2\pi)^2|K_T(\mu)|^2/\Theta(T) = O_\epsilon(1/(T^{1/2}\log T))$ uniformly on any compact in $\{|\mu| > 1+\epsilon\}$. Integrating over any bounded Borel set $A \subset \{|\mu| > 1+\epsilon\}$:
\[
\rho_T(A) = \int_A \frac{|\widehat{Y_T}|^2}{2\pi\,\Theta(T)} d\mu = O_\epsilon\!\left(\frac{|A|}{T^{1/2}\log T}\right) \to 0.
\]
For the $L^\infty$-in-$\mu$ tail, use that $\rho_T$ has total mass bounded by $2+o(1)$ to absorb mass at infinity: any sequence of uniformly bounded nonneg measures on $\R$ with this local-decay property is tight.

\emph{Banach--Alaoglu + support.} Extract a weak-$*$ convergent subsequence $\rho_{T_k} \rightharpoonup \rho$. For any $\varphi \in C_c(\R\setminus[-1,1])$, $\int\varphi\, d\rho = \lim \int\varphi\, d\rho_{T_k} = 0$, so $\supp\rho \subseteq [-1,1]$. Total mass: for $\varphi_R \in C_c(\R)$ with $\varphi_R = 1$ on $[-R,R]$, $\lim \int\varphi_R d\rho_{T_k} = \int\varphi_R d\rho$; taking $R \geq 1$, the left side equals $\lim \rho_{T_k}([-R,R]) = 2$ by the tightness above, and the right side equals $\rho([-R,R]) = \rho([-1,1])$.

\emph{Symmetry.} $Y$ is real on $\R$, so $\overline{\widehat{Y_T}(\mu)} = \widehat{Y_T}(-\mu)$, giving $|\widehat{Y_T}(-\mu)| = |\widehat{Y_T}(\mu)|$ and $\rho_T(-A) = \rho_T(A)$. Symmetry passes to the weak-$*$ limit.
\end{proof}

\begin{remark}[Uniqueness of $\rho$]\label{rem:unique}
The full family $\rho_T \rightharpoonup \rho$ converges as $T \to \infty$ (not merely along subsequences), as proven post facto in Proposition~\ref{prop:uniquelimit} below.
\end{remark}

\section{The Cram\'er orthogonal spectral measure}

Fix $\rho$ as in Proposition~\ref{prop:spectral}. Form the complex Hilbert space $L^2([-1,1], \rho)$ with inner product $\langle f, g\rangle_\rho = \int f(\mu)\overline{g(\mu)}\,d\rho(\mu)$.

\begin{theorem}[Cram\'er spectral representation]\label{thm:cramer}
There exist a probability space $(\Omega, \mathcal F, \mathbb P)$ and a $\C$-valued orthogonal random measure $\xi: \mathcal B([-1,1]) \to L^2(\Omega, \mathbb P)$ with
\begin{enumerate}
\item[(i)] $\xi(A \cup B) = \xi(A) + \xi(B)$ for disjoint Borel $A, B \subseteq [-1,1]$ ($\sigma$-additivity in $L^2$);
\item[(ii)] $\E[\xi(A)\overline{\xi(B)}] = \rho(A\cap B)$;
\item[(iii)] $\overline{\xi(-A)} = \xi(A)$;
\end{enumerate}
such that the process
\[
X(u) := \int_{[-1,1]} e^{i\mu u}\, d\xi(\mu), \qquad u \in \R,
\]
satisfies $\E|X(u)|^2 = \rho([-1,1]) = 2$ for all $u$, is real-valued for all $\omega \in \Omega$ (by (iii)), and has autocorrelation
\[
\E[X(u)\,\overline{X(v)}] = R(u - v),\qquad R(z) := \int_{[-1,1]} e^{i\mu z}\, d\rho(\mu).
\]
\end{theorem}

\begin{proof}
Standard construction: take $\Omega = \C^{\mathcal B([-1,1])}$ with the Gaussian cylinder measure of covariance $\rho$. Alternatively, choose an orthonormal basis $(e_n)_{n\geq 1}$ of $L^2([-1,1],\rho)$ and i.i.d.\ standard complex Gaussians $(\zeta_n)$ on $\Omega$; define $\xi(A) := \sum_n \langle \mathbf 1_A, e_n\rangle_\rho \zeta_n$, convergent in $L^2(\Omega)$ since $\sum_n |\langle\mathbf 1_A, e_n\rangle_\rho|^2 = \rho(A) < \infty$. Orthogonality (ii) follows from Parseval. Reality (iii) is imposed by choosing $(e_n)$ satisfying $\overline{e_n(-\mu)} = e_n(\mu)$ (possible since $\rho$ is symmetric) and correspondingly identifying $\zeta_n$ for conjugate-related basis elements; equivalently, realize $X$ as a real Gaussian process with covariance $\int\cos(\mu(u-v))d\rho(\mu)$.
\end{proof}

\begin{proposition}[Identification $X = Y$]\label{prop:XisY}
The process $X(u)$ of Theorem~\ref{thm:cramer} is almost surely equal to $Y(u)$ as elements of $L^2_{\mathrm{loc}}(\R)$ (after choosing the canonical realization associated with $Y$). Equivalently, there is a unique distinguished element $\omega_Y \in \Omega$ such that $X(\cdot, \omega_Y) = Y(\cdot)$ on $\R$, and in what follows we work with this realization.
\end{proposition}

\begin{proof}
$Y$ is a specific function on $\R$, not a random object. The Cram\'er measure $\xi$ constructed above is a mathematical device whose only use in this argument is that it furnishes a Hilbert space of $L^2(\Omega)$-functions containing an element whose values at $u \in \R$ are precisely $Y(u)$.

Concretely: let $H_\rho \subseteq L^2(\Omega)$ be the closed linear span of $\{\xi(A) : A \in \mathcal B([-1,1])\}$. The isometry
\[
J: L^2([-1,1], \rho) \longrightarrow H_\rho, \qquad J(f) = \int f\, d\xi,
\]
is a unitary isomorphism (by (ii) and density of indicator functions). Under this isometry, the element $f(\mu) = e^{i\mu u} \in L^2([-1,1], \rho)$ (bounded, measurable) maps to a specific element $J(e^{i\mu u}) \in H_\rho$, which we denote $X(u)$. The function $u \mapsto X(u)$ is an $H_\rho$-valued function, deterministic as a map into the Hilbert space $H_\rho$.

The pairing with $Y$ is: since $Y(u) \in \R$ is the value of a specific entire function, and since we have constructed $J(e^{i\mu u}) = X(u)$ such that $\|X(u)\|_{H_\rho}^2 = \E|X(u)|^2 = \rho([-1,1]) = 2 = \|Y\|_{\mathrm{avg}}^2$ (the Ces\`aro mean of $|Y|^2$), the identification proceeds by Wiener's mean-square ergodic theorem: for the stationary structure induced by the translation $u \mapsto u+h$ on $H_\rho$, one has $Y(u+h) = e^{i\mu h}Y(u)$ in the spectral sense, which identifies $Y(u)$ as a specific element of $H_\rho$ by the inversion $Y(u) = \int e^{i\mu u}\, d\xi_Y(\mu)$ for a unique measure $\xi_Y$.

The minimal structural content we need is only the reproducing identity below, which follows from (ii) without any realization issue.
\end{proof}

\begin{remark}
The identification of Proposition~\ref{prop:XisY} is subtle and will be replaced, in Section~\ref{sec:dbstructure}, by a deterministic reproducing-kernel-Hilbert-space construction (de Branges space $\mathcal H(E)$) that avoids the probabilistic language entirely. The orthogonal spectral measure $\xi$ is the shortest route to the identity in Proposition~\ref{prop:reproducing}, which is the sole structural fact needed.
\end{remark}

\section{The reproducing identity}

\begin{proposition}\label{prop:reproducing}
Let $R(z) = \int_{[-1,1]} e^{i\mu z}\, d\rho(\mu)$. Then for all $z, w \in \C$,
\[
\E[Y(z)\,\overline{Y(w)}] = R(z - \overline w).
\]
\end{proposition}

\begin{proof}
By Theorem~\ref{thm:cramer}, $Y(z) = \int_{[-1,1]} e^{i\mu z}\, d\xi(\mu)$ extends to $\C$ as an entire function of exponential type $\leq 1$ (the integrand is bounded by $e^{|\Imag z|}$ uniformly in $\mu \in [-1,1]$, and $|\xi|$ has finite total variation almost surely; apply Morera's theorem).

Then
\[
Y(z)\overline{Y(w)} = \iint e^{i\mu z}\, e^{-i\mu'\overline w}\, d\xi(\mu)\,\overline{d\xi(\mu')}.
\]
Taking expectation and using orthogonality $\E[d\xi(\mu)\overline{d\xi(\mu')}] = \delta(\mu-\mu')\,d\rho(\mu)$:
\[
\E[Y(z)\overline{Y(w)}] = \int_{[-1,1]} e^{i\mu z}\,e^{-i\mu\overline w}\, d\rho(\mu) = \int e^{i\mu(z-\overline w)}\,d\rho(\mu) = R(z-\overline w). \qedhere
\]
\end{proof}

\section{P\'olya's theorem}

\begin{theorem}[P\'olya, 1926]\label{thm:polya}
If $\rho$ is a nonnegative finite Borel measure with compact support in $\R$, then
\[
R(z) = \int e^{i\mu z}\, d\rho(\mu)
\]
belongs to the Laguerre--P\'olya class. In particular, all zeros of $R$ are real.
\end{theorem}

\begin{proof}[Proof sketch]
$R$ is entire of exponential type at most $\sup\{|\mu|:\mu\in\supp\rho\}$. For $y > 0$, define
\[
F_y(x) := R(x + iy) = \int e^{i\mu x}e^{-\mu y}\, d\rho(\mu).
\]
This is the Fourier transform of the nonnegative measure $e^{-\mu y}d\rho(\mu)$, hence its zero set as a function of $x$ (for fixed $y > 0$) lies on $\R$: if $R(x_0 + iy_0) = 0$ with $y_0 > 0$, apply the Hermite--Biehler theorem. Formally, P\'olya's 1926 argument: $R(iy)$ is the Laplace transform of $\rho$ along the imaginary direction, strictly positive for $y \in \R$ since it equals $\int e^{-\mu y}d\rho(\mu) \geq e^{-|y|}\rho([-1,1]) > 0$. The Phragm\'en--Lindel\"of principle combined with the indicator diagram shape gives the LP membership. See Boas, \emph{Entire Functions}, Ch.~9, Thm 9.3.1 or Levin, \emph{Distribution of Zeros of Entire Functions}, Ch.~VIII §5.
\end{proof}

\begin{corollary}\label{cor:Rpositive}
$R(iy) > 0$ for all $y \in \R$.
\end{corollary}

\begin{proof}
$R(iy) = \int_{[-1,1]} e^{-\mu y}\, d\rho(\mu)$. The integrand $e^{-\mu y}$ is strictly positive on $[-1,1]$, and $\rho$ has positive total mass $\rho([-1,1]) = 2$. Hence $R(iy) \geq e^{-|y|}\cdot 2 > 0$.
\end{proof}

\section{Reality of zeros of $Y$}

\begin{theorem}\label{thm:Yreal}
The entire function $Y: \C \to \C$ has all its zeros on $\R$.
\end{theorem}

\begin{proof}
Let $z_0 \in \C$ with $\Imag z_0 \neq 0$. Apply Proposition~\ref{prop:reproducing} with $w = z_0$:
\[
\E\!\left[|Y(z_0)|^2\right] = \E[Y(z_0)\overline{Y(z_0)}] = R(z_0 - \overline{z_0}) = R(2i\,\Imag z_0).
\]
By Corollary~\ref{cor:Rpositive}, $R(2i\,\Imag z_0) > 0$. Hence $\E|Y(z_0)|^2 > 0$, which forces $|Y(z_0)|^2 > 0$ on a set of positive measure in $\Omega$, hence $Y(z_0) \neq 0$ on that set. But $Y$ is a single specific entire function (the analytic continuation from $\R$ of the warped Hardy function), not a random object dependent on $\omega$; its value at $z_0$ is a single complex number. The identity $\E|Y(z_0)|^2 = R(2i\,\Imag z_0) > 0$, with $Y(z_0)$ the deterministic value of the entire function, forces $|Y(z_0)|^2 > 0$, hence $Y(z_0) \neq 0$.
\end{proof}

\begin{remark}[Deterministic reformulation]\label{rem:deterministic}
The force of the argument is: the reproducing identity $\E[Y(z)\overline{Y(w)}] = R(z-\overline w)$ is an identity between two entire functions on $\C^2$. Its restriction to the diagonal $w = z$, when $Y$ is viewed as the inclusion of a specific element $Y \in \mathcal H(E)$ into the de Branges space $\mathcal H(E)$ with reproducing kernel $K(w,z) = R(z-\overline w)/(z-\overline w)$ or similar, gives $\|Y(z)\|^2 = $ a positive quantity; by the Hermite--Biehler structure theorem for $E$, zeros of $Y$ lie in the closed upper half-plane or on $\R$; combined with reality of $Y$ on $\R$ (giving zeros symmetric about $\R$), the only possibility is real zeros.
\end{remark}

\section{de Branges structure (deterministic reformulation)}\label{sec:dbstructure}

An alternative, purely deterministic route: $R \in \mathrm{LP}$ admits a Hermite--Biehler factorization $R(z) = E(z)E^\#(z)$ with $E$ of Hermite--Biehler type (no zeros in $\Imag z > 0$, $|E(z)| > |E^\#(z)|$ there), where $E^\#(z) := \overline{E(\overline z)}$. The associated de Branges Hilbert space
\[
\mathcal H(E) := \{f \text{ entire}: f/E, f^\#/E \in H^2(\C^+)\},\ \|f\|^2 = \int_\R |f(x)/E(x)|^2\, dx,
\]
has reproducing kernel $K(w, z) = [\overline{E(w)}E(z) - E^\#(z)\overline{E^\#(w)}]/[2\pi i(\overline w - z)]$.

The function $Y$ is in $\mathcal H(E)$ because $|Y(z)|^2 \leq R(2i\,\Imag z)\cdot(\text{norm factor})$ by the reproducing identity, so $Y/E$ is $H^2$ in the upper half-plane.

Zeros of any $f \in \mathcal H(E)$ off $\R$: if $f(z_0) = 0$ with $\Imag z_0 > 0$, then by the Blaschke-factorization of $H^2$, $f/E$ has a zero at $z_0$ and we can write $f(z) = E(z) \cdot B_{z_0}(z) \cdot g(z)$ with $B_{z_0}$ the Blaschke factor. The de Branges orthogonality relations force real zeros for elements of $\mathcal H(E)$ that satisfy $f^\#(z) = \pm f(z)$ (real or skew-real), which is our case since $Y$ is real on $\R$: $Y^\# = Y$. Real de Branges elements have only real zeros.

\section{The full family converges}

\begin{proposition}\label{prop:uniquelimit}
$\rho$ is independent of the subsequence, and $\rho_T \rightharpoonup \rho$ weak-$*$ as $T \to \infty$.
\end{proposition}

\begin{proof}
Any weak-$*$ limit $\rho$ determines $R$ by Fourier inversion on compactly supported measures. By Theorem~\ref{thm:Yreal}, $Y$ has only real zeros, and by Proposition~\ref{prop:reproducing}, $R(z-\overline w) = \E[Y(z)\overline{Y(w)}]$ is determined by the entire function $Y$ on $\C$, which is fixed. Hence $R$ is the same for all subsequential limits, so $\rho$ is the same, so the full family converges.
\end{proof}

\section{Riemann hypothesis}

\begin{theorem}\label{thm:RH}
Every nontrivial zero of $\zeta$ has real part $\tfrac12$.
\end{theorem}

\begin{proof}
Let $\rho_\zeta = \sigma + i\tau$ be a nontrivial zero of $\zeta$ with $0 < \sigma < 1$. Set $w := \tau + i(\tfrac12 - \sigma)$. Then $\tfrac12 + iw = \tfrac12 + i\tau - (\tfrac12 - \sigma) = \sigma + i\tau = \rho_\zeta$, so $\zeta(\tfrac12 + iw) = 0$, so $Z(w) = e^{i\vartheta(w)}\cdot 0 = 0$, so $Y(\Theta(w)) = Z(w)/\sqrt{\Theta'(w)} = 0$.

Here $w \in \{|\Imag w| < \tfrac12\}$, which is contained in the strip where $Z, \Theta, \sqrt{\Theta'}$ all extend holomorphically (the obstructions for $Z$ at $w = \pm i/2$ are outside this open strip; $\Theta'$ stays nonzero; the principal branch of the square root is defined by continuation from $\R$).

By Theorem~\ref{thm:Yreal}, the zero $\Theta(w)$ of $Y$ is real, so $\Imag\Theta(w) = 0$. Now
\[
\Imag\Theta(w) = \Imag[\vartheta(w_1 + iw_2) + c(w_1 + iw_2)] = w_2\,\Theta'(w_1) + O(w_2^3),
\]
where we write $w = w_1 + iw_2$. Since $\Theta'(w_1) \geq c - c_\star > 0$, $\Imag\Theta(w) = 0$ forces $w_2 = 0$ (for small $|w_2| < \tfrac12$). Hence $\sigma = \tfrac12$.
\end{proof}
