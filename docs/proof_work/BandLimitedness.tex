\title{Band-Limitedness of the Warped Pullback of the Hardy $Z$-Function}
\author{Stephen Crowley}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Let $Z$ denote the Hardy $Z$-function and $\Theta(t) = \vartheta(t) + ct$ with $c > c_\star := -\inf_\R \vartheta'$. The warped pullback
\[
Y(u) := Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}
\]
has windowed Fourier transform $\widehat{Y_T}$ satisfying $|K_T(\mu)|^2/\Theta(T) = O_\mu(T^{-1/2})$ for every fixed $|\mu| > 1$, where $K_T = \widehat{Y_T}/(2\pi)$ and $\Theta(T) = \tfrac12 T\log(T/2\pi) + O(T)$. The proof uses Gabcke's second-order Riemann--Siegel remainder bound $R_{\mathrm{RS}}(t) = O(t^{-3/4})$. Consequently the windowed second-moment spectral measures $d\rho_T(\mu) = |\widehat{Y_T}(\mu)|^2/(2\pi\Theta(T))\,d\mu$ concentrate on $[-1,1]$ with total mass tending to $2$, and any weak-$*$ subsequential limit $\rho$ has $\mathrm{supp}\,\rho \subseteq [-1,1]$.
\end{abstract}

\section{Setup}

\begin{definition}[Hardy function]
$\vartheta(t) := \arg\Gamma(\tfrac14 + \tfrac{it}{2}) - \tfrac{t}{2}\log\pi$ (continuous branch, $\vartheta(0) = 0$), and $Z(t) := e^{i\vartheta(t)}\zeta(\tfrac12 + it)$, real-valued on $\R$.
\end{definition}

\begin{definition}[Warp]
Fix $c > c_\star := -\inf_{t \in \R}\vartheta'(t)$. Set $\Theta(t) := \vartheta(t) + ct$. Then $\Theta'(t) = \vartheta'(t) + c \geq c - c_\star > 0$ on $\R$, so $\Theta: \R \to \R$ is a $C^\infty$ diffeomorphism.
\end{definition}

\begin{definition}[Warped pullback]
$Y(u) := Z(\Theta^{-1}(u))/\sqrt{\Theta'(\Theta^{-1}(u))}$ for $u \in \R$, principal branch of the positive square root.
\end{definition}

\begin{definition}[Windowed Fourier transform]
For $T \geq 2$ and $\mu \in \R$,
\[
\widehat{Y_T}(\mu) := \int_0^{\Theta(T)} Y(u)\,e^{-i\mu u}\,du, \qquad K_T(\mu) := \widehat{Y_T}(\mu)/(2\pi).
\]
\end{definition}

\paragraph{Axioms used.}

\textbf{(RS)} Riemann--Siegel with Gabcke remainder \cite{Gabcke}: $Z(t) = 2\sum_{n \leq N(t)} n^{-1/2}\cos(\vartheta(t) - t\log n) + R_{\mathrm{RS}}(t)$ with $N(t) = \lfloor\sqrt{t/2\pi}\rfloor$ and
\[
R_{\mathrm{RS}}(t) = O(t^{-3/4}),
\]
obtained by retaining the second-order term of the Riemann--Siegel asymptotic series. Gabcke's 1979 thesis establishes the bound with an explicit effective constant.

\textbf{(HL)} Hardy--Littlewood: $\int_0^T|Z|^2\,dt = T\log(T/2\pi) + (2\gamma - 1)T + O(T^{1/2}\log T)$.

\textbf{($\vartheta$)} $\vartheta'(t) = \tfrac12\log(t/2\pi) + O(t^{-2})$; $\vartheta''(t) = 1/(2t) + O(t^{-3})$.

\section{Plancherel mass}

\begin{lemma}[Plancherel via warp]\label{lem:plancherel}
$\displaystyle\int_0^{\Theta(T)}|Y(u)|^2\,du = \int_0^T|Z(t)|^2\,dt$, hence
\[
\int_\R |\widehat{Y_T}(\mu)|^2\,d\mu = 2\pi \int_0^T|Z|^2\,dt.
\]
\end{lemma}

\begin{proof}
Change of variables $u = \Theta(t)$, $du = \Theta'(t)\,dt$: $\int|Y|^2 du = \int (|Z|^2/\Theta')\Theta'\,dt = \int|Z|^2\,dt$. The second identity is Plancherel on $L^2([0,\Theta(T)])$.
\end{proof}

\begin{corollary}[Total mass]\label{cor:mass}
$\displaystyle\frac{1}{2\pi\Theta(T)}\int_\R |\widehat{Y_T}(\mu)|^2\,d\mu \longrightarrow 2$ as $T \to \infty$, where $\Theta(T) = \tfrac12 T\log(T/2\pi) + O(T)$.
\end{corollary}

\begin{proof}
Combine Lemma~\ref{lem:plancherel} with (HL): $\int_0^T|Z|^2\,dt = T\log(T/2\pi) + O(T) \sim 2\Theta(T)$.
\end{proof}

\section{Mode factorization}

\begin{lemma}[Riemann--Siegel factorization of $\widehat{Y_T}$]\label{lem:modes}
For $T \geq 2$, $\mu \in \R$,
\[
\widehat{Y_T}(\mu) = \int_0^T Z(t)\sqrt{\Theta'(t)}\,e^{-i\mu\Theta(t)}\,dt = \sum_{n=1}^{N(T)}\sum_{\sigma \in \{\pm 1\}}\mathcal J_{n,\sigma}(T,\mu) + \mathcal R_T(\mu),
\]
\[
\mathcal J_{n,\sigma}(T,\mu) := n^{-1/2}\int_{t_n}^T \sqrt{\Theta'(t)}\,e^{i\Phi_{n,\sigma,\mu}(t)}\,dt,
\qquad \Phi_{n,\sigma,\mu}(t) := \sigma(\vartheta(t) - t\log n) - \mu\Theta(t),
\]
with $t_n := \max(2, 2\pi n^2)$.
\end{lemma}

\begin{proof}
Substitute $u = \Theta(t)$ in the Fourier integral: $du = \Theta'(t)dt$, $Y(u)du = Z(t)\Theta'(t)/\sqrt{\Theta'(t)}\,dt = Z(t)\sqrt{\Theta'(t)}\,dt$. Insert (RS) and expand $\cos = (e^{i\cdot} + e^{-i\cdot})/2$; the $\sigma = \pm 1$ index tracks the two exponentials. Modes $n > N(t)$ do not contribute at time $t$, giving the cutoff $t \geq 2\pi n^2$.
\end{proof}

\begin{lemma}[Phase derivative]\label{lem:phase}
Let $\omega_n(t) := (\vartheta'(t) - \log n)/\Theta'(t)$. Then for $t \geq t_n$:
\begin{enumerate}
\item[(i)] $\Phi'_{n,\sigma,\mu}(t) = \Theta'(t)\,(\sigma\omega_n(t) - \mu)$.
\item[(ii)] $\omega_n(t) \in [0, 1)$.
\item[(iii)] For $|\mu| \geq 1 + \epsilon$, $|\Phi'_{n,\sigma,\mu}(t)| \geq \epsilon\,\Theta'(t)$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Direct: $\Phi' = \sigma\vartheta' - \sigma\log n - \mu\Theta' = \Theta'(\sigma\omega_n - \mu)$.
(ii) For $t \geq 2\pi n^2$, $\vartheta'(t) \geq \log n$, so $\omega_n \geq 0$; and $\Theta' - (\vartheta' - \log n) = c + \log n > 0$, so $\omega_n < 1$.
(iii) $\sigma\omega_n \in [-1, 1)$, so $|\sigma\omega_n - \mu| \geq |\mu| - 1 \geq \epsilon$.
\end{proof}

\section{Band-limit bound}

\begin{lemma}[Single-mode off-band bound]\label{lem:mode}
Fix $\epsilon > 0$ and $|\mu| \geq 1 + \epsilon$. For each $1 \leq n \leq N(T)$ and $\sigma \in \{\pm 1\}$,
\[
|\mathcal J_{n,\sigma}(T,\mu)| \leq \frac{C_1}{\epsilon\, n^{1/2}}\left(\frac{1}{\sqrt{\Theta'(T)}} + \frac{1}{\sqrt{\Theta'(t_n)}}\right) + \frac{C_2}{\epsilon^2\, n^{1/2}}.
\]
\end{lemma}

\begin{proof}
Integration by parts with $e^{i\Phi} = (i\Phi')^{-1}(d/dt)e^{i\Phi}$:
\[
\mathcal J_{n,\sigma} = n^{-1/2}\left[\frac{\sqrt{\Theta'}}{i\Phi'}e^{i\Phi}\right]_{t_n}^T - n^{-1/2}\int_{t_n}^T \frac{d}{dt}\!\left(\frac{\sqrt{\Theta'}}{i\Phi'}\right)e^{i\Phi}\,dt.
\]
Boundary: $|\sqrt{\Theta'}/\Phi'| \leq 1/(\epsilon\sqrt{\Theta'})$ by Lemma~\ref{lem:phase}(iii).

Integrated part: $\frac{d}{dt}(\sqrt{\Theta'}/\Phi') = \Theta''/(2\sqrt{\Theta'}\Phi') - \sqrt{\Theta'}\Phi''/\Phi'^2$, with $\Phi'' = \sigma\vartheta'' - \mu\vartheta''$ (since $\Theta'' = \vartheta''$) so $|\Phi''| \leq (1+|\mu|)|\vartheta''|$. Both pieces are bounded by $C|\vartheta''|/(\epsilon^2\Theta'^{3/2})$; integrating, $\int_2^\infty |\vartheta''|/\Theta'^{3/2}\,dt \leq C\int_2^\infty dt/(t(\log t)^{3/2}) < \infty$.
\end{proof}

\begin{lemma}[Remainder bound via Gabcke]\label{lem:remainder}
Let $\mathcal R_T(\mu) := \int_0^T R_{\mathrm{RS}}(t)\sqrt{\Theta'(t)}e^{-i\mu\Theta(t)}\,dt$. Uniformly in $\mu \in \R$,
\[
|\mathcal R_T(\mu)| \leq \int_0^T |R_{\mathrm{RS}}(t)|\sqrt{\Theta'(t)}\,dt = O(T^{1/4}\sqrt{\log T}).
\]
\end{lemma}

\begin{proof}
By (RS), $|R_{\mathrm{RS}}(t)| = O(t^{-3/4})$ for $t \geq 2$, and $\sqrt{\Theta'(t)} = O(\sqrt{\log t})$. Triangle inequality on the integral defining $\mathcal R_T$ gives
\[
|\mathcal R_T(\mu)| \leq C + C'\int_2^T t^{-3/4}\sqrt{\log t}\,dt = O(T^{1/4}\sqrt{\log T}),
\]
uniformly in $\mu$.
\end{proof}

\begin{theorem}[Band-limit]\label{thm:bandlimit}
For every $\mu \in \R$ with $|\mu| > 1$,
\[
\frac{|K_T(\mu)|^2}{\Theta(T)} = O_\mu\!\left(\frac{1}{T^{1/2}}\right) \longrightarrow 0,\qquad T \to \infty,
\]
locally uniformly on $\{|\mu| > 1\}$.
\end{theorem}

\begin{proof}
Fix $\epsilon > 0$, $|\mu| \geq 1 + \epsilon$. By Lemma~\ref{lem:mode},
\[
\sum_{n=1}^{N(T)}\sum_\sigma |\mathcal J_{n,\sigma}| \leq \frac{2C_1}{\epsilon}\sum_{n=1}^{N(T)}\frac{1}{n^{1/2}\sqrt{\Theta'(t_n)}} + \frac{2C_1}{\epsilon\sqrt{\Theta'(T)}}\sum_{n=1}^{N(T)}\frac{1}{n^{1/2}} + \frac{2C_2}{\epsilon^2}\sum_{n=1}^{N(T)}\frac{1}{n^{1/2}}.
\]
Using $\Theta'(t_n) \asymp \log n$ for $n \geq 2$, $\Theta'(T) \asymp \log T$, $\sum_{n \leq N}n^{-1/2} = O(\sqrt N)$, $\sum_{n \leq N}1/(n^{1/2}\sqrt{\log n}) = O(\sqrt{N/\log N})$, the mode sum is $O_\epsilon(T^{1/4})$. Combined with the remainder bound $|\mathcal R_T(\mu)| = O(T^{1/4}\sqrt{\log T})$ of Lemma~\ref{lem:remainder},
\[
|\widehat{Y_T}(\mu)| = O_\epsilon(T^{1/4}\sqrt{\log T}).
\]
Hence
\[
\frac{|K_T(\mu)|^2}{\Theta(T)} = \frac{|\widehat{Y_T}(\mu)|^2}{(2\pi)^2\Theta(T)} = \frac{O_\epsilon(T^{1/2}\log T)}{T\log T/2} = O_\epsilon\!\left(\frac{1}{T^{1/2}}\right).
\]
Uniformity on compact subsets of $\{|\mu| > 1\}$ is immediate from the uniform lower bound on $\epsilon = |\mu| - 1$.
\end{proof}

\section{Weak-$*$ concentration of the spectral measure}

\begin{definition}
$d\rho_T(\mu) := |\widehat{Y_T}(\mu)|^2/(2\pi\Theta(T))\,d\mu$.
\end{definition}

\begin{corollary}\label{cor:support}
The family $\{\rho_T\}_{T \geq 2}$ of nonnegative Borel measures on $\R$ has total mass $\rho_T(\R) \to 2$, and for every $\epsilon > 0$,
\[
\rho_T\big(\R\setminus[-1-\epsilon,1+\epsilon]\big) \longrightarrow 0, \qquad T \to \infty.
\]
Consequently, any weak-$*$ subsequential limit $\rho$ is a nonnegative even Borel measure with $\mathrm{supp}\,\rho \subseteq [-1,1]$ and $\rho([-1,1]) = 2$.
\end{corollary}

\begin{proof}
Total mass $\rho_T(\R) \to 2$ by Corollary~\ref{cor:mass}. Evenness of $\rho_T$ from $|\widehat{Y_T}(-\mu)| = |\widehat{Y_T}(\mu)|$ (reality of $Y$).

\emph{Middle off-band window.} Fix $\epsilon > 0$ and $M > 1+\epsilon$. By Theorem~\ref{thm:bandlimit} with constants uniform on $[1+\epsilon, M]$,
\[
\rho_T\big(\{1+\epsilon \leq |\mu| \leq M\}\big) \leq \int_{1+\epsilon \leq |\mu| \leq M} O_\epsilon(T^{-1/2})\, d\mu = O_\epsilon\!\left(\frac{M}{T^{1/2}}\right).
\]

\emph{Tail $|\mu| > M$ via iterated integration by parts in $u$.} Since $Z$ is entire and $\Theta$ is a $C^\infty$ diffeomorphism with $\Theta' \geq c - c_\star > 0$, the function $Y$ is $C^\infty$ on $\R$. Integration by parts $k$ times in $u$ on the integral defining $\widehat{Y_T}(\mu) = \int_0^{\Theta(T)} Y(u) e^{-i\mu u}\,du$ gives
\[
|\widehat{Y_T}(\mu)| \leq \frac{C_k(T)}{|\mu|^k}, \qquad C_k(T) := \sum_{j=0}^{k-1}\big(|Y^{(j)}(0)| + |Y^{(j)}(\Theta(T))|\big) + \int_0^{\Theta(T)} |Y^{(k)}(u)|\, du.
\]
Each $Y^{(j)}$ is bounded on $[0, \Theta(T)]$ by a polynomial in $T$: the convexity bound $|Z(t)| = O(t^{1/4+\delta})$ and its derivative analogues give $|Y^{(j)}(u)| = O(T^{1/4+\delta})$ for $u \leq \Theta(T)$, hence $C_k(T) = O(T^{5/4 + \delta})$ for every fixed $k \geq 1$.

Therefore, with $k \geq 2$ fixed,
\[
\int_{|\mu| > M} |\widehat{Y_T}(\mu)|^2\, d\mu \leq \frac{2\, C_k(T)^2}{(2k-1)\, M^{2k-1}}, \qquad \rho_T\big(\{|\mu| > M\}\big) \leq \frac{C_k(T)^2}{\pi\,(2k-1)\,\Theta(T)\, M^{2k-1}}.
\]

\emph{Cutoff choice.} Set $M = M(T) := T^{1/4}$ and $k := 4$. Middle off-band: $O_\epsilon(T^{1/4}/T^{1/2}) = O_\epsilon(T^{-1/4}) \to 0$. Tail: $C_4(T)^2 = O(T^{5/2 + 2\delta})$, $M^{2k-1} = T^{7/4}$, $\Theta(T)\,M^{2k-1} \asymp T \log T \cdot T^{7/4} = T^{11/4}\log T$, giving $\rho_T(\{|\mu| > T^{1/4}\}) = O(T^{5/2 + 2\delta}/T^{11/4}\log T) = O(T^{-1/4 + 2\delta}/\log T) \to 0$. With $M = T^\alpha$ and $\alpha \in (0, 1/2)$, the middle rate is $T^{\alpha - 1/2}$ and the tail rate is $T^{3/2 - \alpha(2k-1) + 2\delta}/\log T$; balancing gives $\alpha = 1/k$, so $(\alpha, k) = (1/4, 4)$ is the joint optimum at this $k$, with common rate $T^{-1/4 + 2\delta}$.

Hence $\rho_T(\{|\mu| > 1 + \epsilon\}) \to 0$ for every $\epsilon > 0$.

\emph{Weak-$*$ limit.} Banach--Alaoglu extracts a weak-$*$ convergent subsequence $\rho_{T_k} \rightharpoonup \rho$. For $\varphi \in C_c(\R \setminus [-1,1])$ supported in $\{|\mu| > 1 + \epsilon\}$ for some $\epsilon > 0$, $\int\varphi\, d\rho = \lim \int\varphi\, d\rho_{T_k} = 0$ (since total off-band mass $\to 0$ dominates any bounded continuous test function supported there). Hence $\mathrm{supp}\,\rho \subseteq [-1,1]$. Total mass in the limit: $\rho([-1,1]) = \lim\rho_{T_k}([-1,1]) = \lim(\rho_{T_k}(\R) - \rho_{T_k}(\{|\mu|>1\})) = 2 - 0 = 2$. Evenness passes from $\rho_T$ to $\rho$.
\end{proof}
