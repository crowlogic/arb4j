{\titlestar{Salem--Rao Orthogonal Polynomials}{Orthogonal Polynomials
Associated with the Salem--Rao Kernel Weight}}

{\author{S. Crowley}}{\address{Dallas, Texas}} 3amrc4 23, 2026

\begin{abstract}
  We introduce and study the family of orthogonal polynomials associated with
  the Salem--Rao kernel weight $w (u) = e^u / (e^{e^u} + 1)$ on $\mathbb{R}$.
  The moments of this weight are shown to be derivatives of the entire
  function $F (s) = \Gamma (s)  (1 - 2^{1 - s}) \zeta (s)$ evaluated at $s =
  1$, yielding closed-form expressions in terms of the Euler--Mascheroni
  constant, the Stieltjes constants, and special values of the Riemann zeta
  function. Using these moments we carry out the Gram--Schmidt process
  explicitly, derive the first three monic orthogonal polynomials, establish
  the three-term recurrence via Favard's theorem, and verify the nondegeneracy
  of the associated Hankel determinants.
\end{abstract}

\section{Introduction}\label{sec:intro}

The Fermi--Dirac distribution $(e^x + 1)^{- 1}$ and its associated Mellin
transform are classical objects in analytic number theory and mathematical
physics. Under the logarithmic change of variable $x = e^u$, the kernel $x^{s
- 1} / (e^x + 1)$ pulls back to the function
\[ e^{su} / (e^{e^u} + 1), \]
whose restriction to $s = 1$ defines a rapidly decaying, strictly positive
weight on the real line. We call this the {\tmem{Salem--Rao kernel weight}},
denoted $w (u) = e^u / (e^{e^u} + 1)$, and develop in this paper the theory of
the orthogonal polynomial sequence $\{P_n \}_{n \ge 0}$ generated by $w$.

The moments of $w$ are expressible in closed form through the function $F (s)
\assign \Gamma (s)  (1 - 2^{1 - s}) \zeta (s)$, which is entire: the simple
pole of the Riemann zeta function at $s = 1$ is cancelled by the simple zero
of the Dirichlet eta prefactor $1 - 2^{1 - s}$. Specifically, $\mu_k = F^{(k)}
(1)$. This link to the gamma function, Riemann zeta function, and Stieltjes
constants endows the Salem--Rao polynomials with a rich arithmetic structure.

The paper is organized as follows. Section~\ref{sec:weight} establishes basic
analytic properties of the weight and defines the moment sequence.
Section~\ref{sec:momentformula} proves the moment--derivative identity $\mu_k
= F^{(k)} (1)$. Section~\ref{sec:explicit} computes the first four moments in
closed form. Section~\ref{sec:OP} carries out Gram--Schmidt orthogonalization
and derives the polynomials $P_0, P_1, P_2$ explicitly.
Section~\ref{sec:recurrence} establishes the three-term recurrence and
computes the first recurrence coefficients.

\section{The Salem--Rao Weight}\label{sec:weight}

\begin{definition}
  [Salem--Rao kernel weight]\label{def:weight} Let $\ell \assign \log 2$. The
  {\tmem{Salem--Rao kernel weight}} is the function
  \begin{equation}
    \label{eq:weight} w (u) \assign \frac{e^u}{e^{e^u} + 1}, \qquad u \in
    \mathbb{R}.
  \end{equation}
\end{definition}

\begin{proposition}
  [Integrability]\label{prop:integrability} The function $w$ is strictly
  positive on $\mathbb{R}$, belongs to $L^p (\mathbb{R})$ for every $1 \le p
  \le \infty$, and the measure $w (u)  \hspace{0.17em} du$ has finite moments
  of all orders. In particular, $w (u)  \hspace{0.17em} du$ defines a finite
  positive Borel measure on $\mathbb{R}$.
\end{proposition}

\begin{proof}
  Strict positivity is immediate from the formula: $e^u > 0$ and $e^{e^u} + 1
  > 1$.
  
  {\noindent}\tmtextbf{Decay at $\tmmathbf{- \infty}$.} As $u \to - \infty$,
  $e^u \to 0$ and $e^{e^u} \to e^0 = 1$, so $w (u) \sim e^u / (1 + 1) =
  \tfrac{1}{2} e^u$. Hence $w$ decays exponentially as $u \to - \infty$.
  
  {\noindent}\tmtextbf{Decay at $\tmmathbf{+ \infty}$.} As $u \to + \infty$,
  $e^{e^u} \gg 1$ and $w (u) \sim e^u \cdot e^{- e^u} = e^{u - e^u}$. Since
  $e^u$ grows much faster than $u$, this is super-exponential decay.
  
  {\noindent}\tmtextbf{$L^p$ membership.} For $1 \le p < \infty$, $|w (u) |^p$
  inherits the same exponential and super-exponential tails, so $w \in L^p
  (\mathbb{R})$. Since $w$ is continuous and decays to $0$ at both ends, $w
  \in L^{\infty} (\mathbb{R})$ as well.
  
  {\noindent}\tmtextbf{Finite moments.} For any $k \ge 0$, $|u|^k 
  \hspace{0.17em} w (u) \le C_k  \hspace{0.17em} e^{u / 2}$ for $u < 0$ (since
  $w (u) \sim \tfrac{1}{2} e^u$) and $|u|^k  \hspace{0.17em} w (u) \le C_k' 
  \hspace{0.17em} e^{u - e^u}$ for $u > 0$. Both tails are integrable, so
  $\mu_k \assign \int_{\mathbb{R}} u^k  \hspace{0.17em} w (u)  \hspace{0.17em}
  du < \infty$ for every $k \ge 0$.
\end{proof}

\begin{definition}
  [Moments]\label{def:moments} The {\tmem{moment sequence}} of the Salem--Rao
  weight is
  \begin{equation}
    \label{eq:moments} \mu_k \assign \int_{\mathbb{R}} u^k  \hspace{0.17em} w
    (u)  \hspace{0.17em} du, \qquad k = 0, 1, 2, \ldots
  \end{equation}
\end{definition}

\section{The Moment--Derivative Identity}\label{sec:momentformula}

\begin{definition}
  \label{def:F}Define
  \begin{equation}
    \label{eq:Fdef} F (s) \assign \Gamma (s)  (1 - 2^{1 - s}) \zeta (s),
    \qquad s \in \mathbb{C}.
  \end{equation}
  The simple pole of $\zeta (s)$ at $s = 1$ is cancelled by the simple zero of
  $1 - 2^{1 - s}$ at $s = 1$ (since $1 - 2^{1 - s} = - \ell \hspace{0.17em} (s
  - 1) + O ((s - 1)^2)$ and $\zeta (s) = (s - 1)^{- 1} + O (1)$), so $F$
  extends to an entire function.
\end{definition}

\begin{theorem}
  [Moment--derivative identity]\label{thm:momentF} For every $k \ge 0$,
  \begin{equation}
    \label{eq:momentF} \mu_k = F^{(k)} (1) .
  \end{equation}
\end{theorem}

\begin{proof}
  Substitute $x = e^u$ in~\eqref{eq:moments}: with $u = \log x$ and $du = dx /
  x$,
  \begin{equation}
    \label{eq:Mellin_moment} \mu_k = \int_0^{\infty} \frac{(\log x)^k}{e^x +
    1}  \hspace{0.17em} dx.
  \end{equation}
  The Mellin transform of the Fermi--Dirac kernel is the classical identity
  \begin{equation}
    \label{eq:Mellin_FD} F (s) = \int_0^{\infty} \frac{x^{s - 1}}{e^x + 1} 
    \hspace{0.17em} dx, \qquad \Re (s) > 0.
  \end{equation}
  Differentiating $k$ times under the integral sign with respect to $s$ gives
  \begin{equation}
    \label{eq:Fk} F^{(k)} (s) = \int_0^{\infty} \frac{(\log x)^k 
    \hspace{0.17em} x^{s - 1}}{e^x + 1}  \hspace{0.17em} dx, \qquad \Re (s) >
    0.
  \end{equation}
  The interchange of differentiation and integration is justified by dominated
  convergence: for any compact set $K \subset \{s : \Re (s) > 0\}$ and any $k
  \ge 0$, the integrand $| (\log x)^k  \hspace{0.17em} x^{\sigma - 1} / (e^x +
  1) |$ is dominated by an integrable function on $(0, \infty)$ uniformly for
  $s \in K$.
  
  Setting $s = 1$ in~\eqref{eq:Fk} and comparing with~\eqref{eq:Mellin_moment}
  yields $\mu_k = F^{(k)} (1)$.
\end{proof}

\begin{remark}
  \label{rem:generating}The identity~\eqref{eq:momentF} can be restated as
  saying that $F (1 + \epsilon)$ is the exponential generating function of the
  moment sequence: $F (1 + \epsilon) = \sum_{k = 0}^{\infty} \frac{\mu_k}{k!} 
  \hspace{0.17em} \epsilon^k$.
\end{remark}

\section{Explicit Moment Formulae}\label{sec:explicit}

To compute $\mu_k = F^{(k)} (1)$, we expand $F (1 + \epsilon)$ as a Cauchy
product of the Taylor expansions of its three factors around $\epsilon = 0$.
Throughout this section, $\gamma$ denotes the Euler--Mascheroni constant and
$\gamma_1, \gamma_2$ the Stieltjes constants defined by the Laurent expansion
\begin{equation}
  \label{eq:zeta_Laurent} \zeta (1 + \epsilon) = \frac{1}{\epsilon} + \gamma +
  \gamma_1 \epsilon + \gamma_2 \epsilon^2 + \cdots
\end{equation}
\begin{lemma}
  [Gamma factor expansion]\label{lem:gamma} Write $\Gamma (1 + \epsilon) =
  \sum_{n = 0}^{\infty} g_n \epsilon^n$. Then
  \begin{equation}
    \label{eq:gamma_expand} g_0 = 1, \quad g_1 = - \gamma, \quad g_2 =
    \frac{\gamma^2}{2} + \frac{\pi^2}{12}, \quad g_3 = - \frac{\gamma^3}{6} -
    \frac{\gamma \pi^2}{12} - \frac{\zeta (3)}{3} .
  \end{equation}
\end{lemma}

\begin{proof}
  From the well-known expansion $\log \Gamma (1 + \epsilon) = - \gamma
  \epsilon + \frac{\pi^2}{12} \epsilon^2 - \frac{\zeta (3)}{3} \epsilon^3 +
  \cdots$, exponentiate and collect powers of $\epsilon$. Specifically, write
  $\Gamma (1 + \epsilon) = \exp \hspace{-0.17em} \left( - \gamma \epsilon +
  \frac{\pi^2}{12} \epsilon^2 - \frac{\zeta (3)}{3} \epsilon^3 + \cdots
  \right)$ and expand the exponential to third order.
\end{proof}

\begin{lemma}
  [Eta factor expansion]\label{lem:eta} Define $\eta (\epsilon) \assign (1 -
  2^{- \epsilon})  \hspace{0.17em} \zeta (1 + \epsilon) = \sum_{n =
  0}^{\infty} a_n \epsilon^n$. Then
  
  \begin{align}
    a_0 & = \ell,  \label{eq:a0}\\
    a_1 & = \gamma \ell - \frac{\ell^2}{2},  \label{eq:a1}\\
    a_2 & = \gamma_1 \ell - \frac{\gamma \ell^2}{2} + \frac{\ell^3}{6}, 
    \label{eq:a2}\\
    a_3 & = \gamma_2 \ell - \frac{\gamma_1 \ell^2}{2} + \frac{\gamma
    \ell^3}{6} - \frac{\ell^4}{24} .  \label{eq:a3}
  \end{align}
\end{lemma}

\begin{proof}
  The Taylor expansion of $1 - 2^{- \epsilon} = 1 - e^{- \ell \epsilon}$ is
  $\sum_{j = 1}^{\infty} \frac{(- 1)^{j + 1} \ell^j}{j!} \epsilon^j$. The
  Laurent expansion of $\zeta (1 + \epsilon)$ is given
  by~\eqref{eq:zeta_Laurent}. The product $\eta (\epsilon) = \left( \sum_{j
  \ge 1} c_j \epsilon^j \right)  (\epsilon^{- 1} + \gamma + \gamma_1 \epsilon
  + \cdots)$ is computed via Cauchy product, collecting terms through
  $\epsilon^3$.
  
  For $a_0$: the only contribution at order $\epsilon^0$ is $c_1 \cdot
  \epsilon^{- 1} \cdot \epsilon = \ell \cdot 1 = \ell$.
  
  For $a_1$: $c_1 \cdot \gamma + c_2 \cdot \epsilon^{- 1}$, giving $\gamma
  \ell + (- \ell^2 / 2) = \gamma \ell - \ell^2 / 2$.
  
  The remaining coefficients follow by the same convolution.
\end{proof}

\begin{theorem}
  [Explicit moments]\label{thm:moments} The first four moments of the
  Salem--Rao weight are:
  
  \begin{align}
    \mu_0 & = \ell,  \label{eq:mu0}\\
    \mu_1 & = - \frac{\ell^2}{2},  \label{eq:mu1}\\
    \mu_2 & = 2 \gamma_1 \ell - \gamma^2 \ell + \frac{\ell^3}{3} +
    \frac{\pi^2 \ell}{6},  \label{eq:mu2}\\
    \mu_3 & = (6 \gamma_2 - 6 \gamma \gamma_1 + 2 \gamma^3 - 2 \zeta (3))
    \ell + \left( - 3 \gamma_1 + \frac{3 \gamma^2}{2} - \frac{\pi^2}{4}
    \right) \ell^2 - \frac{\ell^4}{4} .  \label{eq:mu3}
  \end{align}
\end{theorem}

\begin{proof}
  Write $F (1 + \epsilon) = \Gamma (1 + \epsilon) \cdot \eta (\epsilon)$, with
  notation as in Lemmas~\ref{lem:gamma} and~\ref{lem:eta}. By
  Remark~\ref{rem:generating}, the coefficient of $\epsilon^n$ in $F (1 +
  \epsilon)$ equals $\mu_n / n$!. Set $b_n \assign \sum_{j = 0}^n g_j 
  \hspace{0.17em} a_{n - j}$.
  
  {\noindent}\tmtextbf{$\tmmathbf{\mu_0}$:} $b_0 = g_0 a_0 = \ell$, so $\mu_0
  = 0! \cdot b_0 = \ell$.
  
  {\noindent}\tmtextbf{$\tmmathbf{\mu_1}$:} $b_1 = g_0 a_1 + g_1 a_0 = (\gamma
  \ell - \ell^2 / 2) - \gamma \ell = - \ell^2 / 2$, so $\mu_1 = 1! \cdot b_1 =
  - \ell^2 / 2$.
  
  {\noindent}\tmtextbf{$\tmmathbf{\mu_2}$:}
  
  \begin{align*}
    b_2 & = g_0 a_2 + g_1 a_1 + g_2 a_0\\
    & = \left( \gamma_1 \ell - \frac{\gamma \ell^2}{2} + \frac{\ell^3}{6}
    \right) + (- \gamma)  \left( \gamma \ell - \frac{\ell^2}{2} \right) +
    \left( \frac{\gamma^2}{2} + \frac{\pi^2}{12} \right) \ell\\
    & = \gamma_1 \ell - \frac{\gamma \ell^2}{2} + \frac{\ell^3}{6} - \gamma^2
    \ell + \frac{\gamma \ell^2}{2} + \frac{\gamma^2 \ell}{2} + \frac{\pi^2
    \ell}{12}\\
    & = \gamma_1 \ell - \frac{\gamma^2 \ell}{2} + \frac{\ell^3}{6} +
    \frac{\pi^2 \ell}{12} .
  \end{align*}
  
  Hence $\mu_2 = 2! \cdot b_2 = 2 \gamma_1 \ell - \gamma^2 \ell + \ell^3 / 3 +
  \pi^2 \ell / 6$.
  
  {\noindent}\tmtextbf{$\tmmathbf{\mu_3}$:} $b_3 = g_0 a_3 + g_1 a_2 + g_2 a_1
  + g_3 a_0$. Expanding and collecting powers of $\ell$:
  
  \begin{align*}
    b_3 & = \left( \gamma_2 - \gamma \gamma_1 + \frac{\gamma^3}{3} -
    \frac{\zeta (3)}{3} \right) \ell + \left( - \frac{\gamma_1}{2} +
    \frac{\gamma^2}{4} - \frac{\pi^2}{24} \right) \ell^2 - \frac{\ell^4}{24} .
  \end{align*}
  
  Hence $\mu_3 = 3! \cdot b_3$, yielding~\eqref{eq:mu3}.
\end{proof}

\section{Orthogonal Polynomials via Gram--Schmidt}\label{sec:OP}

\begin{definition}
  [Inner product]\label{def:ip} For $f, g \in L^2 (\mathbb{R}, w (u)
  \hspace{0.17em} du)$, define
  \begin{equation}
    \label{eq:ip} \langle f, g \rangle_w \assign \int_{\mathbb{R}} f (u) 
    \hspace{0.17em} g (u)  \hspace{0.17em} w (u)  \hspace{0.17em} du.
  \end{equation}
\end{definition}

We construct the monic orthogonal polynomial sequence $\{P_n \}_{n \ge 0}$ by
applying Gram--Schmidt to the monomials $\{1, u, u^2, \ldots\}$ with respect
to $\langle \cdummy, \cdummy \rangle_w$.

\begin{theorem}
  [First three Salem--Rao polynomials]\label{thm:polys} The first three monic
  orthogonal polynomials with respect to the weight $w (u) = e^u / (e^{e^u} +
  1)$ on $\mathbb{R}$ are:
  
  \begin{align}
    P_0 (u) & = 1,  \label{eq:P0}\\
    P_1 (u) & = u + \frac{\ell}{2},  \label{eq:P1}\\
    P_2 (u) & = u^2 - \alpha \hspace{0.17em} u - \beta,  \label{eq:P2}
  \end{align}
  
  where
  \begin{equation}
    \label{eq:alpha} \alpha = \frac{\mu_0 \mu_3 - \mu_1 \mu_2}{\mu_0 \mu_2 -
    \mu_1^2},
  \end{equation}
  \begin{equation}
    \label{eq:beta} \beta = \frac{\mu_2^2 - \mu_1 \mu_3}{\mu_0 \mu_2 -
    \mu_1^2},
  \end{equation}
  with $\mu_0, \ldots, \mu_3$ given by~\eqref{eq:mu0}--\eqref{eq:mu3}.
  Equivalently, $P_2$ admits the Hankel determinantal representation
  \begin{equation}
    \label{eq:P2det} P_2 (u) = \frac{1}{\Delta_1} \left|\begin{array}{ccc}
      \mu_0 & \mu_1 & \mu_2\\
      \mu_1 & \mu_2 & \mu_3\\
      1 & u & u^2
    \end{array}\right|, \qquad \Delta_1 \assign \mu_0 \mu_2 - \mu_1^2 .
  \end{equation}
  The Hankel determinant $\Delta_1$ is strictly positive:
  \begin{equation}
    \label{eq:Delta1} \Delta_1 = \left( 2 \gamma_1 - \gamma^2 +
    \frac{\pi^2}{6} \right) \ell^2 + \frac{\ell^4}{12} > 0.
  \end{equation}
\end{theorem}

\begin{proof}
  \tmtextbf{Step 1 ($P_0$ and $P_1$).}{\hspace{0.27em}} Set $P_0 (u) \assign
  1$. Then
  \[ P_1 (u) = u - \frac{\langle u, P_0 \rangle_w}{\langle P_0, P_0 \rangle_w}
     = u - \frac{\mu_1}{\mu_0} = u - \frac{- \ell^2 / 2}{\ell} = u +
     \frac{\ell}{2} . \]
  {\noindent}\tmtextbf{Step 2 ($P_2$).}{\hspace{0.27em}} Write $P_2 (u) = u^2
  - \alpha u - \beta$ and impose the orthogonality conditions $\langle P_2,
  P_0 \rangle_w = 0$ and $\langle P_2, P_1 \rangle_w = 0$.
  
  The first condition gives
  \begin{equation}
    \label{eq:sys1} \mu_2 - \alpha \mu_1 - \beta \mu_0 = 0.
  \end{equation}
  For the second, expand using $P_1 (u) = u + \ell / 2$:
  
  \begin{align*}
    \langle P_2, P_1 \rangle_w & = \langle u^2 - \alpha u - \beta,
    \hspace{0.27em} u + \tfrac{\ell}{2} \rangle_w\\
    & = (\mu_3 - \alpha \mu_2 - \beta \mu_1) + \frac{\ell}{2}  (\mu_2 -
    \alpha \mu_1 - \beta \mu_0) .
  \end{align*}
  
  The second group vanishes by~\eqref{eq:sys1}, leaving
  \begin{equation}
    \label{eq:sys2} \mu_3 - \alpha \mu_2 - \beta \mu_1 = 0.
  \end{equation}
  Solving~\eqref{eq:sys1}--\eqref{eq:sys2}: from~\eqref{eq:sys1}, $\beta =
  (\mu_2 - \alpha \mu_1) / \mu_0$. Substituting into~\eqref{eq:sys2}:
  \[ \mu_3 - \alpha \mu_2 - \frac{\mu_1  (\mu_2 - \alpha \mu_1)}{\mu_0} = 0
     \Longrightarrow \alpha (\mu_0 \mu_2 - \mu_1^2) = \mu_0 \mu_3 - \mu_1
     \mu_2, \]
  yielding~\eqref{eq:alpha}. Back-substitution gives~\eqref{eq:beta}. The
  determinantal form~\eqref{eq:P2det} follows by cofactor expansion along the
  third row.
  
  {\noindent}\tmtextbf{Step 3 (Positivity of $\Delta_1$).}{\hspace{0.27em}}
  Direct computation:
  
  \begin{align*}
    \Delta_1 & = \mu_0 \mu_2 - \mu_1^2\\
    & = \ell \left( 2 \gamma_1 \ell - \gamma^2 \ell + \frac{\ell^3}{3} +
    \frac{\pi^2 \ell}{6} \right) - \frac{\ell^4}{4}\\
    & = \left( 2 \gamma_1 - \gamma^2 + \frac{\pi^2}{6} \right) \ell^2 +
    \frac{\ell^4}{3} - \frac{\ell^4}{4}\\
    & = \left( 2 \gamma_1 - \gamma^2 + \frac{\pi^2}{6} \right) \ell^2 +
    \frac{\ell^4}{12} .
  \end{align*}
  
  Numerically, $\gamma_1 \approx - 0.0728$, $- \gamma^2 + \pi^2 / 6 \approx
  1.3131$, $2 \gamma_1 - \gamma^2 + \pi^2 / 6 \approx 1.1675$, and $\ell^4 /
  12 \approx 0.0192$, so $\Delta_1 > 0$. This confirms nondegeneracy of the
  Gram matrix and the linear independence of $\{P_0, P_1, P_2 \}$.
\end{proof}

\section{Three-Term Recurrence}\label{sec:recurrence}

\begin{theorem}
  [Favard recurrence]\label{thm:recurrence} The monic orthogonal polynomials
  $\{P_n \}_{n \ge 0}$ for the Salem--Rao weight satisfy the three-term
  recurrence
  \begin{equation}
    \label{eq:recurrence} P_{n + 1} (u) = (u - \alpha_n)  \hspace{0.17em} P_n
    (u) - \beta_n  \hspace{0.17em} P_{n - 1} (u), \qquad n \ge 0,
  \end{equation}
  with $P_{- 1} \equiv 0$, $P_0 \equiv 1$, and recurrence coefficients
  \begin{equation}
    \label{eq:recurrence_coeffs} \alpha_n = \frac{\langle u \hspace{0.17em}
    P_n, P_n \rangle_w}{\langle P_n, P_n \rangle_w}, \qquad \beta_n =
    \frac{\langle P_n, P_n \rangle_w}{\langle P_{n - 1}, P_{n - 1} \rangle_w}
    .
  \end{equation}
  Every inner product appearing here is computable in closed form via $\langle
  u^j, u^k \rangle_w = \mu_{j + k} = F^{(j + k)} (1)$.
  
  The first recurrence coefficients are
  
  \begin{align}
    \alpha_0 & = - \frac{\ell}{2},  \label{eq:alpha0}\\
    \beta_1 & = 2 \gamma_1 - \gamma^2 + \frac{\pi^2}{6} + \frac{\ell^2}{12} .
    \label{eq:beta1}
  \end{align}
\end{theorem}

\begin{proof}
  The existence and uniqueness of the three-term
  recurrence~\eqref{eq:recurrence} with
  coefficients~\eqref{eq:recurrence_coeffs} is the content of Favard's theorem
  applied to the measure $w (u)  \hspace{0.17em} du$ on $\mathbb{R}$. The
  measure is determinate (i.e. uniquely determined by its moments) by
  Carleman's criterion: since $w$ has super-exponential decay at $+ \infty$
  and exponential decay at $- \infty$, the moments grow at most factorially,
  so $\sum_{k = 1}^{\infty} \mu_{2 k}^{- 1 / (2 k)} = + \infty$.
  
  {\noindent}\tmtextbf{Computation of $\tmmathbf{\alpha_0}$:}{\hspace{0.27em}}
  $\alpha_0 = \langle u, 1 \rangle_w / \langle 1, 1 \rangle_w = \mu_1 / \mu_0
  = (- \ell^2 / 2) / \ell = - \ell / 2$.
  
  {\noindent}\tmtextbf{Computation of $\tmmathbf{\beta_1}$:}{\hspace{0.27em}}
  We need $\langle P_1, P_1 \rangle_w / \langle P_0, P_0 \rangle_w$. Using
  $P_1 (u) = u + \ell / 2$:
  
  \begin{align*}
    \langle P_1, P_1 \rangle_w & = \mu_2 + \ell \hspace{0.17em} \mu_1 +
    \frac{\ell^2}{4}  \hspace{0.17em} \mu_0 = \mu_2 - \frac{\ell^3}{2} +
    \frac{\ell^3}{4} = \mu_2 - \frac{\ell^3}{4} .
  \end{align*}
  
  Hence
  
  \begin{align*}
    \beta_1 & = \frac{\mu_2 - \ell^3 / 4}{\mu_0} = \frac{1}{\ell}  \left( 2
    \gamma_1 \ell - \gamma^2 \ell + \frac{\ell^3}{3} + \frac{\pi^2 \ell}{6} -
    \frac{\ell^3}{4} \right)\\
    & = 2 \gamma_1 - \gamma^2 + \frac{\pi^2}{6} + \frac{\ell^2}{12} .
  \end{align*}
\end{proof}

\begin{remark}
  \label{rem:beta1_Delta1}The positivity of $\beta_1$ is equivalent to
  $\Delta_1 > 0$: indeed $\beta_1 = \Delta_1 / \mu_0^2 = \Delta_1 / \ell^2$.
  More generally, the positivity of every $\beta_n$ is guaranteed by the
  strict positivity of all Hankel determinants $\Delta_n \assign \det (\mu_{i
  + j})_{0 \le i, j \le n}$, which holds because $w > 0$ a.e. and the support
  of $w \hspace{0.17em} du$ is all of $\mathbb{R}$ (hence contains infinitely
  many points).
\end{remark}
