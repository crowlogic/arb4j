\section*{Exact Closed-Form European Call Price under Rough Heston via
Three-Term Orthogonal Pad\'e of the Cumulant Generator}

\begin{theorem}[Exact rough Heston call price via TTRR Pad\'e]
\label{thm:main}
Let $S_t$ follow rough Heston with $H\in(0,\tfrac12)$,
$\mu=H+\tfrac12$, parameters $\lambda,\theta,\nu,\rho,V_0,r$. Fix
$K>0,T>0$, and set
\[
P(v)=\tfrac12(-v^2-iv),\quad
Q(v)=\lambda(iv\rho\nu-1),\quad
R=\tfrac{\lambda^2\nu^2}{2},\quad
k=\log(K/S_0)-rT.
\]
Let $a_j(v)\in\C[v]$, $j\ge 1$, be the M\"untz--Tau coefficients
\[
a_1=\frac{P(v)}{\Gamma(\mu+1)},\quad
a_j=\frac{\Gamma((j-1)\mu+1)}{\Gamma(j\mu+1)}\!
\Big(Qa_{j-1}+R\!\sum_{\ell=1}^{j-2}a_\ell a_{j-1-\ell}\Big),\quad j\ge 2,
\]
and the cumulant lattice coefficients
\[
d_k(v)=u_k a_k(v)+w_k a_{k+1}(v),\quad a_0:=0,\quad
\Phi(v,T)=\sum_{k\ge 0}d_k(v)T^{k\mu+1}=T\cdot\widetilde\Phi(T^\mu;v),
\]
\[
u_k=\begin{cases}0&k=0\\ \dfrac{\lambda\theta}{k\mu+1}&k\ge 1\end{cases},\qquad
w_k=\begin{cases}V_0\Gamma(\mu+1)&k=0\\[4pt]
V_0\dfrac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+2)}&k\ge 1\end{cases}.
\]

\medskip
\noindent\textbf{Three-term orthogonal Pad\'e construction.}
Treat $\widetilde\Phi(z;v)=\sum_{k\ge 0}d_k(v)z^k$ as a formal moment
sequence in $z$ with moments $\{d_k(v)\}_{k\ge 0}$ and moment functional
$\mathcal{L}_v[z^k]:=d_k(v)$ on $\C[z]$. Run the Chebyshev/Stieltjes
algorithm for $M+1$ steps to produce the monic formal orthogonal
polynomial sequence $\{p_n(z;v)\}_{n=0}^{M+1}$ via the three-term
recurrence
\begin{equation}\label{eq:TTRR}
p_{n+1}(z;v)=(z-\alpha_n(v))\,p_n(z;v)-\beta_n(v)\,p_{n-1}(z;v),\qquad
p_{-1}\equiv 0,\ p_0\equiv 1,
\end{equation}
with $\alpha_n(v)=\mathcal{L}_v[z\,p_n^2]/\mathcal{L}_v[p_n^2]$,
$\beta_n(v)=\mathcal{L}_v[p_n^2]/\mathcal{L}_v[p_{n-1}^2]$, and the
associated polynomials of the first kind $\{p_n^{(1)}(z;v)\}$
satisfying \eqref{eq:TTRR} with shifted initial data
$p_{-1}^{(1)}\equiv -1$, $p_0^{(1)}\equiv 0$. We adopt throughout the
convention that $p_n$ has degree $n$, that the FOPS index runs to
$M+1$, and that
\begin{equation}\label{eq:padeFOPS}
S_M(z;v):=d_0(v)\cdot\frac{p_{M+1}^{(1)}(z;v)}{p_{M+1}(z;v)}
\end{equation}
is the diagonal $[M/M]$ Pad\'e element of $\widetilde\Phi(\cdot;v)$,
matching $2M+2$ moments.

\medskip
Define
\[
\Phi_M(v,T):=T\cdot S_M(T^\mu;v)
=\sum_{j=1}^{N}\frac{c_j(T)}{v-\eta_j(T)},\qquad N\le 2M,
\]
$\varphi_M=\exp\Phi_M$,\quad
$H_j(v)=\exp\!\big(\sum_{i\ne j}c_i/(v-\eta_i)\big)$,\quad
$G_j(v)=e^{-ivk}H_j(v)/[v(v+i)]$,
\[
\mathcal{R}_j=\sum_{n=0}^\infty\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\]
For $M$ large, $\Imag\eta_j<0$ for all $j$; pick
$\alpha\in(1,p^\ast)$ with $p^\ast=\min_j(-\Imag\eta_j)>0$. Then
\begin{equation}\label{eq:price}
\boxed{\;
C_M=\begin{cases}
(S_0-Ke^{-rT})-Ke^{-rT}\,\Real\!\Big[i\!\!\!\sum_{\Imag\eta_j>-\alpha}\!\!\!\mathcal{R}_j\Big],&k\le 0,\\[1.3ex]
-\,Ke^{-rT}\,\Real\!\Big[i\!\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\!\mathcal{R}_j\Big],&k>0.
\end{cases}\;}
\end{equation}
$C_M\to C$ at the logarithmic-capacity geometric rate; bit-precision
$2^{-B}$ is achieved by the smallest $M$ satisfying the a-posteriori
bound on $|\widetilde\Phi-S_M|$.
\end{theorem}

\begin{proof}
\textbf{Step 1: M\"untz-lattice expansion.}
The power rule
$I^r t^{j\mu}=\Gamma(j\mu+1)/\Gamma(j\mu+r+1)\cdot t^{j\mu+r}$
applied termwise to
$\Phi(v,T)=\lambda\theta\!\int_0^T\!h\,ds+V_0 I^{1-\mu}h(v,T)$,
with reindexing $j=k+1$ on the second sum, gives
$\Phi(v,T)=\sum_k d_k(v)T^{k\mu+1}=T\widetilde\Phi(T^\mu;v)$.

\textbf{Step 2: TTRR Pad\'e of $\widetilde\Phi$.}
The Chebyshev algorithm, run on the moment sequence $\{d_k(v)\}$,
produces $\alpha_n(v),\beta_n(v)\in\C(v)$ recursively, generating
$\{p_n\}$ and $\{p_n^{(1)}\}$ through \eqref{eq:TTRR}. By the
classical Pad\'e--FOPS correspondence (Brezinski, Gautschi, Chihara),
the convergent
$d_0(v)\,p_{n}^{(1)}(z;v)/p_n(z;v)$ of the associated continued
fraction is the diagonal $[n-1/n]$ Pad\'e element of
$\widetilde\Phi$, matching $2n$ moments; with $n=M+1$ this is the
$[M/M]$ element of \eqref{eq:padeFOPS}, matching $2M+2$ moments
(equivalently, $S_M$ is the even part of the J-fraction associated
with $\mathcal{L}_v$ truncated at level $M+1$).

Stahl's theorem on $\widetilde\Phi(\cdot;v)$ gives convergence in
logarithmic capacity outside the minimal-capacity singular set; the
geometric tail bound
$|\widetilde\Phi-S_M|\le|\Delta_M|^2/(|\Delta_{M-1}|-|\Delta_M|)$ with
$\Delta_k=S_k-S_{k-1}$ follows from $\Delta_M=O(z^{2M+1})$.

Rationality of $S_M(z;v)$ in $v$ at fixed $z$ is preserved because
$\alpha_n(v),\beta_n(v)\in\C(v)$. Hence $\Phi_M(v,T)$ is rational in
$v$ with partial fraction $\sum c_j/(v-\eta_j)$, $N\le 2M$.

\textbf{Step 3: Polynomial structure, normalization, pole location.}

\emph{(a) Vanishing.} $a_1(v)=(-v^2-iv)/(2\Gamma(\mu+1))$ has roots
$0,-i$; the recurrence preserves degree $j+1$ and roots $0,-i$.
Hence $d_k(0)=d_k(-i)=0$ for all $k$, so
$\widetilde\Phi(z;0)=\widetilde\Phi(z;-i)=0$ identically in $z$,
$\Phi_M(0,T)=\Phi_M(-i,T)=0$, and $\varphi_M(0)=\varphi_M(-i)=1$.

\emph{(b) Symmetry.} $P(-\bar v)=\overline{P(v)}$,
$Q(-\bar v)=\overline{Q(v)}$, $\overline R=R$ propagate to
$a_j(-\bar v)=\overline{a_j(v)}$,
$d_k(-\bar v)=\overline{d_k(v)}$, hence
$\Phi_M(-\bar v,T)=\overline{\Phi_M(v,T)}$.

\emph{(c) Pole location.} El Euch--Rosenbaum moment stability gives
$h(v_0,t)$ bounded on $[0,T]$ uniformly for $v_0$ in compact subsets
of $\overline{\{\Imag v\ge 0\}}$, hence $\Phi(v_0,T)$ finite there.
By Step 2 and Hurwitz on $1/\Phi_M\to 1/\Phi$ in
$\overline{\{\Imag v\ge 0\}}\setminus\{0,-i\}$, no pole of $\Phi_M$
accumulates in the closed upper half-plane for $M$ large; Volterra
contraction at real $v_0$ rules out $\Imag\eta_j=0$. Hence
$\Imag\eta_j<0$ and $p^\ast>0$.

\emph{(d) Strip holomorphy.} $\varphi_M$ is holomorphic and bounded
on $-p^\ast<\Imag v<0$.

\textbf{Step 4: Plancherel and payoff transform.}
With $X_T=\log(S_T/S_0)-rT$, $w(x)=(S_0e^{x+rT}-K)^+$, and Plancherel,
\[
C_M=\frac{e^{-rT}}{2\pi}\!\int_{\Imag v=-\alpha}\!\!\varphi_M\widehat w\,dv,\qquad
\widehat w(v)=-\frac{Ke^{-ivk}}{v(v+i)}\quad(\Imag v<-1),
\]
hence $C_M=-\frac{Ke^{-rT}}{2\pi}\!\int_{\Imag v=-\alpha}\!
\frac{e^{-ivk}\varphi_M(v)}{v(v+i)}\,dv$.
The contour shift from $-1-\varepsilon$ to $-\alpha$ is valid by
Step~3(d).

\textbf{Step 5: Contour closure.}
Set $F(v)=e^{-ivk}\varphi_M(v)/[v(v+i)]$.

\emph{$k\le 0$:} close upward (counterclockwise);
$\Res_{v=0}F=-i$, $\Res_{v=-i}F=ie^{-k}$, kernel piece equals
$Ke^{-rT}(e^{-k}-1)=S_0-Ke^{-rT}$; model piece is
$-iKe^{-rT}\!\sum_{\Imag\eta_j>-\alpha}\!\mathcal{R}_j$. Reality of
$C_M$ gives the first branch of \eqref{eq:price}.

\emph{$k>0$:} close downward (clockwise); kernel poles not enclosed.
$\int F\,dv=-2\pi i\sum_{\Imag\eta_j<-\alpha}\mathcal{R}_j$, and
\[
C_M=-\frac{Ke^{-rT}}{2\pi}(-2\pi i)\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\mathcal{R}_j
=-iKe^{-rT}\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\mathcal{R}_j,
\]
so $C_M=-Ke^{-rT}\,\Real[i\sum\mathcal{R}_j]$, the second branch of
\eqref{eq:price}.

\textbf{Step 6: Residue series.}
$\varphi_M(v)=\exp(c_j/(v-\eta_j))H_j(v)$ near $\eta_j$, $G_j$
holomorphic at $\eta_j$. Coefficient extraction at $p=n+1$:
$\mathcal{R}_j=\sum_{n\ge 0}c_j^{n+1}G_j^{(n)}(\eta_j)/[(n+1)!\,n!]$.
Cauchy estimate with $\rho=\min_{i\ne j}|\eta_i-\eta_j|$ gives
absolute convergence bounded by $M_\rho|c_j|e^{|c_j|/\rho}$.

\textbf{Step 7: Convergence.} $\Phi_M\to\Phi$ at the
logarithmic-capacity geometric rate uniformly on $\Imag v=-\alpha$;
local Lipschitzness of $\exp$ yields $\varphi_M\to\varphi$ uniformly
on the contour, hence $C_M\to C$.
\end{proof}

\begin{remark}[Pure recurrence pipeline]
The entire computation is recurrence-driven:
\begin{enumerate}\itemsep0pt
\item M\"untz--Tau recurrence $\to$ $\{a_j(v)\}$.
\item Linear combination $\to$ $\{d_k(v)\}$ (the cumulant moments).
\item Chebyshev algorithm on $\{d_k(v)\}$ for $M+1$ steps $\to$
TTRR coefficients $\alpha_n(v),\beta_n(v)$.
\item TTRR \eqref{eq:TTRR} $\to$ FOPS $\{p_n\}_{n=0}^{M+1}$ and
associated $\{p_n^{(1)}\}_{n=0}^{M+1}$.
\item $S_M=d_0\,p_{M+1}^{(1)}/p_{M+1}$, evaluated at $z=T^\mu$,
factored in $v$ to extract $\eta_j,c_j$.
\item Residue series $\mathcal{R}_j$ assembled into \eqref{eq:price}.
\end{enumerate}
\end{remark}
