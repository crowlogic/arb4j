\title{Rough Heston European Call: Implementation Specification}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

Fix parameters $H\in(0,\tfrac12)$, $\mu=H+\tfrac12$,
$\lambda,\theta,\nu,\rho,V_0,r,S_0,K,T$ and truncation order $M$.

\bigskip
\noindent\textbf{Step 1.} Set
\begin{align}
P(v) &= \tfrac12(-v^2-iv), &
Q(v) &= \lambda(iv\rho\nu-1), &
R &= \tfrac12\lambda^2\nu^2, &
k &= \log(K/S_0)-rT.
\end{align}

\bigskip
\noindent\textbf{Step 2.} Compute $a_j(v)$ for $j=1,\ldots,2M+2$:
\begin{align}
a_1 &= \frac{P(v)}{\Gamma(\mu+1)}, \\
a_j &= \frac{\Gamma((j-1)\mu+1)}{\Gamma(j\mu+1)}
\!\left(Q\,a_{j-1}+R\sum_{\ell=1}^{j-2}a_\ell\,a_{j-1-\ell}\right),
\quad j\ge2.
\end{align}

\bigskip
\noindent\textbf{Step 3.} Form cumulant moments $d_k(v)$ for $k=0,\ldots,2M$,
with $a_0:=0$:
\begin{align}
u_k &= \begin{cases}0&k=0\\[2pt]\dfrac{\lambda\theta}{k\mu+1}&k\ge1\end{cases}, &
w_k &= \begin{cases}V_0\Gamma(\mu+1)&k=0\\[4pt]
V_0\dfrac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+2)}&k\ge1\end{cases},
\end{align}
\begin{equation}
d_k(v) = u_k\,a_k(v)+w_k\,a_{k+1}(v).
\end{equation}

\bigskip
\noindent\textbf{Step 4.} Run the Chebyshev--Stieltjes algorithm on
$\{d_k(v)\}_{k=0}^{2M}$. Initialize $\sigma(0,\ell)=d_\ell$,
$\sigma(-1,\ell)=0$, $\alpha_0=d_1/d_0$, $\beta_0=d_0$.
For $n=1,\ldots,M$:
\begin{align}
\sigma(n,\ell) &= \sigma(n-1,\ell+1)
-\alpha_{n-1}\,\sigma(n-1,\ell)-\beta_{n-1}\,\sigma(n-2,\ell), \\
\alpha_n &= \frac{\sigma(n,n+1)}{\sigma(n,n)}
-\frac{\sigma(n-1,n)}{\sigma(n-1,n-1)}, \\
\beta_n &= \frac{\sigma(n,n)}{\sigma(n-1,n-1)}.
\end{align}

\bigskip
\noindent\textbf{Step 5.} Evaluate the three-term recurrence at $z=T^\mu$
($p_{-1}=0$, $p_0=1$, $p_{-1}^{(1)}=-1$, $p_0^{(1)}=0$):
\begin{align}
p_{n+1} &= (T^\mu-\alpha_n)\,p_n-\beta_n\,p_{n-1}, \\
p_{n+1}^{(1)} &= (T^\mu-\alpha_n)\,p_n^{(1)}-\beta_n\,p_{n-1}^{(1)}.
\end{align}
Let $A(v):=d_0(v)\cdot p_{M+1}^{(1)}$ and $B(v):=p_{M+1}(v)$
(polynomials in $v$); then
\begin{equation}
\Phi_M(v,T) = T\cdot\frac{A(v)}{B(v)}.
\end{equation}

\bigskip
\noindent\textbf{Step 6.} Factor $B(v)=\prod_{j=1}^N(v-\eta_j)$
to get the poles $\{\eta_j\}$, then compute residues by polynomial
evaluation:
\begin{equation}
c_j = \frac{T\cdot A(\eta_j)}{B'(\eta_j)},\qquad j=1,\ldots,N,
\end{equation}
so that
\begin{equation}
\Phi_M(v,T)=\sum_{j=1}^N\frac{c_j}{v-\eta_j},\qquad N\le2M.
\end{equation}

\bigskip
\noindent\textbf{Step 7.} For each $j$ define
\begin{align}
H_j(v) &= \exp\!\left(\sum_{i\ne j}\frac{c_i}{v-\eta_i}\right), \\
G_j(v) &= \frac{e^{-ivk}H_j(v)}{v(v+i)}.
\end{align}

\bigskip
\noindent\textbf{Step 8.} Sum until the tail bound
$M_{\rho_j}|c_j|^{n+2}/(\rho_j^n(n+2)!)$ is below tolerance,
where $\rho_j=\min_{i\ne j}|\eta_i-\eta_j|$:
\begin{equation}
\mathcal{R}_j = \sum_{n=0}^{\infty}
\frac{c_j^{n+1}\,G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\end{equation}

\bigskip
\noindent\textbf{Step 9.} Pick any $\alpha\in(1,\,p^*)$ where
$p^*=\min_j(-\Imag\,\eta_j)$:
\begin{equation}
C_M=\begin{cases}
(S_0-Ke^{-rT})
-Ke^{-rT}\,\Real\!\left[i\displaystyle
\sum_{\Imag\,\eta_j>-\alpha}\!\!\mathcal{R}_j\right],
& k\le0,\\[1.8ex]
-\,Ke^{-rT}\,\Real\!\left[i\displaystyle
\sum_{\Imag\,\eta_j<-\alpha}\!\!\mathcal{R}_j\right],
& k>0.
\end{cases}
\end{equation}
