\title{The Fractional Riccati--M\"{u}ntz--Pad\'{e} Theorem:\\
\large Exact Rough Heston Option Pricing}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Let $\mu=H+\tfrac12\in(\tfrac12,1)$ and let $h(v,t)$ solve the rough
Heston fractional Riccati equation.  Define scalar weights
$u_k=\lambda\theta/(k\mu+1)$ ($k\ge1$, $u_0=0$) and
$w_k=V_0\Gamma((k+1)\mu+1)/\Gamma(k\mu+2)$ ($k\ge0$), and set
$d_k(v):=u_k a_k(v)+w_k a_{k+1}(v)$ where $\{a_k(v)\}$ are the
M\"{u}ntz--Tau coefficients of $h$.  Then
$\Phi(v,T)=\sum_{k=0}^\infty d_k(v)T^{k\mu+1}$
is a power series in $z=T^\mu$ with moments $d_k(v)$, and its
diagonal $[M/M]$ Pad\'{e} approximant $\Phi_M(v,T)$,
constructed by the Wheeler algorithm applied to
$\Lk[z^k]=d_k(v)$, converges uniformly to $\Phi$ and admits
the partial-fraction form $\Phi_M=\sum_j c_j/(v-\eta_j)$
whose poles $\eta_j$ are eigenvalues of the associated Jacobi
matrix.  The European call price is then the exact finite sum
given in Theorem~\ref{thm:price}.
\end{abstract}

\section{Fractional calculus and the M\"{u}ntz basis}
\label{sec:fc}
% ------------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
\begin{align}
\Imu f(t)&:=\frac{1}{\Gamma(\mu)}\int_0^t(t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t)&:=\frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$ and $r>0$,
$I^r t^s = \tfrac{\Gamma(s+1)}{\Gamma(s+r+1)}\,t^{s+r}$.
\end{lemma}
\begin{proof}
Substitute $\sigma=t\tau$; the beta integral gives
$B(r,s+1)=\Gamma(r)\Gamma(s+1)/\Gamma(s+r+1)$.
\end{proof}

% ------------------------------------------------------------------
\section{M\"{u}ntz--Tau recurrence and the $d_k$ moments}
\label{sec:recurrence}
% ------------------------------------------------------------------

Set $\gamma_k:=\Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$.

\begin{theorem}[M\"{u}ntz--Tau recurrence]\label{thm:recurrence}
The unique solution of
$D^\mu h=c_0+c_1 h+c_2 h^2$, $\Ico h(0)=0$,
is $h(v,t)=\sum_{k\ge1}a_k(v)t^{k\mu}$ with
\begin{align}
a_1&=\frac{c_0(v)}{\Gamma(\mu+1)},\label{eq:a1}\\
a_k&=\gamma_k\!\left[c_1(v)a_{k-1}+c_2\sum_{j=1}^{k-2}a_ja_{k-1-j}\right],
\quad k\ge2.\label{eq:ak}
\end{align}
Each $a_k(v)\in\C[v]$ has degree $k+1$ and $a_k(0)=a_k(-i)=0$.
\end{theorem}
\begin{proof}
Match coefficients of $t^{k\mu}$ using the power rule and the Cauchy
product; uniqueness by induction on the error sequence.
\end{proof}

\begin{definition}[$d_k$ moments]\label{def:dk}
Set $u_0=0$, $u_k=\lambda\theta/(k\mu+1)$ for $k\ge1$,
and $w_k=V_0\Gamma((k+1)\mu+1)/\Gamma(k\mu+2)$ for $k\ge0$.
Define
\begin{equation}\label{eq:dk}
d_k(v):=u_k a_k(v)+w_k a_{k+1}(v)\in\C[v],\qquad k\ge0.
\end{equation}
\end{definition}

\begin{theorem}[M\"{u}ntz-lattice form of the cumulant]\label{thm:Phi_series}
With $a_0(v):=0$,
\begin{equation}\label{eq:Phi_series}
\Phi(v,T)
:=\lambda\theta\int_0^T h(v,s)\,ds+V_0\,\Ico h(v,\cdot)(T)
=\sum_{k=0}^\infty d_k(v)\,T^{k\mu+1}.
\end{equation}
\end{theorem}
\begin{proof}
By the power rule with $r=1$:
$\lambda\theta\int_0^T s^{k\mu}\,ds=u_k T^{k\mu+1}$.
By the power rule with $r=1-\mu$:
$V_0 I^{1-\mu}[s^{k\mu}](T)=V_0\tfrac{\Gamma(k\mu+1)}{\Gamma((k-1)\mu+2)}T^{(k-1)\mu+1}$;
reindex $k\mapsto k+1$ to land on $T^{k\mu+1}$ with coefficient $w_k$.
Both operations act diagonally on the M\"{u}ntz coefficients; summing
gives~\eqref{eq:Phi_series}.
\end{proof}

% ------------------------------------------------------------------
\section{Pad\'{e} approximant of $\Phi$ via Wheeler}
\label{sec:wheeler}
% ------------------------------------------------------------------

Write $\Phi(v,T)=T\cdot\hat\Phi(v,T^\mu)$ where
$\hat\Phi(v,z)=\sum_{k=0}^\infty d_k(v)z^k$.
The diagonal $[M/M]$ Pad\'{e} approximant of $\hat\Phi$ in $z$ at
fixed $v$ is determined by the moment functional
$\Lk[z^k]:=d_k(v)$.

\subsection{Wheeler algorithm}

Initialize $\sigma(-1,\ell)=0$, $\sigma(0,\ell)=d_{\ell}(v)$.
For $k=1,\ldots,M-1$:
\begin{equation}\label{eq:sigma_rec}
\sigma(k,\ell)=\sigma(k-1,\ell+1)
-\alpha(k-1)\sigma(k-1,\ell)-\beta(k-1)\sigma(k-2,\ell),
\end{equation}
\begin{align}
\alpha(0)&=d_1/d_0,\quad\beta(0)=d_0,\label{eq:ab0}\\
\alpha(k)&=\sigma(k,k+1)/\sigma(k,k)-\sigma(k-1,k)/\sigma(k-1,k-1),
\label{eq:alpha_k}\\
\beta(k)&=\sigma(k,k)/\sigma(k-1,k-1).\label{eq:beta_k}
\end{align}
The three-term recurrence
$\pi_{k+1}=(z-\alpha(k))\pi_k-\beta(k)\pi_{k-1}$,
$\pi_{-1}=0$, $\pi_0=1$,
produces the denominator $\pi_M=Q_M$ of the Pad\'{e} approximant.

\subsection{Jacobi matrix and poles}

Assemble the $M\times M$ tridiagonal Jacobi matrix
\begin{equation}\label{eq:JM}
J_M=\begin{pmatrix}
\alpha(0)&1&&\\
\beta(1)&\alpha(1)&1&\\
&\ddots&\ddots&\ddots\\
&&\beta(M-1)&\alpha(M-1)
\end{pmatrix}.
\end{equation}
The eigenvalues $\{\lambda_j\}$ of $J_M$ are the zeros of $\pi_M$;
the poles of $\Phi_M(v,T)=T\cdot P_M(v,T^\mu)/Q_M(v,T^\mu)$
in $z=T^\mu$ are $\{z_j^\#=1/\lambda_j\}$.
With unit eigenvectors $u_j$ of $J_M$ and $e_1=(1,0,\ldots,0)^\top$,
the partial-fraction form is
\begin{equation}\label{eq:PhiM_pf_z}
\Phi_M(v,T)=T\sum_{j=1}^M\frac{w_j^\#}{T^\mu-z_j^\#},\qquad
w_j^\#=\frac{(e_1^\top u_j)^2}{\lambda_j^2}.
\end{equation}

% ------------------------------------------------------------------
\section{Convergence (Theorem T1)}
\label{sec:conv}
% ------------------------------------------------------------------

\begin{theorem}[Convergence of $\Phi_M$]\label{thm:conv}
Let $T^*$ be the radius of convergence of $\hat\Phi(v,\cdot)$
in $z=T^\mu$.  Then $\Phi_M(v,T)\to\Phi(v,T)$ uniformly on
every compact $[0,T]\subset[0,T^*)$, with a-posteriori bound
\begin{equation}\label{eq:aposteriori}
|\Phi(v,T)-\Phi_M(v,T)|\le
\frac{|\Delta_M|^2}{|\Delta_{M-1}|-|\Delta_M|},\qquad
\Delta_k:=\Phi_k-\Phi_{k-1},
\end{equation}
whenever $|\Delta_M|<|\Delta_{M-1}|$.
\end{theorem}
\begin{proof}
$\hat\Phi(v,\cdot)$ is meromorphic on $\C\setminus K$ for the Stahl
compact $K$~\cite{StahlBranch}; the eigenvalues of $J_M$ converge
in capacity to $K$ and are bounded away from $[0,(T^*)^\mu)$ for
all large $M$.  Capacity convergence upgrades to uniform convergence
on compacta by normal families.  The bound follows from summing a
geometric tail with ratio $|\Delta_M|/|\Delta_{M-1}|<1$.
\end{proof}

% ------------------------------------------------------------------
\section{The $v$-plane partial fraction and the price}
\label{sec:pricing}
% ------------------------------------------------------------------

For fixed $T$, $\Phi_M(v,T)$ is a rational function of $v$ (since
each $d_k(v)\in\C[v]$).  By Theorem~\ref{thm:Phi_series} and the
closed form~\eqref{eq:PhiM_pf_z}, collect over a common denominator
in $v$ to obtain
\begin{equation}\label{eq:PhiM_v_pf}
\Phi_M(v,T)=\sum_{j=1}^N\frac{c_j(T)}{v-\eta_j(T)},\qquad N\le 2M,
\end{equation}
where the poles $\{\eta_j(T)\}$ and residues $\{c_j(T)\}$ are
obtained by a second application of the Wheeler algorithm to the
Laurent coefficients of $\Phi_M(\cdot,T)$ at $v=\infty$
(moments $b_k(T):=[v^{-(k+1)}]\Phi_M(v,T)$, computable from~\eqref{eq:PhiM_pf_z}):
the eigenvalues of the resulting Jacobi matrix $J_N^{(v)}$ are the
$\eta_j(T)$ and its Christoffel numbers are the $c_j(T)$.

\begin{lemma}\label{lem:norm}
$\Phi_M(0,T)=\Phi_M(-i,T)=0$, hence
$\varphi_M(v):=\exp\Phi_M(v,T)$ satisfies $\varphi_M(0)=\varphi_M(-i)=1$.
\end{lemma}
\begin{proof}
$a_k(0)=a_k(-i)=0$ for all $k$ (Theorem~\ref{thm:recurrence}),
so $d_k(0)=d_k(-i)=0$, hence $\Phi_M(0,T)=\Phi_M(-i,T)=0$.
\end{proof}

\begin{lemma}\label{lem:herm}
$\varphi_M(-\bar v)=\overline{\varphi_M(v)}$.
The poles $\{\eta_j\}$ lie off $\R$ and come in conjugate pairs;
$p^*:=\min_j(-\operatorname{Im}\eta_j)>0$.
\end{lemma}
\begin{proof}
$c_0(-\bar v)=\overline{c_0(v)}$, $c_1(-\bar v)=\overline{c_1(v)}$,
$c_2\in\R$; induction on~\eqref{eq:ak} gives
$a_k(-\bar v)=\overline{a_k(v)}$, hence $d_k(-\bar v)=\overline{d_k(v)}$
and $\Phi_M(-\bar v,T)=\overline{\Phi_M(v,T)}$.
A real pole $\eta_j\in\R$ would force a real blow-up of $h_M$ on
$(0,T^*)$, excluded by Theorem~\ref{thm:conv}.
\end{proof}

\begin{theorem}[Exact European call price]\label{thm:price}
Let $k_0=\log(K/S_0)-rT$,
$G_j(v)=e^{-ivk_0}H_j(v)/[v(v+i)]$,
$H_j(v)=\exp(\sum_{\ell\ne j}c_\ell/(v-\eta_\ell))$, and
\begin{equation}\label{eq:Rj}
\mathcal{R}_j=\sum_{n=0}^\infty
\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\end{equation}
For any $\alpha\in(1,p^*)$,
\begin{equation}\label{eq:price_formula}
C_M=\begin{cases}
(S_0-Ke^{-rT})-Ke^{-rT}\operatorname{Re}\!\bigl[i\!\!\displaystyle
\sum_{\operatorname{Im}\eta_j>-\alpha}\!\!\mathcal{R}_j\bigr],
& k_0\le0,\\[1.5ex]
-Ke^{-rT}\operatorname{Re}\!\bigl[i\!\!\displaystyle
\sum_{\operatorname{Im}\eta_j<-\alpha}\!\!\mathcal{R}_j\bigr],
& k_0>0.
\end{cases}
\end{equation}
\end{theorem}
\begin{proof}
\textit{Step 1 (Parseval).}
Let $w(x)=(S_0e^{x+rT}-K)^+$.  For $\operatorname{Im}v<-1$,
$\int_\R e^{-ivx}w(x)\,dx=-Ke^{-ivk_0}/[v(v+i)]$
(integrate $\int_{k_0}^\infty$; use $S_0e^{k_0+rT}=K$ and
$1/(iv)-1/(1-iv)=-1/[v(v+i)]$).
Risk-neutral pricing and Plancherel give
\[
C_M=-\frac{Ke^{-rT}}{2\pi}\int_{\operatorname{Im}v=-\alpha}
\frac{e^{-ivk_0}\varphi_M(v)}{v(v+i)}\,dv.
\]
The contour shift from $\operatorname{Im}v=-1-\varepsilon$ to
$\operatorname{Im}v=-\alpha$ is valid: $\varphi_M$ is holomorphic
and bounded on $-p^*<\operatorname{Im}v<0$ (Lemma~\ref{lem:herm})
and $\Phi_M(v,T)=O(1/|v|)$ as $|v|\to\infty$, so Jordan's lemma
applies.

\textit{Step 2 (Contour closure and kernel residues).}
For $k_0\le0$: $|e^{-ivk_0}|\to0$ as $\operatorname{Im}v\to+\infty$;
close upward (counterclockwise).  For $k_0>0$: close downward
(clockwise, contributing $-2\pi i$).
Kernel residues using Lemma~\ref{lem:norm}:
\[
\operatorname{Res}_{v=0}F=-i,\quad
\operatorname{Res}_{v=-i}F=ie^{-k_0},\quad
-Ke^{-rT}\cdot i[(-i)+ie^{-k_0}]=S_0-Ke^{-rT}.
\]
Kernel poles $v=0,-i$ lie above $\operatorname{Im}v=-\alpha$ ($\alpha>1$)
and are enclosed only for $k_0\le0$.

\textit{Step 3 (Model residues).}
Write $\varphi_M(v)=e^{c_j/(v-\eta_j)}H_j(v)$ near $\eta_j$;
expand both factors as Laurent/Taylor series in $(v-\eta_j)$;
the coefficient of $(v-\eta_j)^{-1}$ comes from $p=n+1$:
$\mathcal{R}_j=\operatorname{Res}_{v=\eta_j}F$
gives~\eqref{eq:Rj}.  Absolute convergence: Cauchy's estimate
gives $|G_j^{(n)}(\eta_j)/n!|\le M_\rho\rho^{-n}$ where
$\rho=\min_{\ell\ne j}|\eta_\ell-\eta_j|$, so
$\sum_n|c_j|^{n+1}M_\rho\rho^{-n}/(n+1)!\le M_\rho|c_j|e^{|c_j|/\rho}<\infty$.

\textit{Step 4 (Reality).}
Conjugate-pair symmetry
(Lemma~\ref{lem:herm}) gives
$\mathcal{R}_{\bar\eta_j}=\overline{\mathcal{R}_{\eta_j}}$,
so $\operatorname{Re}[i\sum_j\mathcal{R}_j]\in\R$.

\textit{Step 5 (Signs).}
For $k_0\le0$ the contour is counterclockwise and the factor is
$-Ke^{-rT}\cdot\frac{2\pi i}{2\pi}$, giving net coefficient
$-iKe^{-rT}$ on $\sum\mathcal{R}_j$.
For $k_0>0$ the contour is clockwise ($-2\pi i$) and kernel poles
are excluded; net coefficient is $+iKe^{-rT}$ on
$\sum_{\operatorname{Im}\eta_j<-\alpha}\mathcal{R}_j$;
taking $\operatorname{Re}[i\sum\mathcal{R}_j]=-\operatorname{Re}[i\sum(-\mathcal{R}_j)]$
yields $-Ke^{-rT}\operatorname{Re}[i\sum_{\operatorname{Im}\eta_j<-\alpha}\mathcal{R}_j]$.
\end{proof}

\begin{proposition}[Convergence rate]\label{prop:pconv}
$|C_M-C|\le C'r_*^M$ for some $r_*\in(0,1)$; the rate is
determined by the logarithmic capacity of the singular set of
$\hat\Phi(v,\cdot)$.
\end{proposition}
\begin{proof}
Theorem~\ref{thm:conv} gives geometric decay of
$|\Phi_M-\Phi|$ on $\operatorname{Im}v=-\alpha$; the
kernels $e^{-ivk_0}/(v(v+i))$ are integrable there.
\end{proof}
