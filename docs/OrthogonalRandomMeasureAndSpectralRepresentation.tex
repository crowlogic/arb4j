\begin{abstract}
Let $\theta$ be the Riemann--Siegel theta function, $Z(t)=e^{i\theta(t)}\zeta(\tfrac12+it)$ the Hardy $Z$-function, and $H(x):=Z(\theta^{-1}(x))/\sqrt{\theta'(\theta^{-1}(x))}$ the warped Hardy signal. Given as input the companion results
\begin{itemize}
\item Theorem~A of \emph{Three Missing Theorems}: $C(\eta):=\lim_{X\to\infty}C_X(\eta)=2\cos|\eta|$,
\item the Bochner factorization $C(\eta)=\int_{\mathbb{R}} e^{i\eta\xi}\,d\nu(\xi)$ with $\nu=\delta_{-1}+\delta_{+1}$ from \emph{Existence of the Spectral Measure in the Limit},
\end{itemize}
we construct, without any probabilistic hypothesis added beyond these two inputs, the complex orthogonal measure $\Phi$ of $H$ and prove the spectral representation
\[
  H(x) \;=\; \int_{\mathbb{R}} e^{i\xi x}\,d\Phi(\xi)
\]
as an identity in the Hilbert space generated by the shifts of $H$ under the time-average inner product. The measure $\Phi$ is a Hilbert-space-valued orthogonally-scattered measure on the Borel $\sigma$-algebra of $\mathbb{R}$, supported on $\{-1,+1\}$, with intensity $\langle\Phi(A),\Phi(B)\rangle=\nu(A\cap B)$, and the two atoms $\Phi(\{-1\}),\Phi(\{+1\})$ are orthonormal in that Hilbert space. The corresponding spectral representation of the Hardy $Z$-function is
\[
  Z(t) \;=\; \sqrt{\theta'(t)}\int_{\mathbb{R}} e^{i\xi\theta(t)}\,d\Phi(\xi)\qquad\text{in the same Hilbert space.}
\]
\end{abstract}

\section{Inputs and notation}\label{sec:inputs}

Fix $T_0\ge 200$ and set $X_0:=\theta(T_0)$. The Riemann--Siegel theta function
\[
  \theta(t)\;=\;\Im\log\Gamma\!\left(\tfrac14+\tfrac{it}{2}\right)-\tfrac{t}{2}\log\pi
\]
is a $C^\infty$-diffeomorphism of $[T_0,\infty)$ onto $[X_0,\infty)$ with $\theta'>0$ there. The Hardy $Z$-function is $Z(t)=e^{i\theta(t)}\zeta(\tfrac12+it)$, which is real-valued on $\mathbb{R}$. The warped Hardy signal is
\[
  H(x)\;:=\;\frac{Z(\theta^{-1}(x))}{\sqrt{\theta'(\theta^{-1}(x))}},\qquad x\in[X_0,\infty),
\]
and $H$ is a locally integrable (in fact continuous) real-valued function on $[X_0,\infty)$. For $\eta\in\mathbb{R}$ and $X>X_0+|\eta|$ define the one-sided time-averaged covariance
\begin{equation}\label{eq:CX}
  C_X(\eta)\;:=\;\frac{1}{X-X_0}\int_{X_0}^{X-|\eta|} H(x)\,\overline{H(x+|\eta|)}\,dx.
\end{equation}
Since $H$ is real, $C_X(\eta)=C_X(-\eta)$.

\paragraph{Inputs cited.}
\begin{description}
\item[\textbf{Input 1 (Theorem~A).}] For every $\eta\in\mathbb{R}$,
\[
  C(\eta)\;:=\;\lim_{X\to\infty} C_X(\eta)\;=\;2\cos|\eta|,
\]
with an explicit quantitative rate $|C_X(\eta)-C(\eta)|\le 8\pi(1+|\eta|)\log^3(X/X_0)/(X-X_0)+0.349/\sqrt{T_0}$.
\item[\textbf{Input 2 (Bochner measure).}] There is a unique finite positive Borel measure $\nu$ on $\mathbb{R}$ with $C(\eta)=\int_{\mathbb{R}} e^{i\eta\xi}\,d\nu(\xi)$ for all $\eta\in\mathbb{R}$, and
\[
  \nu\;=\;\delta_{-1}+\delta_{+1},\qquad \nu(\mathbb{R})=2.
\]
\end{description}

\paragraph{Shift operator.} For $s\in\mathbb{R}$ and a function $f$ on $[X_0,\infty)$, we write $(T_s f)(x):=f(x+s)$, defined on $[X_0-s,\infty)\cap[X_0,\infty)$; since our inner product below integrates over an interval that eventually contains any bounded translation set, all finite shifts are well-defined members of the Hilbert space constructed.

\section{The time-average Hilbert space}\label{sec:hilbert}

We construct a complex Hilbert space $\mathcal{H}$ in which the shifts $T_s H$, $s\in\mathbb{R}$, live and in which Input~1 reads as a positive-definite kernel.

\begin{definition}[Time-average sesquilinear form on trigonometric combinations]\label{def:inner}
Let $\mathcal{V}$ be the complex vector space spanned by the formal symbols $T_sH$, $s\in\mathbb{R}$: its elements are finite formal linear combinations $f=\sum_{j=1}^{n}a_jT_{s_j}H$ with $a_j\in\mathbb{C}$, $s_j\in\mathbb{R}$, modulo the obvious linearity relations (so $T_sH+T_sH=2T_sH$ etc.). Every such $f$ determines an actual function on $[X_0+\max_j|s_j|^-,\infty)$ where $|s_j|^-:=\max(-s_j,0)$ is the amount of leftward shift, by
\[
  f(x)\;=\;\sum_{j=1}^{n}a_jH(x+s_j),\qquad x\ge X_0+\max_j(-s_j,0).
\]
For $f=\sum_j a_j\,T_{s_j}H$ and $g=\sum_k b_k\,T_{u_k}H$ in $\mathcal{V}$, set
\begin{equation}\label{eq:inner-def}
  \langle f,g\rangle\;:=\;\sum_{j,k} a_j\,\overline{b_k}\,C(u_k-s_j).
\end{equation}
\end{definition}

\begin{lemma}[The form is well-defined, sesquilinear, and equal to a time average]\label{lem:time-avg}
For every pair of formal symbols $T_sH,T_uH$ with $s,u\ge 0$,
\begin{equation}\label{eq:time-avg}
  C(u-s)\;=\;\lim_{X\to\infty}\frac{1}{X-X_0}\int_{X_0}^{X-\max(s,u)} H(x+s)\,\overline{H(x+u)}\,dx,
\end{equation}
and the inner product of two elements $f=\sum_j a_j T_{s_j}H$, $g=\sum_k b_k T_{u_k}H$ with all $s_j,u_k\ge 0$ is
\begin{equation}\label{eq:inner-timeavg}
  \langle f,g\rangle\;=\;\lim_{X\to\infty}\frac{1}{X-X_0}\int_{X_0+M}^{X} f(x)\overline{g(x)}\,dx
\end{equation}
where $M:=\max_j s_j\vee\max_k u_k$. The form \eqref{eq:inner-def} is sesquilinear and Hermitian, and the value \eqref{eq:inner-def} does not depend on the representation of $f$ and $g$.
\end{lemma}

\begin{proof}
For \eqref{eq:time-avg}: by the substitution $y=x+s$ (with $s\ge 0$),
\[
  \int_{X_0}^{X-\max(s,u)}H(x+s)\overline{H(x+u)}\,dx\;=\;\int_{X_0+s}^{X-\max(s,u)+s}H(y)\overline{H(y+(u-s))}\,dy.
\]
Set $\eta:=u-s$ (whose sign is arbitrary); WLOG $\eta\ge 0$ (else use $C(-\eta)=C(\eta)$ and swap $s\leftrightarrow u$, $f\leftrightarrow g$). Then $\max(s,u)=u=s+\eta$ and the integral becomes $\int_{X_0+s}^{X-\eta+s}H(y)\overline{H(y+\eta)}\,dy$. Comparing with $(X-X_0)C_X(\eta)=\int_{X_0}^{X-\eta}H(y)\overline{H(y+\eta)}\,dy$, the two integration intervals differ by two sets each of length $s$ (the left window $[X_0,X_0+s]$ removed, the right window $[X-\eta,X-\eta+s]$ added). Denote the integral of $H(y)\overline{H(y+\eta)}$ over such a window $J$ of length $s$ by $E_J$. Cauchy--Schwarz gives
\[
|E_J|\;\le\;\Bigl(\int_J|H(y)|^2\,dy\Bigr)^{1/2}\Bigl(\int_{J+\eta}|H(y)|^2\,dy\Bigr)^{1/2}.
\]
Input~1 at $\eta=0$ says
\begin{equation}\label{eq:L2growth}
\int_{X_0}^{X}|H(y)|^2\,dy\;=\;(X-X_0)\bigl(2+o(1)\bigr)\qquad(X\to\infty),
\end{equation}
and consequently for any interval $J$ of fixed length $s$ and right endpoint at distance $\le s+\eta$ from $X$,
\[
\int_J|H(y)|^2\,dy\;\le\;\int_{X_0}^{X+s+\eta}|H|^2-\int_{X_0}^{X-s-\eta}|H|^2\;=\;2(2s+2\eta)+o(X)\;=\;O(s+\eta)+o(X).
\]
For an interval $J\subseteq[X_0,X_0+2s]$ of length $s$, $\int_J|H|^2\,dy$ is bounded by the finite constant $\int_{X_0}^{X_0+2s}|H|^2\,dy$, which is a fixed number (continuity of $H$). In either case $|E_J|\le O(s+\eta)+o(X)$, and this divided by $X-X_0$ tends to $0$. Summing over the two endpoint windows,
\[
\Bigl|\int_{X_0+s}^{X-\eta+s}H(y)\overline{H(y+\eta)}\,dy\;-\;\int_{X_0}^{X-\eta}H(y)\overline{H(y+\eta)}\,dy\Bigr|\;=\;O(s+\eta)+o(X).
\]
Dividing by $X-X_0$ and using $\lim_XC_X(\eta)=C(\eta)$ (Input~1),
\[
\lim_{X\to\infty}\frac{1}{X-X_0}\int_{X_0}^{X-\max(s,u)}H(x+s)\overline{H(x+u)}\,dx\;=\;C(u-s).
\]

For \eqref{eq:inner-timeavg}: expand $f\overline g=\sum_{j,k}a_j\overline{b_k}H(x+s_j)\overline{H(x+u_k)}$, integrate from $X_0+M$ to $X$, divide by $X-X_0$, and take $X\to\infty$. The limit of each summand is $a_j\overline{b_k}\cdot C(u_k-s_j)$ by \eqref{eq:time-avg} applied with $X$ replaced by $X-M$ (the $M$-offset is absorbed into the $o(X)$ error). Summing gives \eqref{eq:inner-def}.

Sesquilinearity on both slots is immediate from \eqref{eq:inner-def}. Hermitian symmetry $\langle f,g\rangle=\overline{\langle g,f\rangle}$ amounts to $C(-\eta)=\overline{C(\eta)}$, which holds because $C(\eta)=2\cos|\eta|$ is real.

Independence of the representation of $f$ and $g$: by \eqref{eq:inner-timeavg} the value $\langle f,g\rangle$ depends only on the functions $x\mapsto f(x),g(x)$ on any right half-line $[X_0+M,\infty)$, not on the formal representation.

For symbols $T_sH$ with $s<0$, extend the inner product to $\mathcal{V}$ by the formula \eqref{eq:inner-def}, which is defined for any $s_j,u_k\in\mathbb{R}$ because $C$ is. Sesquilinearity and Hermitian symmetry are again immediate. The time-average formula \eqref{eq:inner-timeavg} then extends to arbitrary real shifts, integrating from $X_0+\max_j(-s_j,-u_k,0)$: the proof is identical, with $s$ replaced by $|s|$ in the endpoint-window bound.
\end{proof}

\begin{lemma}[Positive-semidefiniteness]\label{lem:psd}
The form $\langle\cdot,\cdot\rangle$ on $\mathcal{V}$ is positive semidefinite: $\langle f,f\rangle\ge 0$ for every $f\in\mathcal{V}$.
\end{lemma}

\begin{proof}
By Input~2, $C(\eta)=\int_{\mathbb{R}}e^{i\eta\xi}\,d\nu(\xi)$. For $f=\sum_j a_j T_{s_j}H\in\mathcal{V}$,
\[
\langle f,f\rangle=\sum_{j,k}a_j\overline{a_k}\,C(s_k-s_j)
=\sum_{j,k}a_j\overline{a_k}\int_{\mathbb{R}}e^{i(s_k-s_j)\xi}\,d\nu(\xi)
=\int_{\mathbb{R}}\Bigl|\sum_j a_j e^{-is_j\xi}\Bigr|^2 d\nu(\xi)\ge 0.
\]
\end{proof}

\begin{definition}[The Hilbert space $\mathcal{H}$]\label{def:H}
Let $\mathcal{N}:=\{f\in\mathcal{V}:\langle f,f\rangle=0\}$; by Cauchy--Schwarz for the positive semidefinite form, $\mathcal{N}$ is a linear subspace of $\mathcal{V}$. Equip the quotient $\mathcal{V}/\mathcal{N}$ with the induced inner product $\langle\cdot,\cdot\rangle$. Let $\mathcal{H}$ be the Hilbert-space completion of $\mathcal{V}/\mathcal{N}$.
\end{definition}

By construction, the map $\mathcal{V}\ni f\mapsto [f]\in\mathcal{V}/\mathcal{N}\subseteq\mathcal{H}$ is linear and length-preserving for $\langle\cdot,\cdot\rangle$. We denote the class $[T_sH]\in\mathcal{H}$ simply by $H_s$ and write $H_0=H$ when we mean $[H]$. Then
\begin{equation}\label{eq:cov}
  \langle H_s,H_t\rangle_{\mathcal{H}}\;=\;C(s-t)\;=\;2\cos(s-t),\qquad s,t\in\mathbb{R}.
\end{equation}

\begin{lemma}[Strong continuity of the shift group]\label{lem:shift-cts}
The map $\mathbb{R}\ni s\mapsto H_s\in\mathcal{H}$ is continuous; more precisely,
\[
  \|H_s-H_t\|_{\mathcal{H}}^2\;=\;2C(0)-2\Re C(s-t)\;=\;4-4\cos(s-t)\;=\;4\sin^2\!\tfrac{s-t}{2},
\]
so $\|H_s-H_t\|_{\mathcal{H}}=2\bigl|\sin\tfrac{s-t}{2}\bigr|$.
\end{lemma}

\begin{proof}
Expand $\|H_s-H_t\|_{\mathcal{H}}^2=\langle H_s-H_t,H_s-H_t\rangle$ using \eqref{eq:cov} and $C(0)=2$, $C(s-t)=2\cos(s-t)$.
\end{proof}

\section{The Kolmogorov isometry and the complex orthogonal measure}

\begin{definition}[The closed span]\label{def:cspan}
Let $\mathcal{H}_H:=\overline{\operatorname{span}}\{H_s:s\in\mathbb{R}\}\subseteq\mathcal{H}$, the closed linear span in $\mathcal{H}$ of the shifts of $H$.
\end{definition}

\begin{theorem}[Cramér/Kolmogorov isometry]\label{thm:iso}
There is a unique isometric linear isomorphism
\[
  V\colon\mathcal{H}_H\;\longrightarrow\;L^2(\mathbb{R},\nu)
\]
such that $V(H_s)(\xi)=e^{i\xi s}$ for every $s\in\mathbb{R}$.
\end{theorem}

\begin{proof}
Define $V_0\colon\operatorname{span}\{H_s:s\in\mathbb{R}\}\to L^2(\nu)$ by linear extension of $H_s\mapsto e^{i\xi s}$. This is well-defined: for $f=\sum_j a_jH_{s_j}\in\operatorname{span}\{H_s\}$,
\[
  \|f\|_{\mathcal{H}}^2\;=\;\sum_{j,k}a_j\overline{a_k}C(s_k-s_j)\;=\;\int_{\mathbb{R}}\Bigl|\sum_j a_j e^{is_j\xi}\Bigr|^2 d\nu(\xi)\;=\;\Bigl\|\sum_j a_j e^{i\cdot s_j}\Bigr\|_{L^2(\nu)}^2,
\]
so $V_0$ is well-defined and isometric (in particular, whenever $\sum_j a_j H_{s_j}=0$ in $\mathcal{H}$, also $\sum_j a_j e^{i\xi s_j}=0$ in $L^2(\nu)$, and conversely). By the bounded linear extension theorem, $V_0$ extends uniquely to an isometry $V\colon\mathcal{H}_H\to L^2(\nu)$.

Surjectivity. Since $\nu=\delta_{-1}+\delta_{+1}$ is supported on two points, $L^2(\nu)\simeq\mathbb{C}^2$ under $f\mapsto(f(-1),f(+1))$. The two vectors $V(H_0)=(1,1)$ and $V(H_\pi)=(e^{-i\pi},e^{i\pi})=(-1,-1)$ are parallel, so we need another point. Take $V(H_{\pi/2})=(e^{-i\pi/2},e^{i\pi/2})=(-i,i)$. The vectors $(1,1)$ and $(-i,i)$ are linearly independent in $\mathbb{C}^2$, so the image $V(\operatorname{span}\{H_s\})$ contains a two-dimensional subspace of $\mathbb{C}^2$, hence equals $\mathbb{C}^2=L^2(\nu)$. Since $V$ is isometric with dense range in the complete space $L^2(\nu)$, it is surjective.
\end{proof}

\begin{definition}[The complex orthogonal measure $\Phi$]\label{def:Phi}
For every Borel set $A\subseteq\mathbb{R}$, define
\[
  \Phi(A)\;:=\;V^{-1}(\mathbf{1}_A)\;\in\;\mathcal{H}_H\subseteq\mathcal{H}.
\]
\end{definition}

By construction,
\begin{equation}\label{eq:Phi-norm}
  \|\Phi(A)\|_{\mathcal{H}}^2\;=\;\|\mathbf{1}_A\|_{L^2(\nu)}^2\;=\;\nu(A),
\end{equation}
so $\Phi(A)\in\mathcal{H}$ has finite norm for every Borel set $A$ and $\Phi(A)=0$ whenever $\nu(A)=0$.

\begin{theorem}[Properties of $\Phi$]\label{thm:Phi}
The map $\Phi\colon\mathcal{B}(\mathbb{R})\to\mathcal{H}$ defined by $\Phi(A):=V^{-1}(\mathbf{1}_A)$ satisfies
\begin{enumerate}
\item[\textup{(i)}] $\langle\Phi(A),\Phi(B)\rangle_{\mathcal{H}}=\nu(A\cap B)$ for all $A,B\in\mathcal{B}(\mathbb{R})$ (orthogonal intensity);
\item[\textup{(ii)}] $\Phi$ is finitely additive: $\Phi(A\sqcup B)=\Phi(A)+\Phi(B)$;
\item[\textup{(iii)}] $\Phi$ is countably additive in the norm of $\mathcal{H}$: if $A=\bigsqcup_{n\ge 1}A_n$ is a Borel partition, then $\sum_{n=1}^{N}\Phi(A_n)\to\Phi(A)$ in $\mathcal{H}$ as $N\to\infty$;
\item[\textup{(iv)}] $\Phi(\mathbb{R})=\Phi(\{-1\})+\Phi(\{+1\})$, with $\|\Phi(\{-1\})\|=\|\Phi(\{+1\})\|=1$ and $\langle\Phi(\{-1\}),\Phi(\{+1\})\rangle_{\mathcal{H}}=0$.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) By the polarisation of \eqref{eq:Phi-norm},
\[
\langle\Phi(A),\Phi(B)\rangle_{\mathcal{H}}=\langle V^{-1}\mathbf{1}_A,V^{-1}\mathbf{1}_B\rangle_{\mathcal{H}}=\langle\mathbf{1}_A,\mathbf{1}_B\rangle_{L^2(\nu)}=\int_{\mathbb{R}}\mathbf{1}_A\mathbf{1}_B\,d\nu=\nu(A\cap B).
\]
(ii) For disjoint Borel $A,B$, $\mathbf{1}_{A\sqcup B}=\mathbf{1}_A+\mathbf{1}_B$, so applying $V^{-1}$ (which is linear) gives $\Phi(A\sqcup B)=\Phi(A)+\Phi(B)$. 

(iii) Set $S_N:=\sum_{n=1}^N\mathbf{1}_{A_n}=\mathbf{1}_{A_1\sqcup\cdots\sqcup A_N}\uparrow\mathbf{1}_A$ monotonically. Since $|\mathbf{1}_A-S_N|\le 1\in L^2(\nu)$ (a constant function is in $L^2(\nu)$ because $\nu$ is finite) and $\mathbf{1}_A-S_N\to 0$ pointwise, dominated convergence on $L^2(\nu)$ gives $\|\mathbf{1}_A-S_N\|_{L^2(\nu)}\to 0$. Applying the isometry $V^{-1}$,
\[
\Bigl\|\Phi(A)-\sum_{n=1}^N\Phi(A_n)\Bigr\|_{\mathcal{H}}=\|\mathbf{1}_A-S_N\|_{L^2(\nu)}\xrightarrow{N\to\infty}0.
\]

(iv) Since $\nu=\delta_{-1}+\delta_{+1}$ is supported on $\{-1,+1\}$ and $\mathbb{R}=\{-1\}\sqcup\{+1\}\sqcup(\mathbb{R}\setminus\{-1,+1\})$ with $\nu(\mathbb{R}\setminus\{-1,+1\})=0$, part (iii) together with \eqref{eq:Phi-norm} (which forces $\Phi(\mathbb{R}\setminus\{-1,+1\})=0$) yields $\Phi(\mathbb{R})=\Phi(\{-1\})+\Phi(\{+1\})$. The norms $\|\Phi(\{\pm1\})\|_{\mathcal{H}}^2=\nu(\{\pm1\})=1$ give unit length, and orthogonality $\langle\Phi(\{-1\}),\Phi(\{+1\})\rangle=\nu(\{-1\}\cap\{+1\})=\nu(\emptyset)=0$ comes from (i).
\end{proof}

\section{The spectral representation of $H$}\label{sec:rep}

\begin{definition}[Stochastic integral against $\Phi$]\label{def:stoch-int}
For $g\in L^2(\nu)$, define
\[
  I(g)\;:=\;V^{-1}(g)\;\in\;\mathcal{H}_H.
\]
We write this equivalently as
\[
  I(g)\;=\;\int_{\mathbb{R}} g(\xi)\,d\Phi(\xi).
\]
\end{definition}

The notation is justified by: for any simple function $g=\sum_j c_j\mathbf{1}_{A_j}\in L^2(\nu)$ with disjoint $A_j\in\mathcal{B}(\mathbb{R})$, linearity of $V^{-1}$ gives
\[
  I(g)\;=\;\sum_j c_j\Phi(A_j),
\]
which is the classical definition of the integral of a simple function against a vector-valued orthogonally-scattered measure; and simple functions are dense in $L^2(\nu)$, so $I$ as defined extends the simple-function integral by continuity, agreeing with the general definition of a Dunford--Schwartz/Riesz integral against an orthogonal measure.

\begin{theorem}[Spectral representation of $H$]\label{thm:spec-rep}
For every $x\in\mathbb{R}$,
\begin{equation}\label{eq:spec-rep}
  H_x\;=\;\int_{\mathbb{R}} e^{i\xi x}\,d\Phi(\xi)\qquad\text{in }\mathcal{H}.
\end{equation}
In particular, the function $\xi\mapsto e^{i\xi x}$ belongs to $L^2(\nu)$ for each $x\in\mathbb{R}$ (it is bounded, $\nu$ is finite), and the identity \eqref{eq:spec-rep} holds in the norm of $\mathcal{H}$.
\end{theorem}

\begin{proof}
By definition of $V$ (Theorem~\ref{thm:iso}), $V(H_x)(\xi)=e^{i\xi x}$, i.e.\ $V(H_x)=e^{i\xi x}$ as an element of $L^2(\nu)$. Applying $V^{-1}$ to both sides and using Definition~\ref{def:stoch-int},
\[
  H_x\;=\;V^{-1}(e^{i\xi x})\;=\;I(e^{i\xi x})\;=\;\int_{\mathbb{R}}e^{i\xi x}\,d\Phi(\xi).
\]
\end{proof}

\begin{corollary}[Explicit two-atom form]\label{cor:two-atom}
Writing $\Phi_{\pm}:=\Phi(\{\pm1\})$, the spectral representation \eqref{eq:spec-rep} reduces to
\begin{equation}\label{eq:two-atom}
  H_x\;=\;e^{-ix}\,\Phi_{-}\;+\;e^{+ix}\,\Phi_{+}\qquad\text{in }\mathcal{H},
\end{equation}
where $\{\Phi_{-},\Phi_{+}\}$ is orthonormal in $\mathcal{H}$.
\end{corollary}

\begin{proof}
Since $\nu=\delta_{-1}+\delta_{+1}$, for any $g\in L^2(\nu)$ the integral $\int g\,d\nu=g(-1)+g(+1)$, and the corresponding simple function representation of $g$ in $L^2(\nu)$ is $g=g(-1)\mathbf{1}_{\{-1\}}+g(+1)\mathbf{1}_{\{+1\}}$ (they agree $\nu$-a.e.). Applying $V^{-1}$,
\[
  I(g)=g(-1)\Phi_{-}+g(+1)\Phi_{+}.
\]
Taking $g(\xi)=e^{i\xi x}$ gives \eqref{eq:two-atom}. Orthonormality of $\{\Phi_{-},\Phi_{+}\}$ is Theorem~\ref{thm:Phi}(iv).
\end{proof}

\begin{corollary}[Reality constraint]\label{cor:reality}
For every $s\in\mathbb{R}$,
\[
\langle H_s,\Phi_{-}\rangle_{\mathcal{H}}\;=\;e^{-is},\qquad
\langle H_s,\Phi_{+}\rangle_{\mathcal{H}}\;=\;e^{+is},
\]
and in particular
\[
\langle H_s,\Phi_{-}\rangle_{\mathcal{H}}\;=\;\overline{\langle H_s,\Phi_{+}\rangle_{\mathcal{H}}}.
\]
\end{corollary}

\begin{proof}
Using the isometry $V$ of Theorem~\ref{thm:iso} and the definition $\Phi_{\pm}=V^{-1}(\mathbf{1}_{\{\pm 1\}})$,
\[
\langle H_s,\Phi_{-}\rangle_{\mathcal{H}}\;=\;\langle V H_s,V\Phi_{-}\rangle_{L^2(\nu)}\;=\;\langle e^{i\xi s},\mathbf{1}_{\{-1\}}\rangle_{L^2(\nu)}\;=\;\int_{\mathbb{R}}e^{i\xi s}\mathbf{1}_{\{-1\}}(\xi)\,d\nu(\xi)\;=\;e^{-is}.
\]
Similarly $\langle H_s,\Phi_{+}\rangle_{\mathcal{H}}=e^{+is}$, and $\overline{e^{+is}}=e^{-is}$.
\end{proof}

\section{The spectral representation of the Hardy \texorpdfstring{$Z$}{Z}-function}

Recall $Z(t)=\sqrt{\theta'(t)}\,H(\theta(t))$ on $[T_0,\infty)$. Under the coordinate change $x=\theta(t)$, the shift $T_s$ on the $x$-axis becomes a pullback along $\theta$; in $\mathcal{H}$ this is expressed simply as substitution.

\begin{theorem}[Spectral representation of the Hardy $Z$-function]\label{thm:Z-rep}
For every $t\ge T_0$, define $Z_t:=\sqrt{\theta'(t)}\cdot H_{\theta(t)}\in\mathcal{H}$, where $H_{\theta(t)}:=V^{-1}(e^{i\xi\theta(t)})$. Then
\begin{equation}\label{eq:Z-rep}
  Z_t\;=\;\sqrt{\theta'(t)}\int_{\mathbb{R}} e^{i\xi\theta(t)}\,d\Phi(\xi)\;=\;\sqrt{\theta'(t)}\bigl(e^{-i\theta(t)}\Phi_{-}+e^{+i\theta(t)}\Phi_{+}\bigr)\qquad\text{in }\mathcal{H}.
\end{equation}
Moreover, the limit covariance $\mathcal{C}_Z(t,s):=\langle Z_t,Z_s\rangle_{\mathcal{H}}$ is
\[
  \mathcal{C}_Z(t,s)\;=\;\sqrt{\theta'(t)\theta'(s)}\cdot C(\theta(t)-\theta(s))\;=\;2\sqrt{\theta'(t)\theta'(s)}\cos(\theta(t)-\theta(s)).
\]
\end{theorem}

\begin{proof}
The defining identity $Z_t=\sqrt{\theta'(t)}H_{\theta(t)}$ is an equation in $\mathcal{H}$ by scalar multiplication and the element $H_{\theta(t)}=V^{-1}(e^{i\xi\theta(t)})\in\mathcal{H}_H$ is well-defined because $\theta(t)\in\mathbb{R}$ for every $t$ and the exponential is in $L^2(\nu)$. Substituting Theorem~\ref{thm:spec-rep} applied at $x=\theta(t)$ and Corollary~\ref{cor:two-atom} yields the two forms in \eqref{eq:Z-rep}.

For the covariance, use bilinearity:
\begin{align*}
\mathcal{C}_Z(t,s)&=\langle \sqrt{\theta'(t)}H_{\theta(t)},\sqrt{\theta'(s)}H_{\theta(s)}\rangle_{\mathcal{H}}\\
&=\sqrt{\theta'(t)\theta'(s)}\,\langle H_{\theta(t)},H_{\theta(s)}\rangle_{\mathcal{H}}\\
&=\sqrt{\theta'(t)\theta'(s)}\,C(\theta(t)-\theta(s))
\end{align*}
by \eqref{eq:cov}, and $C(\theta(t)-\theta(s))=2\cos(\theta(t)-\theta(s))$ by Input~1.
\end{proof}

\begin{corollary}[Diagonal variance]\label{cor:Zvar}
$\langle Z_t,Z_t\rangle_{\mathcal{H}}=2\theta'(t)$ for every $t\ge T_0$.
\end{corollary}

\begin{proof}
Set $s=t$ in Theorem~\ref{thm:Z-rep}; $C(0)=2$, $\theta(t)-\theta(t)=0$.
\end{proof}

\section{Remarks on terminology}

\begin{remark}[The word ``random'']\label{rem:random}
The object $\Phi$ constructed in Definition~\ref{def:Phi} is a finitely-additive, countably-additive-in-norm map from the Borel $\sigma$-algebra of $\mathbb{R}$ into the Hilbert space $\mathcal{H}$, with $\langle\Phi(A),\Phi(B)\rangle_{\mathcal{H}}=\nu(A\cap B)$. This is exactly the definition of a complex orthogonally-scattered vector measure on $\mathcal{B}(\mathbb{R})$ with intensity $\nu$ and values in $\mathcal{H}$, in the sense of Masani--Rosenberg. It is called the \emph{complex orthogonal measure} of the signal $H$.

Every such vector-valued measure is isomorphic, via the Kolmogorov construction, to the complex orthogonal increment measure of a second-order centered process on a probability space $(\Omega,\mathcal{F},\mathbb{P})$ whose covariance is $\nu(A\cap B)$: namely, take $\Omega$ to be a Gaussian measure space with reproducing kernel Hilbert space $\mathcal{H}$. Under that isomorphism $\mathcal{H}\ni f\mapsto \widetilde f\in L^2(\Omega,\mathcal{F},\mathbb{P})$ with $\widetilde f$ a complex Gaussian and $\mathbb{E}[\widetilde f\,\overline{\widetilde g}]=\langle f,g\rangle_{\mathcal{H}}$, the map $\Phi$ becomes a genuine complex Gaussian orthogonal random measure; this is the sense of ``random'' used in the companion literature. The present paper does not require this extra step: the identities \eqref{eq:spec-rep} and \eqref{eq:Z-rep} hold in $\mathcal{H}$ without invoking any probability space.
\end{remark}

\begin{remark}[Two atoms is the content]\label{rem:two-atoms}
The support of $\nu$ is $\{-1,+1\}$ and $\mathcal{H}_H$ is two-dimensional; under $V$, $\mathcal{H}_H\simeq L^2(\nu)\simeq\mathbb{C}^2$ via $f\mapsto(Vf(-1),Vf(+1))$. In particular $\{H_s:s\in\mathbb{R}\}$ spans only a two-dimensional subspace of $\mathcal{H}$: the limit structure of $H$ records only the phase pair $(\pm 1)$. This is consistent with the band-edge calculation of the companion paper \emph{Existence of the Spectral Measure in the Limit}, where the same two atoms appeared as the Bochner measure of $C(\eta)=2\cos|\eta|$.
\end{remark}

\begin{remark}[Relation to the pre-limit harmonizable representation]\label{rem:prelimit}
At finite $X$, the pre-limit measure $\nu_X$ is absolutely continuous with density $|\widehat{H_X}|^2/(X-X_0)$ where $\widehat{H_X}(\xi)=\int_{X_0}^{X}H(x)e^{-i\xi x}\,dx$; see \emph{Existence of the Spectral Measure in the Limit}, Proposition~9. By Theorem~\ref{thm:Phi}, the complex orthogonal measure $\Phi$ constructed here is the limit-in-$\mathcal{H}$ of the pre-limit complex orthogonal measures $\Phi_X$ obtained by applying the same Kolmogorov isometry to $\nu_X$ in place of $\nu$: the isometries $V_X\colon\mathcal{H}_{H,X}\to L^2(\nu_X)$ exist at every $X$ (for every positive finite Borel measure $\nu_X$ there is such an isometry sending $H_s\mapsto e^{i\xi s}$), and under the distributional convergence $\nu_X\to\nu$ in $\mathcal{S}'(\mathbb{R})$ their transported images in a common Hilbert space converge to $\Phi$. The two-atom limit $\nu=\delta_{-1}+\delta_{+1}$ then concentrates the pre-limit continuous densities onto the atoms $\Phi_{\pm}$.
\end{remark}

\section*{Summary of proved statements}

\begin{itemize}
\item \textbf{Hilbert space exists} (Definitions~\ref{def:inner}--\ref{def:H}, Lemmas~\ref{lem:time-avg}--\ref{lem:psd}): the shifts $T_sH$ of the warped Hardy signal lie in a Hilbert space $\mathcal{H}$ where the inner product is the time-average covariance $\langle H_s,H_t\rangle=C(s-t)=2\cos(s-t)$, with $C$ from Input~1.
\item \textbf{Complex orthogonal measure exists} (Theorem~\ref{thm:iso}, Definition~\ref{def:Phi}, Theorem~\ref{thm:Phi}): there is a unique Hilbert-space-valued orthogonally-scattered measure $\Phi$ on $\mathcal{B}(\mathbb{R})$ with intensity $\nu=\delta_{-1}+\delta_{+1}$, supported on $\{-1,+1\}$, whose two atoms $\Phi_{\pm}$ are orthonormal in $\mathcal{H}$.
\item \textbf{Spectral representation of $H$} (Theorem~\ref{thm:spec-rep}, Corollary~\ref{cor:two-atom}): $H_x=\int e^{i\xi x}d\Phi(\xi)=e^{-ix}\Phi_{-}+e^{+ix}\Phi_{+}$ in $\mathcal{H}$.
\item \textbf{Spectral representation of $Z$} (Theorem~\ref{thm:Z-rep}): $Z_t=\sqrt{\theta'(t)}\int e^{i\xi\theta(t)}d\Phi(\xi)=\sqrt{\theta'(t)}\bigl(e^{-i\theta(t)}\Phi_{-}+e^{+i\theta(t)}\Phi_{+}\bigr)$ in $\mathcal{H}$, with covariance $\langle Z_t,Z_s\rangle_{\mathcal{H}}=2\sqrt{\theta'(t)\theta'(s)}\cos(\theta(t)-\theta(s))$.
\end{itemize}
