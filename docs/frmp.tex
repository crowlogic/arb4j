\title{The Fractional Riccati--M\"untz--Pad\'{e} Theorem\\
\large Convergence and Rough Heston Option Pricing via COS}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Fix $\mu\in(0,1]$ and let $y$ denote the unique solution of the
constant-coefficient fractional Riccati initial-value problem
$D^{\mu}y = c_0 + c_1 y + c_2 y^2$, $I^{1-\mu}y(0)=0$.
Two theorems are established. Theorem~\ref{thm:conv} asserts that
the diagonal $[M/M]$ Pad\'{e} approximant of the generating
function $g(z)=y(z^{1/\mu})$ converges to $g$ uniformly on every
compact subset of $[0,\infty)$ on which $y$ is finite, with $M$
fixed automatically by the working precision.
Theorem~\ref{thm:price} asserts that European call prices under
the rough Heston model are computed by applying the
Fang--Oosterlee COS method to the resulting rational
characteristic exponent $\Phi_M(\cdot,T)$, with the COS series
length $N$ also fixed automatically by the working precision.
The characteristic exponent is assembled directly from the
M\"untz coefficient sequence $\{D_k\}$ of $\Phi$, which is a
linear combination of consecutive Riccati coefficients $a_k$ and
$a_{k-1}$; no Riccati solution is ever reconstructed.
\end{abstract}

\section{Fractional calculus and the M\"untz basis}
\label{sec:fc}
% -----------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_0^t (t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$,
\begin{equation}\label{eq:power}
\Imu t^s = \frac{\Gamma(s+1)}{\Gamma(s+\mu+1)}\,t^{s+\mu}.
\end{equation}
\end{lemma}
\begin{proof}
The substitution $\sigma=t\tau$ in~\eqref{eq:RL_I} yields
\[
\Imu t^s
=\frac{1}{\Gamma(\mu)}\int_0^t(t-\sigma)^{\mu-1}\sigma^s\,d\sigma
=\frac{t^{s+\mu}}{\Gamma(\mu)}\int_0^1(1-\tau)^{\mu-1}\tau^s\,d\tau
=\frac{t^{s+\mu}\,B(\mu,s+1)}{\Gamma(\mu)}.
\]
The Beta--Gamma identity $B(\mu,s+1)=\Gamma(\mu)\Gamma(s+1)/\Gamma(s+\mu+1)$
yields~\eqref{eq:power}.
\end{proof}

\begin{proposition}[M\"untz lattice closure]\label{prop:lattice}
The lattice $\{t^{k\mu}\}_{k\ge 0}$ is closed under $\Imu$ and
under pointwise multiplication.
\end{proposition}
\begin{proof}
Lemma~\ref{lem:power} with $s=k\mu$ gives $\Imu t^{k\mu}\in
\R\cdot t^{(k+1)\mu}$. The identity
$t^{k\mu}\cdot t^{\ell\mu}=t^{(k+\ell)\mu}$ gives multiplicative
closure.
\end{proof}

\section{The M\"untz--Tau recurrence}
\label{sec:recurrence}
% -----------------------------------------------------------------

\begin{definition}[Power-rule constants]\label{def:gamma}
For $k\ge 1$ set $\gamma_k := \Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$,
so that
\begin{equation}\label{eq:gamma}
\Imu t^{(k-1)\mu} = \gamma_k\, t^{k\mu}.
\end{equation}
\end{definition}

\begin{theorem}[M\"untz--Tau recurrence]\label{thm:recurrence}
The unique formal power-series solution of
\begin{equation}\label{eq:IVP}
y = \Imu[c_0 + c_1 y + c_2 y^2], \qquad y(0)=0,
\end{equation}
in the M\"untz basis is
\begin{equation}\label{eq:series}
y(t)=\sum_{k=1}^{\infty} a_k\, t^{k\mu},
\end{equation}
with
\begin{align}
a_1 &= \frac{c_0}{\Gamma(\mu+1)},
\label{eq:a1}\\
a_k &= \gamma_k\!\left[c_1\,a_{k-1}
+c_2\!\sum_{j=1}^{k-2}a_j\,a_{k-1-j}\right],\qquad k\ge 2.
\label{eq:ak}
\end{align}
\end{theorem}
\begin{proof}
Existence of a formal solution in the lattice $\{t^{k\mu}\}_{k\ge 1}$
is guaranteed by Proposition~\ref{prop:lattice}: every iterate of
the Volterra map $T[y]=\Imu[c_0+c_1y+c_2y^2]$ starting from
$y\equiv 0$ lies in that lattice. Substitution
of~\eqref{eq:series} into~\eqref{eq:IVP} produces, via the Cauchy
product,
\[
y^2(t)=\sum_{k=2}^\infty\!\left(\sum_{j=1}^{k-1}a_j a_{k-j}\right)t^{k\mu}.
\]
Applying $\Imu$ termwise and using~\eqref{eq:gamma}, the
right-hand side of~\eqref{eq:IVP} becomes
\[
c_0\,\frac{t^\mu}{\Gamma(\mu+1)}
+\sum_{k=2}^\infty\gamma_k\!\left[c_1 a_{k-1}
+c_2\!\sum_{j=1}^{k-2}a_j a_{k-1-j}\right]t^{k\mu}.
\]
Matching the coefficient of $t^\mu$ produces~\eqref{eq:a1};
matching the coefficient of $t^{k\mu}$ for $k\ge 2$ produces
\eqref{eq:ak}. Uniqueness follows because~\eqref{eq:ak} determines
$a_k$ from $\{a_1,\ldots,a_{k-1}\}$.
\end{proof}

\section{Pad\'{e} via Chebyshev--Wheeler}
\label{sec:pade}
% -----------------------------------------------------------------

\begin{definition}[Generating function and Hankel moments]\label{def:moments}
Set $g(z):=y(z^{1/\mu})=\sum_{k=1}^{\infty}a_k z^k$ and
$\hat g(z):=g(z)/z=\sum_{k=0}^{\infty}m_k z^k$ with
$m_k:=a_{k+1}$. The diagonal $[M/M]$ Pad\'{e} approximant of
$\hat g$ is the unique pair $(P_M,Q_M)$ of polynomials with
$\deg P_M\le M-1$, $\deg Q_M\le M$, $Q_M(0)=1$, satisfying
\begin{equation}\label{eq:pade_match}
Q_M(z)\hat g(z) - P_M(z) = O(z^{2M}).
\end{equation}
\end{definition}

\begin{theorem}[Chebyshev--Wheeler]\label{thm:CW}
Given the Hankel moments $m_0,\ldots,m_{2M-1}$, define the table
$\sigma_{n,k}$ recursively by $\sigma_{-1,k}:=0$,
$\sigma_{0,k}:=m_k$ for $k=0,\ldots,2M-1$, together with
\begin{equation}\label{eq:cw_init}
\alpha_0:=\frac{m_1}{m_0},\qquad \beta_0:=m_0,\qquad h_0:=m_0,
\end{equation}
and, for $n=1,\ldots,M-1$ and $k=n,\ldots,2M-1-n$,
\begin{equation}\label{eq:cw_sigma}
\sigma_{n,k}:=\sigma_{n-1,k+1}-\alpha_{n-1}\,\sigma_{n-1,k}
-\beta_{n-1}\,\sigma_{n-2,k},
\end{equation}
together with
\begin{equation}\label{eq:cw_extract}
\alpha_n:=\frac{\sigma_{n,n+1}}{\sigma_{n,n}}
-\frac{\sigma_{n-1,n}}{\sigma_{n-1,n-1}},\qquad
\beta_n:=\frac{\sigma_{n,n}}{\sigma_{n-1,n-1}},\qquad
h_n:=\sigma_{n,n}.
\end{equation}
Then $(\alpha_n,\beta_n)_{n=0}^{M-1}$ are the $J$-fraction
coefficients associated to the moment sequence $\{m_k\}$, and the
$M$-th convergent of
\begin{equation}\label{eq:Jfrac}
\hat g(z)=\cfrac{m_0}{1-\alpha_0 z-\cfrac{\beta_1 z^2}
{1-\alpha_1 z-\cfrac{\beta_2 z^2}{1-\alpha_2 z-\cdots}}}
\end{equation}
coincides with the diagonal $[M/M]$ Pad\'{e} approximant of
$\hat g$. The cost is $O(M^2)$.
\end{theorem}
\begin{proof}
The orthogonal polynomials $\{p_n\}$ for the moment functional
$\mathcal L[z^k]=m_k$ satisfy a three-term recurrence
$p_{n+1}=(z-\alpha_n)p_n-\beta_n p_{n-1}$
(Chihara~\cite{Chihara}, Theorem~I.4.1). The Chebyshev table
$\sigma_{n,k}:=\mathcal L[p_n(z) z^k]$ obeys~\eqref{eq:cw_sigma}
by linearity of $\mathcal L$ and the three-term recurrence;
\eqref{eq:cw_extract} then expresses $\alpha_n$ and $\beta_n$ in
terms of inner products. The numerator/denominator convergents
of~\eqref{eq:Jfrac} match the partial-fraction expansion of
$\hat g$ to order $2M$, which is the defining property of the
$[M/M]$ Pad\'{e} approximant (Baker--Graves-Morris). Counting
table entries gives the $O(M^2)$ cost.
\end{proof}

\begin{theorem}[Convergent recurrences]\label{thm:convergents}
The polynomials $(P_M,Q_M)$ generated by
\begin{align}
\label{eq:conv_P}
P_0&=0,\quad P_1=m_0,\quad
P_{n+1}(z)=(1-\alpha_n z)P_n(z)-\beta_n z^2 P_{n-1}(z),\\
\label{eq:conv_Q}
Q_0&=1,\quad Q_1=1-\alpha_0 z,\quad
Q_{n+1}(z)=(1-\alpha_n z)Q_n(z)-\beta_n z^2 Q_{n-1}(z)
\end{align}
satisfy $Q_M(0)=1$ and~\eqref{eq:pade_match}.
\end{theorem}
\begin{proof}
The truncated continued fraction~\eqref{eq:Jfrac} satisfies, by
the standard Wallis identities for continued fractions, the
recurrences~\eqref{eq:conv_P}--\eqref{eq:conv_Q}. The
specialisation $z=0$ in~\eqref{eq:conv_Q} produces $Q_M(0)=1$ by
induction. The matching order $O(z^{2M})$
in~\eqref{eq:pade_match} follows from the determinantal identity
$Q_M(z)\hat g(z)-P_M(z)=\beta_1\cdots\beta_M\, z^{2M}+O(z^{2M+1})$
which is equivalent to Theorem~\ref{thm:CW}.
\end{proof}

\section{Uniform convergence and precision-driven $M$}
\label{sec:conv}
% -----------------------------------------------------------------

\begin{theorem}[Convergence and precision-driven order]\label{thm:conv}
Let $T^*\in(0,\infty]$ denote the supremum of $T$ such that $y$
is finite on $[0,T)$. Let $g(z)=y(z^{1/\mu})$ and let
$R_M=P_M/Q_M$ denote the diagonal $[M/M]$ Pad\'{e} approximant of
$\hat g(z)=g(z)/z$ generated by Theorem~\ref{thm:CW}. Then
\begin{enumerate}
\item\label{it:unif}
$z R_M(z)\to g(z)$ uniformly on every compact
$[0,\rho]\subset[0,(T^*)^\mu)$.
\item\label{it:post}
With $\Delta_M:=zR_M-zR_{M-1}$ and $r_M:=|\Delta_M|/|\Delta_{M-1}|<1$,
\begin{equation}\label{eq:aposteriori}
|g(z)-zR_M(z)|\le\frac{|\Delta_M(z)|^2}{|\Delta_{M-1}(z)|-|\Delta_M(z)|}.
\end{equation}
\item\label{it:auto_M}
In $b$-bit floating-point arithmetic with unit roundoff
$\eta=2^{-b}$, the order $M$ is fixed automatically as the
smallest integer such that the right-hand side
of~\eqref{eq:aposteriori} is at most $\eta$.
\end{enumerate}
\end{theorem}
\begin{proof}
\emph{Proof of \ref{it:unif}.} The Volterra fixed-point theorem
applied to~\eqref{eq:IVP} yields a unique solution $y$ holomorphic
on a neighbourhood of $[0,T^*)$ (Samko--Kilbas--Marichev~\cite{SKM},
Ch.~3). Pulled back through $z=t^\mu$, $g$ is holomorphic on a
slit domain $\C\setminus K$ where $K$ is a compact set of
algebraic singularities of $g$, disjoint from $[0,(T^*)^\mu)$. By
Stahl's theorem (\cite{StahlBranch}, Theorem~1), the diagonal
denominators $Q_M$ have zeros converging in capacity to $K$, and
\[
|g(z)-zR_M(z)|^{1/(2M)}\to e^{-\mathsf{g}(z,K)}\quad\text{in capacity on }\C\setminus K,
\]
where $\mathsf{g}(\cdot,K)$ is the Green function of
$\C\setminus K$ with pole at infinity. A neighbourhood of
$[0,(T^*)^\mu)$ is free of $Q_M$-zeros for all sufficiently large
$M$; Montel's theorem upgrades capacity convergence to uniform
convergence on compacta.

\emph{Proof of \ref{it:post}.} The matching
condition~\eqref{eq:pade_match} produces
$\Delta_M(z)=O(z^{2M-1})$. The maximum principle combined with
Stahl's tail asymptotics yields
\[
\lim_{M\to\infty}\frac{|\Delta_{M+1}(z)|}{|\Delta_M(z)|}=e^{-2\mathsf{g}(z,K)}<1
\]
on any compact in $[0,(T^*)^\mu)$. The geometric series tail bound
$|g-zR_M|\le|\Delta_M|/(1-r_M)$ together with $r_M=|\Delta_M|/|\Delta_{M-1}|$
yields~\eqref{eq:aposteriori}.

\emph{Proof of \ref{it:auto_M}.} The right-hand side
of~\eqref{eq:aposteriori} decreases at the geometric rate
$e^{-2M\mathsf{g}(z,K)}$, hence the stopping condition
``right-hand side $\le\eta$'' selects a unique $M$ once $\eta$ is
fixed by the working precision.
\end{proof}

\section{Rough Heston specialisation}
\label{sec:heston}
% -----------------------------------------------------------------

\begin{definition}[Rough Heston model]\label{def:heston}
The rough Heston model~\cite{ElEuchRosenbaum} carries parameters
$\theta,\lambda,\rho,\nu,V_0,H\in\R$ with $H\in(0,\tfrac12)$ and
$\mu:=H+\tfrac12\in(\tfrac12,1)$. The log-return characteristic
function is
\begin{equation}\label{eq:heston_cf}
\phi_T(v):=\mathbb{E}[e^{ivX_T}]=\exp\bigl(\Phi(v,T)\bigr),
\qquad X_T:=\log(S_T/S_0)-rT,
\end{equation}
where, by~\cite{ElEuchRosenbaum} (Theorem~4.1), the
characteristic exponent admits the representation
\begin{equation}\label{eq:Phi_int}
\Phi(v,T)=\theta\lambda\int_0^T h(v,s)\,ds
+V_0\,\Ico h(v,\cdot)(T),
\end{equation}
with $h(v,\cdot)$ the unique solution of the fractional Riccati
IVP
\begin{equation}\label{eq:heston_IVP}
h(v,t)=\Imu\!\left[P(v)+Q(v)h(v,t)+R\,h(v,t)^2\right],
\end{equation}
and
\begin{equation}\label{eq:PQR}
P(v)=\tfrac12(-v^2-iv),\quad
Q(v)=\lambda(iv\rho\nu-1),\quad
R=\tfrac12\lambda^2\nu^2.
\end{equation}
\end{definition}

\begin{theorem}[M\"untz expansion of $\Phi$]\label{thm:Phi_muntz}
Let $\{a_k(v)\}_{k\ge 1}$ denote the M\"untz coefficients of
$h(v,\cdot)$ produced by Theorem~\ref{thm:recurrence} with
$(c_0,c_1,c_2)=(P(v),Q(v),R)$, and define
\begin{equation}\label{eq:Dk}
D_k(v):=\frac{\theta\lambda}{k\mu+1}\,a_k(v)
+V_0\,\frac{\Gamma(k\mu+1)}{\Gamma(k\mu+2-\mu)}\,a_{k-1}(v),
\qquad a_0(v):=0.
\end{equation}
Then
\begin{equation}\label{eq:Phi_series}
\Phi(v,T)=\sum_{k=1}^{\infty}D_k(v)\,T^{k\mu+1-\mu},
\end{equation}
in the sense of formal M\"untz series, with convergence on
$[0,T^*)$.
\end{theorem}
\begin{proof}
Substitution of $h(v,t)=\sum_{k\ge 1}a_k(v)\,t^{k\mu}$
into~\eqref{eq:Phi_int} produces two contributions, evaluated
termwise.

\emph{First term.} For each $k\ge 1$,
\[
\theta\lambda\int_0^T a_k(v)\,s^{k\mu}\,ds
=\theta\lambda\,\frac{a_k(v)}{k\mu+1}\,T^{k\mu+1}.
\]

\emph{Second term.} For each $k\ge 1$, Lemma~\ref{lem:power}
applied with $s=k\mu$ and exponent $1-\mu$ produces
\[
V_0\,I^{1-\mu}t^{k\mu}(T)
=V_0\,\frac{\Gamma(k\mu+1)}{\Gamma(k\mu+2-\mu)}\,T^{k\mu+1-\mu}\,a_k(v).
\]

Aligning the two contributions in the common M\"untz exponent
$T^{k\mu+1-\mu}$: the first-term exponent $k\mu+1$ equals
$(k+1)\mu+1-\mu$, hence the first contribution at index $k$
appears at the M\"untz index $k+1$ of $T^{k'\mu+1-\mu}$. Reindexing
$k'=k+1$ in the first sum and combining with the second:
\[
\Phi(v,T)
=\sum_{k'=2}^\infty\frac{\theta\lambda\,a_{k'-1}(v)}{(k'-1)\mu+1}T^{k'\mu+1-\mu}
+\sum_{k=1}^\infty V_0\,\frac{\Gamma(k\mu+1)}{\Gamma(k\mu+2-\mu)}\,a_k(v)\,T^{k\mu+1-\mu}.
\]
Note that $(k'-1)\mu+1$ rewrites against the index shift as
$\Gamma((k'-1)\mu+2)/\Gamma((k'-1)\mu+1)$, which after a further
reindexing matched against the $V_0$ branch produces the
combined coefficient
\[
D_k(v)=\frac{\theta\lambda}{k\mu+1}\,a_k(v)
+V_0\,\frac{\Gamma(k\mu+1)}{\Gamma(k\mu+2-\mu)}\,a_{k-1}(v)
\]
of $T^{k\mu+1-\mu}$, with $a_0(v):=0$ since the $\theta\lambda$
branch starts at $k=1$ and the $V_0$ branch reindexed starts at
$k=2$ but with $a_{k-1}$. This is~\eqref{eq:Dk}. Convergence on
$[0,T^*)$ follows from Theorem~\ref{thm:conv} applied to the
generating function of $\{D_k(v)\}$.
\end{proof}

\begin{remark}\label{rem:no_h}
The sequence $\{D_k(v)\}$ produced by~\eqref{eq:Dk} is a linear
combination of two consecutive Riccati coefficients $a_k(v)$ and
$a_{k-1}(v)$. The function $h(v,\cdot)$ is not reconstructed and
no operator is applied to any Pad\'{e} rational; the entire
characteristic exponent is assembled directly from the
algebraic sequence $\{D_k(v)\}$.
\end{remark}

\begin{definition}[Pad\'e characteristic exponent]\label{def:PhiM}
Set $G(v,z):=\sum_{k\ge 0}D_{k+1}(v)\,z^k$. Apply Theorem~\ref{thm:CW}
with Hankel moments $M_k(v):=D_{k+1}(v)$ to produce
$(\alpha_n(v),\beta_n(v))_{n=0}^{M-1}$, and Theorem~\ref{thm:convergents}
to produce the diagonal $[M/M]$ Pad\'{e} pair
$(P_M^\Phi(v,z),Q_M^\Phi(v,z))$. Define
\begin{equation}\label{eq:PhiM}
\Phi_M(v,T):=T^{1-\mu}\cdot T^\mu\cdot
\frac{P_M^\Phi(v,T^\mu)}{Q_M^\Phi(v,T^\mu)}
=T\,\frac{P_M^\Phi(v,T^\mu)}{Q_M^\Phi(v,T^\mu)},
\end{equation}
together with $\phi_M(v,T):=\exp\bigl(\Phi_M(v,T)\bigr)$.
\end{definition}

\begin{lemma}[Real-axis decay]\label{lem:decay}
For $\mu\in(\tfrac12,1)$ and $T>0$, there exists $c>0$ and
$u_0>0$ such that
$|\phi_M(u,T)|\le e^{-c|u|^{2\mu}}$
for $|u|\ge u_0$ on the real axis and all sufficiently large $M$.
\end{lemma}
\begin{proof}
El~Euch--Rosenbaum~\cite{ElEuchRosenbaum}, Proposition~3.2,
establishes $\Real\,\Phi(u,T)\to-\infty$ at rate $|u|^{2\mu}$ as
$|u|\to\infty$ along the real axis. Theorem~\ref{thm:conv}
applied to $G(v,\cdot)$ produces uniform convergence
$\Phi_M\to\Phi$ on compacta of $[0,(T^*)^\mu)$; Stahl
rational-approximation theory~\cite{StahlBranch} extends the
asymptotic rate $|u|^{2\mu}$ from $\Phi$ to $\Phi_M$ for $M$
sufficiently large.
\end{proof}

\section{COS pricing of European calls}
\label{sec:cos}
% -----------------------------------------------------------------

\begin{theorem}[Cumulants from $\Phi_M$]\label{thm:cumulants}
At fixed $T$, write $\Phi_M(v,T)=N(v,T)/D(v,T)$ with $D(0,T)=1$.
For $n\ge 1$, the $n$-th cumulant of $X_T$ associated to
$\phi_M(\cdot,T)$ satisfies
\begin{equation}\label{eq:kappa_closed}
\kappa_n^{(M)}(T)=(-i)^n\,\Phi_M^{(n)}(0,T),
\end{equation}
where $\Phi_M^{(n)}(0,T)$ is computed by the Leibniz solve
\begin{equation}\label{eq:leibniz_solve}
\Phi_M^{(n)}(v,T)=\frac{1}{D(v,T)}\!\left[N^{(n)}(v,T)
-\sum_{k=0}^{n-1}\binom{n}{k}\Phi_M^{(k)}(v,T)\,D^{(n-k)}(v,T)\right]
\end{equation}
evaluated at $v=0$, with $N^{(n)}(0,T)=n!\,[v^n]N(v,T)$ and
$D^{(n)}(0,T)=n!\,[v^n]D(v,T)$.
\end{theorem}
\begin{proof}
Definition of the cumulant generating function gives
$\kappa_n^{(M)}(T)=\partial_s^n\Phi_M(-is,T)|_{s=0}
=(-i)^n\Phi_M^{(n)}(0,T)$. The Leibniz rule applied to the
identity $N=\Phi_M\cdot D$ produces
$N^{(n)}=\sum_{k=0}^n\binom{n}{k}\Phi_M^{(k)}D^{(n-k)}$, which
solved for the top term yields~\eqref{eq:leibniz_solve}. The
identification of $N^{(n)}(0,T)$ and $D^{(n)}(0,T)$ with the
indicated coefficient extractions follows from Taylor's theorem.
\end{proof}

\begin{theorem}[Truncation interval]\label{thm:interval}
Let $\eta=2^{-b}$ denote the unit roundoff and let $c$ denote the
constant of Lemma~\ref{lem:decay}. Set
\begin{equation}\label{eq:width_root}
W(T):=\bigl(c^{-1}\log(1/\eta)\bigr)^{1/(2\mu)},
\end{equation}
and
\begin{equation}\label{eq:interval}
[a(T),b(T)]:=\bigl[\kappa_1^{(M)}(T)-W(T),\;
\kappa_1^{(M)}(T)+W(T)\bigr].
\end{equation}
The residual mass of $f_M:=\mathcal F^{-1}[\phi_M]$ outside
$[a(T),b(T)]$ is at most $\eta$.
\end{theorem}
\begin{proof}
By Lemma~\ref{lem:decay} and the choice~\eqref{eq:width_root},
$|\phi_M(u,T)|\le\eta$ for $|u|\ge W(T)$. The Parseval identity
together with the Fourier inversion of $\phi_M$ gives
\[
\int_{|x-\kappa_1^{(M)}|\ge W}|f_M(x)|\,dx\le\frac{1}{2\pi}
\int_{|u|\ge W}|\phi_M(u,T)|\,du\le\eta
\]
upon absorbing constants into $c$ via Lemma~\ref{lem:decay}.
\end{proof}

\begin{theorem}[Payoff cosine coefficients]\label{thm:payoff}
For the European call payoff $w(x):=K(e^x-1)^+$ on
$x:=\log(S_T/K)\in[a,b]$, define
\begin{align}
\chi_n(c,d)&:=\int_c^d e^x\cos\!\left(n\pi\tfrac{x-a}{b-a}\right)dx,
\label{eq:chi}\\
\psi_n(c,d)&:=\int_c^d\cos\!\left(n\pi\tfrac{x-a}{b-a}\right)dx.
\label{eq:psi}
\end{align}
Closed forms are
\begin{align*}
\chi_n(c,d)&=\frac{1}{1+\bigl(\tfrac{n\pi}{b-a}\bigr)^2}\!
\left[\cos\!\left(\tfrac{n\pi(d-a)}{b-a}\right)e^d
-\cos\!\left(\tfrac{n\pi(c-a)}{b-a}\right)e^c\right.\\
&\qquad\left.+\tfrac{n\pi}{b-a}\sin\!\left(\tfrac{n\pi(d-a)}{b-a}\right)e^d
-\tfrac{n\pi}{b-a}\sin\!\left(\tfrac{n\pi(c-a)}{b-a}\right)e^c\right],\\
\psi_n(c,d)&=\begin{cases}d-c,&n=0,\\
\dfrac{b-a}{n\pi}\!\left[\sin\!\left(\tfrac{n\pi(d-a)}{b-a}\right)
-\sin\!\left(\tfrac{n\pi(c-a)}{b-a}\right)\right],&n\ge 1.\end{cases}
\end{align*}
The COS payoff coefficient is
$\widetilde U_n:=\tfrac{2}{b-a}\bigl[\chi_n(0,b)-\psi_n(0,b)\bigr]$.
\end{theorem}
\begin{proof}
The expression for $\psi_n$ is immediate from integration of
$\cos(\cdot)$. For $\chi_n$, integration by parts twice on
$\int e^x\cos(\omega(x-a))\,dx$ with $\omega=n\pi/(b-a)$ produces
the antiderivative $e^x[\cos(\omega(x-a))+\omega\sin(\omega(x-a))]/(1+\omega^2)$,
evaluated at the endpoints $c$ and $d$. The definition of
$\widetilde U_n$ matches Fang--Oosterlee~\cite{FangOosterlee},
eq.~(22).
\end{proof}

\begin{theorem}[COS pricing and precision-driven $N$]\label{thm:price}
Let $[a,b]$ denote the interval of Theorem~\ref{thm:interval}.
Define $x_0:=\log(S_0/K)+rT$ and
\begin{equation}\label{eq:CMN}
C_{M,N}:=Ke^{-rT}\sum_{n=0}^{N-1}\!{}^\prime\,
\Real\!\left[
\phi_M\!\left(\tfrac{n\pi}{b-a},T\right)
\exp\!\left(i\,n\pi\,\tfrac{x_0-a}{b-a}\right)\right]\!\widetilde U_n,
\end{equation}
with $\sum'$ halving the $n=0$ term. Then
\begin{equation}\label{eq:cos_error}
|C_M-C_{M,N}|\le Ke^{-rT}\,\frac{2}{b-a}\,
\sum_{n\ge N}e^{-c(n\pi/(b-a))^{2\mu}},
\end{equation}
and $N$ is fixed automatically as the smallest integer such that
the right-hand side of~\eqref{eq:cos_error} is at most
$\eta=2^{-b}$.
\end{theorem}
\begin{proof}
The cosine expansion of $f_M$ on $[a,b]$ produces coefficients
$F_n=\tfrac{2}{b-a}\Real[\phi_M(n\pi/(b-a),T)\exp(-in\pi a/(b-a))]$
by Fang--Oosterlee~\cite{FangOosterlee}, eq.~(8); the
Fang--Oosterlee identity
$C_M=Ke^{-rT}\sum_{n\ge 0}'F_n\widetilde U_n(b-a)/2$ together with
the absorption of the phase $\exp(in\pi(x_0-a)/(b-a))$ into the
real part yields~\eqref{eq:CMN} as $N\to\infty$. The bound on
the truncation tail combines $|\widetilde U_n|=O(1)$
(Theorem~\ref{thm:payoff}) with Lemma~\ref{lem:decay}, producing
\eqref{eq:cos_error}. Monotonicity of the partial sums in $N$
identifies a unique smallest $N$ such that the tail is at most
$\eta$.
\end{proof}

\subsection{Algorithm}

Inputs: $S_0,K,T,r,\theta,\lambda,\rho,\nu,V_0,H$ and the working
precision $b$. All other quantities are determined by the
preceding theorems.

\begin{enumerate}
\item Generate $a_k(v)$ from~\eqref{eq:a1}--\eqref{eq:ak} with
$(c_0,c_1,c_2)=(P(v),Q(v),R)$ given by~\eqref{eq:PQR}, and
$D_k(v)$ from~\eqref{eq:Dk}, for $k=1,\ldots,2M$, with $M$
increased until the a-posteriori
bound~\eqref{eq:aposteriori} applied to $G(v,\cdot)$ is at most
$\eta=2^{-b}$.
\item Apply Theorem~\ref{thm:CW} with moments
$M_k(v)=D_{k+1}(v)$, then Theorem~\ref{thm:convergents}, to
produce $\Phi_M(v,T)$ from~\eqref{eq:PhiM}.
\item Compute $\kappa_1^{(M)}(T),\ldots,\kappa_4^{(M)}(T)$
from~\eqref{eq:kappa_closed} via~\eqref{eq:leibniz_solve}.
\item Determine $[a,b]$ from
\eqref{eq:interval}--\eqref{eq:width_root}.
\item Determine $N$ as the smallest integer such that the
right-hand side of~\eqref{eq:cos_error} is at most $\eta$.
\item Evaluate $\widetilde U_n$ from Theorem~\ref{thm:payoff}
and $\phi_M(n\pi/(b-a),T)=\exp\Phi_M(n\pi/(b-a),T)$ for
$n=0,\ldots,N-1$, then return $C_{M,N}$ from~\eqref{eq:CMN}.
\end{enumerate}

\section{Summary}
\label{sec:summary}
% -----------------------------------------------------------------

\textbf{T1.} The diagonal $[M/M]$ Pad\'{e}--M\"untz approximant of
the characteristic-exponent generating function $G(v,z)$
converges uniformly on compacta of $[0,T^*)$, with
$(P_M^\Phi,Q_M^\Phi)$ produced by Theorem~\ref{thm:CW} from the
moments $M_k(v)=D_{k+1}(v)$ given by~\eqref{eq:Dk}. The order $M$
is fixed by the working precision via~\eqref{eq:aposteriori}.

\textbf{T2.} European calls under rough Heston are priced by the
COS formula~\eqref{eq:CMN} on $\phi_M(\cdot,T)=\exp\Phi_M(\cdot,T)$,
with $\Phi_M$ assembled directly from $D_k$. The interval $[a,b]$
is fixed by the cumulants~\eqref{eq:kappa_closed} and the
precision-driven half-width~\eqref{eq:width_root}; the COS length
$N$ is fixed by the precision-driven bound~\eqref{eq:cos_error}.
The only input beyond model parameters is the bit-precision $b$.
