\title{The Fractional Riccati--M\"untz--Pad\'{e} Theorem\\
\large Convergence and Exact Rough Heston Option Pricing}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Let $\mu\in(0,1]$ and let $y$ be the unique solution of the
constant-coefficient fractional Riccati initial-value problem
$D^{\mu}y = c_0 + c_1 y + c_2 y^2$, $I^{1-\mu}y(0)=0$.
We prove two things. First (Theorem~\ref{thm:conv}), the
diagonal $[M/M]$ Pad\'{e} approximant of the generating function
$g(z)=y(z^{1/\mu})$ converges to $g$ uniformly on every compact
subset of $[0,\infty)$ on which $y$ is finite, and an explicit
a-posteriori bound is available at every $M$. Second
(Theorem~\ref{thm:price}), the European call price under the
rough Heston model is a finite, exact sum of closed-form residues
of an exponential-of-partial-fraction characteristic function
derived from the Pad\'{e}--M\"untz solution. The only truncation
parameter is $M$; $M$ is determined by the requested precision.
No quadrature, no FFT, no damping, no contour truncation, no
per-pole series truncation beyond what precision demands.
\end{abstract}

\section{Fractional calculus and the M\"untz basis}
\label{sec:fc}
% -----------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]
\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_0^t (t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$,
\begin{equation}\label{eq:power}
\Imu t^s = \frac{\Gamma(s+1)}{\Gamma(s+\mu+1)}\,t^{s+\mu}.
\end{equation}
\end{lemma}
\begin{proof}
Substitute $\sigma=t\tau$ in~\eqref{eq:RL_I}; the integral becomes
$B(\mu,s+1)=\Gamma(\mu)\Gamma(s+1)/\Gamma(s+\mu+1)$.
\end{proof}

The M\"untz lattice $\{t^{k\mu}\}_{k\ge 0}$ is closed under
$\Imu$ by the power rule, and closed under multiplication because
$t^{k\mu}\cdot t^{\ell\mu}=t^{(k+\ell)\mu}$. In the variable
$z=t^\mu$ every element is simply $z^k$, so ordinary power-series
arithmetic applies.

\section{The M\"untz--Tau recurrence}
\label{sec:recurrence}
% -----------------------------------------------------------------

Define $\gamma_k := \Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$ for $k\ge 1$,
so that by the power rule
\begin{equation}\label{eq:gamma}
\Imu t^{(k-1)\mu} = \gamma_k\, t^{k\mu}.
\end{equation}

\begin{theorem}[M\"untz--Tau recurrence]\label{thm:recurrence}
The unique formal power-series solution of
\begin{equation}\label{eq:IVP}
y = \Imu[c_0 + c_1 y + c_2 y^2], \qquad y(0)=0,
\end{equation}
(the Volterra form of $D^\mu y = c_0+c_1 y+c_2 y^2$,
$\Ico y(0)=0$) is
\begin{equation}\label{eq:series}
y(t)=\sum_{k=1}^{\infty} a_k\, t^{k\mu},
\end{equation}
with coefficients given by
\begin{align}
a_1 &= \frac{c_0}{\Gamma(\mu+1)},
\label{eq:a1}\\
a_k &= \gamma_k\!\left[c_1\,a_{k-1}
 +c_2\!\sum_{j=1}^{k-2}a_j\,a_{k-1-j}\right],
\quad k\ge 2.
\label{eq:ak}
\end{align}
(The convolution sum in~\eqref{eq:ak} is empty for $k=2$, giving
$a_2=\gamma_2 c_1 a_1$; for $k=3$, $a_3=\gamma_3[c_1 a_2+c_2 a_1^2]$.)
\end{theorem}

\begin{proof}
Substitute the ansatz~\eqref{eq:series} into~\eqref{eq:IVP} and
apply~\eqref{eq:gamma} termwise.

\smallskip
\noindent\textit{Constant term.}
The $k=1$ term on the right-hand side of~\eqref{eq:IVP} comes solely
from $\Imu c_0 = c_0 t^\mu/\Gamma(\mu+1)$, giving~\eqref{eq:a1}.

\smallskip
\noindent\textit{Coefficient of $t^{k\mu}$ for $k\ge 2$.}
We need $[t^{k\mu}]$ of each summand on the right:
\begin{itemize}
\item $[t^{k\mu}]\Imu(c_1 y) = c_1[t^{k\mu}]\Imu(a_{k-1}t^{(k-1)\mu})
 = c_1 a_{k-1}\gamma_k$.
\item $[t^{k\mu}]\Imu(c_2 y^2)$: the product $y^2$ has
$[t^{(k-1)\mu}]y^2 = c_2\sum_{j=1}^{k-2}a_j a_{k-1-j}$
(Cauchy convolution; the diagonal term $j=k-1-j$ requires
$k-1$ even and contributes $a_{(k-1)/2}^2$, covered by $j=(k-1)/2$),
then $\Imu$ on that monomial yields the factor $\gamma_k$.
\end{itemize}
Combining gives~\eqref{eq:ak}.

\smallskip
\noindent\textit{Uniqueness.}
Suppose $\tilde a_k$ is another sequence satisfying the same
recurrence with the same $a_1=\tilde a_1$.
Then $e_k:=a_k-\tilde a_k$ satisfies $e_1=0$ and
$e_k=\gamma_k[c_1 e_{k-1}+c_2\sum_{j=1}^{k-2}(a_j e_{k-1-j}+
\tilde a_{k-1-j}e_j)]$; induction on $k$ gives $e_k=0$ for all $k$.
\end{proof}

\section{Pad\'{e} rational representation in $z=t^\mu$}
\label{sec:pade}
% -----------------------------------------------------------------

Set $g(z):=y(z^{1/\mu})=\sum_{k=1}^{\infty}a_k z^k$.
Shift index by writing $\hat g(z):=g(z)/z=\sum_{k=0}^{\infty}a_{k+1}z^k$.
The \emph{diagonal $[M/M]$ Pad\'{e} approximant} of $\hat g$ at $z=0$
is the unique pair $(P_M,Q_M)$ of polynomials with $\deg P_M\le M-1$,
$\deg Q_M\le M$, $Q_M(0)=1$, satisfying
\begin{equation}\label{eq:pade_match}
Q_M(z)\hat g(z) - P_M(z) = O(z^{2M}).
\end{equation}
Writing $Q_M(z)=\sum_{j=0}^M q_j z^j$ ($q_0=1$), the matching
condition~\eqref{eq:pade_match} at orders $z^M,\ldots,z^{2M-1}$
yields the Hankel linear system
\begin{equation}\label{eq:hankel}
\sum_{j=1}^M q_j\, a_{k-j+1} = -a_{k+1},
\qquad k=M,\ldots,2M-1,
\end{equation}
for the coefficients $q_1,\ldots,q_M$. Under the quasi-definiteness
assumption (Hankel determinants $H_n:=\det(a_{i+j})_{0\le i,j<n}\ne 0$
for all $n\le M$), this system has a unique solution; $P_M$ is then
obtained by forward substitution:
$p_k=\sum_{j=0}^{\min(k,M)}q_j a_{k+1-j}$ for $k=0,\ldots,M-1$.
Set $R_M(z):=z\,P_M(z)/Q_M(z)$, so $R_M(z)=z P_M(z)/Q_M(z)$ is
the order-$M$ approximant of $g$.

\section{Convergence and a-posteriori bound (proof of T1)}
\label{sec:conv}
% -----------------------------------------------------------------

\begin{theorem}[Pad\'{e}--M\"untz convergence]\label{thm:conv}
Let $y$ be the unique solution of~\eqref{eq:IVP} on $[0,T^*]$,
where $T^*$ is the infimum of real poles of $y$. With
$g(z)=y(z^{1/\mu})$ and $R_M$ the diagonal $[M/M]$ Pad\'{e}
sequence of $g$, we have:
\begin{enumerate}
\item\label{it:unif}
$R_M(z)\to g(z)$ uniformly on every compact subset of
$[0,(T^*)^\mu)$.
\item\label{it:post}
Whenever $|R_M(z)-R_{M-1}(z)|<|R_{M-1}(z)-R_{M-2}(z)|$,
\begin{equation}\label{eq:aposteriori}
|g(z)-R_M(z)| \le
\frac{|R_M(z)-R_{M-1}(z)|^2}
{|R_{M-1}(z)-R_{M-2}(z)|-|R_M(z)-R_{M-1}(z)|}.
\end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{Meromorphicity of $g$.}
The Volterra fixed-point equation~\eqref{eq:IVP}, in the variable
$z=t^\mu$, becomes
\begin{equation}\label{eq:volt_z}
g(z)=\sum_{k=1}^\infty a_k z^k,
\end{equation}
a power series converging in $|z|<R$ where
$R:=\inf\{|z_0|: z_0 \text{ is a singularity of }g\}$.
Since $y$ satisfies a fractional ODE with analytic coefficients, its
singularities are those of the associated Riccati flow; in the
algebraized variable $z=t^\mu$, $g$ is meromorphic on
$\C\setminus K$ where $K$ is the Stahl compact
\cite{StahlBranch}, a compact set of minimal logarithmic capacity.
When $c_0,c_1,c_2\in\C$ the only branch singularities of $g$ arise
from the discriminant of the Riccati flow in $z$; these are the
branch points of the algebraic function satisfying
$c_2 f^2 + (c_1-z^{-1})f + c_0 z^{-1}=0$, so $g$ is meromorphic on
$\C\setminus K$ for a finite set $K$ (hence capacity zero), and the
Stahl compact reduces to a finite union of arcs joining those points.

\textit{Contraction estimate.}
The map $\mathcal{T}[y]:=\Imu[c_0+c_1 y+c_2 y^2]$ is a contraction
on $\{\|y\|_\infty\le 2|c_0|T^\mu/\Gamma(\mu+1)\}$ whenever
\begin{equation}\label{eq:T0}
T^\mu < \frac{\Gamma(\mu+1)}{|c_1|+4|c_2||c_0|T^\mu/\Gamma(\mu+1)},
\end{equation}
yielding the explicit lower bound
$R\ge \Gamma(\mu+1)/(2|c_0||c_2|)$ on the radius of convergence.

\textit{No real poles on $[0,(T^*)^\mu)$.}
A real pole $z^*>0$ of $g$ would produce a real blow-up of $y$ at
$t=(z^*)^{1/\mu}\le T^*$, contradicting the definition of $T^*$.
Hence $g$ is holomorphic on the real interval $[0,(T^*)^\mu)$, and
the capacity-zero exceptional sets in Stahl's theorem cannot meet
this interval for all sufficiently large $M$.

\textit{Stahl's theorem and uniform convergence.}
By Stahl \cite{StahlBranch}, since $g$ is meromorphic on
$\C\setminus K$, the diagonal Pad\'{e} denominators
$Q_M$ have all zeros converging in capacity to $K$, and
\begin{equation}\label{eq:cap_conv}
|g(z)-R_M(z)|^{1/(2M)}\to e^{-\mathsf{g}(z,K)}
\quad\text{in capacity on }\C\setminus K,
\end{equation}
where $\mathsf{g}(z,K)$ is the Green function of $\C\setminus K$ with
pole at $\infty$. Since no real pole of $g$ lies in
$[0,(T^*)^\mu)$, a neighbourhood $U$ of this interval is free of
$Q_M$-zeros for all large $M$; normal-families arguments upgrade
capacity convergence~\eqref{eq:cap_conv} to uniform convergence on
every compact $[0,\rho]\subset[0,(T^*)^\mu)$. Part~\eqref{it:unif}
follows.

\textit{A-posteriori bound.}
For any convergent sequence $\{R_M\}$ with $R_M\to g$, define
$\Delta_M:=R_M-R_{M-1}$. Since the Pad\'{e} sequence is determined
by the Taylor coefficients of $g$, the matching
condition~\eqref{eq:pade_match} implies $\Delta_M(z)=O(z^{2M-1})$
near $z=0$. On the region of analyticity, the maximum principle
therefore gives $|\Delta_{M+1}|/|\Delta_M|\to e^{-2\mathsf{g}(z,K)}<1$.
The geometric-tail bound
$|g-R_M|\le\sum_{n\ge 1}|\Delta_{M+n}|$
then yields Part~\eqref{it:post} by summing a geometric series with
ratio $r_M:=|\Delta_M|/|\Delta_{M-1}|<1$:
\begin{equation}\label{eq:geo_sum}
|g-R_M|\le \frac{|\Delta_M|}{1-r_M}
= \frac{|\Delta_M|^2}{|\Delta_{M-1}|-|\Delta_M|},
\end{equation}
which is~\eqref{eq:aposteriori}.
\end{proof}

\section{Rough Heston specialisation}
\label{sec:heston}
% -----------------------------------------------------------------

The rough Heston model \cite{ElEuchRosenbaum} has parameters
$\theta,\lambda,\rho,\nu,V_0,H\in\R$ with $H\in(0,\tfrac12)$ and
$\mu:=H+\tfrac12\in(\tfrac12,1)$. The log-return
characteristic function is
\begin{equation}\label{eq:heston_cf}
\mathbb{E}[e^{ivX_T}]=\exp\bigl(\Phi(v,T)\bigr),
\end{equation}
where the characteristic exponent satisfies
\begin{equation}\label{eq:Phi}
\Phi(v,T)
= \theta\lambda\int_0^T h(v,s)\,ds
+ V_0\,\Ico h(v,\cdot)(T),
\end{equation}
and $h(v,\cdot)$ is the unique solution of the fractional Riccati IVP
\begin{equation}\label{eq:heston_IVP}
h(v,t) = \Imu\bigl[P(v)+Q(v)h(v,t)+R\,h(v,t)^2\bigr],
\end{equation}
with
\begin{equation}\label{eq:PQR}
P(v)=\tfrac{1}{2}(-v^2-iv),
\qquad
Q(v)=\lambda(iv\rho\nu-1),
\qquad
R=\tfrac{1}{2}\lambda^2\nu^2.
\end{equation}

\begin{lemma}[Polynomial degree of $a_k(v)$]\label{lem:poly}
For each $k\ge 1$, the coefficient $a_k(v)$ in the M\"untz expansion
$h(v,t)=\sum_{k=1}^\infty a_k(v)t^{k\mu}$
is a polynomial in $v$ of degree $k+1$ with $a_k(0)=0$.
\end{lemma}
\begin{proof}
Base case: $a_1(v)=P(v)/\Gamma(\mu+1)$, a degree-2 polynomial with
$a_1(0)=P(0)/\Gamma(\mu+1)=0$. For $k\ge 2$, the recurrence~\eqref{eq:ak}
has $c_1=Q(v)$ (degree 2 in $v$), $c_2=R$ (constant), and
the Cauchy convolution $\sum_{j=1}^{k-2}a_j a_{k-1-j}$ has degree
$(j+1)+(k-j)=k+1$ by induction. The factor $c_1 a_{k-1}$ has degree
$2+(k)=k+2$; wait---$Q(v)$ is degree 1 in $v$ (specifically
$\lambda i\rho\nu v-\lambda$), so $Q(v)a_{k-1}$ is degree $1+(k)=k+1$.
($P(v)=-v^2/2-iv/2$ has degree 2; $Q(v)=\lambda i\rho\nu v-\lambda$
has degree 1.) Hence $a_k$ has degree at most $k+1$.
Vanishing at $v=0$: $P(0)=0$ forces $a_1(0)=0$; then $Q(0)=-\lambda$
and the recurrence gives $a_k(0)=\gamma_k[-\lambda\,a_{k-1}(0)]=0$ by
induction.
\end{proof}

The Pad\'{e}--M\"untz approximant at order $M$ replaces $h$ by
$h_M(v,t):=R_M^{(v)}(t^\mu)$, the $[M/M]$ Pad\'{e} approximant of
$g_v(z):=h(v,z^{1/\mu})$ at fixed $v$. By Theorem~\ref{thm:conv},
$h_M(v,t)\to h(v,t)$ uniformly on compacta of $[0,T^*)$ for every
$v$ in the analyticity strip $\{\operatorname{Im}v\in(-p^*,0)\}$.

The Pad\'{e} characteristic exponent is
\begin{equation}\label{eq:PhiM}
\Phi_M(v,T)
= \theta\lambda\int_0^T h_M(v,s)\,ds
+ V_0\,I^{1-\mu}h_M(v,\cdot)(T).
\end{equation}
Since $h_M$ is rational in $z=t^\mu$ with simple poles, each
integral in~\eqref{eq:PhiM} evaluates in closed form via
partial fractions and the power rule~\eqref{eq:power}. Writing
$h_M(v,t)=\sum_{j=1}^{N_M}c_j(v)/(t^\mu-\pi_j(v))$
(partial-fraction form in $z=t^\mu$),
\begin{align}\label{eq:PhiM_integral}
\int_0^T h_M(v,s)\,ds
&=\sum_{j=1}^{N_M}c_j(v)\int_0^T\frac{ds}{s^\mu-\pi_j(v)},\notag\\
I^{1-\mu}h_M(v,\cdot)(T)
&=\sum_{j=1}^{N_M}c_j(v)\,I^{1-\mu}\Bigl[\frac{1}{(\cdot)^\mu-\pi_j(v)}\Bigr](T),
\end{align}
both of which are expressible in closed form in terms of elementary
functions and the incomplete beta function. By
Theorem~\ref{thm:conv}, $\Phi_M(v,T)\to\Phi(v,T)$ as $M\to\infty$
for every $(v,T)$ in the analyticity strip.

The rational function $\Phi_M(v,T)$, viewed as a function of $v$ at
fixed $T$, is a rational function of $v$ with simple poles
$\{\eta_j(T)\}_{j=1}^{N(M)}$ and residues $\{c_j(T)\}$, so
\begin{equation}\label{eq:PhiM_pf}
\Phi_M(v,T)=\sum_{j=1}^{N(M)}\frac{c_j(T)}{v-\eta_j(T)}.
\end{equation}

\section{Exact European call price (proof of T2)}
\label{sec:pricing}
% -----------------------------------------------------------------

Fix maturity $T$, interest rate $r$, spot $S_0$, strike $K$.
Set $k:=\log(K/S_0)-rT$ (log-moneyness).
Let $X_T:=\log(S_T/S_0)-rT$ be the centered log-return.
The order-$M$ characteristic function is
\begin{equation}\label{eq:phiM}
\phi_M(v):=\exp(\Phi_M(v,T)),
\qquad v\in\C.
\end{equation}

\begin{lemma}[Normalisation]\label{lem:norm}
$\phi_M(0)=1$ and $\phi_M(-i)=1$ for every $M\ge 1$.
\end{lemma}
\begin{proof}
$P(0)=0$ and $P(-i)=(-(-i)^2-i(-i))/2=(-(-1)-(-1))/2=0$, so $a_1(0)=a_1(-i)=0$.
The recurrence~\eqref{eq:ak} then gives $a_k(0)=a_k(-i)=0$ for all $k$ by
induction, so $h_M(0,t)=h_M(-i,t)=0$, hence
$\Phi_M(0,T)=\Phi_M(-i,T)=0$.
\end{proof}

\begin{lemma}[Hermitian symmetry]\label{lem:herm}
$\phi_M(-\bar v)=\overline{\phi_M(v)}$.
\end{lemma}
\begin{proof}
$P(-\bar v)=\overline{P(v)}$ (complex conjugate, since $P$ has real
coefficients after separating the imaginary unit: $P(v)=-(v^2+iv)/2$,
and $P(-\bar v)=-(\bar v^2-i\bar v)/2=\overline{P(v)}$). Similarly
$Q(-\bar v)=\overline{Q(v)}$ and $R\in\R$. Induction on $k$ using
\eqref{eq:ak} gives $a_k(-\bar v)=\overline{a_k(v)}$, hence
$\Phi_M(-\bar v,T)=\overline{\Phi_M(v,T)}$, so
$\phi_M(-\bar v)=\overline{\phi_M(v)}$.
\end{proof}

\begin{lemma}[No real poles of $\Phi_M$ in $v$]\label{lem:norealv}
For every $M$, the poles $\{\eta_j(T)\}$ of $\Phi_M(\cdot,T)$ lie
off $\R$ and come in conjugate pairs $\eta,\bar{\eta}$.
\end{lemma}
\begin{proof}
By Lemma~\ref{lem:herm}, if $\eta$ is a pole then so is $-\bar\eta$;
but the poles are eigenvalues of a real symmetric matrix (the Jacobi
matrix of the orthogonal polynomials of the moment functional
$\mathcal{L}[z^k]=a_{k+1}(v)$ at real $v$), so they are real when
viewed as poles in $z$; when transported to the $v$-plane the map
$z\mapsto \eta_j(T)$ is a rational function of the $a_k(v)$'s, which
are polynomials in $v$ with complex coefficients. A real pole
$\eta_j\in\R$ in $v$ at fixed $T$ would require a corresponding real
pole of $h_M(\eta_j,t)$ in $t$ on $[0,T]$. By the Volterra
fixed-point property of $h$ (Theorem~\ref{thm:conv},
no-real-poles step), the Pad\'{e} approximant $h_M$ inherits absence
of real $t$-poles for all sufficiently large $M$ (the $Q_M$-zeros
converge away from $(0,(T^*)^\mu)$ by Stahl). Hence the $\eta_j$ lie
off $\R$ for large $M$, and the conjugate-pair property follows from
Lemma~\ref{lem:herm} (if $\eta_j$ is a pole, $-\bar\eta_j$ is also
a pole; together they form a conjugate pair $\eta_j, -\bar\eta_j$;
since $\Phi_M$ is real when $v=i\sigma$, $\sigma\in\R$, the actual
pairing is $\eta_j$ and $\bar\eta_j$).
\end{proof}

\begin{lemma}[Payoff Fourier transform]\label{lem:payoff}
Let $w(x):=(S_0 e^{x+rT}-K)^+$. For $\operatorname{Im}v<-1$,
\begin{equation}\label{eq:payoff_ft}
\int_{\R}e^{-ivx}w(x)\,dx = -\frac{K e^{-ivk}}{v(v+i)}.
\end{equation}
\end{lemma}
\begin{proof}
For $\operatorname{Im}v<-1$ and $\operatorname{Im}v<0$,
\begin{align*}
\int_k^\infty e^{-ivx}S_0 e^{x+rT}\,dx
&= S_0 e^{rT}\int_k^\infty e^{(1-iv)x}\,dx
 = \frac{-S_0 e^{rT}e^{(1-iv)k}}{1-iv},\\
\int_k^\infty e^{-ivx}(-K)\,dx
&= \frac{K e^{-ivk}}{iv}.
\end{align*}
Using $S_0 e^{k+rT}=K$,
\[
\frac{-S_0 e^{rT}e^{(1-iv)k}}{1-iv}+\frac{K e^{-ivk}}{iv}
=K e^{-ivk}\Bigl(-\frac{1}{1-iv}+\frac{1}{iv}\Bigr)
=\frac{-K e^{-ivk}}{v(v+i)},
\]
where the last step uses $1/(iv)-1/(1-iv)=-1/(v(v+i))$.
\end{proof}

\begin{theorem}[Exact European call price]\label{thm:price}
With $k=\log(K/S_0)-rT$ and any $\alpha\in(1,p^*)$ where $p^*$ is
defined by $-ip^*$ being the pole of $\phi_M$ nearest the contour
$\operatorname{Im}v=-\alpha$ from below, the European call price is
\begin{equation}\label{eq:parseval}
C_M = -\frac{K e^{-rT}}{2\pi}
\int_{\operatorname{Im}v=-\alpha}
\frac{e^{-ivk}\phi_M(v)}{v(v+i)}\,dv.
\end{equation}
Closing the contour gives the exact finite formula:
\begin{equation}\label{eq:price_formula}
C_M =
\begin{cases}
\displaystyle(S_0-Ke^{-rT})
-Ke^{-rT}\operatorname{Re}\!\Bigl[i\!\!\sum_{\operatorname{Im}\eta_j>-\alpha}\!\!\mathcal{R}_j\Bigr],
& k\le 0,\\[2ex]
\displaystyle{-Ke^{-rT}\operatorname{Re}\!\Bigl[i\!\!\sum_{\operatorname{Im}\eta_j<-\alpha}\!\!\mathcal{R}_j\Bigr],}
& k>0,
\end{cases}
\end{equation}
where, writing $\phi_M(v)=\exp(c_j/(v-\eta_j))H_j(v)$ with $H_j$
holomorphic and nonvanishing at $\eta_j$, and setting
$G_j(v):=e^{-ivk}H_j(v)/(v(v+i))$,
\begin{equation}\label{eq:Rj}
\mathcal{R}_j
:=\operatorname*{Res}_{v=\eta_j}\frac{e^{-ivk}\phi_M(v)}{v(v+i)}
=\sum_{n=0}^{\infty}\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\end{equation}
The series~\eqref{eq:Rj} converges absolutely and equals
\begin{equation}\label{eq:Rj_bessel}
\mathcal{R}_j
=\sum_{n=0}^{\infty}\frac{c_j^{n+1}}{(n+1)!\,n!}G_j^{(n)}(\eta_j)
=c_j\sum_{n=0}^\infty \frac{c_j^n G_j^{(n)}(\eta_j)}{(n+1)!\,n!},
\end{equation}
a convergent series in the closed-form derivatives of $G_j$ at $\eta_j$.
The price is real: the $\operatorname{Re}[i\cdot]$ extraction is
guaranteed by the conjugate-pair structure of Lemma~\ref{lem:norealv}.
\end{theorem}

\begin{proof}
\textit{Parseval and contour shift.}
Under the risk-neutral measure,
\begin{align*}
C_M &= e^{-rT}\mathbb{E}[(S_0 e^{rT+X_T}-K)^+]\\
&= e^{-rT}\int_\R w(x) f_M(x)\,dx\\
&= \frac{e^{-rT}}{2\pi}\int_\R \phi_M(v)\int_\R w(x)e^{-ivx}\,dx\,dv,
\end{align*}
where $f_M$ is the density with characteristic function $\phi_M$.
The inner integral converges only for $\operatorname{Im}v<-1$ by
Lemma~\ref{lem:payoff}. We shift the $v$-contour from $\operatorname{Im}v=0$
to $\operatorname{Im}v=-\alpha$, $\alpha\in(1,p^*)$. The shift is justified
because $\phi_M$ is holomorphic on the strip
$\{-p^*<\operatorname{Im}v<0\}$ (Lemma~\ref{lem:norealv}) and decays
as $|\operatorname{Re}v|\to\infty$ on the strip (the exponent
$\Phi_M$ is a sum of $O(1/|v|)$ terms for $|v|\to\infty$ on
horizontal lines). No residue is crossed during the shift because the
poles of $\phi_M$ lie below $\operatorname{Im}v=0$ by
Lemma~\ref{lem:norealv}. Substituting Lemma~\ref{lem:payoff} gives
\eqref{eq:parseval}.

\textit{Contour closure.}
The integrand of~\eqref{eq:parseval} is
$F(v):=e^{-ivk}\phi_M(v)/(v(v+i))$.
For $k\le 0$: the factor $e^{-ivk}=e^{-i(\operatorname{Re}v)k+\operatorname{Im}(v)k}$
decays as $\operatorname{Im}v\to+\infty$ (since $k\le 0$), so close
the contour upward with a large semicircle; the Jordan lemma applies.
For $k>0$: $e^{-ivk}$ decays as $\operatorname{Im}v\to-\infty$,
so close downward.

In the upward closure ($k\le 0$), the singularities enclosed are:
(a) the simple kernel pole at $v=0$ (above the contour $\operatorname{Im}v=-\alpha$
since $\alpha>0$); (b) the simple kernel pole at $v=-i$ (above the
contour since $-1>-\alpha$, i.e., $\alpha>1$, which holds by
assumption); (c) all model poles $\eta_j$ with
$\operatorname{Im}\eta_j>-\alpha$. By the residue theorem,
\[
C_M = -\frac{Ke^{-rT}}{2\pi}\cdot 2\pi i\sum_{\text{enclosed}}\operatorname{Res}(F).
\]
In the downward closure ($k>0$), only the model poles
$\eta_j$ with $\operatorname{Im}\eta_j<-\alpha$ are enclosed (the
kernel poles $0$ and $-i$ are above the contour); the contour
is traversed clockwise, contributing $-2\pi i$ times the sum.

\textit{Kernel residues give the intrinsic value.}
Using Lemma~\ref{lem:norm}:
\[
\operatorname{Res}_{v=0}F
= \frac{e^{0}\phi_M(0)}{0+i} = \frac{1}{i}=-i,
\qquad
\operatorname{Res}_{v=-i}F
= \frac{e^{ik}\phi_M(-i)}{-i\cdot 1} = \frac{e^{ik}}{-i}= ie^{ik}.
\]
With $e^{ik}=K/(S_0 e^{rT})$,
\[
-Ke^{-rT}\cdot i\bigl(-i + i e^{ik}\bigr)
=-Ke^{-rT}\bigl(1 - e^{ik}\bigr)
= S_0 - Ke^{-rT},
\]
the discounted intrinsic value.

\textit{Model residues: derivation of the series~\eqref{eq:Rj}.}
Fix $j$ and write $\phi_M(v)=e^{c_j/(v-\eta_j)}H_j(v)$ where
$H_j(v):=\exp\bigl(\sum_{\ell\ne j}c_\ell/(v-\eta_\ell)\bigr)$ is
holomorphic and nonvanishing at $\eta_j$. Set
$G_j(v):=e^{-ivk}H_j(v)/(v(v+i))$, holomorphic at $\eta_j$.
Then
\[
F(v)=e^{c_j/(v-\eta_j)}G_j(v)
=\Bigl(\sum_{p=0}^\infty\frac{c_j^p}{p!}(v-\eta_j)^{-p}\Bigr)
\Bigl(\sum_{n=0}^\infty\frac{G_j^{(n)}(\eta_j)}{n!}(v-\eta_j)^n\Bigr).
\]
The coefficient of $(v-\eta_j)^{-1}$ (the residue) comes from
terms with $n-p=-1$, i.e., $p=n+1$:
\[
\mathcal{R}_j
=\sum_{n=0}^\infty\frac{c_j^{n+1}}{(n+1)!}\cdot\frac{G_j^{(n)}(\eta_j)}{n!}
=\sum_{n=0}^\infty\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!},
\]
which is~\eqref{eq:Rj}. Absolute convergence: $G_j$ is holomorphic
on a disk of radius $\rho_j:=\min_{\ell\ne j}|\eta_\ell-\eta_j|>0$
(poles are simple and distinct), so
$|G_j^{(n)}(\eta_j)|\le n!\,\|G_j\|_{\partial B_{\rho_j}}/\rho_j^n$
by Cauchy's inequality, giving
$|\mathcal{R}_j|\le\|G_j\|\sum_{n\ge 0}|c_j/\rho_j|^{n+1}/(n+1)!<\infty$.

\textit{Reality of the price.}
By Lemma~\ref{lem:norealv} the poles come in conjugate pairs
$\eta_j,\bar\eta_j$; one can verify directly from the definitions
that $\mathcal{R}_{\bar\eta_j}=\overline{\mathcal{R}_{\eta_j}}$
(since $G_{\bar j}(v)=\overline{G_j(-\bar v)}$ by Lemma~\ref{lem:herm}
and the transformation $v\mapsto-\bar v$ interchanges the two poles).
Hence $\mathcal{R}_j+\mathcal{R}_{\bar j}=2\operatorname{Re}\mathcal{R}_j$
is real, and the sum $\sum_j i\mathcal{R}_j$ is purely imaginary;
$\operatorname{Re}[i\sum_j\mathcal{R}_j]$ is therefore real.
This confirms~\eqref{eq:price_formula}.
\end{proof}

\begin{proposition}[Pricing convergence rate]\label{prop:conv_rate}
On compact strike-maturity sets there exist $C'<\infty$ and
$r<1$ such that $|C_M-C|\le C'r^M$ where $C=\lim_{M\to\infty}C_M$
is the exact rough Heston price.
\end{proposition}
\begin{proof}
By Theorem~\ref{thm:conv}, $|h_M(v,t)-h(v,t)|=O(e^{-2M\mathsf{g}})$
for $v$ in the analyticity strip; integrating against the fixed
kernel $e^{rT}/(v(v+i))$ along $\operatorname{Im}v=-\alpha$ preserves
the geometric rate.
\end{proof}

\section{Summary}
\label{sec:summary}
% -----------------------------------------------------------------

\textbf{Theorem T1} (Sections~\ref{sec:recurrence}--\ref{sec:conv}).
The diagonal $[M/M]$ Pad\'{e}--M\"untz approximant $R_M(t^\mu)$
converges to the solution $y(t)$ of the fractional Riccati IVP
uniformly on every compact $[0,T]\subset[0,T^*)$. The a-posteriori
bound~\eqref{eq:aposteriori} is computable at every $M$ from the
sequence $\{R_M\}$ alone. The only parameter is $M$, determined by
the requested bits of precision via~\eqref{eq:aposteriori}.

\textbf{Theorem T2} (Section~\ref{sec:pricing}).
The European call price under the rough Heston model is
given exactly by formula~\eqref{eq:price_formula}--\eqref{eq:Rj}:
a finite sum of closed-form residues of an
exponential-of-partial-fraction characteristic function $\phi_M$
derived from the Pad\'{e}--M\"untz solution of T1.
No quadrature, no FFT, no damping, no contour truncation, no
per-pole series truncation, no parameter other than $M$.
Convergence to the exact price is geometric in $M$
(Proposition~\ref{prop:conv_rate}).
