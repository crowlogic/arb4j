\title{The Fractional Riccati--M\"untz--Pad\'{e} Theorem\\
\large Closed-Form Density and European Call Price under Rough Heston via Fox $H$, with COS Pricing as Alternative}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Three results are established for the rough Heston model. First
(Theorem~\ref{thm:conv}), the diagonal $[M/M]$ Pad\'{e} resummation
of the local Puiseux series of the cumulant generating function
converges on every compact subset of the complement of the Stahl
branch/polar set, including beyond the Puiseux radius of the
original series, at Stahl rate. The Pad\'{e} denominator is the
orthogonal polynomial of degree $M$ under the Hankel moment
functional, produced directly from the M\"untz moments by the
Chebyshev algorithm in arbitrary-precision arithmetic, equivalent
to solving the Hankel system but without matrix formation or
inversion. Second (Theorems~\ref{thm:foxH_density},
\ref{thm:foxH_price}), both the density of the log-return and the
European call price admit strictly closed-form representations as
finite sums of Fox $H$-functions in the partial-fraction data of
the rational characteristic exponent $\Phi_M$, evaluable in
$O(M\log^2 M)$ operations with no quadrature. The price formula
is derived by two analytically independent routes (convolution
chain and Mellin--Barnes inversion) converging on identical
parameter data. The Pad\'{e} characteristic function has Gaussian
real-axis decay, sharper than the stretched-exponential decay of
the exact rough Heston cf. Third (Theorem~\ref{thm:price}), the
same prices may alternatively be computed by applying the
Fang--Oosterlee COS method to $\Phi_M(\cdot,T)$, with Gaussian-rate
truncation error.
\end{abstract}

\section{Fractional calculus and the M\"untz basis}
\label{sec:fc}
% -----------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_0^t (t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$ and $\alpha>0$,
\begin{equation}\label{eq:power}
I^{\alpha} t^s = \frac{\Gamma(s+1)}{\Gamma(s+\alpha+1)}\,t^{s+\alpha}.
\end{equation}
\end{lemma}
\begin{proof}
Substitution $\sigma=t\tau$ in $I^\alpha$ gives
\[
I^{\alpha} t^s
=\frac{1}{\Gamma(\alpha)}\int_0^t(t-\sigma)^{\alpha-1}\sigma^s\,d\sigma
=\frac{t^{s+\alpha}}{\Gamma(\alpha)}\int_0^1(1-\tau)^{\alpha-1}\tau^s\,d\tau
=\frac{t^{s+\alpha}\,B(\alpha,s+1)}{\Gamma(\alpha)}.
\]
The Beta-function identity
$B(\alpha,s+1)=\Gamma(\alpha)\Gamma(s+1)/\Gamma(s+\alpha+1)$
gives~\eqref{eq:power}.
\end{proof}

The M\"untz lattice $\{t^{k\mu}\}_{k\ge 0}$ is closed under
$\Imu$ by~\eqref{eq:power} and closed under multiplication. In
the variable $z=t^\mu$ every element is $z^k$, so ordinary
power-series arithmetic applies.

\section{The M\"untz--Tau recurrence}
\label{sec:recurrence}
% -----------------------------------------------------------------

Define $\gamma_k:=\Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$ for $k\ge 1$,
so that $\Imu t^{(k-1)\mu}=\gamma_k\,t^{k\mu}$.

\begin{theorem}[M\"untz--Tau recurrence]\label{thm:recurrence}
The unique formal power-series solution of
\begin{equation}\label{eq:IVP}
y=\Imu[c_0+c_1y+c_2y^2],\qquad y(0)=0,
\end{equation}
is $y(t)=\sum_{k\ge 1}a_kt^{k\mu}$ with
\begin{align}
a_1 &= \frac{c_0}{\Gamma(\mu+1)},
\label{eq:a1}\\
a_k &= \gamma_k\!\left[c_1\,a_{k-1}
+c_2\!\sum_{j=1}^{k-2}a_j\,a_{k-1-j}\right],\qquad k\ge 2.
\label{eq:ak}
\end{align}
\end{theorem}

\begin{proof}
Substitution of the ansatz into~\eqref{eq:IVP} and termwise
application of~\eqref{eq:power} produces $a_1=c_0/\Gamma(\mu+1)$.
For $k\ge 2$, $[t^{k\mu}]\Imu(c_1y)=c_1a_{k-1}\gamma_k$ and the
Cauchy product
$[t^{(k-1)\mu}]y^2=\sum_{j=1}^{k-2}a_ja_{k-1-j}$ combine to
yield~\eqref{eq:ak}. Uniqueness from linearity in $\{e_{<k}\}$
where $e_k:=a_k-\tilde a_k$.
\end{proof}

\section{Rough Heston cumulant moments}
\label{sec:cumulant}
% -----------------------------------------------------------------

The rough Heston model has parameters
$\theta,\lambda,\rho,\nu,V_0,H\in\R$ with $H\in(0,\tfrac12)$ and
$\mu:=H+\tfrac12\in(\tfrac12,1)$. The log-return characteristic
function is $\phi_T(v):=\mathbb{E}[e^{ivX_T}]=\exp(\Phi(v,T))$
with $X_T:=\log(S_T/S_0)-rT$ and
\begin{equation}\label{eq:Phi}
\Phi(v,T)=\theta\lambda\int_0^Th(v,s)\,ds+V_0\,\Ico h(v,\cdot)(T),
\end{equation}
$h(v,\cdot)$ solving
\begin{equation}\label{eq:heston_IVP}
h(v,t)=\Imu\bigl[P(v)+Q(v)h(v,t)+R\,h(v,t)^2\bigr],
\end{equation}
with $P(v)=\tfrac12(-v^2-iv)$, $Q(v)=\lambda(iv\rho\nu-1)$,
$R=\tfrac12\lambda^2\nu^2$.

\begin{lemma}[Riccati Puiseux series]\label{lem:h_series}
The unique formal solution of~\eqref{eq:heston_IVP} is
$h(v,t)=\sum_{k\ge 1}a_k(v)t^{k\mu}$ with $\{a_k(v)\}$ from
Theorem~\ref{thm:recurrence} applied to $c_0=P(v),c_1=Q(v),c_2=R$.
Each $a_k(v)$ is a polynomial in $v$ of degree $\le k+1$
satisfying $a_k(0)=a_k(-i)=0$.
\end{lemma}
\begin{proof}
Series solution from Theorem~\ref{thm:recurrence}; degree by
induction with $\deg P=2,\deg Q=1,\deg R=0$; vanishing from
$P(0)=P(-i)=0$ and induction.
\end{proof}

\begin{theorem}[Cumulant generating function as Puiseux series]\label{thm:cumulant_moments}
With $z:=T^\mu$, $\Phi(v,T)=T\cdot\hat\Phi(v,z)$,
\begin{equation}\label{eq:Phi_hat_def}
\hat\Phi(v,z)=\sum_{k=0}^{\infty}M_k(v)\,z^k,
\end{equation}
\begin{equation}\label{eq:cumulant_moments}
M_k(v)=\frac{\theta\lambda}{k\mu+1}\,a_k(v)
+V_0\,\frac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+2)}\,a_{k+1}(v),
\quad k\ge 0,
\end{equation}
$a_0(v)\equiv 0$.
\end{theorem}
\begin{proof}
Substitute $h(v,s)=\sum_{k\ge 1}a_k(v)s^{k\mu}$
into~\eqref{eq:Phi}.
$\int_0^Ts^{k\mu}ds=T^{k\mu+1}/(k\mu+1)$ contributes
$\theta\lambda\sum_{k\ge 1}a_k(v)/(k\mu+1)\cdot T^{k\mu+1}$.
By~\eqref{eq:power}, $\Ico t^{k\mu}|_{t=T}=\Gamma(k\mu+1)/\Gamma((k-1)\mu+2)\cdot T^{(k-1)\mu+1}$,
contributing $V_0\sum_{k\ge 1}a_k(v)\Gamma(k\mu+1)/\Gamma((k-1)\mu+2)\cdot T^{(k-1)\mu+1}$.
Reindexing by $j=k-1$ and combining in $z=T^\mu$
gives~\eqref{eq:cumulant_moments}.
\end{proof}

\begin{remark}[Index bookkeeping]\label{rem:index}
$\{M_k(v)\}_{k=0}^{2M-1}$ via~\eqref{eq:cumulant_moments} requires
$\{a_k(v)\}_{k=1}^{2M}$, since the $V_0$ term in $M_k$ uses
$a_{k+1}$. The M\"untz-coefficient budget is $2M$ Riccati
coefficients per strike.
\end{remark}

% -----------------------------------------------------------------
\section{Pad\'{e} resummation via the Chebyshev algorithm}
\label{sec:pade}
% -----------------------------------------------------------------

The diagonal $[M/M]$ Pad\'{e} approximant of $\hat\Phi$ is the
unique pair $(P_M,Q_M)$ with $\deg P_M\le M-1$, $\deg Q_M\le M$,
$Q_M(0)=1$, satisfying
\begin{equation}\label{eq:pade_match}
Q_M(z)\hat\Phi(v,z)-P_M(z)=O(z^{2M}).
\end{equation}

\subsection{Moment functional}

Define $\mathcal{L}:\C[w]\to\C$ by $\mathcal{L}[w^k]:=M_k(v)$,
extended linearly. Let $H_n:=\det(M_{i+j})_{0\le i,j<n}$.
Quasi-definiteness $H_n\ne 0$ ($n\ge 1$) is assumed throughout.

\subsection{Three-term recurrence and orthogonality}

\begin{theorem}[Three-term recurrence]\label{thm:three_term}
Under quasi-definiteness there is a unique sequence of monic
polynomials $\{p_n\}_{n\ge 0}$, $\deg p_n=n$, satisfying
\begin{equation}\label{eq:OP_orthog}
\mathcal{L}[w^k\,p_n(w)]=h_n\,\delta_{k,n}\quad(0\le k\le n),
\quad h_n:=\mathcal{L}[w^np_n]\ne 0,
\end{equation}
\begin{equation}\label{eq:three_term}
p_{n+1}(w)=(w-\alpha_n)p_n(w)-\beta_np_{n-1}(w),
\quad p_0\equiv 1,p_{-1}\equiv 0,
\end{equation}
with $\alpha_n=\mathcal{L}[wp_n^2]/h_n$, $\beta_n=h_n/h_{n-1}$
($n\ge 1$), $\beta_0:=0$, $h_n=H_{n+1}/H_n$.
\end{theorem}
\begin{proof}
Heine determinantal construction
\[
p_n(w)=\frac{1}{H_n}\det\begin{pmatrix}
M_0 & M_1 & \cdots & M_n\\
M_1 & M_2 & \cdots & M_{n+1}\\
\vdots & & & \vdots\\
M_{n-1} & M_n & \cdots & M_{2n-1}\\
1 & w & \cdots & w^n
\end{pmatrix}
\]
is monic, orthogonal to $1,\ldots,w^{n-1}$. Recurrence forced by
$\deg(wp_n-p_{n+1})\le n$ and orthogonality.
\end{proof}

\subsection{The Chebyshev algorithm}

\begin{theorem}[Chebyshev algorithm]\label{thm:chebyshev}
$\sigma_{k,l}:=\mathcal{L}[p_k(w)w^l]$ ($k\ge -1$, $l\ge 0$).
\begin{equation}\label{eq:cheb_recurrence}
\sigma_{-1,l}=0,\qquad \sigma_{0,l}=M_l(v),
\end{equation}
\begin{equation}\label{eq:cheb_recurrence2}
\sigma_{k+1,l}=\sigma_{k,l+1}-\alpha_k\sigma_{k,l}-\beta_k\sigma_{k-1,l},
\qquad k\ge 0,\;l\ge k+1,
\end{equation}
\begin{equation}\label{eq:cheb_extract}
\alpha_0=\frac{M_1(v)}{M_0(v)},\quad \beta_0=0,\quad h_0=M_0(v),
\end{equation}
\begin{equation}\label{eq:cheb_extract_k}
\alpha_k=\frac{\sigma_{k,k+1}}{\sigma_{k,k}}-\frac{\sigma_{k-1,k}}{\sigma_{k-1,k-1}},\quad
\beta_k=\frac{\sigma_{k,k}}{\sigma_{k-1,k-1}},\quad
h_k=\sigma_{k,k},\qquad k\ge 1.
\end{equation}
Cost is $O(M^2)$ ring operations on $\{M_l(v)\}_{l=0}^{2M-1}$,
no Hankel matrix formed and no linear system solved.
Arbitrary-precision exact.
\end{theorem}
\begin{proof}
Apply $\mathcal{L}$ to
$p_{k+1}(w)w^l=p_k(w)w^{l+1}-\alpha_kp_k(w)w^l-\beta_kp_{k-1}(w)w^l$.
Equivalence to Hankel solve: $\sum_{j=0}^Mq_jM_{j+l}=0$ for
$l=0,\ldots,M-1$, $q_M=1$ has the monomial coefficients of $p_M$
as unique solution.
\end{proof}

\subsection{Pad\'{e} numerator from associated polynomials}

\begin{definition}[Associated polynomial of the second kind]\label{def:assoc}
\begin{equation}\label{eq:p1_def}
p_n^{(1)}(w):=\mathcal{L}_v\!\left[\frac{p_n(w)-p_n(v)}{w-v}\right].
\end{equation}
\end{definition}

\begin{lemma}[Same recurrence]\label{lem:p1_recur}
$p_{n+1}^{(1)}=(w-\alpha_n)p_n^{(1)}-\beta_np_{n-1}^{(1)}$
($n\ge 1$), $p_0^{(1)}\equiv 0$, $p_1^{(1)}\equiv M_0(v)$.
\end{lemma}
\begin{proof}
Subtract~\eqref{eq:three_term} at $w$ and $v$, divide by $w-v$,
apply $\mathcal{L}_v$. Cross-term $\mathcal{L}[p_n]$ vanishes for
$n\ge 1$ by orthogonality.
\end{proof}

\subsection{Pad\'{e} matching}

\begin{lemma}[Pad\'{e} denominator and numerator]\label{lem:match}
$Q_M(z):=z^Mp_M(1/z)$, $P_M(z):=z^{M-1}p_M^{(1)}(1/z)$. Then
$Q_M(0)=1$, $\deg P_M\le M-1$,
\begin{equation}\label{eq:pade_proved}
Q_M(z)\hat\Phi(v,z)-P_M(z)=h_Mz^{2M}+O(z^{2M+1}).
\end{equation}
\end{lemma}
\begin{proof}
Cauchy representation
$\hat\Phi(v,z)=\mathcal{L}_w[(1-wz)^{-1}]$ and the algebraic
identity $z^Mp_M(1/z)/(1-wz)=z^M(p_M(1/z)-p_M(w))/(1-wz)+z^Mp_M(w)/(1-wz)$
split $Q_M\hat\Phi$ into a polynomial-of-degree-$\le M-1$ part
(equal to $z^{M-1}p_M^{(1)}(1/z)=P_M(z)$) plus
$z^M\sum_{k\ge 0}\mathcal{L}[w^kp_M]z^k$, starting at $k=M$ with
leading $h_M$.
\end{proof}

\begin{corollary}[Rational characteristic exponent]\label{cor:rational_Phi}
\begin{equation}\label{eq:PhiM_def}
\Phi_M(v,T):=T\cdot z\cdot\frac{P_M(z;v,T)}{Q_M(z;v,T)}\bigg|_{z=T^\mu}
\end{equation}
is rational in $v$ at fixed $T$ with $2M$ poles in $v$ from
$Q_M(z;v,T)|_{z=T^\mu}=0$.
\end{corollary>

\subsection{J-fraction}

\begin{theorem}[J-fraction]\label{thm:Jfrac}
\begin{equation}\label{eq:Jfrac}
\hat\Phi(v,z)=\cfrac{M_0(v)}{1-\alpha_0z-\cfrac{\beta_1z^2}{1-\alpha_1z-\cfrac{\beta_2z^2}{1-\alpha_2z-\cdots}}}
\end{equation}
with $M$-th convergent $P_M(z)/Q_M(z)$.
\end{theorem}
\begin{proof}
Standard correspondence between J-fractions and orthogonal
polynomial sequences satisfying~\eqref{eq:three_term}.
\end{proof}

\section{Convergence and a-posteriori bound}
\label{sec:conv}
% -----------------------------------------------------------------

\begin{theorem}[Stahl convergence]\label{thm:conv}
$\rho(v)>0$ Puiseux radius; $K(v)\subset\C\setminus[0,\rho(v))$
Stahl branch/polar set. Then
\begin{equation}\label{eq:stahl}
|\hat\Phi(v,z)-R_M(z;v)|^{1/(2M)}\to e^{-\mathsf{g}(z,K(v))}
\end{equation}
in capacity on $\C\setminus K(v)$, uniform on real compacta. In
particular $\Phi_M(v,T)\to\Phi(v,T)$ for every $T\ge 0$ with
$T^\mu\notin K(v)$, including $T$ beyond the Puiseux radius.
\end{theorem}
\begin{proof}
Volterra fixed-point applied to~\eqref{eq:heston_IVP} extends
$h(v,\cdot)$ holomorphically beyond the initial Puiseux disc;
$\hat\Phi(v,z)$ is meromorphic on a slit domain omitting compact
$K(v)$. Stahl's theorem gives~\eqref{eq:stahl} in capacity;
normal-families upgrades to uniform on real compacta.
\end{proof}

\begin{theorem}[A-posteriori bound, supremum-ratio]\label{thm:aposteriori}
$\Delta_M:=\Phi_M-\Phi_{M-1}$,
$q_M:=\sup_{n\ge M}|\Delta_{n+1}|/|\Delta_n|\to e^{-2\mathsf{g}(z,K(v))}<1$
on $\C\setminus K(v)$. Whenever $q_M<1$,
\begin{equation}\label{eq:aposteriori}
|\Phi(v,T)-\Phi_M(v,T)|\le\frac{|\Delta_M|}{1-q_M}.
\end{equation}
\end{theorem}
\begin{proof}
$|\Phi-\Phi_M|\le\sum_{n\ge 1}|\Delta_{M+n}|\le|\Delta_M|/(1-q_M)$.
\end{proof}

\begin{remark}[Aitken form]\label{rem:aitken}
Under exact geometric decay $|\Delta_{M+1}|/|\Delta_M|=q$ the
sharper Aitken form $|\Phi-\Phi_M|\le|\Delta_M|^2/(|\Delta_{M-1}|-|\Delta_M|)$
applies; in general it is heuristic.
\end{remark>

\begin{lemma}[Normalisation]\label{lem:norm}
$\Phi_M(0,T)=\Phi_M(-i,T)=0$, hence $\phi_M(0,T)=\phi_M(-i,T)=1$.
\end{lemma}
\begin{proof}
$P(0)=P(-i)=0$ and Lemma~\ref{lem:h_series} give
$a_k(0)=a_k(-i)=0$, hence $M_k(0)=M_k(-i)=0$.
\end{proof}

\begin{lemma}[Hermitian symmetry]\label{lem:herm}
$\phi_M(-\bar v,T)=\overline{\phi_M(v,T)}$.
\end{lemma}
\begin{proof}
$P(-\bar v)=\overline{P(v)}$, $Q(-\bar v)=\overline{Q(v)}$,
$R\in\R$, induction via~\eqref{eq:ak},
\eqref{eq:cumulant_moments}.
\end{proof}

\begin{lemma}[Gaussian real-axis decay]\label{lem:decay}
With $(\sigma_T,\mu_T,\{u_j(T),A_j(T)\}_{j=1}^{2M})$ from
PFE~\eqref{eq:PFE},
\begin{equation}\label{eq:gauss_decay}
|\phi_M(u,T)|\le e^{C_T}\,e^{-\tfrac12\sigma_T^2u^2},\qquad u\in\R,
\end{equation}
$C_T:=\sum_{j=1}^{2M}|A_j(T)|/|\Imag u_j(T)|<\infty$.
\end{lemma}
\begin{proof}
On $\R$, $|u-u_j(T)|\ge|\Imag u_j(T)|>0$ since $u_j\notin\R$, so
$\bigl|\sum_jA_j/(u-u_j)\bigr|\le C_T$ uniformly in $u$. The
linear term $-i\mu_Tu$ contributes a unimodular phase factor
$e^{-i\mu_Tu}$ of modulus one for $u,\mu_T\in\R$. The Gaussian
factor gives~\eqref{eq:gauss_decay}.
\end{proof}

\begin{remark}[Pre-asymptotic regime]\label{rem:CT}
The bound~\eqref{eq:gauss_decay} is governed by $C_T$, which
blows up if any pole approaches $\R$. The Gaussian envelope
dominates $e^{C_T}$ for $|u|\gtrsim\sqrt{2C_T/\sigma_T^2}$. For
smaller $|u|$ the trivial bound
$|\phi_M(u,T)|\le 1+\sum_j|A_j(T)|/|\Imag u_j(T)|$ from the
$L^\infty$ norm of $\Phi_M$ along $\R$ is used. The COS error
rate~\eqref{eq:cos_error} is asymptotic in $N$; the constant
$C'$ absorbs $C_T$.
\end{remark}

\begin{remark}[Tail comparison]\label{rem:tail_compare}
Pad\'{e} $\phi_M$ has Gaussian decay
$e^{-\tfrac12\sigma_T^2u^2}$, sharper than the
stretched-exponential $e^{-c|u|^{2\mu}}$ ($\mu<1$) of the exact
rough Heston cf.
\end{remark>

\begin{proposition}[Strip admissibility]\label{prop:strip}
For rough Heston with $\mu\in(\tfrac12,1)$ and Pad\'{e} order
$M$, the PFE poles $\{u_j(T)\}_{j=1}^{2M}$ split by Hermitian
symmetry (Lemma~\ref{lem:herm}) into $M$ poles in
$\{\Imag u_j<0\}$ and $M$ in $\{\Imag u_j>0\}$. Consider the
strip hypothesis
\begin{equation}\label{eq:strip_hypothesis}
\min_{j=1,\ldots,2M}\Imag u_j(T)>c+1\tag{$\mathrm{H}_c$}
\end{equation}
for $c\in\R$. Any $c\le\min_j\Imag u_j(T)-1$
satisfies~\eqref{eq:strip_hypothesis}; such $c$ exists because
the PFE has finitely many poles off $\R$, so
$\min_j\Imag u_j(T)>-\infty$. The corresponding damping
$\alpha=-c-1\ge -\min_j\Imag u_j(T)$ is positive and finite.
\end{proposition}

\begin{remark}[Independence from $c$]\label{rem:cauchy_independence}
The standard Carr--Madan range $\alpha\in(0,1)$ corresponds to
$c\in(-2,-1)$; Proposition~\ref{prop:strip} permits arbitrarily
negative $c$ and arbitrarily large $\alpha$, all integrable by
Lemma~\ref{lem:decay}. The Fox $H$ price
formula~\eqref{eq:foxH_price} is independent of the choice of
$c$ within the admissible region: by Cauchy's theorem the Lewis
contour~\eqref{eq:lewis} can be deformed between any two
admissible vertical lines without crossing poles, so the integral
value is unchanged. The formula is a contour-invariant analytic
identity.
\end{remark}

\section{Closed-form density via Fox $H$}
\label{sec:foxH_density}
% -----------------------------------------------------------------

\begin{definition}[Fox $H$-function]\label{def:foxH}
For nonnegative integers $m,n,p,q$ with $0\le n\le p$, $0\le m\le q$,
\begin{equation}\label{eq:foxH_def}
H^{m,n}_{p,q}\!\left[z\,\bigg|\,\begin{array}{c}(a_i,\alpha_i)_1^p\\(b_j,\beta_j)_1^q\end{array}\right]
:=\frac{1}{2\pi i}\int_{\mathcal{L}}
\frac{\prod_{j=1}^m\Gamma(b_j+\beta_js)\prod_{i=1}^n\Gamma(1-a_i-\alpha_is)}
{\prod_{j=m+1}^q\Gamma(1-b_j-\beta_js)\prod_{i=n+1}^p\Gamma(a_i+\alpha_is)}\,z^{-s}\,ds.
\end{equation}
\end{definition}

\begin{theorem}[Closed-form density]\label{thm:foxH_density}
With PFE
\begin{equation}\label{eq:PFE}
\Phi_M(u,T)=-\tfrac12\sigma_T^2u^2-i\mu_Tu+\sum_{j=1}^{2M}\frac{A_j(T)}{u-u_j(T)},
\end{equation}
$\sigma_T^2>0$, $\mu_T\in\R$, $u_j\notin\R$,
$\epsilon_j:=-\sgn(\Imag u_j(T))$, the density
$f_M(x,T):=\Fc^{-1}[\phi_M(\cdot,T)](x)$ admits
\begin{equation}\label{eq:foxH_density}
\boxed{\;f_M(x,T)=\sum_{S\in\Pc(\{1,\ldots,2M\})}d_S(T)\,
H^{m_S^\circ,n_S^\circ}_{p_S^\circ,q_S^\circ}\!\left[\eta_S(x,T)\,\bigg|\,
\begin{array}{c}(\mathbf{a}_S^\circ,\boldsymbol{\alpha}_S^\circ)\\(\mathbf{b}_S^\circ,\boldsymbol{\beta}_S^\circ)\end{array}\right]\;}
\end{equation}
\begin{equation}\label{eq:density_counts}
(m_S^\circ,n_S^\circ,p_S^\circ,q_S^\circ)=(|S|+1,|S|,|S|+1,2|S|+1),
\end{equation}
\begin{equation}\label{eq:density_prefactors}
d_S(T)=\frac{1}{\sqrt{2\pi\sigma_T^2}}\prod_{j\in S}\epsilon_j\sqrt{A_j(T)/i},
\end{equation}
\begin{equation}\label{eq:density_argument}
\eta_S(x,T)=\frac{(x-\mu_T)\,\mathbf{1}_{\bigwedge_{j\in S}\epsilon_j(x-\mu_T)>0}}{\sqrt{2\sigma_T^2}}\prod_{j\in S}(iA_j(T))\cdot\exp\!\left(-i\sum_{j\in S}u_j(T)(x-\mu_T)\right).
\end{equation}
The parameter vectors are
\begin{equation}\label{eq:density_params}
\begin{aligned}
\mathbf{a}_S^\circ &= (\underbrace{0,\ldots,0}_{|S|}),
&\boldsymbol{\alpha}_S^\circ &= (\underbrace{1,\ldots,1}_{|S|}),\\
\mathbf{b}_S^\circ &= (\underbrace{0}_{\text{Gauss}},\underbrace{0,\ldots,0}_{|S|},\underbrace{-1,\ldots,-1}_{|S|}),
&\boldsymbol{\beta}_S^\circ &= (\tfrac12,\underbrace{1,\ldots,1}_{|S|},\underbrace{1,\ldots,1}_{|S|}),
\end{aligned}
\end{equation}
all entries constant in $j\in S$. The $j$-dependent data
$\{A_j(T),u_j(T):j\in S\}$ enters only through $d_S(T)$ and
$\eta_S(x,T)$, both symmetric in $S$.
\end{theorem}
\begin{proof}
\textit{Step 1.} Absolutely convergent inverse Fourier transform
of the right side of~\eqref{eq:PFE} by Lemma~\ref{lem:decay}.

\textit{Step 2.} $\Fc^{-1}[\exp(A_j/(u-u_j))](x)=\delta(x)+B_j(x;T)$
with
\begin{equation}\label{eq:Bj_explicit}
B_j(x;T)=\epsilon_je^{-iu_jx}\mathbf{1}_{\epsilon_jx>0}\sqrt{A_j/(ix)}\,J_1(2\sqrt{iA_jx}),
\end{equation}
from $\Fc^{-1}[(u-u_j)^{-n}](x)$ and the Bessel generating
function.

\textit{Step 3.} Centring $\xi:=x-\mu_T$;
$g_0(x)=(2\pi\sigma_T^2)^{-1/2}\exp(-\xi^2/(2\sigma_T^2))$.

\textit{Step 4.}
$f_M(x,T)=\sum_S\bigl(g_0\,\ast\,\bigast_{j\in S}B_j(\cdot;T)\bigr)(x)$.

\textit{Step 5.} Mellin transforms: Gaussian rescaled gives
$\Mc[g_0](s)\propto\Gamma(s/2)$ ($m$-class $(0,1/2)$); each
$B_j$ contributes $\Gamma(s)$ ($m$-class $(0,1)$) and $\Gamma(2-s)$
($q$-class $(-1,1)$), independent of $j$ in form.

\textit{Step 6.} Identification with Fox $H$ via
Definition~\ref{def:foxH} gives~\eqref{eq:density_counts} and
\eqref{eq:density_params}.

\textit{Step 7.}
$\prod_{j\in S}\mathbf{1}_{\epsilon_j\xi>0}=\mathbf{1}_{\bigwedge_{j\in S}\epsilon_j\xi>0}$.

\textit{Step 8.} $d_S(T)$ and $\eta_S(x,T)$ from Gaussian
normalisation, Bessel amplitudes, $\xi$ shift,
$\sqrt{2\sigma_T^2}$ rescaling, $iA_j(T)$ Bessel arguments,
combined exponential prefactor $\exp(-i\sum_{j\in S}u_j\xi)$.
\end{proof}

\begin{corollary}[Newton--Girard symmetric collapse]\label{cor:density_symmetric}
The $j$-dependent data in~\eqref{eq:density_argument} enters
through $\prod_{j\in S}(iA_j(T))$ and
$\exp(-i\sum_{j\in S}u_j(T)\xi)$, both symmetric in
$\{j\in S\}$. The level-$r$ consolidations are
\begin{equation}\label{eq:NG_amplitude}
\sum_{|S|=r}\prod_{j\in S}(iA_j(T))=i^r\,e_r\bigl(A_1(T),\ldots,A_{2M}(T)\bigr),
\end{equation}
\begin{equation}\label{eq:NG_phase}
\sum_{|S|=r}\exp\!\Bigl(-i\sum_{j\in S}u_j(T)\xi\Bigr)
=e_r\bigl(e^{-iu_1(T)\xi},\ldots,e^{-iu_{2M}(T)\xi}\bigr),
\end{equation}
where $e_r$ is the $r$-th elementary symmetric polynomial in
$2M$ variables, computable in $O(M^2)$ by the recurrence
$e_r(x_1,\ldots,x_n)=e_r(x_1,\ldots,x_{n-1})+x_ne_{r-1}(x_1,\ldots,x_{n-1})$,
or in $O(M\log^2M)$ by FFT-based polynomial multiplication of
$\prod_j(1+tx_j)$ with coefficient extraction. The
$2^{2M}$-subset sum~\eqref{eq:foxH_density} reduces to $2M+1$
symmetric groups, evaluable in $O(M\log^2M)$ overall.
\end{corollary>

\begin{corollary}[Tail asymptotics]\label{cor:tail}
For $|x|\to\infty$, $\xi=x-\mu_T$,
\begin{equation}\label{eq:tail}
f_M(x,T)\sim D_T\,\xi^{-3/2}\exp\!\left(-\frac{\xi^2}{2\sigma_T^2}-i\sum_{j=1}^{2M}u_j(T)\xi+2\sqrt{i\xi\sum_{j=1}^{2M}A_j(T)}\right).
\end{equation}
\end{corollary>

\section{Closed-form European call price via Fox $H$}
\label{sec:foxH_price}
% -----------------------------------------------------------------

\begin{theorem}[Closed-form European call price]\label{thm:foxH_price}
$\tilde k:=\log(K/S_0)-rT$.
\begin{equation}\label{eq:foxH_price}
\boxed{\;C_M(K,T)=S_0\sum_{S\in\Pc(\{1,\ldots,2M\})}c_S(T)\,
H^{m_S,n_S}_{p_S,q_S}\!\left[\zeta_S(K,T)\,\bigg|\,
\begin{array}{c}(\mathbf{a}_S,\boldsymbol{\alpha}_S)\\(\mathbf{b}_S,\boldsymbol{\beta}_S)\end{array}\right]\;}
\end{equation}
$(m_S,n_S,p_S,q_S)=(|S|+1,|S|+1,|S|+2,2|S|+2)$,
\begin{align}
c_S(T)&=\frac{1}{\sqrt{2\pi\sigma_T^2}}\prod_{j\in S}\epsilon_j\sqrt{A_j(T)/i},
\label{eq:price_prefactor}\\
\zeta_S(K,T)&=\frac{(K/S_0)e^{-rT-\mu_T}}{\sqrt{2\sigma_T^2}}\prod_{j\in S}(iA_j(T)).
\label{eq:price_argument}
\end{align}
Parameter vectors extend~\eqref{eq:density_params} by appending
two fixed entries (independent of $S,j$) — $(a,\alpha)=(0,1)$
in the numerator and $(b,\beta)=(1,1)$ in the denominator —
from~\eqref{eq:Hint_const}. After Newton--Girard
(Corollary~\ref{cor:density_symmetric}) the sum reduces to
$2M+1$ symmetric groups, evaluable in $O(M\log^2M)$.
\end{theorem}

\begin{proof}[Proof I: convolution chain]
Theorem~\ref{thm:foxH_density} gives~\eqref{eq:foxH_density}.
Payoff splits as
$(S_T-K)^+=S_0e^{x+rT}\mathbf{1}_{x>\tilde k}-K\mathbf{1}_{x>\tilde k}$.
Per Fox $H$-summand,
\begin{equation}\label{eq:Hint_const}
\int_{\tilde k}^\infty H^{m,n}_{p,q}\!\bigl[\zeta x\big|\cdots\bigr]dx
=\zeta^{-1}H^{m+1,n}_{p+1,q+1}\!\bigl[\zeta\tilde k\big|(0,1),\ldots,(1,1)\bigr],
\end{equation}
adding one $\Gamma$-factor each in $(m,p,q)$. Mellin shift
$\Mc[e^xf](s)=\Mc[f](s)$ followed by~\eqref{eq:Hint_const} adds
one in $n$. Increment $(m_S^\circ,n_S^\circ,p_S^\circ,q_S^\circ)\to(m_S,n_S,p_S,q_S)$
by $(1,1,1,1)$ giving $(|S|+1,|S|+1,|S|+2,2|S|+2)$. Discount
and prefactors combine into $c_S(T),\zeta_S(K,T)$. Newton--Girard
reduces $2^{2M}$ to $2M+1$; residue series $O(n!^{-1})$ gives
$O(M\log^2M)$ per price.
\end{proof}

\begin{proof}[Proof II: direct Mellin--Barnes inversion]
\textit{Step 1: Strip hypothesis.}
Assume~\eqref{eq:strip_hypothesis} for some $c\in\R$
(admissibility: Proposition~\ref{prop:strip}). Set
$\alpha:=-c-1$. Lemma~\ref{lem:decay} ensures absolute
integrability of $\phi_M(\cdot+ci,T)$ on $\R$ via Gaussian
envelope. The Lewis contour $\Real s=c$ lies in
$\bigcap_j\{\Real s>\Real\tilde u_j\}=\bigcap_j\{\Real s>\Imag u_j\}$
by~\eqref{eq:strip_hypothesis}.
By Remark~\ref{rem:cauchy_independence} the resulting formula
is independent of $c$ within the admissible region.

\textit{Step 2: Lewis representation.}
\begin{equation}\label{eq:lewis}
C_M(K,T)=\frac{S_0e^{-rT}}{2\pi i}\int_{c-i\infty}^{c+i\infty}\frac{\phi_M(iw,T)e^{-w\tilde k}}{w(w-1)}\,dw.
\end{equation}

\textit{Step 3: Mellin form.} $y:=e^{-\tilde k}=(S_0/K)e^{rT}$,
\begin{equation}\label{eq:mellin_form}
C_M(K,T)=S_0e^{-rT}\cdot\frac{1}{2\pi i}\int_{\mathcal{L}}\widetilde\Phi_M(s,T)y^{-s}ds,
\quad \widetilde\Phi_M(s,T):=\frac{\phi_M(is,T)}{s(s-1)}.
\end{equation}

\textit{Step 4.} $\tilde A_j:=-iA_j$, $\tilde u_j:=-iu_j$,
\begin{equation}\label{eq:phi_is}
\phi_M(is,T)=e^{\tfrac12\sigma_T^2s^2+\mu_Ts}\prod_{j=1}^{2M}\exp(\tilde A_j/(s-\tilde u_j)).
\end{equation}

\textit{Step 5.}
\begin{equation}\label{eq:bessel_resum}
\sum_{n\ge 1}\frac{c^n}{n!(s-u)_n}=\frac{\Gamma(s-u)I_{s-u-1}(2\sqrt{c})}{c^{(s-u-1)/2}}-1
\end{equation}
converges absolutely on $\Real(s-u)>0$. Strip hypothesis places
the Lewis contour strictly inside the simultaneous convergence
region; no residues crossed, termwise resummation licensed.

\textit{Step 6.} Multi-series with $S=\{j:n_j\ge 1\}$,
combination with Gaussian Mellin transform
$\Gamma(s/2+\tfrac12)$ and payoff factor $1/(s(s-1))$.

\textit{Step 7.} Identification as Fox $H$ with
$(|S|+1,|S|+1,|S|+2,2|S|+2)$, matching Proof~I.

\textit{Step 8.} Constants and argument identical to
\eqref{eq:price_prefactor}, \eqref{eq:price_argument}.

\textit{Step 9.} Independence: Proof~I in $x$-space via half-line
Mellin convolution; Proof~II in Fourier/Mellin space via Lewis +
Bessel resummation. No shared intermediate machinery.
\end{proof}

\begin{corollary}[Two-route verification]\label{cor:two_routes}
\eqref{eq:foxH_price} is established by two analytically
independent routes converging on identical
$(m_S,n_S,p_S,q_S),c_S(T),\zeta_S(K,T)$, parameter vectors. The
representation is independent of derivation technique.
\end{corollary>

\begin{corollary}[Quadrature-free pricing]\label{cor:no_quadrature}
$C_M(K,T)$ is a finite sum of Fox $H$ evaluations, each via
absolutely convergent residue series with $n$-th term
$O(n!^{-1})$, no Fourier inversion quadrature, no cosine series,
no truncation interval.
\end{corollary>

\section{COS pricing as alternative}
\label{sec:cos}
% -----------------------------------------------------------------

\subsection{Cumulants}

$\psi_M(s,T):=\Phi_M(-is,T)$,
\begin{equation}\label{eq:cumulants}
\kappa_n^{(M)}(T):=\frac{d^n\psi_M(s,T)}{ds^n}\bigg|_{s=0},
\quad n=1,2,3,4.
\end{equation}

\subsection{Truncation interval}

\begin{equation}\label{eq:interval}
[a(T),b(T)]:=\Bigl[\kappa_1^{(M)}-L\sqrt{\kappa_2^{(M)}+\sqrt{\kappa_4^{(M)}}},\,
\kappa_1^{(M)}+L\sqrt{\kappa_2^{(M)}+\sqrt{\kappa_4^{(M)}}}\Bigr],
\quad L\in[8,12].
\end{equation}

\begin{remark}[Single-interval term structure]\label{rem:single_interval}
$\kappa_2^{(M)}(T),\kappa_4^{(M)}(T)$ nondecreasing in $T$, so
$[a,b]:=[a(T_{\max}),b(T_{\max})]$ is admissible for every
$T\in(0,T_{\max}]$.
\end{remark}

\subsection{Payoff coefficients}

$w(x)=K(e^x-1)^+$,
$\widetilde U_n=\frac{2}{b-a}[\chi_n(0,b)-\psi_n(0,b)]$,
\begin{align}
\chi_n(c,d)&=\frac{1}{1+(n\pi/(b-a))^2}\Bigl[\cos(n\pi(d-a)/(b-a))e^d-\cos(n\pi(c-a)/(b-a))e^c
\nonumber\\
&\quad+\tfrac{n\pi}{b-a}(\sin(n\pi(d-a)/(b-a))e^d-\sin(n\pi(c-a)/(b-a))e^c)\Bigr],\\
\psi_n(c,d)&=\begin{cases}d-c,&n=0,\\
\frac{b-a}{n\pi}[\sin(n\pi(d-a)/(b-a))-\sin(n\pi(c-a)/(b-a))],&n\ge 1.\end{cases}
\end{align}

\subsection{COS price with Gaussian-rate error}

\begin{theorem}[COS pricing]\label{thm:price}
$x_0:=\log(S_0/K)+rT$,
\begin{equation}\label{eq:CMN}
C_{M,N}:=Ke^{-rT}\sum_{n=0}^{N-1}\!{}^\prime\,\Real\!\left[\phi_M\!\left(\tfrac{n\pi}{b-a},T\right)\exp\!\left(in\pi\tfrac{x_0-a}{b-a}\right)\right]\widetilde U_n.
\end{equation}
$C_{M,N}\to C_M$ as $N\to\infty$, with Gaussian-rate truncation
\begin{equation}\label{eq:cos_error}
|C_M-C_{M,N}|\le C'\exp\!\left(-\tfrac12\sigma_T^2(N\pi/(b-a))^2\right).
\end{equation}
\end{theorem}
\begin{proof}
Cosine expansion of $f_M$ on $[a,b]$, Parseval against the
cosine basis, Lemma~\ref{lem:decay} bounds the tail by its first
term, yielding~\eqref{eq:cos_error}.
\end{proof}

\section{Algorithm and computational summary}
\label{sec:algorithm}
% -----------------------------------------------------------------

For each strike $K$ and Fourier node $v$ dual to $\log K$:
\begin{enumerate}
\item M\"untz coefficients $\{a_k(v)\}_{k=1}^{2M}$
from~\eqref{eq:ak}, $O(M^2)$ per node
(Remark~\ref{rem:index}).
\item Cumulant moments $\{M_k(v)\}_{k=0}^{2M-1}$
from~\eqref{eq:cumulant_moments}, $O(M)$.
\item Chebyshev algorithm for
$\{\alpha_k,\beta_k,h_k,p_k\}_{k=0}^M$ in $O(M^2)$; assemble
$Q_M(z;v),P_M(z;v)$ via Lemma~\ref{lem:match}.
\item Per-$T$ specialisation $z=T^\mu$, $O(M)$ per $T$.
\item PFE: $\{u_j(T),A_j(T)\}_{j=1}^{2M},\sigma_T,\mu_T$ by
cover-up, $O(M^2)$ per $T$.
\item Pricing endpoint:
\begin{enumerate}
\item Fox $H$: \eqref{eq:foxH_price} via Newton--Girard
(Corollary~\ref{cor:density_symmetric}) and residue summation,
$O(M\log^2M)$ per $(K,T)$, no quadrature.
\item COS: cumulants~\eqref{eq:cumulants}, $[a,b]$
via~\eqref{eq:interval} or Remark~\ref{rem:single_interval},
sum~\eqref{eq:CMN}, $O(NM)$ per $(K,T)$.
\end{enumerate}
\end{enumerate}
Front-end (steps 1--3) is $O(M^2)$ per strike, amortised across
the entire term structure since the Chebyshev construction in
$\C[v]$ is $T$-independent.

\section{Summary}
\label{sec:summary}
% -----------------------------------------------------------------

\textbf{T1.} Pad\'{e} resummation of the cumulant generating
function performed directly on $\{M_k(v)\}$ via Chebyshev
algorithm in arbitrary precision, bypassing numerical integration
of $h(v,t)$. Orthogonal polynomial sequence produced by
recurrence in $O(M^2)$ ring operations, no Hankel matrix
inversion. Stahl convergence (Theorem~\ref{thm:conv}) extends
beyond the Puiseux radius. Supremum-ratio a-posteriori bound
(Theorem~\ref{thm:aposteriori}) fixes $M$ from precision.

\textbf{T2.} Density (Theorem~\ref{thm:foxH_density}) and call
price (Theorem~\ref{thm:foxH_price}) admit strictly closed-form
Fox $H$ representations with explicit symmetric parameter
vectors~\eqref{eq:density_params}, evaluable in $O(M\log^2M)$ per
evaluation with no quadrature, established by two analytically
independent routes (Corollary~\ref{cor:two_routes}). Pad\'{e}
characteristic function has Gaussian real-axis decay
(Lemma~\ref{lem:decay}). Strip admissibility for the
Mellin--Barnes contour is Proposition~\ref{prop:strip}; the
formula is contour-invariant
(Remark~\ref{rem:cauchy_independence}).

\textbf{T3.} COS pricing (Theorem~\ref{thm:price}) on rational
$\Phi_M(\cdot,T)$ provides a fallback endpoint of cost $O(NM)$
per $(K,T)$ with Gaussian-rate truncation error
$\exp(-\tfrac12\sigma_T^2(N\pi/(b-a))^2)$.
