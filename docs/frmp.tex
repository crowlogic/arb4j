\title{The Fractional Riccati--M\"untz--Pad\'{e} Theorem\\
\large Convergence and Rough Heston Option Pricing via COS}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Let $\mu\in(0,1]$ and let $y$ be the unique solution of the
constant-coefficient fractional Riccati initial-value problem
$D^{\mu}y = c_0 + c_1 y + c_2 y^2$, $I^{1-\mu}y(0)=0$.
We prove two things. First (Theorem~\ref{thm:conv}), the
diagonal $[M/M]$ Pad\'{e} approximant of the generating function
$g(z)=y(z^{1/\mu})$ converges to $g$ uniformly on every compact
subset of $[0,\infty)$ on which $y$ is finite, with $M$ fixed
automatically by the working precision. Second
(Theorem~\ref{thm:price}), European call prices under the rough
Heston model are computed by applying the Fang--Oosterlee COS
method to the resulting rational characteristic exponent
$\Phi_M(\cdot,T)$, with the COS series length $N$ also fixed
automatically by the working precision.
\end{abstract}

\section{Fractional calculus and the M\"untz basis}
\label{sec:fc}
% -----------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_0^t (t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$,
\begin{equation}\label{eq:power}
\Imu t^s = \frac{\Gamma(s+1)}{\Gamma(s+\mu+1)}\,t^{s+\mu}.
\end{equation}
\end{lemma}
\begin{proof}
Substitute $\sigma=t\tau$ in~\eqref{eq:RL_I}:
\[
\Imu t^s
=\frac{1}{\Gamma(\mu)}\int_0^t(t-\sigma)^{\mu-1}\sigma^s\,d\sigma
=\frac{t^{s+\mu}}{\Gamma(\mu)}\int_0^1(1-\tau)^{\mu-1}\tau^s\,d\tau
=\frac{t^{s+\mu}\,B(\mu,s+1)}{\Gamma(\mu)}.
\]
Using $B(\mu,s+1)=\Gamma(\mu)\Gamma(s+1)/\Gamma(s+\mu+1)$
gives~\eqref{eq:power}.
\end{proof}

The M\"untz lattice $\{t^{k\mu}\}_{k\ge 0}$ is closed under $\Imu$
by the power rule and closed under multiplication because
$t^{k\mu}\cdot t^{\ell\mu}=t^{(k+\ell)\mu}$. In the variable
$z=t^\mu$ every element is simply $z^k$, so ordinary power-series
arithmetic applies.

\section{The M\"untz--Tau recurrence}
\label{sec:recurrence}
% -----------------------------------------------------------------

Define $\gamma_k := \Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$ for $k\ge 1$,
so that
\begin{equation}\label{eq:gamma}
\Imu t^{(k-1)\mu} = \gamma_k\, t^{k\mu}.
\end{equation}

\begin{theorem}[M\"untz--Tau recurrence]\label{thm:recurrence}
The unique formal power-series solution of
\begin{equation}\label{eq:IVP}
y = \Imu[c_0 + c_1 y + c_2 y^2], \qquad y(0)=0,
\end{equation}
is
\begin{equation}\label{eq:series}
y(t)=\sum_{k=1}^{\infty} a_k\, t^{k\mu},
\end{equation}
with
\begin{align}
a_1 &= \frac{c_0}{\Gamma(\mu+1)},
\label{eq:a1}\\
a_k &= \gamma_k\!\left[c_1\,a_{k-1}
+c_2\!\sum_{j=1}^{k-2}a_j\,a_{k-1-j}\right],\qquad k\ge 2.
\label{eq:ak}
\end{align}
\end{theorem}

\begin{proof}
Substitute~\eqref{eq:series} into~\eqref{eq:IVP} and apply
\eqref{eq:gamma} termwise. Matching $t^\mu$ gives~\eqref{eq:a1};
for $k\ge 2$, $[t^{k\mu}]\Imu(c_1 y)=c_1\gamma_k a_{k-1}$ and
$[t^{k\mu}]\Imu(c_2 y^2)=c_2\gamma_k\sum_{j=1}^{k-2}a_j a_{k-1-j}$
by the Cauchy product, giving~\eqref{eq:ak}. Uniqueness is
immediate from the recurrence.
\end{proof}

\section{Pad\'{e} via Chebyshev--Wheeler}
\label{sec:pade}
% -----------------------------------------------------------------

Set $g(z):=y(z^{1/\mu})=\sum_{k=1}^{\infty}a_k z^k$ and
$\hat g(z):=g(z)/z=\sum_{k=0}^{\infty}m_k z^k$ with
$m_k:=a_{k+1}$. The diagonal $[M/M]$ Pad\'{e} approximant of
$\hat g$ is the unique pair $(P_M,Q_M)$ of polynomials with
$\deg P_M\le M-1$, $\deg Q_M\le M$, $Q_M(0)=1$, satisfying
\begin{equation}\label{eq:pade_match}
Q_M(z)\hat g(z) - P_M(z) = O(z^{2M}).
\end{equation}

\subsection{Chebyshev--Wheeler algorithm}

Given Hankel moments $m_0,\ldots,m_{2M-1}$, the Pad\'{e}
$J$-fraction coefficients $(\alpha_n,\beta_n)$ and the squared
norms $h_n$ are produced by Chebyshev's algorithm, a pure cross-recurrence
on a single two-dimensional table $\sigma_{n,k}$ with no
polynomial evaluations and no linear system.

\begin{enumerate}
\item Set $\sigma_{-1,k}:=0$ and $\sigma_{0,k}:=m_k$ for
$k=0,\ldots,2M-1$, and
\begin{equation}\label{eq:cw_init}
\alpha_0:=\frac{m_1}{m_0},\qquad \beta_0:=m_0,\qquad h_0:=m_0.
\end{equation}
\item For $n=1,\ldots,M-1$ and $k=n,\ldots,2M-1-n$,
\begin{equation}\label{eq:cw_sigma}
\sigma_{n,k}:=\sigma_{n-1,k+1}-\alpha_{n-1}\,\sigma_{n-1,k}
-\beta_{n-1}\,\sigma_{n-2,k},
\end{equation}
and then
\begin{equation}\label{eq:cw_extract}
\alpha_n:=\frac{\sigma_{n,n+1}}{\sigma_{n,n}}
-\frac{\sigma_{n-1,n}}{\sigma_{n-1,n-1}},\qquad
\beta_n:=\frac{\sigma_{n,n}}{\sigma_{n-1,n-1}},\qquad
h_n:=\sigma_{n,n}.
\end{equation}
\end{enumerate}
The cost is $O(M^2)$.

\subsection{Pad\'{e} from $(\alpha_n,\beta_n)$}

The numerator $P_M(z)$ and denominator $Q_M(z)$ are read directly
off the $J$-fraction
\begin{equation}\label{eq:Jfrac}
\hat g(z)=\cfrac{m_0}{1-\alpha_0 z-\cfrac{\beta_1 z^2}
{1-\alpha_1 z-\cfrac{\beta_2 z^2}{1-\alpha_2 z-\cdots}}},
\end{equation}
truncated after $M$ levels, by the standard convergent recurrences
\begin{align}
\label{eq:conv_P}
P_0&=0,\quad P_1=m_0,\quad
P_{n+1}(z)=(1-\alpha_n z)P_n(z)-\beta_n z^2 P_{n-1}(z),\\
\label{eq:conv_Q}
Q_0&=1,\quad Q_1=1-\alpha_0 z,\quad
Q_{n+1}(z)=(1-\alpha_n z)Q_n(z)-\beta_n z^2 Q_{n-1}(z).
\end{align}
The pair $(P_M,Q_M)$ obtained from~\eqref{eq:conv_P}--\eqref{eq:conv_Q}
satisfies $Q_M(0)=1$ and~\eqref{eq:pade_match}; this is the
$M$-th convergent of~\eqref{eq:Jfrac} and equals the diagonal
$[M/M]$ Pad\'{e} of $\hat g$.

\section{Uniform convergence and precision-driven $M$}
\label{sec:conv}
% -----------------------------------------------------------------

\begin{theorem}[Convergence and precision-driven order]\label{thm:conv}
Let $T^*\in(0,\infty]$ be the supremum of $T$ such that $y$ is
finite on $[0,T)$. Set $g(z)=y(z^{1/\mu})$ and let $R_M=P_M/Q_M$
be the diagonal $[M/M]$ Pad\'{e} approximant of $\hat g(z)=g(z)/z$,
computed via Chebyshev--Wheeler. Then:
\begin{enumerate}
\item\label{it:unif}
$z R_M(z)\to g(z)$ uniformly on every compact
$[0,\rho]\subset[0,(T^*)^\mu)$.
\item\label{it:post}
With $\Delta_M:=zR_M-zR_{M-1}$ and $r_M:=|\Delta_M|/|\Delta_{M-1}|<1$,
\begin{equation}\label{eq:aposteriori}
|g(z)-zR_M(z)|\le\frac{|\Delta_M(z)|^2}{|\Delta_{M-1}(z)|-|\Delta_M(z)|}.
\end{equation}
\item\label{it:auto_M}
Working in $b$-bit floating-point arithmetic with unit roundoff
$\eta=2^{-b}$, the order $M$ is fixed automatically as the
smallest $M$ such that the right-hand side
of~\eqref{eq:aposteriori} is at most $\eta$. No tunable
parameter is involved.
\end{enumerate}
\end{theorem}
\begin{proof}
By the Volterra fixed-point theorem on~\eqref{eq:IVP}, $y$ extends
holomorphically to a neighbourhood of $[0,T^*)$, and $g(z)$ is
holomorphic on a slit domain omitting a compact set
$K\subset\C\setminus[0,(T^*)^\mu)$ of algebraic singularities.
By Stahl~\cite{StahlBranch}, the diagonal denominators $Q_M$ have
zeros converging in capacity to $K$, and
$|g(z)-zR_M(z)|^{1/(2M)}\to e^{-\mathsf{g}(z,K)}$ in capacity on
$\C\setminus K$, where $\mathsf{g}(z,K)$ is the Green function
with pole at infinity. A neighbourhood of $[0,(T^*)^\mu)$ is free
of $Q_M$-zeros for all large $M$; normal-families upgrades capacity
convergence to uniform convergence on compacta of
$[0,(T^*)^\mu)$, proving~\ref{it:unif}.

The differences $\Delta_M=zR_M-zR_{M-1}$ satisfy
$\Delta_M(z)=O(z^{2M-1})$ from~\eqref{eq:pade_match}; the maximum
principle gives $|\Delta_{M+1}|/|\Delta_M|\to e^{-2\mathsf{g}(z,K)}<1$,
so the geometric tail bound $|g-zR_M|\le|\Delta_M|/(1-r_M)$
yields~\eqref{eq:aposteriori}, proving~\ref{it:post}.

For~\ref{it:auto_M}, the right-hand side of~\eqref{eq:aposteriori}
decreases at the geometric rate $e^{-2M\mathsf{g}(z,K)}$, so the
stopping condition ``$\le\eta$'' selects $M$ uniquely once $\eta$
is fixed by the working precision.
\end{proof}

\section{Rough Heston specialisation}
\label{sec:heston}
% -----------------------------------------------------------------

The rough Heston model~\cite{ElEuchRosenbaum} has parameters
$\theta,\lambda,\rho,\nu,V_0,H\in\R$ with $H\in(0,\tfrac12)$ and
$\mu:=H+\tfrac12\in(\tfrac12,1)$. The log-return characteristic
function is
\begin{equation}\label{eq:heston_cf}
\phi_T(v):=\mathbb{E}[e^{ivX_T}]=\exp\bigl(\Phi(v,T)\bigr),
\qquad X_T:=\log(S_T/S_0)-rT,
\end{equation}
\begin{equation}\label{eq:Phi}
\Phi(v,T)=\theta\lambda\int_0^T h(v,s)\,ds
+ V_0\,\Ico h(v,\cdot)(T),
\end{equation}
where $h(v,\cdot)$ solves the fractional Riccati IVP
\begin{equation}\label{eq:heston_IVP}
h(v,t)=\Imu\bigl[P(v)+Q(v)h(v,t)+R\,h(v,t)^2\bigr],
\end{equation}
with
\begin{equation}\label{eq:PQR}
P(v)=\tfrac12(-v^2-iv),\quad
Q(v)=\lambda(iv\rho\nu-1),\quad
R=\tfrac12\lambda^2\nu^2.
\end{equation}

The Pad\'{e}--M\"untz approximant at order $M$ is
$h_M(v,t):=R_M^{(v)}(t^\mu)\cdot t^\mu$, with $R_M^{(v)}$ the
$[M/M]$ Pad\'{e} from Section~\ref{sec:pade} applied to the
$v$-parametrised moments. The Pad\'{e} characteristic exponent is
\begin{equation}\label{eq:PhiM}
\Phi_M(v,T)=\theta\lambda\int_0^T h_M(v,s)\,ds
+V_0\,I^{1-\mu}h_M(v,\cdot)(T),
\end{equation}
which is rational in $v$ at fixed $T$, and
$\phi_M(v,T):=\exp(\Phi_M(v,T))$.

\begin{lemma}[Real-axis decay]\label{lem:decay}
For $\mu\in(\tfrac12,1)$ and all $T>0$,
$|\phi_M(u,T)|\le e^{-c|u|^{2\mu}}$ for some $c>0$ and all
sufficiently large $|u|$ on the real axis.
\end{lemma}
\begin{proof}
$\mathrm{Re}\,\Phi(u,T)\to-\infty$ at rate $|u|^{2\mu}$ along the
real axis (El Euch--Rosenbaum~\cite{ElEuchRosenbaum}, Prop.~3.2).
Theorem~\ref{thm:conv} gives uniform convergence
$\Phi_M\to\Phi$ on compacta, and Stahl rational-approximation
theory matches the tail behaviour for $M$ sufficiently large.
\end{proof}

\section{COS pricing of European calls}
\label{sec:cos}
% -----------------------------------------------------------------

\subsection{Cumulants of $X_T$}

With $\psi_M(s,T):=\Phi_M(-is,T)$,
\begin{equation}\label{eq:cumulants}
\kappa_n^{(M)}(T):=\left.\frac{d^n\psi_M(s,T)}{ds^n}\right|_{s=0}
=(-i)^n\left.\frac{d^n\Phi_M(v,T)}{dv^n}\right|_{v=0}.
\end{equation}
At fixed $T$, $\Phi_M(v,T)=N(v,T)/D(v,T)$ is a ratio of
polynomials in $v$ with $D(0,T)=1$. The Leibniz identity
$N^{(n)}=\sum_{k=0}^n\binom{n}{k}\Phi_M^{(k)}D^{(n-k)}$ solved for
the top term gives
\begin{equation}\label{eq:leibniz_solve}
\Phi_M^{(n)}(v,T)=\frac{1}{D(v,T)}\!\left[N^{(n)}(v,T)
-\sum_{k=0}^{n-1}\binom{n}{k}\Phi_M^{(k)}(v,T)\,D^{(n-k)}(v,T)\right],
\end{equation}
evaluated at $v=0$, where $N^{(n)}(0,T)$ and $D^{(n)}(0,T)$ are
$n!$ times the coefficient of $v^n$ in $N$ and $D$. Hence
\begin{equation}\label{eq:kappa_closed}
\kappa_n^{(M)}(T)=(-i)^n\,\Phi_M^{(n)}(0,T).
\end{equation}

\subsection{Truncation interval}

The Fang--Oosterlee interval is determined by the cumulants and
the working precision $\eta=2^{-b}$. Set
\begin{equation}\label{eq:interval}
[a(T),b(T)]:=\bigl[\kappa_1^{(M)}(T)-W(T),\;
\kappa_1^{(M)}(T)+W(T)\bigr],
\end{equation}
where the half-width $W(T)$ is the unique positive root of
\begin{equation}\label{eq:width_root}
e^{-c\,W(T)^{2\mu}}=\eta,
\qquad\text{i.e.}\qquad
W(T)=\bigl(c^{-1}\log(1/\eta)\bigr)^{1/(2\mu)},
\end{equation}
with $c$ the constant of Lemma~\ref{lem:decay}. By
Lemma~\ref{lem:decay}, $|\phi_M(u,T)|\le\eta$ for $|u|\ge W(T)$,
so the residual mass of $f_M$ outside $[a(T),b(T)]$ is at most
$\eta$ by Fourier inversion. No free parameter enters.

\subsection{Payoff cosine coefficients}

For the call payoff $w(x):=K(e^x-1)^+$ on
$x:=\log(S_T/K)\in[a,b]$, the COS payoff coefficients are
\begin{align}
\chi_n(c,d)&:=\int_c^d e^x\cos\!\left(n\pi\frac{x-a}{b-a}\right)dx
\nonumber\\
&=\frac{1}{1+\bigl(\tfrac{n\pi}{b-a}\bigr)^2}\!
\left[\cos\!\left(\tfrac{n\pi(d-a)}{b-a}\right)e^d
-\cos\!\left(\tfrac{n\pi(c-a)}{b-a}\right)e^c\right.\nonumber\\
&\qquad\left.+\tfrac{n\pi}{b-a}\sin\!\left(\tfrac{n\pi(d-a)}{b-a}\right)e^d
-\tfrac{n\pi}{b-a}\sin\!\left(\tfrac{n\pi(c-a)}{b-a}\right)e^c\right],
\label{eq:chi}\\
\psi_n(c,d)&:=\int_c^d\cos\!\left(n\pi\frac{x-a}{b-a}\right)dx
=\begin{cases}d-c,&n=0,\\
\dfrac{b-a}{n\pi}\!\left[\sin\!\left(\tfrac{n\pi(d-a)}{b-a}\right)
-\sin\!\left(\tfrac{n\pi(c-a)}{b-a}\right)\right],&n\ge1,\end{cases}
\label{eq:psi}
\end{align}
and $\widetilde U_n:=\tfrac{2}{b-a}[\chi_n(0,b)-\psi_n(0,b)]$.

\subsection{COS price and precision-driven $N$}

\begin{theorem}[COS pricing]\label{thm:price}
With $[a,b]$ from~\eqref{eq:interval}, the $N$-term COS truncation
\begin{equation}\label{eq:CMN}
C_{M,N}:=Ke^{-rT}\sum_{n=0}^{N-1}\!{}^\prime\,
\Real\!\left[
\phi_M\!\left(\frac{n\pi}{b-a},T\right)
\exp\!\left(i\,n\pi\,\frac{x_0-a}{b-a}\right)\right]\!\widetilde U_n,
\end{equation}
where $x_0:=\log(S_0/K)+rT$ and $\sum'$ halves the $n=0$ term,
satisfies
\begin{equation}\label{eq:cos_error}
|C_M-C_{M,N}|\le K e^{-rT}\,\frac{2}{b-a}\,
\sum_{n\ge N}e^{-c(n\pi/(b-a))^{2\mu}}.
\end{equation}
The COS length $N$ is fixed automatically as the smallest $N$
such that the right-hand side of~\eqref{eq:cos_error} is at most
$\eta=2^{-b}$. No free parameter is involved.
\end{theorem}
\begin{proof}
The cosine expansion of $f_M$ on $[a,b]$ together with the
Fang--Oosterlee identity yields~\eqref{eq:CMN} in the limit
$N\to\infty$. The bound~\eqref{eq:cos_error} follows from
Lemma~\ref{lem:decay} and $|\widetilde U_n|=O(1)$. Setting the
right-hand side equal to $\eta$ uniquely determines $N$.
\end{proof}

\subsection{Algorithm}

Inputs: $S_0,K,T,r,\theta,\lambda,\rho,\nu,V_0,H$ and the working
precision $b$ (bits). All other quantities are determined.

\begin{enumerate}
\item Generate $m_k=a_{k+1}$ from~\eqref{eq:ak} for
$k=0,1,2,\ldots$, increasing $M$ until the a-posteriori
bound~\eqref{eq:aposteriori} drops to $\eta=2^{-b}$.
\item Run Chebyshev--Wheeler~\eqref{eq:cw_init}--\eqref{eq:cw_extract}
to produce $(\alpha_n,\beta_n)_{n=0}^{M-1}$, then build
$(P_M,Q_M)$ from~\eqref{eq:conv_P}--\eqref{eq:conv_Q}, and
$\Phi_M(v,T)$ from~\eqref{eq:PhiM}.
\item Compute $\kappa_1^{(M)}(T),\ldots,\kappa_4^{(M)}(T)$
from~\eqref{eq:kappa_closed} via the Leibniz
solve~\eqref{eq:leibniz_solve}.
\item Set $[a,b]$ from~\eqref{eq:interval}--\eqref{eq:width_root}.
\item Determine $N$ as the smallest integer making the
right-hand side of~\eqref{eq:cos_error} $\le\eta$.
\item Evaluate $\widetilde U_n$ from~\eqref{eq:chi}--\eqref{eq:psi}
and $\phi_M(n\pi/(b-a),T)=\exp\Phi_M(n\pi/(b-a),T)$ for
$n=0,\ldots,N-1$, then return $C_{M,N}$ from~\eqref{eq:CMN}.
\end{enumerate}

\section{Summary}
\label{sec:summary}
% -----------------------------------------------------------------

\textbf{T1.} The diagonal $[M/M]$ Pad\'{e}--M\"untz approximant of
the fractional Riccati solution converges uniformly on compacta of
$[0,T^*)$, with $(P_M,Q_M)$ produced by Chebyshev--Wheeler from
the moments $m_k=a_{k+1}$ and the convergent
recurrences~\eqref{eq:conv_P}--\eqref{eq:conv_Q}. The order $M$
is fixed by the working precision via~\eqref{eq:aposteriori}.

\textbf{T2.} European calls under rough Heston are priced by the
COS formula~\eqref{eq:CMN} on $\phi_M(\cdot,T)=\exp\Phi_M(\cdot,T)$.
The interval $[a,b]$ is fixed by the
cumulants~\eqref{eq:kappa_closed} and the precision-driven
half-width~\eqref{eq:width_root}; the COS length $N$ is fixed by
the precision-driven bound~\eqref{eq:cos_error}. The only input
beyond model parameters is the bit-precision $b$.
