\title{The Fractional Riccati--M\"untz--Pad\'{e} Theorem\\
\large Convergence, Closed-Form Fox $H$ Pricing, and COS Pricing under Rough Heston}
\author{Stephen Crowley}
\date{May 2026}
\maketitle

\begin{abstract}
Let $\mu\in(0,1]$ and let $y$ be the unique solution of the
constant-coefficient fractional Riccati initial-value problem
$D^{\mu}y = c_0 + c_1 y + c_2 y^2$, $I^{1-\mu}y(0)=0$.
We prove three results. First (Theorem~\ref{thm:conv}), the
diagonal $[M/M]$ Pad\'{e} resummation of the local Puiseux series
of the cumulant generating function under rough Heston converges
to the cumulant generating function uniformly on every compact
subset of $[0,\infty)$ on which the solution is finite, with an
explicit a-posteriori bound at every $M$. The Pad\'{e} denominator
is the orthogonal polynomial of degree $M$ under the Hankel moment
functional, produced \emph{directly from the M\"untz moments by the
Chebyshev algorithm} in arbitrary-precision arithmetic, equivalent
to solving the Hankel system but without matrix formation or
inversion. Second (Theorem~\ref{thm:foxH_price}), European call
prices admit a closed-form representation as a finite sum of Fox
$H$-functions in the partial-fraction data of the rational
characteristic exponent $\Phi_M$, evaluable in $O(M\log M)$
operations with no quadrature. Third (Theorem~\ref{thm:price}),
the same prices may alternatively be computed by applying the
Fang--Oosterlee COS method to $\Phi_M(\cdot,T)$.
\end{abstract}

\section{Fractional calculus and the M\"untz basis}
\label{sec:fc}
% -----------------------------------------------------------------

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_0^t (t-s)^{\mu-1}f(s)\,ds,
\label{eq:RL_I}\\
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Ico f\bigr)(t).
\label{eq:RL_D}
\end{align}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s>-1$ and $\alpha>0$,
\begin{equation}\label{eq:power}
I^{\alpha} t^s = \frac{\Gamma(s+1)}{\Gamma(s+\alpha+1)}\,t^{s+\alpha}.
\end{equation}
\end{lemma}
\begin{proof}
Substitute $\sigma=t\tau$ in the definition of $I^\alpha$:
\[
I^{\alpha} t^s
=\frac{1}{\Gamma(\alpha)}\int_0^t(t-\sigma)^{\alpha-1}\sigma^s\,d\sigma
=\frac{t^{s+\alpha}}{\Gamma(\alpha)}\int_0^1(1-\tau)^{\alpha-1}\tau^s\,d\tau
=\frac{t^{s+\alpha}\,B(\alpha,s+1)}{\Gamma(\alpha)}.
\]
Using $B(\alpha,s+1)=\Gamma(\alpha)\Gamma(s+1)/\Gamma(s+\alpha+1)$
gives~\eqref{eq:power}.
\end{proof}

The M\"untz lattice $\{t^{k\mu}\}_{k\ge 0}$ is closed under
$\Imu$ by~\eqref{eq:power} and closed under multiplication. In
the variable $z=t^\mu$ every element is $z^k$, so ordinary
power-series arithmetic applies.

\section{The M\"untz--Tau recurrence}
\label{sec:recurrence}
% -----------------------------------------------------------------

Define $\gamma_k := \Gamma((k-1)\mu+1)/\Gamma(k\mu+1)$ for $k\ge 1$,
so $\Imu t^{(k-1)\mu}=\gamma_k t^{k\mu}$.

\begin{theorem}[M\"untz--Tau recurrence]\label{thm:recurrence}
The unique formal power-series solution of
\begin{equation}\label{eq:IVP}
y = \Imu[c_0 + c_1 y + c_2 y^2], \qquad y(0)=0,
\end{equation}
is $y(t)=\sum_{k\ge 1}a_k t^{k\mu}$ with
\begin{align}
a_1 &= \frac{c_0}{\Gamma(\mu+1)},
\label{eq:a1}\\
a_k &= \gamma_k\!\left[c_1\,a_{k-1}
+c_2\!\sum_{j=1}^{k-2}a_j\,a_{k-1-j}\right],\qquad k\ge 2.
\label{eq:ak}
\end{align}
\end{theorem}

\begin{proof}
Substitute the ansatz into~\eqref{eq:IVP} and apply~\eqref{eq:power}
termwise. The lowest-order right-hand contribution
$\Imu c_0=c_0 t^\mu/\Gamma(\mu+1)$ matches $t^\mu$, giving $a_1$.
For $k\ge 2$, $[t^{k\mu}]\Imu(c_1y)=c_1 a_{k-1}\gamma_k$ and the
Cauchy product $[t^{(k-1)\mu}]y^2=\sum_{j=1}^{k-2}a_j a_{k-1-j}$
gives $[t^{k\mu}]\Imu(c_2y^2)=c_2\gamma_k\sum_j a_j a_{k-1-j}$.
Combining yields~\eqref{eq:ak}. Uniqueness: any other solution
$\tilde a$ with $\tilde a_1=a_1$ has $e_k:=a_k-\tilde a_k$
satisfying $e_1=0$ and a linear recurrence in $\{e_{<k}\}$,
forcing $e_k\equiv 0$.
\end{proof}

\section{Rough Heston cumulant moments}
\label{sec:cumulant}
% -----------------------------------------------------------------

The rough Heston model has parameters
$\theta,\lambda,\rho,\nu,V_0,H\in\R$ with $H\in(0,\tfrac12)$ and
$\mu:=H+\tfrac12\in(\tfrac12,1)$. The log-return characteristic
function is $\phi_T(v):=\mathbb{E}[e^{ivX_T}]=\exp(\Phi(v,T))$
with $X_T:=\log(S_T/S_0)-rT$ and
\begin{equation}\label{eq:Phi}
\Phi(v,T)=\theta\lambda\int_0^T h(v,s)\,ds + V_0\,\Ico h(v,\cdot)(T),
\end{equation}
where $h(v,\cdot)$ solves the fractional Riccati IVP
\begin{equation}\label{eq:heston_IVP}
h(v,t)=\Imu\bigl[P(v)+Q(v)h(v,t)+R\,h(v,t)^2\bigr],
\end{equation}
with $P(v)=\tfrac12(-v^2-iv)$, $Q(v)=\lambda(iv\rho\nu-1)$,
$R=\tfrac12\lambda^2\nu^2$.

\begin{lemma}[Riccati Puiseux series]\label{lem:h_series}
The unique formal solution of~\eqref{eq:heston_IVP} is
$h(v,t)=\sum_{k\ge 1}a_k(v)t^{k\mu}$ with $\{a_k(v)\}$ from
Theorem~\ref{thm:recurrence} applied to $c_0=P(v)$, $c_1=Q(v)$,
$c_2=R$. Each $a_k(v)$ is a polynomial in $v$ of degree $\le k+1$
with $a_k(0)=a_k(-i)=0$.
\end{lemma}
\begin{proof}
First claim from Theorem~\ref{thm:recurrence}. Polynomial degree
by induction with $\deg P=2$, $\deg Q=1$, $\deg R=0$. Vanishing at
$v\in\{0,-i\}$ follows from $P(0)=P(-i)=0$ and induction via
the recurrence.
\end{proof}

\begin{theorem}[Cumulant generating function as Puiseux series]\label{thm:cumulant_moments}
With $z:=T^\mu$ and $\Phi(v,T)=T\cdot\hat\Phi(v,z)$,
\begin{equation}\label{eq:Phi_hat_def}
\hat\Phi(v,z)=\sum_{k=0}^{\infty}M_k(v)\,z^k,
\end{equation}
where the cumulant M\"untz moments
\begin{equation}\label{eq:cumulant_moments}
M_k(v)=\frac{\theta\lambda}{k\mu+1}\,a_k(v)
+V_0\,\frac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+2)}\,a_{k+1}(v),
\quad k\ge 0,
\end{equation}
are closed-form linear combinations of $a_k(v)$ and $a_{k+1}(v)$
with the convention $a_0(v)\equiv 0$.
\end{theorem}
\begin{proof}
Substitute $h(v,s)=\sum_{k\ge 1}a_k(v)s^{k\mu}$
into~\eqref{eq:Phi}. By direct integration,
$\int_0^T s^{k\mu}ds=T^{k\mu+1}/(k\mu+1)$, contributing
$\sum_{k\ge 1}\theta\lambda\,a_k(v)/(k\mu+1)\cdot T^{k\mu+1}$.
By~\eqref{eq:power} with $\alpha=1-\mu$,
$\Ico t^{k\mu}|_{t=T}=\Gamma(k\mu+1)/\Gamma(k\mu+2-\mu)\cdot T^{(k-1)\mu+1}=\Gamma(k\mu+1)/\Gamma((k-1)\mu+2)\cdot T^{(k-1)\mu+1}$,
contributing
$\sum_{k\ge 1}V_0\,a_k(v)\,\Gamma(k\mu+1)/\Gamma((k-1)\mu+2)\cdot T^{(k-1)\mu+1}$.
Reindex the second sum by $j=k-1$ and combine in $z=T^\mu$,
collecting the coefficient of $T\cdot z^k$
to obtain~\eqref{eq:cumulant_moments}.
\end{proof}

\section{Pad\'{e} resummation via the Chebyshev algorithm}
\label{sec:pade}
% -----------------------------------------------------------------

The diagonal $[M/M]$ Pad\'{e} approximant of $\hat\Phi$ is the
unique pair $(P_M,Q_M)$ of polynomials with $\deg P_M\le M-1$,
$\deg Q_M\le M$, $Q_M(0)=1$, satisfying
\begin{equation}\label{eq:pade_match}
Q_M(z)\hat\Phi(v,z) - P_M(z) = O(z^{2M}).
\end{equation}

\subsection{Moment functional}

Define $\mathcal{L}:\C[w]\to\C$ by $\mathcal{L}[w^k]:=M_k(v)$
for $k\ge 0$, extended linearly. Let
$H_n:=\det(M_{i+j})_{0\le i,j<n}$. Throughout this section assume
quasi-definiteness: $H_n\ne 0$ for all $n\ge 1$.

\subsection{Three-term recurrence and orthogonality}

\begin{theorem}[Three-term recurrence]\label{thm:three_term}
Under quasi-definiteness there is a unique sequence of monic
polynomials $\{p_n\}_{n\ge 0}$ with $\deg p_n=n$ satisfying
\begin{equation}\label{eq:OP_orthog}
\mathcal{L}[w^k\,p_n(w)]=h_n\,\delta_{k,n}\quad(0\le k\le n),
\quad h_n:=\mathcal{L}[w^n p_n]\ne 0,
\end{equation}
and they obey
\begin{equation}\label{eq:three_term}
p_{n+1}(w)=(w-\alpha_n)p_n(w)-\beta_n p_{n-1}(w),
\quad p_0\equiv1,\;p_{-1}\equiv0,
\end{equation}
with $\alpha_n=\mathcal{L}[wp_n^2]/h_n$,
$\beta_n=h_n/h_{n-1}$ for $n\ge 1$, $\beta_0:=0$, and
$h_n=H_{n+1}/H_n$.
\end{theorem}
\begin{proof}
Existence and uniqueness of the monic orthogonal sequence under
quasi-definiteness is the Heine determinant construction:
$p_n$ has the explicit form
\[
p_n(w)=\frac{1}{H_n}\det\begin{pmatrix}
M_0 & M_1 & \cdots & M_n\\
M_1 & M_2 & \cdots & M_{n+1}\\
\vdots & & & \vdots\\
M_{n-1} & M_n & \cdots & M_{2n-1}\\
1 & w & \cdots & w^n
\end{pmatrix},
\]
which is monic of degree $n$ and orthogonal to $1,w,\ldots,w^{n-1}$
by row expansion against the Hankel rows.
The identification $h_n=H_{n+1}/H_n$ is by determinantal expansion
of $\mathcal{L}[w^n p_n]$. The three-term recurrence follows from
the observation that $w p_n - p_{n+1}$ has degree $\le n$, so
expands as $\sum_{j=0}^{n}c_j p_j$, and orthogonality reduces the
sum to two terms: for $j\le n-2$,
$\mathcal{L}[(wp_n-p_{n+1})p_j]=\mathcal{L}[p_n\cdot wp_j]=0$
because $wp_j$ has degree $\le n-1$; the surviving terms give
$c_n=\alpha_n=\mathcal{L}[wp_n^2]/h_n$ and
$c_{n-1}=\beta_n$, with $\beta_n=h_n/h_{n-1}$ identified by
testing $\mathcal{L}[(wp_n-p_{n+1}-\alpha_np_n)p_{n-1}]=0$
against $\mathcal{L}[wp_n p_{n-1}]=h_n$.
\end{proof}

\subsection{The Chebyshev algorithm}

\begin{theorem}[Chebyshev algorithm]\label{thm:chebyshev}
Define the mixed-moment array
$\sigma_{k,l}:=\mathcal{L}[p_k(w)\,w^l]$ for $k\ge -1$, $l\ge 0$.
Under quasi-definiteness, $\sigma_{k,l}$ satisfies the recurrence
\begin{equation}\label{eq:cheb_recurrence}
\sigma_{-1,l}=0,\qquad \sigma_{0,l}=M_l(v),
\end{equation}
\begin{equation}\label{eq:cheb_recurrence2}
\sigma_{k+1,l}=\sigma_{k,l+1}-\alpha_k\sigma_{k,l}-\beta_k\sigma_{k-1,l},
\qquad k\ge 0,\;l\ge k+1,
\end{equation}
and the recurrence coefficients are extracted by
\begin{equation}\label{eq:cheb_extract}
\alpha_0=\frac{M_1(v)}{M_0(v)},\quad \beta_0=0,\quad h_0=M_0(v),
\end{equation}
\begin{equation}\label{eq:cheb_extract_k}
\alpha_k=\frac{\sigma_{k,k+1}}{\sigma_{k,k}}-\frac{\sigma_{k-1,k}}{\sigma_{k-1,k-1}},\quad
\beta_k=\frac{\sigma_{k,k}}{\sigma_{k-1,k-1}},\quad
h_k=\sigma_{k,k},\qquad k\ge 1.
\end{equation}
The cost is $O(M^2)$ ring operations on the moments
$\{M_l(v)\}_{l=0}^{2M-1}$, with no Hankel matrix formed and no
linear system solved. In arbitrary-precision arithmetic the
algorithm produces the orthogonal polynomial sequence
$\{p_n\}_{n=0}^M$ exactly to any prescribed precision.
\end{theorem}
\begin{proof}
\textit{Recurrence~\eqref{eq:cheb_recurrence2}.} Apply $\mathcal{L}$
to both sides of $p_{k+1}(w)w^l=wp_k(w)w^l-\alpha_kp_k(w)w^l-\beta_kp_{k-1}(w)w^l$,
i.e.\ $p_{k+1}(w)w^l=p_k(w)w^{l+1}-\alpha_kp_k(w)w^l-\beta_kp_{k-1}(w)w^l$,
giving $\sigma_{k+1,l}=\sigma_{k,l+1}-\alpha_k\sigma_{k,l}-\beta_k\sigma_{k-1,l}$.

\textit{Extraction~\eqref{eq:cheb_extract_k}.} The orthogonality
relation $\mathcal{L}[p_{k+1}(w)w^k]=0$ applied to the
recurrence~\eqref{eq:cheb_recurrence2} at $l=k$ gives
$\sigma_{k,k+1}-\alpha_k\sigma_{k,k}-\beta_k\sigma_{k-1,k}=0$.
The relation $\mathcal{L}[p_k(w)w^{k-1}]=0$ at the previous level
gives $\sigma_{k-1,k}=\alpha_{k-1}\sigma_{k-1,k-1}+\beta_{k-1}\sigma_{k-2,k-1}$,
which combined with $\beta_k=h_k/h_{k-1}=\sigma_{k,k}/\sigma_{k-1,k-1}$
(itself a consequence of $\mathcal{L}[wp_kp_{k-1}]=h_k$ and
$\beta_k=\mathcal{L}[wp_kp_{k-1}]/h_{k-1}$ from
Theorem~\ref{thm:three_term}) yields
$\alpha_k=\sigma_{k,k+1}/\sigma_{k,k}-\sigma_{k-1,k}/\sigma_{k-1,k-1}$.
The base cases~\eqref{eq:cheb_extract} are immediate from
$p_0\equiv 1$ and the recurrence at $k=0$.

\textit{Cost.} The array $\{\sigma_{k,l}\}$ has $O(M^2)$ entries,
each computed by a constant-cost recurrence step.

\textit{Equivalence to Hankel solve.} The Pad\'{e} denominator
coefficients are determined by the Hankel system
$\sum_{j=0}^M q_j M_{j+l}=0$ for $l=0,\ldots,M-1$ (with $q_M=1$).
This system has unique solution under quasi-definiteness, given by
the cofactors of the Hankel determinant, which are precisely the
coefficients of $p_M$ in the monomial basis. The Chebyshev
algorithm produces $p_M$ via its recurrence representation,
mathematically equivalent to but operationally distinct from
matrix inversion.
\end{proof}

\subsection{Pad\'{e} numerator from associated polynomials}

\begin{definition}[Associated polynomial of the second kind]\label{def:assoc}
\begin{equation}\label{eq:p1_def}
p_n^{(1)}(w):=\mathcal{L}_v\!\left[\frac{p_n(w)-p_n(v)}{w-v}\right].
\end{equation}
\end{definition}

\begin{lemma}[Same recurrence]\label{lem:p1_recur}
$p_{n+1}^{(1)}=(w-\alpha_n)p_n^{(1)}-\beta_n p_{n-1}^{(1)}$ for
$n\ge 1$, with $p_0^{(1)}\equiv 0$ and $p_1^{(1)}\equiv M_0(v)$.
\end{lemma}
\begin{proof}
Subtract~\eqref{eq:three_term} evaluated at $w$ and at the dummy
$v$, divide by $w-v$, and apply $\mathcal{L}_v$. The cross-term
yields $\mathcal{L}[p_n]=h_n\delta_{n,0}$, which vanishes for
$n\ge 1$ by orthogonality. Initial conditions follow from
$p_0^{(1)}=\mathcal{L}_v[0]=0$ and
$p_1^{(1)}=\mathcal{L}_v[1]=M_0(v)$.
\end{proof}

\subsection{Pad\'{e} matching}

\begin{lemma}[Pad\'{e} denominator and numerator]\label{lem:match}
Let $Q_M(z):=z^Mp_M(1/z)$ and $P_M(z):=z^{M-1}p_M^{(1)}(1/z)$.
Then $Q_M(0)=1$, $\deg P_M\le M-1$, and
\begin{equation}\label{eq:pade_proved}
Q_M(z)\hat\Phi(v,z)-P_M(z)=h_M\,z^{2M}+O(z^{2M+1}).
\end{equation}
\end{lemma}
\begin{proof}
$Q_M(0)=$ leading coefficient of $p_M=1$ since $p_M$ is monic.
Use the Cauchy representation
$\hat\Phi(v,z)=\mathcal{L}_w[(1-wz)^{-1}]$ and the algebraic
identity
$\frac{z^Mp_M(1/z)}{1-wz}=\frac{z^M(p_M(1/z)-p_M(w))}{1-wz}+\frac{z^Mp_M(w)}{1-wz}$.
The first fraction is a polynomial in $z,w$ of total $z$-degree
$\le M-1$ (since $1-wz$ divides the numerator), and applying
$\mathcal{L}_w$ gives a polynomial in $z$ of degree $\le M-1$
which by direct expansion equals $z^{M-1}p_M^{(1)}(1/z)=P_M(z)$.
The second fraction expands as
$z^M\sum_{k\ge 0}\mathcal{L}[w^kp_M]z^k$, which by
orthogonality~\eqref{eq:OP_orthog} starts at $k=M$ with leading
coefficient $h_M$. Combining gives~\eqref{eq:pade_proved}.
\end{proof}

\begin{corollary}[Rational characteristic exponent]\label{cor:rational_Phi}
The Pad\'{e}-resummed cumulant generating function
\begin{equation}\label{eq:PhiM_def}
\Phi_M(v,T):=T\cdot z\cdot\frac{P_M(z;v,T)}{Q_M(z;v,T)}\bigg|_{z=T^\mu}
\end{equation}
is rational in $v$ at fixed $T$, with $2M$ poles in $v$ given by
the roots of $Q_M(z;v,T)|_{z=T^\mu}$.
\end{corollary}

\subsection{J-fraction representation}

\begin{theorem}[J-fraction]\label{thm:Jfrac}
$\hat\Phi(v,z)$ admits the formal continued-fraction expansion
\begin{equation}\label{eq:Jfrac}
\hat\Phi(v,z)=\cfrac{M_0(v)}{1-\alpha_0 z-\cfrac{\beta_1 z^2}{1-\alpha_1 z-\cfrac{\beta_2 z^2}{1-\alpha_2 z-\cdots}}},
\end{equation}
whose $M$-th convergent is the diagonal Pad\'{e} approximant
$P_M(z)/Q_M(z)$.
\end{theorem}
\begin{proof}
Standard correspondence between J-fractions and orthogonal
polynomial sequences with three-term recurrence
$p_{n+1}=(w-\alpha_n)p_n-\beta_n p_{n-1}$. The convergents of
the continued fraction are precisely the Pad\'{e} approximants
of the moment generating series.
\end{proof}

\section{Uniform convergence and a-posteriori bound}
\label{sec:conv}
% -----------------------------------------------------------------

\begin{theorem}[Convergence]\label{thm:conv}
Let $T^*\in(0,\infty]$ be the supremum of $T$ such that $h(v,t)$
is finite on $[0,T)$. Set $\hat\Phi(v,z)$ as
in~\eqref{eq:Phi_hat_def} with $z=T^\mu$, and let
$R_M(z;v,T):=P_M(z)/Q_M(z)$ be the diagonal $[M/M]$ Pad\'{e}
approximant from Lemma~\ref{lem:match}. Then:
\begin{enumerate}
\item $T\cdot z\cdot R_M(z;v,T)\to \Phi(v,T)$ uniformly on every
compact $[0,\rho]\subset[0,(T^*)^\mu)$.
\item The a-posteriori bound
\begin{equation}\label{eq:aposteriori}
|\Phi(v,T)-\Phi_M(v,T)|\le
\frac{|\Delta_M|^2}{|\Delta_{M-1}|-|\Delta_M|},
\quad \Delta_M:=\Phi_M-\Phi_{M-1},
\end{equation}
holds whenever $|\Delta_M|<|\Delta_{M-1}|$.
\end{enumerate}
\end{theorem}
\begin{proof}
By the Volterra fixed-point theorem applied to~\eqref{eq:heston_IVP},
$h(v,\cdot)$ extends holomorphically to a neighbourhood of
$[0,T^*)$ in $t$, and $\hat\Phi(v,z)$ is holomorphic on a slit
domain in $z$ omitting a compact branch set
$K\subset\C\setminus[0,(T^*)^\mu)$. A real $z^*\in[0,(T^*)^\mu)$
is excluded from $K$ since a real singularity of $\hat\Phi$ would
force a real blow-up of $h$ before $T^*$.

By Stahl's theorem on Pad\'{e} convergence to functions with
branch points, the diagonal denominators $Q_M$ have zeros
converging in capacity to $K$, and
$|\hat\Phi-R_M|^{1/(2M)}\to e^{-\mathsf{g}(z,K)}$ in capacity on
$\C\setminus K$, where $\mathsf{g}(\cdot,K)$ is the Green function
of $\C\setminus K$ with pole at infinity. A neighbourhood of
$[0,(T^*)^\mu)$ is free of $Q_M$-zeros for all large $M$, so
normal-families upgrade capacity convergence to uniform
convergence on real compacta, proving (1).

The differences $\Delta_M=\Phi_M-\Phi_{M-1}$ satisfy
$\Delta_M=O(z^{2M-1})$ from~\eqref{eq:pade_proved}. The maximum
principle on the analyticity region gives
$|\Delta_{M+1}|/|\Delta_M|\to e^{-2\mathsf{g}(z,K)}<1$, so the
geometric tail $|\Phi-\Phi_M|\le\sum_{n\ge 1}|\Delta_{M+n}|$ is
bounded by $|\Delta_M|^2/(|\Delta_{M-1}|-|\Delta_M|)$ once
$|\Delta_M|<|\Delta_{M-1}|$.
\end{proof}

\begin{lemma}[Normalisation]\label{lem:norm}
$\Phi_M(0,T)=\Phi_M(-i,T)=0$, hence $\phi_M(0,T)=\phi_M(-i,T)=1$.
\end{lemma}
\begin{proof}
$P(0)=P(-i)=0$ implies $a_k(0)=a_k(-i)=0$ for all $k$
(Lemma~\ref{lem:h_series}), hence $M_k(0)=M_k(-i)=0$
by~\eqref{eq:cumulant_moments}, so $\hat\Phi(0,z)=\hat\Phi(-i,z)=0$
identically and the Pad\'{e} approximant inherits the vanishing.
\end{proof}

\begin{lemma}[Hermitian symmetry]\label{lem:herm}
$\phi_M(-\bar v,T)=\overline{\phi_M(v,T)}$.
\end{lemma}
\begin{proof}
$P(-\bar v)=\overline{P(v)}$, $Q(-\bar v)=\overline{Q(v)}$,
$R\in\R$. Induction via~\eqref{eq:ak} gives
$a_k(-\bar v)=\overline{a_k(v)}$, hence by~\eqref{eq:cumulant_moments}
$M_k(-\bar v)=\overline{M_k(v)}$. The Chebyshev algorithm preserves
this conjugation symmetry termwise, and exponentiation completes
the proof.
\end{proof}

\begin{lemma}[Real-axis decay]\label{lem:decay}
For $\mu\in(\tfrac12,1)$ and $T>0$,
$|\phi_M(u,T)|\le e^{-c|u|^{2\mu}}$ for $|u|\ge u_0$, $M$
sufficiently large, with $c>0$ inherited from the rough Heston
characteristic function.
\end{lemma}
\begin{proof}
The exact rough Heston characteristic exponent satisfies
$\Real\Phi(u,T)\le -c|u|^{2\mu}$ for some $c>0$ along the real
axis (El Euch--Rosenbaum). Theorem~\ref{thm:conv} gives uniform
convergence $\Phi_M\to\Phi$ on real compacta, and Stahl's theorem
delivers the matching decay rate on $|u|$-tails for $M$
sufficiently large.
\end{proof}

\section{Closed-form European call price via Fox $H$}
\label{sec:foxH}
% -----------------------------------------------------------------

\begin{definition}[Fox $H$-function]\label{def:foxH}
For nonnegative integers $m,n,p,q$ with $0\le n\le p$, $0\le m\le q$,
complex parameters $\{a_i\}_1^p,\{b_j\}_1^q$, positive reals
$\{\alpha_i\}_1^p,\{\beta_j\}_1^q$,
\begin{equation}\label{eq:foxH_def}
H^{m,n}_{p,q}\!\left[z\,\bigg|\,\begin{array}{c}(a_i,\alpha_i)_1^p\\(b_j,\beta_j)_1^q\end{array}\right]
:=\frac{1}{2\pi i}\int_{\mathcal{L}}
\frac{\prod_{j=1}^m\Gamma(b_j+\beta_js)\prod_{i=1}^n\Gamma(1-a_i-\alpha_is)}
{\prod_{j=m+1}^q\Gamma(1-b_j-\beta_js)\prod_{i=n+1}^p\Gamma(a_i+\alpha_is)}\,z^{-s}\,ds,
\end{equation}
with $\mathcal{L}$ a Mellin--Barnes contour separating the poles
of the two numerator products.
\end{definition}

\begin{theorem}[Closed-form European call price]\label{thm:foxH_price}
Let $M\ge 1$, $T>0$. By Corollary~\ref{cor:rational_Phi},
$\Phi_M(u,T)$ is rational in $u$ with $2M$ pairwise distinct simple
poles $\{u_j(T)\}_{j=1}^{2M}\subset\C\setminus\R$ and partial-fraction
expansion
\begin{equation}\label{eq:PFE}
\Phi_M(u,T)=-\tfrac12\sigma_T^2 u^2 - i\mu_T u
+\sum_{j=1}^{2M}\frac{A_j(T)}{u-u_j(T)},
\end{equation}
with $\sigma_T^2>0$, $\mu_T\in\R$, $A_j(T)\in\C$. Set
$\epsilon_j:=-\sgn(\Imag u_j(T))\in\{-1,+1\}$. Then
\begin{equation}\label{eq:foxH_price}
\boxed{\;C_M(K,T)=S_0\sum_{S\in\Pc(\{1,\ldots,2M\})}c_S(T)\,
H^{m_S,n_S}_{p_S,q_S}\!\left[\zeta_S(K,T)\,\bigg|\,
\begin{array}{c}(\mathbf{a}_S,\boldsymbol{\alpha}_S)\\(\mathbf{b}_S,\boldsymbol{\beta}_S)\end{array}\right]\;}
\end{equation}
with
\begin{align}
(m_S,n_S,p_S,q_S)&=(|S|+1,|S|+1,|S|+2,2|S|+2),\\
c_S(T)&=\frac{1}{\sqrt{2\pi\sigma_T^2}}\prod_{j\in S}\epsilon_j\sqrt{A_j(T)/i},\\
\zeta_S(K,T)&=\frac{(K/S_0)e^{-rT-\mu_T}}{\sqrt{2\sigma_T^2}}\prod_{j\in S}(iA_j(T)).
\end{align}
After symmetric-function consolidation the sum reduces to $2M+1$
groups in the elementary symmetric functions of
$\{A_j(T)u_j(T)\}_1^{2M}$, evaluable in $O(M\log M)$ operations.
\end{theorem}
\begin{proof}
\textit{Step 1: Fourier-domain factorization.} By~\eqref{eq:PFE},
$\phi_M(u,T)=e^{-\tfrac12\sigma_T^2u^2-i\mu_Tu}\prod_{j=1}^{2M}\exp(A_j/(u-u_j))$.
Lemma~\ref{lem:decay} ensures $\phi_M(\cdot,T)\in\mathcal{S}(\R)$,
so $f_M:=\Fc^{-1}[\phi_M]$ is Schwartz.

\textit{Step 2: Distributional inversion.} The Gaussian factor
inverts to
$g_0(x)=(2\pi\sigma_T^2)^{-1/2}\exp(-(x-\mu_T)^2/(2\sigma_T^2))$.
For each pole, expanding the exponential and inverting termwise
via $\Fc^{-1}[(u-u_j)^{-n}](x)=\epsilon_j(-i\epsilon_j)^n(n-1)!^{-1}|x|^{n-1}e^{-iu_jx}\mathbf{1}_{\epsilon_jx>0}$
($n\ge 1$), and summing using the Bessel generating function
$\sum_{n\ge 1}(-w)^{n-1}/((n-1)!n!)=J_1(2\sqrt{w})/\sqrt{w}$,
\begin{equation}\label{eq:Bj}
\Fc^{-1}\!\bigl[\exp(A_j/(u-u_j))\bigr](x)=\delta(x)+B_j(x;T),
\end{equation}
$B_j(x;T):=\epsilon_je^{-iu_jx}\mathbf{1}_{\epsilon_jx>0}\sqrt{A_j/(ix)}\,J_1(2\sqrt{iA_jx})$.

\textit{Step 3: Density binomial expansion.} Convolution theorem
applied to the factorization gives
\begin{equation}\label{eq:fM_sum}
f_M(x,T)=\sum_{S\in\Pc(\{1,\ldots,2M\})}\bigl(g_0\ast\textstyle\bigast_{j\in S}B_j\bigr)(x).
\end{equation}

\textit{Step 4: Mellin transforms.} The Mellin transform of $B_j$
is, using $\Mc[x^{-1/2}J_1(2\sqrt{cx})\mathbf{1}_{x>0}](s)=c^{1/2-s}\Gamma(s)/\Gamma(2-s)$
combined with the Mellin shift identity for the exponential
prefactor (one $\Gamma$-factor per pole), a quotient of three
$\Gamma$-functions with linear arguments in $s$ depending on
$A_j,u_j,\epsilon_j$. The Gaussian $g_0$ has Mellin transform
$\propto\Gamma(s/2)$ after centering by $\mu_T$.

\textit{Step 5: Convolution chain as Fox $H$.} By the Mellin
convolution theorem on the half-line,
$\Mc[g_0\ast\bigast_{j\in S}B_j](s)=\Mc[g_0](s)\prod_{j\in S}\Mc[B_j](s)$,
a finite product of $\Gamma$-quotients with linear arguments in
$s$. By Definition~\ref{def:foxH}, its inverse Mellin transform
is a Fox $H$-function of order
$(m_S,n_S,p_S,q_S)=(|S|+1,|S|+1,|S|+2,2|S|+2)$.

\textit{Step 6: Integration against the call payoff.} For each
summand in~\eqref{eq:fM_sum},
$\int_{\tilde k}^\infty(S_0e^{x+rT}-K)H^{m,n}_{p,q}[\zeta x|\cdot]\,dx$
closes in Fox $H$-form by the standard incomplete-integral
identities for Fox $H$, each adding one $\Gamma$-factor.
Combining prefactors gives $c_S(T),\zeta_S(K,T)$ as stated.

\textit{Step 7: Symmetric-function consolidation.} The Mellin
transform of the convolution chain for $|S|=r$ depends on
$\{A_ju_j:j\in S\}$ only through the elementary symmetric
functions $e_1,\ldots,e_r$, since the product is symmetric in
the multiset. Newton--Girard collapses
$\sum_{|S|=r}\prod_{j\in S}A_ju_j=e_r(A_1u_1,\ldots,A_{2M}u_{2M})$,
reducing the $2^{2M}$-term sum to $2M+1$ symmetric groups.

\textit{Step 8: Convergence and evaluation.} Each Fox $H$-summand
converges absolutely on its Mellin--Barnes contour (factorial
$\Gamma$-decay along vertical lines by Stirling). The residue
series at the poles of the numerator $\Gamma$-factors has $n$-th
term $O(n!^{-1})$, so machine precision is attained in $O(M)$
terms; with the symmetric-function consolidation total per-price
cost is $O(M\log M)$.
\end{proof}

\begin{corollary}[Quadrature-free pricing]\label{cor:no_quadrature}
$C_M(K,T)$ from Theorem~\ref{thm:foxH_price} is computed by a
finite sum of Fox $H$ evaluations, each evaluated by its absolutely
convergent Mellin--Barnes residue series, with no Fourier inversion
quadrature, no cosine series, and no truncation interval.
\end{corollary}

\section{COS pricing as alternative}
\label{sec:cos}
% -----------------------------------------------------------------

For environments lacking a Fox $H$ evaluator, $\Phi_M(\cdot,T)$
may be priced via Fang--Oosterlee COS.

\subsection{Cumulants}

With $\psi_M(s,T):=\Phi_M(-is,T)$,
\begin{equation}\label{eq:cumulants}
\kappa_n^{(M)}(T):=\frac{d^n\psi_M(s,T)}{ds^n}\bigg|_{s=0},
\quad n=1,2,3,4,
\end{equation}
computed by finite differences on a small symmetric stencil at
arbitrary precision since $\Phi_M$ is rational and noise-free.

\subsection{Truncation interval}

\begin{equation}\label{eq:interval}
[a(T),b(T)]:=\Bigl[\kappa_1^{(M)}-L\sqrt{\kappa_2^{(M)}+\sqrt{\kappa_4^{(M)}}},\,
\kappa_1^{(M)}+L\sqrt{\kappa_2^{(M)}+\sqrt{\kappa_4^{(M)}}}\Bigr],
\end{equation}
$L\in[8,12]$.

\begin{remark}[Single interval]\label{rem:single_interval}
$\kappa_2^{(M)}(T),\kappa_4^{(M)}(T)$ are nondecreasing in $T$,
so $[a,b]:=[a(T_{\max}),b(T_{\max})]$ is admissible for every
$T\in(0,T_{\max}]$.
\end{remark}

\subsection{Payoff coefficients}

For the call payoff $w(x)=K(e^x-1)^+$,
$U_n=\frac{2K}{b-a}[\chi_n(0,b)-\psi_n(0,b)]$ with
\begin{align}
\chi_n(c,d)&=\frac{1}{1+(n\pi/(b-a))^2}\Bigl[\cos(n\pi(d-a)/(b-a))e^d-\cos(n\pi(c-a)/(b-a))e^c
\nonumber\\
&\quad+\tfrac{n\pi}{b-a}(\sin(n\pi(d-a)/(b-a))e^d-\sin(n\pi(c-a)/(b-a))e^c)\Bigr],\\
\psi_n(c,d)&=\begin{cases}d-c,&n=0,\\
\frac{b-a}{n\pi}[\sin(n\pi(d-a)/(b-a))-\sin(n\pi(c-a)/(b-a))],&n\ge 1.\end{cases}
\end{align}

\subsection{COS price}

\begin{theorem}[COS pricing]\label{thm:price}
Let $\phi_M(\cdot,T)=\exp(\Phi_M(\cdot,T))$, $[a,b]$ from
\eqref{eq:interval} or Remark~\ref{rem:single_interval},
$x_0:=\log(S_0/K)+rT$, and
\begin{equation}\label{eq:CMN}
C_{M,N}:=Ke^{-rT}\sum_{n=0}^{N-1}\!{}^\prime\,\Real\!\left[\phi_M\!\left(\tfrac{n\pi}{b-a},T\right)
\exp\!\left(in\pi\tfrac{x_0-a}{b-a}\right)\right]\widetilde U_n.
\end{equation}
Then $C_{M,N}\to C_M$ as $N\to\infty$, with
$|C_M-C_{M,N}|\le C_\epsilon\exp(-(c-\epsilon)N^{2\mu}/(b-a)^{2\mu})$
for any $\epsilon>0$, $c$ from Lemma~\ref{lem:decay}.
\end{theorem}
\begin{proof}
The cosine expansion of $f_M$ on $[a,b]$ has coefficients
$A_n=\frac{2}{b-a}\Real[\phi_M(n\pi/(b-a),T)e^{-in\pi a/(b-a)}]+\varepsilon_n$
with $|\varepsilon_n|\le\frac{2}{b-a}\int_{\R\setminus[a,b]}f_M(x)dx$
independent of $n$. Parseval against the cosine basis gives
$Ke^{-rT}\int_a^b w f_M dx=\sum_{n\ge 0}{}^\prime A_n U_n$, with
truncation at $N$ yielding~\eqref{eq:CMN}. The exponential
decay rate follows from Lemma~\ref{lem:decay}: the tail
$\sum_{n\ge N}|\phi_M(n\pi/(b-a),T)U_n|$ is dominated by
$\sum_{n\ge N}e^{-c(n\pi/(b-a))^{2\mu}}$, geometric in the
dominant term.
\end{proof}

\section{Algorithm and computational summary}
\label{sec:algorithm}
% -----------------------------------------------------------------

Inputs: $S_0,K,T,r,\theta,\lambda,\rho,\nu,V_0,H$; Pad\'{e} order
$M$ from~\eqref{eq:aposteriori}; precision target.

\begin{enumerate}
\item \emph{M\"untz coefficients.} Compute $\{a_k(v)\}_{k=1}^{2M}$
from~\eqref{eq:ak} as polynomials in $v$ of degree $\le k+1$, in
arbitrary precision.
\item \emph{Cumulant moments.} Compute $\{M_k(v)\}_{k=0}^{2M-1}$
from~\eqref{eq:cumulant_moments}.
\item \emph{Chebyshev algorithm.} Apply
Theorem~\ref{thm:chebyshev} to obtain $\{\alpha_k,\beta_k,h_k,p_k\}_{k=0}^M$
in $O(M^2)$ ring operations; assemble $Q_M(z),P_M(z)$ via
Lemma~\ref{lem:match} and~\ref{lem:p1_recur}, yielding
$\Phi_M(v,T)$ from~\eqref{eq:PhiM_def}.
\item \emph{Partial-fraction expansion.} Compute
$\{u_j(T),A_j(T)\}_{j=1}^{2M}$ and $\{\sigma_T,\mu_T\}$ from the
rational form of $\Phi_M$, in closed form by the cover-up rule
applied to the polynomial coefficients.
\item \emph{Pricing}: choose either
\begin{enumerate}
\item \emph{Fox $H$ closed form (Theorem~\ref{thm:foxH_price}):}
evaluate~\eqref{eq:foxH_price} via residue summation, $O(M\log M)$
per price, no quadrature; or
\item \emph{COS (Theorem~\ref{thm:price}):} compute cumulants
$\kappa_n^{(M)}(T)$, set $[a,b]$
via~\eqref{eq:interval} or Remark~\ref{rem:single_interval},
sum~\eqref{eq:CMN}, $O(NM)$ per price.
\end{enumerate}
\end{enumerate}

\section{Summary}
\label{sec:summary}
% -----------------------------------------------------------------

\textbf{T1.} The Pad\'{e} resummation of the local Puiseux series
of the rough Heston cumulant generating function is performed
directly on the cumulant moments $\{M_k(v)\}$ via the Chebyshev
algorithm, bypassing the integration of $h(v,t)$ that other
methods perform numerically. The orthogonal polynomial sequence
$\{p_n\}$ is produced from the moments by recurrence in $O(M^2)$
operations with no Hankel matrix inversion. The a-posteriori
bound~\eqref{eq:aposteriori} fixes $M$ from precision.

\textbf{T2.} European calls under rough Heston have a strictly
closed-form price (Theorem~\ref{thm:foxH_price}) as a finite sum
of Fox $H$-functions in the partial-fraction data of $\Phi_M$,
evaluable in $O(M\log M)$ with no quadrature.

\textbf{T3.} COS (Theorem~\ref{thm:price}) provides an alternative
$O(NM)$ pricing pipeline for environments without a Fox $H$
evaluator, with the single-interval choice of
Remark~\ref{rem:single_interval} amortizing across term structures.
