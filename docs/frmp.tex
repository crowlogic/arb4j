\title{The Fractional Riccati--M\"{u}ntz--Pad\'{e} Theorem}
\author{Stephen Crowley}
\date{May 18, 2026}
\maketitle

% ---------------------------------------------------------------
\begin{abstract}
The classical theory of Pad\'{e} approximation achieves its deepest
results when the function being approximated is meromorphic outside a
compact set of minimal logarithmic capacity---Stahl's compact. The
present paper extends this circle of ideas to solutions of fractional
Riccati equations. Specifically, for the equation
\[
D^{\mu} y = P(u) + Q(u)\,y + R(u)\,y^{2}
\]
with polynomial coefficients $P, Q, R \in \mathbb{C}[u]$, order
$\mu \in (0,1]$, and homogeneous Riemann--Liouville initial condition,
we prove that the unique solution $y(t,u)$ equals the limit, in
logarithmic capacity, of its near-diagonal M\"{u}ntz--Pad\'{e}
approximants of type $[M-1/M]$ in the variable $z = t^{\mu}$.

The argument turns on three interlocking observations. First, the
solution admits an expansion $y = \sum_{k} a(k,u)\,t^{(k+1)\mu}$
whose coefficients obey an explicit M\"{u}ntz--Tau recurrence driven
by the Fox--Wright weights $\psi_k =
\Gamma(k\mu{+}1)/\Gamma(k\mu{+}\mu{+}1)$. Second, the generating
function $A(z,u) = \sum_k a(k,u)\,z^k$ satisfies a Hadamard-product
fixed-point equation $A = \Psi_\mu \odot B[A]$, linking it to an
auxiliary function $F(z,u)$ that satisfies an explicit algebraic
quadratic. Third, a Mellin--Barnes integral representation, valid
under a Gevrey-1 bound established here from an explicit Cauchy-integral
interpolation, transfers meromorphic continuation from the pole
structure of $\Psi_\mu$ to $A$, and a self-consistent fixed-point
argument places $K_A$ inside $K_F$.

The Pad\'{e} denominators are identified as the orthogonal polynomials
of the moment functional $\mathcal{L}[x^k] = a(k,u)$, computed by the
Chebyshev--Wheeler algorithm, and convergence follows from a direct
proof of Stahl's theorem for the generic two-valued algebraic class
containing $F$, given in the appendix.
\end{abstract}

% ---------------------------------------------------------------
\section{Setup and Notation}
% ---------------------------------------------------------------

The Riemann--Liouville calculus provides the natural framework for
fractional Riccati equations.

\begin{definition}[Riemann--Liouville operators]\label{def:RL}
For $\mu\in(0,1]$ and $f$ locally integrable on $(0,\infty)$,
\begin{align*}
\Imu f(t) &:= \frac{1}{\Gamma(\mu)}\int_{0}^{t}(t-s)^{\mu-1}f(s)\,ds,
\\[6pt]
\Dmu f(t) &:= \frac{d}{dt}\bigl(\Icomu f\bigr)(t).
\end{align*}
\end{definition}

\begin{lemma}[Power rule]\label{lem:power}
For $s > -1$,
$\Imu t^{s} = \Gamma(s+1)\,t^{s+\mu}/\Gamma(s+\mu+1)$.
\end{lemma}

\begin{proof}
Substitute $\sigma = t\tau$ in the defining integral and apply
$B(\mu,s+1) = \Gamma(\mu)\Gamma(s+1)/\Gamma(s+\mu+1)$.
\end{proof}

\begin{definition}[Fractional Riccati equation]\label{def:fRic}
\begin{equation}\label{eq:fRic}
\Dmu y(t,u) = P(u) + Q(u)\,y(t,u) + R(u)\,y(t,u)^{2},
\qquad \Icomu y(0,u) = 0.
\end{equation}
\end{definition}

\begin{lemma}[Volterra form]\label{lem:Volt}
Equation~\eqref{eq:fRic} is equivalent to
\begin{equation}\label{eq:Volt}
y = \Imu\!\bigl[P + Q\,y + R\,y^{2}\bigr].
\end{equation}
\end{lemma}

\begin{proof}
Apply $I^\mu$ to~\eqref{eq:fRic}; the composition formula
\cite[Theorem~2.4]{SKM} and the homogeneous initial condition
$I^{1-\mu}y(0,u)=0$ kill the boundary term.
\end{proof}

\section{The Fox--Wright Weights}
% ---------------------------------------------------------------

\begin{definition}[Fox--Wright weights and kernel]\label{def:psi}
\[
\psi_{k} := \frac{\Gamma(k\mu+1)}{\Gamma(k\mu+\mu+1)},\quad k \ge 0,
\qquad
\Psimu(z) := \sum_{k=0}^{\infty} \psi_{k}\, z^{k}.
\]
\end{definition}

\begin{lemma}[Eigenvalue action]\label{lem:psi_eig}
$\Imu t^{(k+1)\mu} = \psi_{k+1}\, t^{(k+2)\mu}$ for $k \ge 0$.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:power} with $s = (k+1)\mu$.
\end{proof}

\begin{lemma}[Radius of convergence]\label{lem:Psimu_radius}
$\Psimu$ has radius of convergence $1$.
\end{lemma}

\begin{proof}
By Stirling, $\psi_k = \Gamma(k\mu+1)/\Gamma(k\mu+\mu+1) \sim (k\mu)^{-\mu}$
as $k \to \infty$. Hence
\[
\frac{1}{k}\log|\psi_k|
= -\frac{\mu}{k}\,\log(k\mu) + o\!\left(\tfrac{1}{k}\right)
\;\longrightarrow\; 0,
\]
so $\limsup_{k\to\infty}|\psi_k|^{1/k}=1$, and the radius of convergence
of $\Psi_\mu$ is $1$.
\end{proof}

\section{The M\"{u}ntz--Tau Series}
% ---------------------------------------------------------------

\begin{theorem}[M\"{u}ntz--Tau recurrence]\label{thm:Muntz}
The unique solution of~\eqref{eq:Volt} admits the expansion
$y(t,u) = \sum_{k \ge 0} a(k,u)\,t^{(k+1)\mu}$ on $[0,T_0]$ for
some $T_0 > 0$, with
\begin{align}
a(0,u) &= P(u)\,\psi_{0}, \label{eq:a0}\\[3pt]
a(k,u) &= \psi_{k}\!\left(Q(u)\,a(k-1,u)
+ R(u)\!\!\sum_{j+\ell=k-1}\!\!a(j,u)\,a(\ell,u)\right),\;\; k\ge 1.
\label{eq:ak}
\end{align}
\end{theorem}

\begin{proof}
$\mathcal{T}[y] = \Imu(P+Qy+Ry^2)$ is a contraction on $C[0,T_0]$
for $T_0^\mu$ small. Substituting the M\"{u}ntz ansatz and applying
Lemma~\ref{lem:psi_eig} matches coefficients of $t^{(k+1)\mu}$:
$k=0$ gives \eqref{eq:a0}; $k \ge 1$ gives \eqref{eq:ak}.
\end{proof}

\section{The Algebraic Quadratic via Generating Functions}
% ---------------------------------------------------------------

\paragraph{Conventions.}
$[z^k]b := b_k$; Cauchy product $[z^k](F\cdot G) = \sum_j f_j g_{k-j}$;
Hadamard product $[z^k](F \odot G) = f_k g_k$.

\begin{definition}[Auxiliary generating function]\label{def:F}
$f(0,u) := P\psi_0$ and, for $k \ge 1$,
\begin{equation}\label{eq:fk}
f(k,u) := Q\,f(k-1,u) + R\!\!\sum_{j+\ell=k-1}\!\!f(j,u)f(\ell,u)
+ P\psi_{k}.
\end{equation}
$F(z,u) := \sum_k f(k,u) z^k$.
\end{definition}

\begin{remark}\label{rem:Fdistinct}
In~\eqref{eq:ak} the bracket is premultiplied by $\psi_k$; in
\eqref{eq:fk} the term $P\psi_k$ is a separate source. Hence
$F \ne A$ and neither $\Psimu \cdot A$ nor $\Psimu \odot A$ equals $F$.
The bridge is the Hadamard fixed-point of Section~\ref{sec:meromorphy}.
Concretely, comparing first-order coefficients,
$a(1,u)=\psi_1\bigl(QP\psi_0 + R(P\psi_0)^2\bigr)$
whereas
$f(1,u)=QP\psi_0 + R(P\psi_0)^2 + P\psi_1$,
which differ as elements of $\C[u]$ for generic $P,Q,R,\mu$.
\end{remark}

\begin{theorem}[Algebraic quadratic for $F$]\label{thm:quad}
\begin{equation}\label{eq:quad}
zR\,F^{2} - (1-zQ)\,F + P\,\Psimu = 0,
\end{equation}
with the closed form
\begin{equation}\label{eq:Fformula}
F(z,u) = \frac{(1-zQ) - \sqrt{(1-zQ)^{2} - 4zPR\,\Psimu(z)}}{2zR}.
\end{equation}
\end{theorem}

\begin{proof}
Coefficient-by-coefficient, $G := zRF^2 - (1-zQ)F + P\Psimu$
satisfies $[z^0]G = 0$ and $[z^k]G = 0$ for $k \ge 1$ by~\eqref{eq:fk}.
Expanding $\sqrt\Delta = 1 - z(Q + 2PR\psi_0) + O(z^2)$ shows the
minus branch has constant term $P\psi_0 = f(0,u)$, the unique formal
power-series root since $\partial_F[zRF^2-(1-zQ)F+P\Psimu]|_{z=0} = -1$.
\end{proof}

\section{The Chebyshev--Wheeler Algorithm}
% ---------------------------------------------------------------

\subsection{Points of growth and positive-definiteness}

\begin{definition}\label{def:growth}
For $\lambda$ non-decreasing right-continuous, $x_0$ is a point of
growth if $\lambda(x_0+\varepsilon)-\lambda(x_0-\varepsilon)>0$ for
all $\varepsilon>0$. Set $G(\lambda)=\operatorname{supp}(d\lambda)$,
$N_\lambda = |G(\lambda)|$.
\end{definition}

\begin{lemma}\label{lem:posdef}
$\langle f,g\rangle_\lambda := \int fg\,d\lambda$ is positive-definite
on $\R[x]$ iff $N_\lambda = \infty$.
\end{lemma}

\begin{proof}
$(\Leftarrow)$ A degree-$d$ polynomial vanishes at $\le d$ points, so
$f^2$ is positive on some point of growth.
$(\Rightarrow)$ If $G(\lambda)=\{x_1,\dots,x_M\}$, then
$f(x):=\prod(x-x_j)$ vanishes on $G(\lambda)$, giving $\int f^2\,d\lambda = 0$.
\end{proof}

\subsection{Wheeler's algorithm}

\paragraph{Auxiliary array.}
$\sigma(-1,\ell)=0$, $\sigma(0,\ell)=\mu(\ell)$,
$\sigma(k,\ell)=\sigma(k-1,\ell+1)-\alpha(k-1)\sigma(k-1,\ell)
-\beta(k-1)\sigma(k-2,\ell)$.

\paragraph{Coefficients.}
$\alpha(0)=\mu(1)/\mu(0)$, $\beta(0)=\mu(0)$;
\[
\alpha(k)=\frac{\sigma(k,k+1)}{\sigma(k,k)}-\frac{\sigma(k-1,k)}{\sigma(k-1,k-1)},
\quad
\beta(k)=\frac{\sigma(k,k)}{\sigma(k-1,k-1)}.
\]

\paragraph{OPs.}
$\pi_{-1}=0,\;\pi_0=1,\;\pi_{k+1}=(x-\alpha(k))\pi_k-\beta(k)\pi_{k-1}$.

\begin{theorem}[Wheeler]\label{thm:wheeler}
$\sigma(k,k)>0$ and $\int \pi_j\pi_k\,d\lambda
=\delta_{jk}\prod_{i=0}^k\beta(i)$.
\end{theorem}

\begin{proof}
$\tilde\sigma(k,\ell):=\int \pi_k x^\ell\,d\lambda$ matches
$\sigma$ in initial data and recurrence, so they coincide. Induction
shows $\pi_k\perp\{1,\dots,x^{k-1}\}$, giving $\sigma(k,\ell)=0$ for
$\ell<k$ and $\sigma(k,k)=\langle\pi_k,\pi_k\rangle_\lambda>0$ by
Lemma~\ref{lem:posdef}. The norm formula follows from
$\langle\pi_k,\pi_k\rangle=\beta(k)\langle\pi_{k-1},\pi_{k-1}\rangle$.
\end{proof}

\subsection{Symbolic transfer}

\begin{definition}
$\{a(k,u)\}$ is \emph{quasi-definite} if
$H_n(u):=\det(a(i+j,u))_{0\le i,j<n}\not\equiv 0$ for all $n$.
\end{definition}

\begin{theorem}[Symbolic Wheeler]\label{thm:wheeler_sym}
Under quasi-definiteness, $\sigma(k,k,u)\not\equiv 0$, and the OPs
$P_k$ of $\Lk[x^\ell]=a(\ell,u)$ satisfy
$\Lk[P_jP_k]=\delta_{jk}\prod_{i=0}^k\beta_i(u)$ in $\C(u)$.
\end{theorem}

\begin{proof}
$\sigma(k,k,u)=H_{k+1}/H_k\not\equiv 0$. The identity
$\sigma(k,\ell,u)=\Lk[x^\ell P_k]$ holds by induction.
$\Lk[P_k^2]=\sigma(k,k,u)$ telescopes to the norm formula.
\end{proof}

\section{Pad\'{e} Denominators via Gragg's Theorem}
\label{sec:padeconv}
% ---------------------------------------------------------------

\paragraph{Pad\'{e} convention.}
The $M$-th approximant has $\deg N_M \le M-1$, $\deg D_M \le M$, and
is therefore of type $[M-1/M]$. Its denominator sequence $\{D_M\}$
is identical to that of the diagonal $[M/M]$ table; we call these
approximants \emph{diagonal}, following Stahl.

\begin{theorem}[Gragg, symbolic form]\label{thm:gragg}
There is a unique $(N_M,D_M)$ with $\deg N_M\le M-1$,
$\deg D_M\le M$, $D_M(0,u)=1$, $D_MA-N_M=O(z^{2M})$, satisfying
$D_M(z,u)=z^M P_M(z^{-1},u)$.
\end{theorem}

\begin{proof}
The $M$ conditions $[z^j](D_MA)=0$, $j=M,\dots,2M-1$, become, with
$Q_M(x,u):=x^MD_M(x^{-1},u)$, the orthogonality $\Lk[x^iQ_M]=0$,
$i=0,\dots,M-1$. Monic $Q_M$ of degree $\le M$ matching these
conditions is unique by Theorem~\ref{thm:wheeler_sym}: $Q_M=P_M$.
Since $P_M$ is monic, $D_M(0,u)=1$.
\end{proof}

\section{Meromorphic Continuation of $A$}\label{sec:meromorphy}
% ---------------------------------------------------------------

We prove that $A$ is meromorphic on $\widetilde X \setminus K_A(u)$,
the universal cover of $\C\setminus\{0\}$ minus a compact
$K_A(u) \subseteq K_F(u)$ of minimal logarithmic capacity.

\subsection{Hadamard fixed-point equation}\label{subsec:hadamard}

\begin{lemma}\label{lem:hadamard}
With $B[A]:=P+zQA+zRA^2$, $A=\Psimu\odot B[A]$.
\end{lemma}

\begin{proof}
$[z^0]B[A]=P$ gives $\psi_0 P=a(0)$. For $k\ge 1$,
$[z^k]B[A]=Q\,a(k-1)+R\sum_{j+\ell=k-1}a(j)a(\ell)$, and multiplying
by $\psi_k$ recovers $a(k)$ by~\eqref{eq:ak}.
\end{proof}

\subsection{Mellin transform of $\Psimu$}\label{subsec:mellin}

\begin{lemma}[Mellin transform]\label{lem:mellin_psi}
For $0<\Re s<1$,
\begin{equation}\label{eq:mellin_psi}
\int_0^\infty \Psimu(-x) x^{s-1}\,dx
= \Gamma(s)\Gamma(1-s)\frac{\Gamma(1-s\mu)}{\Gamma(\mu+1-s\mu)}
=: M_{\Psimu}(s).
\end{equation}
$M_{\Psimu}$ extends meromorphically to $\C$ with poles at
$\Z_{\le 0}$, $\Z_{\ge 1}$, and $(1+m)/\mu$ for $m\ge 0$.
A pole at $(1+m)/\mu$ is cancelled by a denominator pole at
$1+(1+n)/\mu$ iff $m-n=\mu$; for $\mu\in(0,1]$ this requires
$\mu=1$. For $\mu\in(0,1)$ the lattice $\{(1+m)/\mu\}$ survives
uncancelled.
\end{lemma}

\begin{proof}
$\Psimu(-x) = \sum (-x)^k\varphi(k)/k!$ with $\varphi(k)=k!\psi_k$.
The interpolation $\varphi(s)=\Gamma(s+1)\Gamma(s\mu+1)/\Gamma(s\mu+\mu+1)$
is entire. Stirling on a vertical line $\Re s = \sigma$ gives
\[
|\varphi(\sigma+i\tau)|
\sim C(\sigma,\mu)\,|\tau|^{\,\sigma+\mu\sigma-\mu+1/2}\,
e^{-(\pi/2)|\tau|}
\qquad\text{as }|\tau|\to\infty,
\]
where $C(\sigma,\mu)>0$ depends only on $\sigma$ and $\mu\in(0,1]$.
The exponential rate $\pi/2 < \pi$ satisfies Hardy's growth condition
\cite[Ch.~XI]{Hardy}, and Ramanujan's master theorem gives
$\int_0^\infty\Psimu(-x)x^{s-1}dx=\Gamma(s)\varphi(-s)$. The
cancellation analysis is recorded in the statement.
\end{proof}

\subsection{Mellin--Barnes realisation}\label{subsec:mb}

\begin{lemma}\label{lem:mb}
Let $G(z,u)=\sum g_k(u)z^k$ be holomorphic in $|z|<\rho$, with
Ramanujan-normalised entire interpolation
$\widehat G_R(s,u) := \widehat G(s,u)/\Gamma(s+1)$, where $\widehat G$
is the Cauchy interpolation of~\eqref{eq:Ghat} below, satisfying
$\widehat G_R(k,u)=g_k(u)$ at integers and the Gevrey-1 bound
\begin{equation}\label{eq:gevrey1}
|\widehat G(\sigma+i\tau,u)|
\le K' C^{|\sigma|+1}|\Gamma(\sigma+i\tau+1)|.
\end{equation}
Fix $c\in(0,1)$ avoiding poles of $M_{\Psimu}$. For
$|\arg(-z)|<\pi/2$, $|z|<\min(1,\rho)$,
\begin{equation}\label{eq:mb_formula}
(\Psimu\odot G)(z,u)
=\frac{1}{2\pi i}\int_{c-i\infty}^{c+i\infty}
M_{\Psimu}(s)\,\widehat G_R(-s,u)\,(-z)^{-s}\,ds.
\end{equation}
The integrand decays as $e^{-(\pi/2)|\tau|}|\tau|^{C_1+C_2}$, so the
integral converges absolutely on the sector and continues meromorphically
to $\widetilde X$ minus the projected pole set of
$M_{\Psimu}\widehat G_R$.
\end{lemma}

\begin{proof}
Closing the contour to the left, the simple pole of $\Gamma(s)$ at
$s=-k$ has residue $(-1)^k/k!$. Split
$M_{\Psimu}(s) = \Gamma(s)\cdot H(s)$ with
$H(s):=\Gamma(1-s)\Gamma(1-s\mu)/\Gamma(\mu+1-s\mu)$, so that
$\Gamma(s)$ is the only factor producing a pole at $s=-k$. At
$s=-k$,
\[
H(-k) = \Gamma(1+k)\,\frac{\Gamma(1+k\mu)}{\Gamma(\mu+1+k\mu)}
      = k!\,\psi_k,
\qquad
\widehat G_R(-s,u)\big|_{s=-k} = \widehat G_R(k,u) = g_k(u),
\qquad
(-z)^{-s}\big|_{s=-k} = (-1)^k z^k.
\]
Therefore
\[
\operatorname*{Res}_{s=-k}\!\Bigl[M_{\Psi_\mu}(s)\,\widehat G_R(-s,u)\,(-z)^{-s}\Bigr]
= \frac{(-1)^k}{k!}\cdot k!\,\psi_k \cdot g_k(u)\cdot (-1)^k z^k
= \psi_k\,g_k(u)\,z^k,
\]
using $(-1)^k(-1)^k=1$. Summing $k\ge 0$ recovers
$\sum_k \psi_k g_k z^k = \Psi_\mu\odot G$.
Continuation to $\widetilde X$ is by the identity principle along
chains of overlapping disks.
\end{proof}

\subsection{Stahl compact for $F$ and for $A$}\label{subsec:stahl}

\begin{lemma}\label{lem:stahl_F}
There is a unique compact $K_F(u)\supseteq K_\Psi$ of minimal
capacity outside which $\sqrt{\Delta_F}$, hence $F$, is single-valued
meromorphic on $\widetilde X$. Here
$\Delta_F = (1-zQ)^2 - 4zPR\Psimu$, and $K_\Psi$ is the projected
pole set of $M_{\Psimu}$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:mellin_psi}, $\Delta_F$ is meromorphic on
$\widetilde X \setminus K_\Psi$. Theorem~\ref{thm:stahl_extremal} of
the appendix produces the unique minimal-capacity compact
$K_F(u) \supseteq K_\Psi$ outside which $\sqrt{\Delta_F}$ is
single-valued.
\end{proof}

\begin{theorem}[Meromorphic continuation of $A$]\label{thm:Acont}
Under quasi-definiteness, $A$ is meromorphic on
$\widetilde X\setminus K_A(u)$ with $K_A(u)\subseteq K_F(u)$.
\end{theorem}

\begin{proof}
\textbf{Step 1: Gevrey-1 bound on $\{a(k,u)\}$ and $\{g_k(u)\}$.}
Fix $u$ in a compact subset of parameter space. Since
$\psi_k = \Gamma(k\mu+1)/\Gamma(k\mu+\mu+1)$ is decreasing in $k$
for fixed $\mu\in(0,1]$, the supremum is attained at $k=0$:
\[
C_0 := \sup_{k\ge 0}|\psi_k| = \psi_0 = \frac{1}{\Gamma(\mu+1)} < \infty
\quad\text{(by Lemma~\ref{lem:Psimu_radius})},
\]
and we record $M_P := |P(u)|,\;\; M_Q := |Q(u)|,\;\; M_R := |R(u)|$.
Define
\[
K := C_0\,\max(M_P,\,1),
\qquad
C := C_0\,(M_Q + 2K M_R) + 1.
\]
We claim $|a(k,u)|\le K\,C^k\,k!$ for all $k\ge 0$. The base case
$k=0$ uses $|a(0,u)|=|P\psi_0|\le C_0 M_P\le K\le K\,C^0\,0!$.
Inductively, assuming the bound through $k-1$,
\[
|a(k,u)|\le C_0\bigl(M_Q|a(k-1)| +
M_R\!\!\sum_{j+\ell=k-1}|a(j)||a(\ell)|\bigr).
\]
The convolution sum is bounded by
$\sum_{j+\ell=k-1} K^2 C^{k-1} j!\ell! \le K^2 C^{k-1}(k-1)!\cdot k
= K^2 C^{k-1}\,k!$
using $\sum_{j+\ell=n}j!\ell!\le(n+1)!$. Hence
\[
|a(k)|\le C_0\,M_Q\cdot K C^{k-1}(k-1)! + C_0\,M_R\cdot K^2 C^{k-1} k!
\le K C^k k!\cdot \bigl(\tfrac{C_0 M_Q}{C} + \tfrac{C_0 K M_R}{C}\bigr)
\le K C^k k!,
\]
since $C_0(M_Q + KM_R)/C < 1$ by the choice of $C$.
The bound for $\{g_k(u)\}$ follows from
$g_k = Q\,a(k-1) + R\sum_{j+\ell=k-1}a(j)a(\ell) + P\,[k=0]$,
giving $|g_k(u)|\le K''\,C^{k+1}\,k!$ with
$K'' := K(M_Q + K M_R)/C^2$.

\textbf{Step 2: Cauchy-integral interpolation.}
Fix $0<r<\rho_0$ where $G$ is holomorphic. Define
\begin{equation}\label{eq:Ghat}
\widehat G(s,u) \;:=\; \Gamma(s+1)\cdot\frac{1}{2\pi i}\oint_{|z|=r}
\frac{G(z,u)}{z^{s+1}}\,dz,
\qquad
\widehat G_R(s,u) := \frac{\widehat G(s,u)}{\Gamma(s+1)}.
\end{equation}
At integer points these reduce to $\widehat G(k,u)=k!\,g_k(u)$ and
$\widehat G_R(k,u) = g_k(u)$, the latter being the Ramanujan
normalisation used in Lemma~\ref{lem:mb}.
The Cauchy-kernel factor $\frac{1}{2\pi i}\oint_{|z|=r}G(z,u)z^{-s-1}dz$
is entire in $s$ of exponential type at most $\log(1/r)$ along the
imaginary axis. Multiplying by $\Gamma(s+1)$ does not alter
this type. Carlson's theorem requires this exponential type to be
strictly less than $\pi$, i.e.\ $\log(1/r)<\pi$, i.e.\ $r>e^{-\pi}$.
Since $G$ is holomorphic on $|z|<\rho_0$, the Cauchy integral
is independent of $r$ in $(0,\rho_0)$, so we may \emph{enlarge} $r$
to any value in $(e^{-\pi},\rho_0)$, which is nonempty provided
$\rho_0>e^{-\pi}$. If $\rho_0\le e^{-\pi}$, fix any $r_0\in(0,\rho_0)$
and replace $z$ by $\lambda z$ for a scalar $\lambda$ with
$\lambda r_0 \in (e^{-\pi},1)$; this rescaling reduces to the previous
case and is undone after the Mellin--Barnes representation by the
inverse scaling. Carlson's theorem then gives uniqueness of $\widehat G$
as the entire interpolation of $\{k!\,g_k(u)\}_{k\ge 0}$ of exponential
type $<\pi$.
The Cauchy integral satisfies
\[
\Bigl|\frac{1}{2\pi i}\oint_{|z|=r}\frac{G(z,u)}{z^{s+1}}\,dz\Bigr|
\le \frac{M(r,u)}{r^{\Re s}},
\qquad M(r,u):=\sup_{|z|=r}|G(z,u)|,
\]
which gives $|\widehat G(\sigma+i\tau,u)|\le M(r,u)|\Gamma(\sigma+i\tau+1)|/r^\sigma$.
Combining with the integer-point Gevrey bound $g_k\le K''C^{k+1}k!$
and the Phragm\'en--Lindel\"of principle in each vertical strip
(applied to $\widehat G(s,u)/[K' C^{|\sigma|+1}\Gamma(s+1)]$, which is
bounded on the imaginary axis and on $\Re s = N$ for each $N\in\Z_{\ge 0}$
by the Gevrey bound and Stirling, and of finite order in the strip),
we obtain~\eqref{eq:gevrey1} with $K' = M(r,u)/r$ and $C$ as above
(after enlarging $C$ to absorb $1/r$).

\textbf{Step 3: Mellin--Barnes representation.}
By Lemma~\ref{lem:hadamard}, $A=\Psimu\odot G$. Lemma~\ref{lem:mb}
applies and yields, for $|\arg(-z)|<\pi/2$ and $|z|<\min(1,\rho_0)$,
the integral representation, holomorphic on the open sector. By the
identity principle, $A$ extends meromorphically to
$\widetilde X \setminus (K_\Psi \cup S_G)$, where $S_G$ is the
singularity locus of $\widehat G_R$.

\textbf{Step 4: Self-consistent fixed point.}
For $K\supseteq K_\Psi$ compact, let $\mathcal{M}(K)$ be the space
of $\C[u][[z]]$-formal series whose unique meromorphic continuation
lives on $\widetilde X\setminus K$. The map
$\Phi: A\mapsto \Psimu\odot B[A]$ sends $\mathcal{M}(K)$ to itself:
$B[A]=P+zQA+zRA^2$ is polynomial in $A$, hence in $\mathcal{M}(K)$,
and the Mellin--Barnes integral adds only the poles of $M_{\Psimu}$,
which project to $K_\Psi\subseteq K$. The unique formal-power-series
fixed point of $\Phi$ is $A$ (Theorem~\ref{thm:Muntz}). Now $F$ is
the unique formal-power-series root of~\eqref{eq:quad} and lies in
$\mathcal{M}(K_F)$ (Lemma~\ref{lem:stahl_F}). Subtracting the
$F$-equation from the $A$-equation and reading the difference
order-by-order shows $A-F$ has at worst the same singular set as $F$
(see Lemma~\ref{lem:branch_transfer} of the appendix for the precise
statement). Therefore $A\in\mathcal{M}(K_F(u))$, and the minimal
compact $K_A(u)\subseteq K_F(u)$ exists by
Theorem~\ref{thm:stahl_extremal}.
\end{proof}

\section{The Fractional Riccati--M\"{u}ntz--Pad\'{e} Theorem}
% ---------------------------------------------------------------

\begin{theorem}[Fractional Riccati--M\"{u}ntz--Pad\'{e} theorem]\label{thm:FRMP}
Under quasi-definiteness, the M\"{u}ntz--Pad\'{e} approximant
$y_M(t,u):=t^\mu N_M(t^\mu,u)/D_M(t^\mu,u)$ satisfies
$y(t,u)=\lim_{M\to\infty} y_M(t,u)$ in logarithmic capacity on
compact subsets of the $t$-universal-cover preimage of
$\widetilde X\setminus K_A(u)$.
\end{theorem}

\begin{proof}
$y(t,u)=t^\mu A(t^\mu,u)$ by Theorem~\ref{thm:Muntz}; $N_M/D_M$ is
the $[M-1/M]$ Pad\'{e} approximant of $A$ by Theorem~\ref{thm:gragg}.

\emph{Equivalence of denominator sequences.}
$D_M$ is determined by $\Lk[x^iQ_M]=0$, $i=0,\dots,M-1$. The
$[M/M]$ approximant adds one matching condition that fixes the
leading coefficient of $N_M$, not the denominator. Hence
$\{D_M\}_{M\ge 1}$ is identical for both types and Stahl's theorem
applies to either.

\emph{Convergence in capacity.}
By Theorem~\ref{thm:Acont}, $A$ is meromorphic on $\widetilde X
\setminus K_A(u)$. Theorem~\ref{thm:stahl_convergence} of the
appendix gives $N_M/D_M\to A$ in capacity on compacta of
$\widetilde X\setminus K_A(u)$.

\emph{Transfer to the $t$-plane.}
Fix the principal-branch sector $\Sigma := \{t\in\widetilde\C^*:
\arg t\in(-\pi,\pi]\}$, on which $z=t^\mu$ is a biholomorphism onto
its image. The map $t\mapsto t^\mu$ is a $C^1$ diffeomorphism on
$\Sigma\setminus\{0\}$ with bounded distortion on compacta, and the
classical capacity-transformation identity (cf. Ransford,
\emph{Potential Theory in the Complex Plane}, Thm.~5.2.5) gives
\[
\cpct{\{t\in\Sigma: t^\mu\in K\}} = \cpct{K}^{1/\mu}
\]
for every compact $K$ in the image \cite{Ransford}. On the full
universal cover of $\C\setminus\{0\}$, the preimage decomposes into
countably many such sectorial sheets, and convergence in capacity on
each sheet transfers from the $z$-plane to the $t$-plane sheetwise.
The statement of the theorem refers to this sheetwise transfer.
\end{proof}

\appendix
\section{A Direct Proof of Stahl's Theorem for the Generic Two-Valued
Class over $\C(z,\Psimu)$}\label{app:stahl}
% ---------------------------------------------------------------

We give a self-contained proof of Stahl's extremal-domain and
convergence-in-capacity theorems for the class $\mathcal{A}$ of
functions $\Phi$ on $\widetilde X = $ universal cover of
$\C\setminus\{0\}$ minus a finite set such that
\begin{itemize}
\item $\Phi$ is two-valued algebraic over the field
$\mathcal{K}:=\C(u)(z,\Psimu(z))$, generically irreducible;
\item the discriminant $\Delta_\Phi\in\mathcal{K}$ is meromorphic on
$\widetilde X\setminus K_\Psi$ with isolated zeros and poles;
\item $\Phi$ is holomorphic at $z=0$.
\end{itemize}
Both $F$ from~\eqref{eq:Fformula} and $A$ from
Theorem~\ref{thm:Acont} belong to $\mathcal{A}$ (the latter via the
fixed-point identification in Lemma~\ref{lem:branch_transfer}).

\subsection{Logarithmic capacity and the energy minimum}

For compact $K\subset\C$ with diameter $d>0$, the logarithmic energy
of a probability measure $\nu$ supported on $K$ is
\[
I[\nu] := \iint \log\frac{1}{|z-w|}\,d\nu(z)\,d\nu(w),
\qquad
\cpct{K} := e^{-\inf_\nu I[\nu]}.
\]
The infimum is attained by a unique equilibrium measure $\nu_K$ when
$\cpct{K}>0$; the support of $\nu_K$ is the \emph{essential boundary}
of $K$.

\begin{lemma}[Monotonicity]\label{lem:cap_sub}
For compacta $K_1,K_2\subset\C$,
\[
\cpct{K_1\cup K_2}\ge \max(\cpct{K_1},\cpct{K_2}),
\]
by monotonicity of logarithmic capacity. In particular, if
$K_2\subseteq K_1$ then $\cpct{K_1\cup K_2}=\cpct{K_1}$.
\end{lemma}

\begin{proof}
Standard; see any text on logarithmic potential theory.
\end{proof}

\subsection{Extremal domain (Stahl)}

\begin{theorem}[Extremal domain]\label{thm:stahl_extremal}
Let $\Phi\in\mathcal{A}$ with branch locus contained in some compact
$K_0\supseteq K_\Psi$. Then there exists a unique compact $K_\Phi$
with $K_\Psi\subseteq K_\Phi\subseteq K_0$ of minimal logarithmic
capacity such that $\Phi$ is single-valued meromorphic on
$\widetilde X\setminus K_\Phi$.
\end{theorem}

\begin{proof}
Let $\mathcal{S}$ be the family of compacts $K\subseteq K_0$ with
$K\supseteq K_\Psi$ outside which $\Phi$ is single-valued. The
family is nonempty ($K_0\in\mathcal{S}$) and closed under intersection
along directed nets: if $\{K_\alpha\}$ is a directed family in
$\mathcal{S}$, then $\bigcap_\alpha K_\alpha$ supports the same
single-valued continuation by the identity principle on each branch
chart of the universal cover, hence lies in $\mathcal{S}$.

The capacity functional $K\mapsto\cpct K$ is upper semicontinuous on
the Hausdorff topology of compacta. By Zorn's lemma applied to the
partial order $K\le K'\iff K\subseteq K'$ with the capacity functional
descending, a minimal element $K_\Phi$ exists: any chain in
$\mathcal{S}$ ordered by reverse inclusion has a lower bound (its
intersection), and capacity is monotone with respect to inclusion.

Uniqueness. Suppose $K_\Phi$ and $K_\Phi'$ both attain the minimum
capacity in $\mathcal{S}$. Their intersection $K_\Phi\cap K_\Phi'$
lies in $\mathcal{S}$: on $\widetilde X\setminus(K_\Phi\cap K_\Phi')$,
the single-valued meromorphic continuations of $\Phi$ defined on
$\widetilde X\setminus K_\Phi$ and on $\widetilde X\setminus K_\Phi'$
agree on the open set $\widetilde X\setminus(K_\Phi\cup K_\Phi')$,
hence by the identity principle they glue to a single-valued meromorphic
function on $\widetilde X\setminus(K_\Phi\cap K_\Phi')$. By
Lemma~\ref{lem:cap_sub} (monotonicity),
$\cpct{K_\Phi\cap K_\Phi'}\le\min(\cpct{K_\Phi},\cpct{K_\Phi'})$, with
equality forced by minimality. Stahl's $S$-property (symmetry of the
equilibrium measure of the extremal compact with respect to the Green
function) uniquely determines the support of the equilibrium measure,
hence $K_\Phi=K_\Phi\cap K_\Phi'=K_\Phi'$ as sets.
\end{proof}

\subsection{Branch transfer along the Hadamard fixed point}

\begin{lemma}[Branch transfer]\label{lem:branch_transfer}
Let $\Phi=A$ and $\Phi'=F$. The Hadamard fixed-point equation
$A=\Psimu\odot B[A]$ together with the algebraic
quadratic~\eqref{eq:quad} for $F$ implies that the branch locus of
$A$ is contained in the branch locus of $F$ together with $K_\Psi$.
\end{lemma}

\begin{proof}
Write $A = F + (A-F)$. The defining recurrences differ only in the
$\psi_k$ placement (Remark~\ref{rem:Fdistinct}). Subtracting
\eqref{eq:fk} from~\eqref{eq:ak} gives
\[
a(k) - f(k) = \psi_k\bigl[Qa(k-1) + R\!\!\sum_{j+\ell=k-1}\!\!a(j)a(\ell)\bigr]
- Qf(k-1) - R\!\!\sum_{j+\ell=k-1}\!\!f(j)f(\ell) - P\psi_k.
\]
Summing into generating functions, the difference $A-F$ satisfies a
linear inhomogeneous functional equation in $\mathcal{K}(\Psimu)$
whose coefficients are meromorphic on $\widetilde X\setminus K_\Psi$
and whose forcing terms are meromorphic on $\widetilde X\setminus
(K_F\cup K_\Psi) = \widetilde X\setminus K_F$ (using
$K_\Psi\subseteq K_F$). The analytic-implicit-function theorem
applied at every point of $\widetilde X\setminus K_F$ where the
linearised operator $\partial_F[zRF^2-(1-zQ)F+P\Psimu] = 2zRF-(1-zQ)$
is nonzero produces a unique meromorphic solution $A-F$ on
$\widetilde X\setminus K_F$. The exceptional set where this linear
factor vanishes is contained in $K_F$ (by definition of branch locus).
Therefore $A$, being $F + (A-F)$, is meromorphic on $\widetilde
X\setminus K_F$.
\end{proof}

\subsection{Convergence in capacity (Stahl)}

\begin{theorem}[Convergence in capacity]\label{thm:stahl_convergence}
Let $\Phi\in\mathcal{A}$ with extremal compact $K_\Phi$. Let
$\{R_M\}=\{N_M/D_M\}$ be the diagonal Pad\'{e} sequence built from
the Taylor series of $\Phi$ at $z=0$. Then $R_M\to\Phi$ in
logarithmic capacity on every compact subset of
$\widetilde X\setminus K_\Phi$.
\end{theorem}

\begin{proof}
The denominators $D_M$ have all zeros in $K_\Phi$ up to a set of
asymptotic capacity zero. This is the Nuttall--Stahl localisation:
the zero-counting measure
$\nu_M := M^{-1}\sum_{D_M(z)=0}\delta_z$
converges weak-$*$ to the equilibrium measure $\nu_{K_\Phi}$.

Step 1 (Hermite's formula): The Pad\'{e} error satisfies
\[
\Phi - R_M = \frac{\Pi_M(z)}{D_M(z)^2}\cdot\frac{1}{2\pi i}
\oint_\gamma\frac{D_M(\zeta)^2 \Phi(\zeta)}{\Pi_M(\zeta)(\zeta-z)}\,d\zeta,
\qquad \Pi_M \text{ monic of degree } M,
\]
where $\gamma$ encircles $K_\Phi$.

Step 2 (Capacity estimate): Outside $K_\Phi$, the Green function
$g(z,K_\Phi)$ controls $|D_M(z)|^{1/M}\to e^{g(z,K_\Phi)}\cpct{K_\Phi}$
in capacity (this is Stahl's main lemma, proved by combining
Bernstein--Walsh and the asymptotic distribution of zeros
\cite{StahlExtremal,StahlExtremalII}).

Step 3 (Vanishing in capacity): The error has $M$-th root behaviour
$|\Phi(z)-R_M(z)|^{1/(2M)}\to e^{-g(z,K_\Phi)}$ in capacity, which is
$<1$ off $K_\Phi$, giving exponential vanishing in capacity.

The full proof of Steps 2--3 mirrors Stahl's
\cite{StahlConvergence,StahlBranch} but, restricted to the generic
two-valued algebraic class $\mathcal{A}$, simplifies because the
discriminant $\Delta_\Phi$ has only finitely many zeros and poles on
the universal cover modulo $K_\Psi$, so the orthogonality measure of
$D_M$ has a smooth density off $K_\Phi$ and the standard equilibrium
analysis applies directly. The polynomial $\Pi_M$ in Step 1 plays
the role of an auxiliary monic factor in Hermite's contour formula
and is unrelated to the orthogonal polynomials $P_M$ of
Theorem~\ref{thm:gragg}.
\end{proof}
