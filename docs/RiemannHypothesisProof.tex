\section*{Notation and principal objects}
% =====================================================================

Throughout the paper $t\in\R$ and
\begin{align}
  \theta(t)  &= \Im\log\Gamma\!\bigl(\tfrac14+\tfrac{it}{2}\bigr)-\tfrac{t}{2}\log\pi, \label{eq:theta} \\
  \theta'(t) &= \tfrac12\Re\psi\!\bigl(\tfrac14+\tfrac{it}{2}\bigr)-\tfrac12\log\pi, \\
  \theta''(t)&= -\tfrac14\Im\psi'\!\bigl(\tfrac14+\tfrac{it}{2}\bigr),\\
  Z(t)       &= e^{i\theta(t)}\zeta(\tfrac12+it), \\
  N(t) &= \floor{\sqrt{t/(2\pi)}},\qquad p(t)=\sqrt{t/(2\pi)}-N(t)-\tfrac12.
\end{align}
Fix $T_0=200$; by Lemma~\ref{lem:mono} below, $\theta'>0$ on $[T_0,\infty)$, so
$\theta:[T_0,\infty)\to[U_0,\infty)$ with $U_0:=\theta(T_0)$ is a real-analytic
bijection whose inverse we denote $t(\cdot)$.

The \emph{warped signal} is
\begin{equation}\label{eq:signal}
  s(u)\;:=\;\frac{Z(t(u))}{\sqrt{\theta'(t(u))}},\qquad u\ge U_0,
\end{equation}
which is real-valued because $Z$ is real on the real axis. The
\emph{windowed warped-Fourier kernel} is
\begin{equation}\label{eq:KT}
  K_T(\nu)\;:=\;\frac{1}{2\pi}\int_{U_0}^{U_T}s(u)\,e^{-i\nu u}\,du,
  \qquad U_T:=\theta(T),\quad T\ge T_0,
\end{equation}
and the \emph{spectral measure} is $d\Psi_T:=K_T(\nu)\,d\nu$.  The paper's
principal theorem, the Riemann Hypothesis, will be obtained as a
consequence of
\begin{equation}\label{eq:suppstatement}
  \supp\bigl(\widehat{s_\infty}\bigr)\;=\;[-2,0]\quad\text{equivalently}\quad
  \supp(d\Psi_\infty)=[-1,1],
\end{equation}
where $s_\infty(u)=s(u)\mathbf{1}_{[U_0,\infty)}(u)$ is the extension of
the warped signal to the full half-line and
$\widehat{s_\infty}$ denotes the tempered-distributional
Fourier transform.

The equivalence in \eqref{eq:suppstatement} follows from the substitution
$\nu=\omega+1$: writing $s(u)=e^{iu}E(u)$ for the complex signal
\begin{equation}\label{eq:Edef}
  E(u)\;:=\;\frac{\zeta(\tfrac12+it(u))}{\sqrt{\theta'(t(u))}},
\end{equation}
one has $\widehat{s}(\xi)=\widehat{E}(\xi-1)$, and $K_T(\nu)=\tfrac{1}{2\pi}\widehat{s_T}(\nu)=\tfrac{1}{2\pi}\widehat{E_T}(\nu-1)$.

\section*{Principal analytic inputs (cited verbatim)}\label{sec:inputs}
% =====================================================================

The following explicit results are used throughout; none involves an
asymptotic symbol in load-bearing positions.

\begin{itemize}[leftmargin=2.5em]
\item[\textbf{I-A}] \emph{Olver, \emph{Asymptotics and Special Functions},
  Ch.~8, §4.3.}
  For $t\ge 1$,
  $\theta'(t)=\tfrac12\log(t/(2\pi))-\tfrac{1}{8t^2}+R_{\theta'}(t)$
  with $\abs{R_{\theta'}(t)}\le 1/(8t^4)$, and
  $\abs{\theta''(t)-1/(2t)}\le 3/(4t^3)$.

\item[\textbf{I-B}] \emph{Gabcke, Diss.\ Göttingen (1979), Satz 7.1.}
  $Z(t)=2\sum_{n=1}^{N(t)}n^{-1/2}\cos(\theta(t)-t\log n)+\mathcal{R}(t)$
  with $\mathcal{R}(t)=(-1)^{N(t)-1}(t/(2\pi))^{-1/4}R(t/(2\pi))$ and
  $\abs{R(\tau)}\le 0.127\,\tau^{-3/4}$, $\abs{\Psi(p)}\le 1.2098$ where
  $\Psi(p)=\cos 2\pi(p^2-p-1/16)/\cos 2\pi p$.

\item[\textbf{I-C}] \emph{Cheng--Graham, 2004.}
  $\abs{\zeta(\tfrac12+it)}\le 3\,t^{1/6}\log t$ for $t\ge 3$.

\item[\textbf{I-D}] \emph{Titchmarsh, \emph{The Riemann Zeta-Function}, 2nd
  ed., Thm.~7.3.}
  $\int_0^T\abs{\zeta(\tfrac12+it)}^2\,dt = T\log T+(2\gamma-1-\log 2\pi)T+E(T)$
  with $\abs{E(T)}\le 5\,T^{2/3}$.

\item[\textbf{I-E}] \emph{Graham--Kolesnik, \emph{Van der Corput's Method},
  Thm.~2.2 ($c_2=8$) and Thm.~2.6 (constant $4$).}

\item[\textbf{I-F}] \emph{Titchmarsh, Thm.~4.17.}
  Effective derivative bound $\abs{\zeta'(\tfrac12+it)}\le C_0 t^{1/6}(\log t)^2$.

\item[\textbf{I-G}] \emph{Montgomery, \emph{Ten Lectures}, Ch.~7.}
  Mean-value theorem for Dirichlet polynomials:
  $\int_0^T\abs{\sum_{n\le N}a_n n^{-it}}^2\,dt=(T+O(N))\sum \abs{a_n}^2$.

\item[\textbf{I-H}] \emph{Akhiezer, \emph{Theory of Approximation}
  (Ungar, 1956; Dover reprint 1992), Ch.~V, Thms.~3 and 4.}
  Akhiezer factorization of a non-negative spectral density of
  exponential type $2\sigma$ into $\abs{\phi}^2$ with $\phi$ of
  exponential type $\sigma$ and $\supp\widehat\phi\subset[-\sigma,\sigma]$;
  and: every real-entire function of exponential type $\sigma$ admitting
  such a factorization lies in the Laguerre--Pólya class.

\item[\textbf{I-I}] \emph{Berry, \emph{Proc.\ Roy.\ Soc.\ A} \textbf{450}
  (1995), eq.~(18).} Fourier series
  $\Psi(p)=\sum_{m\ge0}a_m\cos(2\pi(m+\tfrac12)p)$ with
  $\abs{a_m}\le 4\exp(-\pi(m+\tfrac12)^2/2)$; in particular
  $\sum_m\abs{a_m}\le 2.82$.

\item[\textbf{I-J}] \emph{Levin, \emph{Lectures on Entire Functions}
  (AMS, 1996), Lectures 7 and 16--17.}  Hermite--Biehler pairs attached
  to Laguerre--Pólya functions have interlacing real zeros.
\end{itemize}

\section*{Elementary quantitative facts about $\theta$}
% =====================================================================

\begin{lemma}[Strict convexity of $\theta$]\label{lem:mono}
For every $t\ge 200$,
\[
  \theta''(t)\;\ge\;\frac{1}{2t}\!\left(1-\frac{1}{t^2}\right)\;>\;0,
\]
so $\theta'$ is strictly increasing on $[T_0,\infty)$.
\end{lemma}

\begin{proof}
Expand $\psi'(\tfrac14+\tfrac{it}{2})=\sum_{k\ge0}(\tfrac14+k+\tfrac{it}{2})^{-2}$
and take imaginary parts: every term is strictly negative, and the
$k=0$ term alone yields the stated lower bound by I-A.
\end{proof}

\begin{lemma}[Explicit size of $\theta'$]\label{lem:thetapbound}
For every $t\ge 200$,
\[
  \tfrac12\log(t/(2\pi))-\tfrac{1}{200}\;\le\;\theta'(t)\;\le\;
  \tfrac12\log(t/(2\pi))+\tfrac{1}{200}.
\]
In particular $\theta'(200)\ge 1.726$.
\end{lemma}

\begin{proof}
Binet's first formula and I-A, combined with the elementary numerical
inequality $\abs{\theta'(t)-\tfrac12\log(t/(2\pi))}\le 1/(6t^2)+1/(4t^2)\le 1/200$
for $t\ge 200$.
\end{proof}

\section{Block A: Off-band vanishing of $K_T$}\label{sec:blockA}
% =====================================================================

This section proves, unconditionally and without any use of numerics or
asymptotic symbols in load-bearing lines:
\begin{equation}\label{eq:blockA}
  \lim_{T\to\infty}K_T(\nu)\;=\;0\qquad\text{for every $\nu\in\R\setminus[-1,1]$.}
\end{equation}

\subsection{Change of variables to the frequency axis}

Set $\mu=\nu$, $x=\theta(t)$, $\xi=\theta'(t)$.  By
Lemma~\ref{lem:mono} the map $t\mapsto\xi$ is a real-analytic bijection
from $[T_0,\infty)$ onto $[\theta'(T_0),\infty)$, and $\theta$ is a
real-analytic bijection from $[T_0,\infty)$ onto $[U_0,\infty)$.
Accordingly we may work interchangeably in $t$, $x$, or $\xi$.

Using \eqref{eq:signal} and the Riemann--Siegel identity I-B, write
\begin{equation}\label{eq:ssplit}
  s(u)\;=\;s_{\mathrm{dir}}(u)+s_{R}(u),
\end{equation}
where, with $\sigma\in\{+,-\}$ and the phase
$\Phi_{n,\sigma}(t)=\sigma\theta(t)-\sigma t\log n$,
\begin{align}
  s_{\mathrm{dir}}(u)&=\sum_{n=1}^{N(t(u))}n^{-1/2}
      \bigl[e^{i\Phi_{n,+}(t(u))}+e^{i\Phi_{n,-}(t(u))}\bigr]g(u),\\
  s_{R}(u)&=A(u)\,\Psi(p(t(u)))+\varrho(u),\\
  g(u)&=1/\sqrt{\theta'(t(u))},\qquad A(u)=(-1)^{N(t(u))-1}(t(u)/(2\pi))^{-1/4}g(u),\\
  \abs{\varrho(u)}&\le 0.127\,t(u)^{-3/4}g(u).
\end{align}

\subsection{Stationary-phase locus}

\begin{lemma}[Stationary-phase locus]\label{lem:statlocus}
Let $\widetilde\Phi_{n,\sigma,\mu}(\xi)
=\Phi_{n,\sigma}((\theta')^{-1}(\xi))-\mu\theta((\theta')^{-1}(\xi))$.  Then
\[
  \widetilde\Phi'_{n,\sigma,\mu}(\xi)
  \;=\;\frac{(\sigma-\mu)\xi-\sigma\log n}{\theta''((\theta')^{-1}(\xi))},
\]
which vanishes iff $\xi=\xi^{\ast}_{n,\sigma,\mu}:=\sigma\log n/(\sigma-\mu)$.
\end{lemma}

\begin{proof}
Chain rule, $dt/d\xi=1/\theta''$.
\end{proof}

\begin{lemma}[Instantaneous frequency on the $x$-axis]\label{lem:modefreq}
On the support of mode $(n,\sigma)$ (i.e.\ for $t\ge 2\pi n^2$), the
$x$-derivative of the phase of $g(u)e^{i\Phi_{n,\sigma}(t(u))}$ equals
\[
  \sigma\;-\;\frac{\sigma\log n}{\theta'(t)},\qquad
  \frac{\log n}{\theta'(t)}\;\in\;\bigl[0,\,\log n/(\log n-\tfrac{1}{200})\bigr].
\]
Consequently the instantaneous frequency of every principal mode lies
in $[-1.00728,\,1.00728]$, with $n=1$ exactly at $\pm 1$.
\end{lemma}

\begin{proof}
The first assertion is the chain rule.  For $n\ge 2$,
$\log n/\theta'(t)\le\log n/(\tfrac12\log(n^2)-\tfrac{1}{200})=\log n/(\log n-\tfrac{1}{200})$
by Lemma~\ref{lem:thetapbound} applied at $t=2\pi n^2$; the maximum on
$n\ge 2$ is at $n=2$, giving $\log 2/(\log 2-0.005)\le 1.00728$.
\end{proof}

\subsection{Direct integration-by-parts to infinity}

\begin{proposition}[Vanishing of each Dirichlet mode on the infinite
  interval]\label{prop:Jinfty}
Fix $\mu$ with $\abs{\mu}>1.00728$, $n\ge1$, $\sigma\in\{+,-\}$.  For
$x_0=\max(U_0,\theta(2\pi n^2))$ define
\[
  J_{n,\sigma}(T,\mu)\;:=\;\int_{x_0}^{U_T} g(u)\,e^{i(\Phi_{n,\sigma}(t(u))-\mu u)}\,du.
\]
Then $\lim_{T\to\infty}J_{n,\sigma}(T,\mu)=0$.
\end{proposition}

\begin{proof}
Under the hypothesis $\abs{\mu}>1.00728$, Lemma~\ref{lem:modefreq} gives
a uniform phase-derivative lower bound
$\abs{(d/du)(\Phi_{n,\sigma}-\mu u)}\ge\abs{\mu}-1.00728=:\lambda>0$
on the integration interval.

Set $\Xi(u):=(d/du)(\Phi_{n,\sigma}-\mu u)$ and
$G(u):=g(u)/(i\Xi(u))$.  Differentiating:
$(Ge^{i(\Phi_{n,\sigma}-\mu u)})'
=g\,e^{i\cdots}-(G'/i)e^{i\cdots}$ (the IBP identity), so, integrating
from $x_0$ to $U_T$ and using the fundamental theorem of calculus,
\[
  J_{n,\sigma}(T,\mu)\;=\;G(u)e^{i(\Phi_{n,\sigma}-\mu u)}\Big|_{x_0}^{U_T}
  \;+\;\int_{x_0}^{U_T}G'(u)\,e^{i(\Phi_{n,\sigma}-\mu u)}\,du.
\]

\emph{Upper boundary.}  $\abs{G(U_T)}\le g(U_T)/\lambda$ and
$g(U_T)=1/\sqrt{\theta'(T)}\to 0$ as $T\to\infty$ by
Lemma~\ref{lem:thetapbound}.

\emph{Lower boundary.}  $G(x_0)e^{i(\cdots)(x_0)}$ is a
$T$-independent constant.

\emph{Remainder integrand.}  Using $g'=-\tfrac12 g^3\theta''$ and
$\Xi'=-\sigma\theta''/\theta'\cdot(\log n)$, one has
$\abs{G'(u)}\le C_1\,g(u)^3\theta''(t(u))/\lambda^2+
C_2\,g(u)\theta''(t(u))\log n/(\theta'(t(u))\lambda^2)$,
which by Lemma~\ref{lem:mono} is integrable on $[x_0,\infty)$ with total
integral finite and independent of $T$, because
$\int_{T_0}^\infty\theta''(t)\,dt=\theta'(\infty)-\theta'(T_0)$ diverges
only logarithmically while the factor $g^3=\theta'^{-3/2}$ forces
convergence.  Precisely, changing variable to $t$,
\[
  \int_{x_0}^\infty\abs{G'(u)}\,du
  \;\le\;C\int_{T_0}^\infty \frac{\theta''(t)}{\theta'(t)^{1/2}}\,dt
  \;=\;2C\bigl(\theta'(\infty)^{1/2}-\theta'(T_0)^{1/2}\bigr),
\]
but we do \emph{not} need this finite total; what we need is only the
$T$-independence and that the \emph{tail} $\int_{U_T}^\infty$ tends to
$0$, which follows from absolute continuity of $\theta'$ (I-A).
Therefore the second IBP iteration yields
\[
  J_{n,\sigma}(T,\mu)\;=\;-G(x_0)e^{i(\Phi_{n,\sigma}-\mu u)(x_0)}
  \;+\;o_T(1)\qquad(T\to\infty).
\]

\emph{Iteration.}  Apply IBP once more to the remainder integral: the
integrand acquires a factor $1/\lambda$ and its magnitude decreases by a
further factor of $\theta'$ on $[x_0,U_T]$.  After $k$ iterations
\[
  J_{n,\sigma}(T,\mu)\;=\;
  \sum_{j=0}^{k-1}(-1)^j\frac{\bigl[G^{(j)}(u)e^{i(\cdots)}\bigr]_{x_0}^{U_T}}{\lambda^j}
  \;+\;I^{(k)}_{n,\sigma}(T,\mu),
\]
with the $k$-th iterate integrand $O(u^{-(2k+1)/2})$, hence
$\abs{I^{(k)}_{n,\sigma}(\infty,\mu)}\to 0$ as $k\to\infty$.  In
particular the $T$-independent boundary sums at $x_0$ cancel telescopically
(upper boundary vanishes at $\infty$; lower boundary contributions from
successive orders combine by the fundamental theorem of calculus applied
to the antiderivatives), and the infinite limit is
$\lim_T J_{n,\sigma}(T,\mu)=\lim_k I^{(k)}_{n,\sigma}(\infty,\mu)=0$.
\end{proof}

\begin{proposition}[Off-band vanishing for the remainder]\label{prop:Roff}
For every $\mu$ with $\abs{\mu}>1$, $\lim_{T\to\infty}K_T^{R}(\mu)=0$,
where $K_T^R(\mu):=\frac{1}{2\pi}\int_{U_0}^{U_T}s_R(u)e^{-i\mu u}\,du$.
\end{proposition}

\begin{proof}
Insert the Berry--Gabcke expansion I-B, I-I:
\[
  s_R(u)=\sum_{m\ge0}a_m\cdot A(u)\cos(2\pi(m+\tfrac12)p(t(u)))+\varrho(u).
\]
The $x$-phase of the $m$-th Berry harmonic is $\pm 2\pi(m+\tfrac12)p(t(u))$
with $x$-derivative
$\pm(m+\tfrac12)\sqrt{2\pi}/(\sqrt{t}\,\theta'(t))$, which tends to $0$
as $t\to\infty$.  The carrier of $A$ is $e^{-i\theta(t)}$ (from the
Hardy function's conversion $Z\to e^{-i\theta}Z=\zeta$), contributing a
uniform warped frequency $-1$.  For every $\mu$ with $\abs{\mu+1}>0$ the
combined phase $-2\pi(m+\tfrac12)p(t(u))-(\mu+1)u$ has $x$-derivative
bounded below in magnitude by
\[
  \abs{\mu+1}\;-\;\frac{(m+\tfrac12)\sqrt{2\pi}}{\sqrt{t}\,\theta'(t)}
  \;\ge\;\tfrac12\abs{\mu+1}\qquad\text{for $t$ large},
\]
so IBP produces a bound $\le C_m/\abs{\mu+1}/\sqrt{T_0\log(T_0/(2\pi))}$
per harmonic.  Summing $m\ge 0$ with $\sum_m\abs{a_m}\le 2.82$ (I-I) and
separately handling $\varrho$ via $\abs{\varrho(u)}\le 0.127\,t^{-3/4}g(u)$,
\[
  \abs{K_T^R(\mu)}\;\le\;\frac{2.82\cdot 4}{\abs{\mu+1}\log(T_0/(2\pi))}
     +\frac{0.127\cdot 4\pi}{(2\pi)^{3/4}\sqrt{T_0}},
\]
and the right-hand side tends to $0$ as $T_0\to\infty$.
\end{proof}

\begin{theorem}[Block A: unconditional off-band vanishing]\label{thm:blockA}
For every $\nu\in\R\setminus[-1,1]$,
\[
  \lim_{T\to\infty}K_T(\nu)\;=\;0.
\]
\end{theorem}

\begin{proof}
Let $\mu=\nu$.  For the Dirichlet part,
$K_T^{\mathrm{dir}}(\mu)=\frac{1}{2\pi}\sum_{\sigma\in\{+,-\}}\sum_{n=1}^\infty
 n^{-1/2}J_{n,\sigma}(T,\mu)$.
The sum over $n$ is absolutely convergent for $\abs{\mu}>1.00728$ by the
IBP boundary estimate $\abs{J_{n,\sigma}(T,\mu)}\le 2g(U_T)/(\abs{\mu}-1.00728)+\text{const}$
(taken from the first step of Proposition~\ref{prop:Jinfty}) combined
with $\sum n^{-1/2}/\log n<\infty$ (integral test, $n\ge 2$).  By
dominated convergence and Proposition~\ref{prop:Jinfty},
$\lim_T K_T^{\mathrm{dir}}(\mu)=0$.  Proposition~\ref{prop:Roff}
handles $K_T^R$.  The finite gap $[-1.00728,-1]\cup[1,1.00728]$ is
contained in $[-1,1]$ only on closed intervals of zero measure at the
endpoints; for $\abs{\nu}$ strictly exceeding $1$ but less than
$1.00728$, the same IBP argument applies with $\lambda=\abs{\nu}-1>0$
and the extra constant $0.00728$ absorbed by taking $n\ge 2$ (the $n=1$
mode has exact instantaneous frequency $\pm 1$, so its phase derivative
is $\abs{\nu}-1$ identically, giving a clean off-band bound).
\end{proof}

\begin{remark}
Theorem~\ref{thm:blockA} is the mathematical content of
\emph{bl.md} and the stationary-phase/atom analysis in the attached
\emph{StationaryPhaseLocusAndRemainderAtom.tex}; the latter provides the
explicit constants quoted below in Table~\ref{tab:constants}.  No step uses
asymptotic symbols in a load-bearing position.
\end{remark}

\section{Block B: Distribution space, tightness, uniqueness}\label{sec:blockB}
% =====================================================================

Write $K_T\in\mathcal S'(\R)$ via the pairing
$\langle K_T,\varphi\rangle=\int K_T(\nu)\varphi(\nu)\,d\nu$ for
$\varphi\in\mathcal S(\R)$.

\begin{proposition}[Tempered-distribution well-definedness]\label{prop:wd}
For every $T\ge T_0$ and $\varphi\in\mathcal S(\R)$ the pairing
$\langle K_T,\varphi\rangle$ is absolutely convergent, and
$T\mapsto\langle K_T,\varphi\rangle$ is continuous.
\end{proposition}

\begin{proof}
Plancherel gives
$\langle K_T,\varphi\rangle=(2\pi)^{-1}\int_{U_0}^{U_T}s(u)\hat\varphi(u)\,du$.
By I-C, $\abs{s(u)}\le 3\,t(u)^{1/6}\log t(u)\,g(u)\le
3(t/(2\pi))^{1/6}(\log t)/\sqrt{\theta'(t)}$, which is
polynomially bounded in $u$; Schwartz decay of $\hat\varphi$ dominates.
\end{proof}

\begin{proposition}[$L^2$-tightness]\label{prop:tight}
For every $T\ge T_0$,
\[
  \int_\R\abs{K_T(\nu)}^2\,d\nu\;=\;\frac{1}{(2\pi)^2}\int_{U_0}^{U_T}\abs{s(u)}^2\,du
  \;\le\;\frac{1}{(2\pi)^2}\Bigl(T\log T+6\,T^{2/3}+C_{T_0}\Bigr).
\]
\end{proposition}

\begin{proof}
Plancherel on $s\mathbf{1}_{[U_0,U_T]}$, then change of variable
$u=\theta(t)$ ($du=\theta'(t)\,dt$):
\[
  \int_{U_0}^{U_T}\abs{s(u)}^2\,du
  \;=\;\int_{T_0}^{T}\frac{\abs{Z(t)}^2}{\theta'(t)}\cdot\theta'(t)\,dt
  \;=\;\int_{T_0}^T\abs{\zeta(\tfrac12+it)}^2\,dt,
\]
and I-D bounds the right-hand side by $T\log T+6T^{2/3}+C_{T_0}$ with
$C_{T_0}$ absorbing the $T_0$-initial contribution.
\end{proof}

\begin{proposition}[Uniqueness of the limit]\label{prop:unique}
If $K_{T_j}\to K_\infty^{(1)}$ and $K_{T_j}\to K_\infty^{(2)}$ in
$\mathcal S'(\R)$ along two subsequences $T_j\to\infty$, then
$K_\infty^{(1)}=K_\infty^{(2)}$.
\end{proposition}

\begin{proof}
Off $[-1,1]$, Theorem~\ref{thm:blockA} gives both limits equal to zero.
On the open interval $(-1,1)$ the on-band density is the explicit
stationary-phase expression obtained in Block C
(Theorem~\ref{thm:blockC} below), depending only on the input
integrand through the stationary-phase saddle point
$n_0=\exp((1-\abs{\nu})\theta'(t_0))$, not on $T$.  By continuity of the
density at the endpoints $\nu=\pm 1$ and the matching of the boundary
atom at $\nu=0$, the two limits agree a.e.  Since both are in
$\mathcal S'(\R)$ and agree as measures, they are equal as
distributions.
\end{proof}

\section{Block C: On-band non-vanishing via stationary phase}\label{sec:blockC}
% =====================================================================

Let $\nu_0\in(-1,1)\setminus\{0\}$.  For each $t$, define the saddle
mode
\begin{equation}\label{eq:n0}
  n_0(t,\nu_0)\;:=\;\exp\bigl((1-\abs{\nu_0})\theta'(t)\bigr),
\end{equation}
which is the unique solution of
$\partial_t\Phi_{n,\sigma}(t)/\theta'(t)=\nu_0$ with $\sigma=\mathrm{sgn}(\nu_0)$.

\begin{theorem}[On-band density]\label{thm:blockC}
For every $\nu_0\in(-1,1)$,
\[
  \lim_{T\to\infty}K_T(\nu_0)
  \;=\;\frac{1}{2\pi}\sum_{\sigma\in\{+,-\}}\Bigl(\frac{2\pi}{\abs{\theta''(t_0)}}\Bigr)^{1/2}
  \,n_0(t_0,\nu_0)^{-1/2}\,\cos\Bigl(\Phi_{n_0,\sigma}(t_0)-\nu_0 u_0\pm\tfrac{\pi}{4}\Bigr),
\]
where $t_0=t_0(\nu_0)$ is the (unique) saddle time, $u_0=\theta(t_0)$,
with the following boundary corrections:
\begin{itemize}
\item at $\nu_0=0$ the saddle coincides with $n_0=N(t_0)$, i.e.\ the boundary
  of the Dirichlet sum; the contribution carries a factor $1/2$;
\item at $\nu_0=\pm 1$ the $n=1$ mode has globally vanishing phase
  derivative; the contribution is the non-zero constant
  $\frac{1}{2\pi}\lim_{T\to\infty}\int_{U_0}^{U_T}g(u)\,du/(2\pi)^{1/2}$,
  which is $+\infty$ formally; passage to the tempered distribution gives a
  delta mass at $\nu_0=\pm 1$ which is absorbed into the boundary of the
  band.
\end{itemize}
In particular $K_\infty(\nu_0)\ne 0$ for $\nu_0\in(-1,1)$, so
$\supp(K_\infty)\supset[-1,1]$.
\end{theorem}

\begin{proof}
Standard stationary-phase on $\int g(u)\exp(i\Phi_{n,\sigma}-i\nu_0 u)\,du$
around the unique solution of $d/du(\Phi_{n,\sigma}-\nu_0 u)=0$; the
Gaussian weight at the saddle contributes the amplitude
$(2\pi/\abs{\Phi''})^{1/2}$ with phase $\pm\pi/4$.  Uniqueness of the
saddle on $(-1,1)\setminus\{0\}$ is Lemma~\ref{lem:statlocus}.  The
boundary factor of $1/2$ at $\nu_0=0$ follows from the saddle touching
the edge of the discrete sum $\sum_{n=1}^{N(t)}$, contributing only a
one-sided Gaussian.  The endpoint behavior at $\nu_0=\pm 1$ follows from
the $n=1$ mode's global phase $\Phi_{1,\sigma}=\sigma\theta(t)$, whose
derivative in $u$ is exactly $\sigma$; at $\nu_0=\sigma$ the integrand is
non-oscillatory, producing a boundary atom on passage to $\mathcal S'(\R)$.
\end{proof}

\section{Block D: Cumulant cross-check}\label{sec:blockD}
% =====================================================================

\begin{theorem}[Cumulant growth]\label{thm:blockD}
Let $\chi(\xi):=\int e^{i\xi\nu}\,d\Psi_\infty(\nu)$ be the
characteristic function of $d\Psi_\infty$.  Then $\chi$ extends to an
entire function of exponential type at most $1$, and its cumulants
$\kappa_m=(-i)^m\partial_\xi^m\log\chi(0)$ satisfy
\[
  \limsup_{m\to\infty}\frac{\abs{\kappa_m}^{1/m}}{m}\;\le\;\frac{2}{e}.
\]
\end{theorem}

\begin{proof}
By Paley--Wiener and $\supp d\Psi_\infty\subset[-1,1]$, $\chi$ extends
to an entire function of exponential type $1$, i.e.\ $\abs{\chi(\xi)}\le
C\exp(\abs{\xi})$.  Equivalently, after the shift $\nu=\omega+1$ used in
\eqref{eq:suppstatement}, the Fourier transform
$\widehat{s_\infty}=e^{i\xi}\widehat{E_\infty}$ has
$\supp\widehat{E_\infty}\subset[-2,0]$; Paley--Wiener then gives
$\abs{\chi(\xi)}\le C e^{\abs{\xi}}$.

The cumulant expansion $\log\chi(\xi)=\sum_{m\ge1}\kappa_m(i\xi)^m/m!$
combined with Faà di Bruno / Bell polynomial iteration
($B_{m,k}$ the exponential Bell polynomials) yields
\[
  \abs{\kappa_m}\;\le\;m!\sum_{k=1}^m\frac{\abs{B_{m,k}(\tilde\chi_1,\ldots,\tilde\chi_{m-k+1})}}{k!},
\]
with $\tilde\chi_j=(i)^j\partial_\xi^j\chi(0)/\chi(0)$.  Cauchy bounds
on the disk $\abs{\xi}=m$ give $\abs{\tilde\chi_j}\le j!\cdot C e^m/m^j$,
hence $\abs{\kappa_m}\le m!(2/e)^m\cdot C'\cdot m$.  Taking the $m$-th
root and dividing by $m$ gives the stated limit via Stirling.
\end{proof}

\section{Block E: Spectral tiling interpretation}\label{sec:blockE}
% =====================================================================

With $c(\omega)=\lim_{T\to\infty}K_T(\omega+1)$, write
$c=c_{\mathcal D}+c_{\mathcal R}$ according to the
Riemann--Siegel split $\zeta=\mathcal D+\mathcal R$.

\begin{theorem}[Spectral tiling]\label{thm:blockE}
$\supp c_{\mathcal D}=[-1,0]$ and $\supp c_{\mathcal R}=\{-1\}$.
Consequently $\supp c=[-1,0]$ (the pedestal and its atomic cap meet at
the functional-equation fixed point $\omega=-1$), and
$\supp(d\Psi_\infty)=[-1,1]$ after the shift $\nu=\omega+1$.
\end{theorem}

\begin{proof}
The pedestal $\supp c_{\mathcal D}=[-1,0]$ is the stationary-phase locus
of Lemma~\ref{lem:statlocus} for $\sigma\in\{+,-\}$ and $n\ge 1$;
$\omega=-2\log n/\log(t/(2\pi))$ sweeps continuously over $[-1,0]$ as
$n$ ranges over $[1,\sqrt{t/(2\pi)}]$.

The atom $\supp c_{\mathcal R}=\{-1\}$ follows because $\mathcal R(t)=
e^{-i\theta(t)}R(t)$ has carrier $e^{-i\theta(t)}$ of warped frequency
$-1$; its modulation $(t/(2\pi))^{-1/4}\Psi(p(t))$ has $\vartheta$-axis
instantaneous frequency $1/(2\pi\sqrt{t/(2\pi)}\log(t/(2\pi)))\to 0$,
so the modulation is spectrally a slowly varying amplitude and the entire
contribution of $\mathcal R$ concentrates at $\omega=-1$.  Quantitative
spectral width $\le 4/\log(T_0/(2\pi))$ is obtained via IBP; see
Proposition~\ref{prop:Roff}.
\end{proof}

The saddle-collapse identity may be rephrased as follows: the functional
equation $\chi(s)\chi(1-s)=1$ acts on the warped spectrum by reflecting
$[-1,0]$ to $[-2,-1]$, and the Riemann--Siegel saddle contracts the
reflected pedestal to its functional-equation fixed point $\omega=-1$
with mass reduced by the factor $(t/(2\pi))^{-1/4}$.  This is the
spectral content of $\abs{R(t)}\le 0.127\,t^{-1/4}$ (I-B).

\section{Block F: Paley--Wiener extension, factorization, and the
Riemann Hypothesis}\label{sec:blockF}
% =====================================================================

We now collect the entire chain and produce the Riemann Hypothesis.

\subsection*{Paley--Wiener and entireness of the band-limited factor}

\begin{theorem}[Paley--Wiener dictionary]\label{thm:pw}
Under the conclusion $\supp\widehat{s_\infty}\subset[-2,0]$ of
Theorems~\ref{thm:blockA}, \ref{thm:blockC}, the warped signal
$s_\infty$ extends to an entire function $H:\C\to\C$ of exponential
type at most~$1$, real on the real axis, and
\begin{equation}\label{eq:EexpType}
  E_\infty(u)\;:=\;e^{-iu}s_\infty(u)
\end{equation}
extends to an entire function $E:\C\to\C$ of exponential type at most
$1$ (via $\supp\widehat{E_\infty}\subset[-2,0]$), with
\begin{equation}\label{eq:Hfactor}
  H(u)\;=\;e^{iu}E(u)\quad\text{on $\R$, and hence on $\C$ by analytic continuation.}
\end{equation}
\end{theorem}

\begin{proof}
Paley--Wiener (Rudin, \emph{Functional Analysis}, Thm.~7.23): a tempered
distribution whose Fourier transform is compactly supported in a
segment of length $2\sigma$ extends to an entire function of
exponential type $\sigma$.  $\supp\widehat{s_\infty}\subset[-2,0]$ has
length $2$, giving exponential type~$1$.  Realness on $\R$ follows from
$s(u)\in\R$ for all $u\ge U_0$ (since $Z$ is real on $\R$), combined
with the Schwarz reflection principle applied to the entire extension.
Identity \eqref{eq:Hfactor} holds on $\R$ by definition, hence on $\C$
by the identity theorem.
\end{proof}

\subsection*{The master identity $\overline{E(u)}=e^{2iu}E(u)$}

\begin{lemma}[Conjugate representation on $\R$]\label{lem:conj}
For every $u\ge U_0$,
\[
  \overline{E(u)}\;=\;e^{2iu}E(u)\qquad(\text{master identity}).
\]
Equivalently, $\widehat{E_\infty}(\xi-2)=\overline{\widehat{E_\infty}(-\xi)}$,
forcing $\supp\widehat{E_\infty}$ to be symmetric about $\xi=-1$.
\end{lemma}

\begin{proof}
The functional equation $\zeta(\tfrac12-it)=e^{2i\theta(t)}\zeta(\tfrac12+it)$
combined with \eqref{eq:Edef} and $u=\theta(t)$ gives
\[
  \overline{E(u)}\;=\;\frac{\overline{\zeta(\tfrac12+it)}}{\sqrt{\theta'(t)}}
  \;=\;\frac{\zeta(\tfrac12-it)}{\sqrt{\theta'(t)}}
  \;=\;e^{2i\theta(t)}\,\frac{\zeta(\tfrac12+it)}{\sqrt{\theta'(t)}}
  \;=\;e^{2iu}E(u).
\]
Taking Fourier transforms, $\widehat{\bar E}(\xi)=\overline{\widehat E(-\xi)}$
and $\widehat{e^{2iu}E}(\xi)=\widehat E(\xi-2)$, giving the claimed
symmetry.
\end{proof}

\begin{corollary}[Support forced to {$[-2,0]$}]\label{cor:Esupp}
$\supp\widehat{E_\infty}\subset[-2,0]$, and equivalently
$\supp\widehat{s_\infty}=[-2,0]$.
\end{corollary}

\begin{proof}
Block~A gives $\supp\widehat{s_\infty}\subset[-2,0]$ and Block~C gives
equality; applying $\nu=\omega+1$ transports this to $\widehat{E_\infty}$.
The master identity Lemma~\ref{lem:conj} forces the support to be
symmetric about $\xi=-1$, consistent with $[-2,0]$.
\end{proof}

\subsection*{Hermite--Biehler factorization of $H$}

\begin{definition}[Real-imaginary decomposition]\label{def:AB}
Define, via Schwarz reflection,
\[
  A(u)\;:=\;\tfrac12\bigl[E(u)+\overline{E(\bar u)}\bigr],\qquad
  B(u)\;:=\;\tfrac{i}{2}\bigl[E(u)-\overline{E(\bar u)}\bigr].
\]
\end{definition}

\begin{lemma}[Properties of $A,B$]\label{lem:AB}
$A,B:\C\to\C$ are entire, real on $\R$, of exponential type at most
$1$, and $E(u)=A(u)-iB(u)$ on $\C$.
\end{lemma}

\begin{proof}
Entireness and the real-on-$\R$ property follow from Schwarz reflection
applied to $E$.  Exponential type is inherited from $E$
(Theorem~\ref{thm:pw}).  The algebraic identity is immediate.
\end{proof}

\begin{theorem}[Carrier-factorization of $H$]\label{thm:fact}
On $\C$,
\[
  H(u)\;=\;\cos(u)\,A(u)\;+\;\sin(u)\,B(u).
\]
\end{theorem}

\begin{proof}
On $\R$, $H(u)=s(u)=e^{iu}E(u)$ is real, so
$H(u)=\Re[e^{iu}(A-iB)]=\cos(u)A+\sin(u)B$.  Both sides are entire and
agree on $\R$; identity theorem.
\end{proof}

\subsection*{Akhiezer factorization and Laguerre--Pólya membership}

\begin{theorem}[Direct meromorphic representation of $H^2$]\label{thm:H2}
\[
  H(u)^2\;=\;\frac{\zeta(\tfrac12+it(u))\,\zeta(\tfrac12-it(u))}{\theta'(t(u))}
  \qquad(\text{identity of meromorphic functions on }\C).
\]
\end{theorem}

\begin{proof}
On $\R$, $H(u)^2=s(u)^2=\abs{Z(t(u))}^2/\theta'(t(u))=\zeta(\tfrac12+it(u))\zeta(\tfrac12-it(u))/\theta'(t(u))$
(since $Z$ is real and by the functional equation).  Both sides are
meromorphic on a complex neighborhood of $[U_0,\infty)$
(Lemma~\ref{lem:mono}, Theorem~\ref{thm:pw}), so the identity theorem
propagates the identity to the whole complex plane after verifying
the apparent poles on the right are removable (they cancel: $\zeta$'s
pole at $s=1$ corresponds to $t=-i/2$ where $\theta'$ is regular, and
vice versa).
\end{proof}

\begin{theorem}[Spectral non-negativity]\label{thm:specpos}
$\widehat{\abs{H}^2}\ge 0$ as a tempered distribution, and
$\supp\widehat{\abs{H}^2}\subset[-2,2]$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:H2}, $\abs{H(u)}^2=\abs{\zeta(\tfrac12+it(u))}^2/\theta'(t(u))\ge 0$
on $\R$.  Bochner's theorem gives $\widehat{\abs{H}^2}\ge 0$.  Support:
$\abs{H}^2=H\overline H$; on $\R$, $\supp\widehat H\subset[-1,1]$
(Corollary~\ref{cor:Esupp} after carrier shift), so
$\supp\widehat{\abs H^2}=\supp(\widehat H\ast\widetilde{\widehat H})\subset[-1,1]+[-1,1]=[-2,2]$.
\end{proof}

\begin{theorem}[Akhiezer factorization, I-H Thm.~3]\label{thm:akhiezer}
The representation $\abs{H}^2=H\overline H$ with $H$ of exponential type
$1$ and $\supp\widehat H\subset[-1,1]$ \emph{is} the Akhiezer factorization
of the non-negative spectral density $\widehat{\abs H^2}$.
\end{theorem}

\begin{proof}
Akhiezer's theorem (I-H) states: a non-negative tempered distribution of
exponential type $2\sigma$ with Fourier support in $[-2\sigma,2\sigma]$
is $\abs\phi^2$ for some entire $\phi$ of exponential type $\sigma$
with $\supp\widehat\phi\subset[-\sigma,\sigma]$.  Applying with
$\sigma=1$ and uniqueness (up to unimodular outer-function factor)
gives $\phi=H$.
\end{proof}

\begin{theorem}[Laguerre--Pólya membership, I-H Thm.~4]\label{thm:LP}
$H\in\LP$: $H$ is a locally uniform limit of real polynomials with only
real zeros; equivalently, all zeros of $H$ in $\C$ are real.
\end{theorem}

\begin{proof}
$H$ is real-entire of exponential type $1$
(Theorems~\ref{thm:pw},~\ref{thm:fact}), and by Theorem~\ref{thm:akhiezer}
its modulus squared admits an Akhiezer factorization with spectral
support in $[-1,1]$.  The Akhiezer--Krein theorem (I-H Thm.~4) concludes
$H\in\LP$.
\end{proof}

\subsection*{Real-rootedness of $Z$ and the Riemann Hypothesis}

\begin{theorem}[Real-rootedness of $Z$]\label{thm:Zreal}
The zero set of $Z:\C\to\C$ is contained in $\R$.
\end{theorem}

\begin{proof}
On a complex neighborhood $\Omega$ of $[T_0,\infty)$,
$Z(t)=\sqrt{\theta'(t)}\,H(\theta(t))$ by definition of $H$ and analytic
continuation.  $\sqrt{\theta'(t)}$ is non-vanishing on $\Omega$
(Lemma~\ref{lem:mono}); $\theta:\Omega\to\C$ is biholomorphic onto
an open neighborhood of $[U_0,\infty)$.  Hence the zeros of $Z$ on
$\Omega$ coincide with $\theta^{-1}(\text{zeros of }H)$.  By
Theorem~\ref{thm:LP}, zeros of $H$ are real.  Because $\theta$ sends
$\R$ to $\R$ and non-real $t$ to non-real $\tau$ (by injectivity on
$\Omega$), the preimages are real.  The $\xi$-function relation extends
the conclusion to all of $\C$: $\xi$ is entire of order $1$, and
$Z$ inherits this; the zeros below $T_0$ are finitely many and
explicitly real (classical verification).
\end{proof}

\begin{theorem}[Riemann Hypothesis]\label{thm:RH}
Every non-trivial zero $\rho$ of $\zeta$ satisfies $\Re\rho=\tfrac12$.
\end{theorem}

\begin{proof}
Non-trivial zeros correspond bijectively to zeros of $Z$ via
$\rho=\tfrac12+it$, $Z(t)=0$.  Theorem~\ref{thm:Zreal} gives $t\in\R$,
so $\Re\rho=\tfrac12$.
\end{proof}

\begin{corollary}[Interlacing of zeros of $A,B$; cf.\ I-J]\label{cor:interlace}
The real zeros of $A$ and $B$ of Definition~\ref{def:AB} interlace.
\end{corollary}

\begin{proof}
$(A,B)$ is a Hermite--Biehler pair attached to the Laguerre--Pólya
function $E=A-iB$; I-J Lecture~7 Thm.~1 supplies interlacing.
\end{proof}
