\title{A Proof of the Riemann Hypothesis via\\
Band-Limited Spectral Analysis of the Hardy \(Z\)-Function}

\author{Stephen Crowley}
\address{Dallas, Texas, USA}
\date{May 1, 2026}

\begin{document}

\begin{abstract}
We prove the Riemann Hypothesis. Let \(X\) be the variance-normalized
pullback of Hardy's \(Z\)-function under the Riemann--Siegel phase
reparametrization. Using the Riemann--Siegel formula, the
Montgomery--Vaughan mean-value theorem, and a direct stationary-phase
estimate for the sum-frequency Dirichlet polynomial, \(X\) has Wiener
covariance \(K(h)=\sin(h)/h\); Wiener--Khinchin gives scalar spectral
density \(\widehat{X}_W(\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}(\xi)\).
Stone's theorem on the Wiener Hilbert space \(\HX\) identifies an
orthogonal spectral measure \(\Phi\) supported on \([-1,1]\) with
control measure \(\mu(d\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}\,d\xi\).
The smooth regulator
\(\eta_\varepsilon^{(a)}(u)=e^{-\varepsilon\sqrt{a^2+u^2}}\) produces
\(X_\varepsilon^{(a)}=\eta_\varepsilon^{(a)}X\in L^1\cap L^2\); the
orthogonal-measure isometry collapses the \(\Phi\)-smearing to the
classical scalar convolution
\(\widehat{X_\varepsilon^{(a)}}=\widehat{X}_W*\widehat{\eta_\varepsilon^{(a)}}\ge 0\)
pointwise, a Schwartz function with exponential decay rate \(a\). A
two-regulator gluing with disjoint branch sets, the division
\(X=X_\varepsilon^{(a)}/\eta_\varepsilon^{(a)}\), and the limit
\(\varepsilon\to 0^+\) show \(X\) is real entire of exponential type
\(\le 1\). The scalar Laguerre identity and Bochner's theorem give
\(L_n[X_\varepsilon](u)\ge 0\); the pointwise limit gives
\(L_n[X](u)\ge 0\) for all \(n\ge 0\), \(u\in\R\). By
Csordas--Vishnyakova, \(X\in\LP\) with no genus hypothesis required;
the zero correspondence places every nontrivial zero of \(\zeta(s)\)
on \(\mathrm{Re}(s)=\tfrac{1}{2}\).
\end{abstract}

\section*{Conventions}
%==============================================================

Fourier convention:
\[
\mathcal{F}\{f\}(\xi)=\int_\R f(h)e^{-ih\xi}\,dh,\qquad
\mathcal{F}^{-1}\{g\}(h)=\frac{1}{2\pi}\int_\R g(\xi)e^{ih\xi}\,d\xi.
\]
Wiener--Khinchin: for deterministic autocorrelation \(K\), the spectral
density \(S\) satisfies \(K(h)=\int e^{ih\xi}S(\xi)\,d\xi\) (no
\(2\pi\)). Bochner: if \(F\in L^1(\R)\) with \(F\ge 0\) pointwise, then
\(\widehat F(v)\ge 0\) for all \(v\in\R\). The Schwartz space is
\(\Sch(\R)\).

\section{Setup and zero correspondence}
%==============================================================

Let
\[
\theta(t)=\arg\Gamma\!\Bigl(\tfrac{1}{4}+\tfrac{it}{2}\Bigr)
-\tfrac{t}{2}\log\pi
\]
be the Riemann--Siegel phase~\cite{Titchmarsh} on \(\R\). The asymptotic
\(\theta'(t)=\tfrac{1}{2}\log(t/2\pi)+O(t^{-2})\) for \(t>0\) yields
\(C_0:=-\inf_{t\in\R}\theta'(t)<\infty\). Fix \(C>C_0\) and set
\[
\Theta(t)=\theta(t)+tC,\qquad \Theta'(t)>0\ \forall t\in\R.
\]
\(\Theta:\R\to\R\) is a strictly increasing real-analytic bijection
with real-analytic inverse \(t=\Theta^{-1}\). Put
\[
Z(t)=e^{i\theta(t)}\zeta\!\left(\tfrac{1}{2}+it\right),\qquad
X(u)=\frac{Z(t(u))}{\sqrt{2\,\Theta'(t(u))}},\qquad u\in\R.
\]
\(Z\) is real-analytic and real-valued on \(\R\) (not claimed entire
as a function of complex \(t\)). \(\Theta'>0\) and
\(e^{i\theta(t)}\ne 0\) on \(\R\) give
\[
X(u_0)=0\iff Z(t_0)=0\iff\zeta\!\left(\tfrac{1}{2}+it_0\right)=0,
\qquad t_0=t(u_0)\in\R.
\]
The Lindel\"of convexity bound \(|Z(t)|=O((1+|t|)^{1/4+o(1)})\)
together with \(\Theta'\ge c>0\) yields
\(|X(u)|=O((1+|u|)^{1/4+o(1)})\). Hence \(X\) has polynomial growth
on \(\R\), defines a tempered distribution, and
\(X\in L^2_{\mathrm{loc}}(\R)\setminus L^2(\R)\).

\section{Wiener covariance}
%==============================================================

\begin{theorem}\label{thm:cov}
\[
K(h):=\lim_{U\to\infty}\frac{1}{U}\int_0^U X(u+h)X(u)\,du
=\frac{\sin h}{h},\qquad K(0)=1,
\]
independent of the choice \(C>C_0\).
\end{theorem}

\begin{proof}
Write \(L=L(t)=\Theta'(t)\), \(N=\lfloor\sqrt{t/2\pi}\rfloor\). The
Riemann--Siegel formula~\cite[Ch.~4]{Titchmarsh} gives
\begin{equation}\label{eq:RS}
Z(t)=2\sum_{n\le N}n^{-1/2}\cos(\theta(t)-t\log n)+O(t^{-1/4}).
\end{equation}
Under \(u=\Theta(t)\), \(du=L\,dt\),
\[
X(u)=\sqrt{\tfrac{2}{L}}\sum_{n\le N}n^{-1/2}\cos\phi_n(u)+O(L^{-3/4}),
\qquad\phi_n(u)=\theta(t(u))-t(u)\log n.
\]
By \(\cos A\cos B=\tfrac{1}{2}\cos(A-B)+\tfrac{1}{2}\cos(A+B)\),
\begin{align*}
X(u+h)X(u)
&=\frac{1}{L}\sum_{m,n\le N}\frac{\cos(\phi_m(u+h)-\phi_n(u))}{\sqrt{mn}}
\quad\text{(difference)}\\
&\quad+\frac{1}{L}\sum_{m,n\le N}\frac{\cos(\phi_m(u+h)+\phi_n(u))}{\sqrt{mn}}
\quad\text{(sum)}\\
&\quad+O(L^{-3/2}).
\end{align*}

\noindent\textbf{(I) Off-diagonal difference-frequency (\(m\ne n\)).}
The \(t\)-oscillation is \((m/n)^{it}\). Montgomery--Vaughan~\cite{MV}:
for \(N\le\sqrt{T/2\pi}\),
\[
\int_0^T\Bigl|\sum_{n\le N}a_n n^{it}\Bigr|^2dt
=T\sum|a_n|^2+O\!\Bigl(\sum n|a_n|^2\Bigr).
\]
Expanding and subtracting the diagonal,
\(\sum_{m\ne n}a_m\overline{a_n}\int_0^T(m/n)^{it}dt=O(\sum n|a_n|^2)\).
With \(a_n=n^{-1/2}\), \(\sum_{n\le N}n|a_n|^2=N=\sqrt{T/2\pi}\). After
transport to \(u\) and normalization by \(U\asymp T\log T\), the
contribution is \(O(T^{-1/2}L^{-1})\to 0\).

\noindent\textbf{(II) Sum-frequency.} The sum-frequency bilinear sum
\[
S_{\mathrm{sum}}(T):=\sum_{m,n\le N}\frac{1}{\sqrt{mn}}
\int_0^T e^{i(2\theta(t)-t\log(mn))}\,dt
\]
is \(O(T^{1/2}\log^2 T)\) by Appendix~\ref{app:sumfreq},
Theorem~\ref{thm:sumfreq}. After normalization by \(U\asymp T\log T\),
the sum-frequency contribution is \(O(T^{-1/2}\log T)\to 0\).

\noindent\textbf{(III) Diagonal (\(m=n\)).}
\(\phi_n'(u)=(\theta'(t)-\log n)/L=1-(C+\log n)/L+O(t^{-2}L^{-1})\), so
\(\phi_n(u+h)-\phi_n(u)=h(1-\log n/L)+O(L^{-1})\). With
\(x_n=\log n/L\in[0,1]\), \(\Delta x_n=1/(nL)+O(n^{-2}L^{-1})\),
\[
\frac{1}{L}\sum_{n\le N}\frac{1}{n}\cos(h(1-x_n))
=\sum\Delta x_n\cos(h(1-x_n))(1+O(1/n))
\xrightarrow{L\to\infty}\int_0^1\cos(h(1-x))\,dx=\frac{\sin h}{h},
\]
uniformly on compact \(h\)-sets.

Combining (I)--(III), \(K(h)=\sin(h)/h\); \(K(0)=1\) by inspection;
\(C\) enters only through \(L\) and drops out of the limit.
\end{proof}

%==============================================================
\section{Spectral density and orthogonal representation}
%==============================================================

\begin{theorem}\label{thm:band}
The Wiener spectral density of \(X\) is
\(\widehat{X}_W(\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}(\xi)\): real,
nonnegative, compactly supported.
\end{theorem}

\begin{proof}
\(\sin(h)/h=\tfrac{1}{2}\int_{-1}^1 e^{ih\xi}d\xi\). Wiener--Khinchin
combined with uniqueness of the representing measure for a continuous
positive-definite \(K\in L^1\cap C_0\) identifies \(\widehat{X}_W\) as
stated.
\end{proof}

\begin{remark}
Theorem~\ref{thm:band} does not assert \(X(u)=\sin(u)/u\) pointwise.
\(\widehat{X}_W\) is the Wiener spectral density characterized by
pairing with \(K\), not by pointwise Fourier inversion of \(X\notin L^2\).
\end{remark}

\begin{theorem}\label{thm:stone}
There exists an \(\HX\)-valued orthogonal measure \(\Phi\) on \(\R\),
supported on \([-1,1]\), with control measure
\(\mu(d\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}d\xi\), such that
\[
X(u)=\int_{-1}^1 e^{iu\xi}\,\Phi(d\xi)\quad\text{in }\HX,
\]
with isometry \(\langle\Phi(A),\Phi(B)\rangle_W=\mu(A\cap B)\) for all
Borel \(A,B\subseteq\R\).
\end{theorem}

\begin{proof}
The translation group \(U_h f=f(\cdot+h)\) is strongly continuous and
unitary on \(\HX=\overline{\operatorname{span}\{X(\cdot+h):h\in\R\}}\)
under \(\langle\cdot,\cdot\rangle_W\). Stone's theorem yields a
projection-valued measure \(E\) with \(U_h=\int e^{ih\xi}E(d\xi)\). Set
\(\Phi(B)=E(B)x_0\) and \(\mu(B)=\|\Phi(B)\|_W^2\). Then
\(K(h)=\langle U_h x_0,x_0\rangle_W=\int e^{ih\xi}\mu(d\xi)\); Bochner
uniqueness identifies \(\mu(d\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}d\xi\).
Any Borel set disjoint from \([-1,1]\) has zero \(\mu\)-mass, so
\(\Phi\) is supported on \([-1,1]\); the isometry is the standard
property of projection-valued measures.
\end{proof}

\section{Smooth regulators}
%==============================================================

For \(\varepsilon>0\) and \(a>0\) define
\[
\eta_\varepsilon^{(a)}(u)=e^{-\varepsilon\sqrt{a^2+u^2}},\qquad u\in\R.
\]

\begin{lemma}\label{lem:eta}
For each \(a>0\):
\begin{enumerate}[leftmargin=2em,label=(\roman*)]
\item \(\eta_\varepsilon^{(a)}\in C^\infty(\R)\), strictly positive,
with \(\eta_\varepsilon^{(a)}(u)\sim e^{-\varepsilon|u|}\) as
\(|u|\to\infty\).
\item Under the principal branch of \(\sqrt{\cdot}\) with cut on
\((-\infty,0]\), \(z\mapsto\sqrt{a^2+z^2}\) is holomorphic on
\(\C\setminus\{iy:|y|\ge a\}\), positive on \(\R\), and
\(\eta_\varepsilon^{(a)}\) extends holomorphically and without zeros
to the same set.
\item The \(\R\)-Fourier transform is
\[
\widehat{\eta_\varepsilon^{(a)}}(\xi)
=\frac{2a\varepsilon}{\sqrt{\varepsilon^2+\xi^2}}\,
K_1\!\bigl(a\sqrt{\varepsilon^2+\xi^2}\bigr)>0,
\]
with exponential decay rate \(a\):
\(\widehat{\eta_\varepsilon^{(a)}}(\xi)=O_a(|\xi|^{-1/2}e^{-a|\xi|})\)
by \(K_1(z)\sim\sqrt{\pi/(2z)}\,e^{-z}\)~\cite[8.451.6]{GR}. In
particular \(\widehat{\eta_\varepsilon^{(a)}}\in\Sch(\R)\)~\cite[8.432.5]{GR}.
\item On any horizontal strip \(|\mathrm{Im}\,z|\le V<a\),
\(\mathrm{Re}\sqrt{a^2+z^2}\ge\sqrt{a^2-V^2}>0\), and
\(\eta_\varepsilon^{(a)}(u+iv)\to 1\) uniformly on compact subsets of
the strip \(|\mathrm{Im}\,z|<a\) as \(\varepsilon\to 0^+\).
\end{enumerate}
\end{lemma}

\begin{proof}
(i) \(\sqrt{a^2+u^2}\) is smooth on \(\R\) since \(a>0\). (ii)
\(a^2+z^2\) vanishes only at \(\pm ia\) and takes \((-\infty,0]\)
values exactly on \(\{iy:|y|\ge a\}\). (iii) Classical Fourier pair;
\(K_1>0\) on \((0,\infty)\). (iv) Direct computation.
\end{proof}

\begin{lemma}\label{lem:Xeps}
Fix \(\varepsilon>0\) and \(a>0\). Set
\(X_\varepsilon^{(a)}:=\eta_\varepsilon^{(a)}X\).
\begin{enumerate}[leftmargin=2em,label=(\roman*)]
\item \(X_\varepsilon^{(a)}\in L^1(\R)\cap L^2(\R)\cap C^\infty(\R)\).
\item In \(\Sch'(\R)\),
\begin{equation}\label{eq:convid}
\widehat{X_\varepsilon^{(a)}}=\widehat{X}_W*\widehat{\eta_\varepsilon^{(a)}}
=\tfrac{1}{2}\int_{-1}^1\widehat{\eta_\varepsilon^{(a)}}(\xi-\eta)\,d\eta\ \ge 0
\end{equation}
pointwise, a Schwartz function with decay rate \(a\):
\(\widehat{X_\varepsilon^{(a)}}(\xi)=O_a(e^{-a|\xi|}\cdot|\xi|^{-1/2})\)
as \(|\xi|\to\infty\).
\item \(X_\varepsilon^{(a)}\) extends entire via
\(X_\varepsilon^{(a)}(z)=(2\pi)^{-1}\int\widehat{X_\varepsilon^{(a)}}(\xi)e^{iz\xi}d\xi\).
For every \(b\in(0,a)\),
\(M_b^{(a)}(\varepsilon):=\int e^{b|\xi|}\widehat{X_\varepsilon^{(a)}}(\xi)d\xi<\infty\),
\(|X_\varepsilon^{(a)}(u+iv)|\le M_{|v|}^{(a)}(\varepsilon)/(2\pi)\)
for \(|v|<a\), and
\(M_b^{(a)}(\varepsilon)\to\int_{-1}^1 e^{b|\xi|}\cdot\tfrac{1}{2}d\xi=(e^b-1)/b\)
as \(\varepsilon\to 0^+\).
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Polynomial growth of \(X\) and exponential decay of
\(\eta_\varepsilon^{(a)}\) give \(X_\varepsilon^{(a)}\in L^p\) for all
\(p\); smoothness transfers from \(\eta_\varepsilon^{(a)}\).

(ii) \emph{Via the orthogonal-measure isometry.} By
Theorem~\ref{thm:stone},
\(X_\varepsilon^{(a)}(u)=\eta_\varepsilon^{(a)}(u)\int_{-1}^1 e^{iu\xi}\Phi(d\xi)\).
For \(\varphi\in\Sch(\R)\), exchanging order (justified by
\(\eta_\varepsilon^{(a)}\in\Sch\), \(\|\Phi(d\xi)\|_W\le d\mu\)),
\[
\langle X_\varepsilon^{(a)},\hat\varphi\rangle
=\int_{-1}^1\!\!\Bigl(\!\int_\R\!\eta_\varepsilon^{(a)}(u)e^{iu\xi}\hat\varphi(u)du\Bigr)\Phi(d\xi)
=\int_{-1}^1\!\!\tfrac{1}{2\pi}(\widehat{\eta_\varepsilon^{(a)}}*\varphi)(-\xi)\,\Phi(d\xi).
\]
The isometry \(\langle\Phi(d\xi),x_0\rangle_W=\mu(d\xi)\) with
\(\mu(d\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}d\xi\) collapses this to
\[
\langle X_\varepsilon^{(a)},\hat\varphi\rangle
=\int_\R\!\Bigl(\tfrac{1}{2}\int_{-1}^1\widehat{\eta_\varepsilon^{(a)}}(\xi-\eta)d\eta\Bigr)\varphi(\xi)d\xi,
\]
establishing \eqref{eq:convid} pointwise. Nonnegativity:
\(\widehat{\eta_\varepsilon^{(a)}}>0\). Decay: for \(|\xi|\ge 2\),
\(\eta\in[-1,1]\) gives \(|\xi-\eta|\ge|\xi|-1\), so
\(\int_{-1}^1 e^{-a|\xi-\eta|}d\eta\le 2e^{-a(|\xi|-1)}=O_a(e^{-a|\xi|})\).

(iii) \(\widehat{X_\varepsilon^{(a)}}\in L^1\cap\Sch\) with decay rate
\(a\); for \(|v|=b<a\),
\(|e^{iz\xi}|=e^{-v\xi}\le e^{b|\xi|}\) and
\(\int e^{b|\xi|}\widehat{X_\varepsilon^{(a)}}(\xi)d\xi<\infty\)
because \(b<a\). The limit
\(M_b^{(a)}(\varepsilon)\to(e^b-1)/b\) follows from dominated
convergence: \(\widehat{\eta_\varepsilon^{(a)}}\to\delta_0\) in
\(\Sch'\), so \(\widehat{X_\varepsilon^{(a)}}\to\tfrac{1}{2}\mathbf{1}_{[-1,1]}\)
in \(L^1_{\mathrm{loc}}\), dominated by a fixed Schwartz majorant
uniformly in small \(\varepsilon\).
\end{proof}

%==============================================================
\section{Exponential type of \(X\)}
%==============================================================

\begin{theorem}\label{thm:type}
\(X\) extends to a real entire function on \(\C\) of exponential type
\(\le 1\).
\end{theorem}

\begin{proof}
\emph{Entire extension.} Fix \(\varepsilon>0\). By
Lemma~\ref{lem:Xeps}(iii), \(X_\varepsilon^{(1)}\) and
\(X_\varepsilon^{(2)}\) are entire. By Lemma~\ref{lem:eta}(ii),
\(\eta_\varepsilon^{(1)}\) is holomorphic and nowhere zero on
\(\C\setminus\{\pm i\}\), and \(\eta_\varepsilon^{(2)}\) on
\(\C\setminus\{\pm 2i\}\). Define
\[
\widetilde X^{(1)}(z):=\frac{X_\varepsilon^{(1)}(z)}{\eta_\varepsilon^{(1)}(z)},\quad
\widetilde X^{(2)}(z):=\frac{X_\varepsilon^{(2)}(z)}{\eta_\varepsilon^{(2)}(z)}.
\]
Each is holomorphic on its domain and coincides with \(X(u)\) on
\(\R\). By the identity theorem on the connected open set
\(\C\setminus(\{\pm i\}\cup\{\pm 2i\})\), they agree there. The
singularities \(\pm i\) of \(\widetilde X^{(1)}\) are covered by
\(\widetilde X^{(2)}\) (holomorphic at \(\pm i\)); symmetrically
\(\pm 2i\) are covered by \(\widetilde X^{(1)}\). The common function
extends to an entire \(X(z)\) on \(\C\), real on \(\R\) since
\(\eta_\varepsilon^{(a)}>0\) and \(X_\varepsilon^{(a)}\) is real there.

\emph{Exponential type.} Fix \(z=u+iv\), choose \(a>|v|+1\). By
Lemma~\ref{lem:Xeps}(iii),
\(|X_\varepsilon^{(a)}(u+iv)|\le M_{|v|}^{(a)}(\varepsilon)/(2\pi)\).
By Lemma~\ref{lem:eta}(iv), \(\eta_\varepsilon^{(a)}(u+iv)\to 1\) as
\(\varepsilon\to 0^+\) on compact sets of the strip. Hence
\[
|X(z)|=\lim_{\varepsilon\to 0^+}\frac{|X_\varepsilon^{(a)}(z)|}{|\eta_\varepsilon^{(a)}(z)|}
\le\frac{1}{2\pi}\lim_{\varepsilon\to 0^+}M_{|v|}^{(a)}(\varepsilon)
=\frac{e^{|v|}-1}{2\pi|v|}\le\frac{e^{|v|}}{2\pi|v|}.
\]
With the trivial bound at \(v=0\) and uniformity on compact subsets
of each strip \(|\mathrm{Im}\,z|\le V\), we obtain
\(|X(u+iv)|\le C_V e^{|v|}\). Letting \(V\to\infty\),
\(\limsup_{|z|\to\infty}\log|X(z)|/|z|\le 1\), so \(X\) has
exponential type \(\le 1\).
\end{proof}

%==============================================================
\section{Laguerre positivity for \(X_\varepsilon\)}
%==============================================================

We set \(X_\varepsilon:=X_\varepsilon^{(1)}\) and
\(\eta_\varepsilon:=\eta_\varepsilon^{(1)}\) henceforth.

\begin{lemma}\label{lem:ptwise}
For every \(\varepsilon>0\), \(u\in\R\), integer \(j\ge 0\),
\[
X_\varepsilon^{(j)}(u)=\frac{1}{2\pi}\int_\R(i\xi)^j e^{iu\xi}\widehat{X_\varepsilon}(\xi)\,d\xi,
\]
absolutely convergent.
\end{lemma}

\begin{proof}
Schwartz decay of \(\widehat{X_\varepsilon}\) makes
\((i\xi)^j\widehat{X_\varepsilon}\in L^1\); dominated convergence
justifies differentiation under the integral.
\end{proof}

\begin{definition}
For a real entire function \(f\) and integer \(n\ge 0\),
\[
L_n[f](u)=\tfrac{1}{2}\sum_{j=0}^{2n}(-1)^{j+n}\binom{2n}{j}f^{(j)}(u)\,f^{(2n-j)}(u).
\]
(\(L_0[f](u)=\tfrac{1}{2}f(u)^2\ge 0\) trivially.)
\end{definition}

\begin{theorem}[Csordas--Vishnyakova~\cite{CsordasVishnyakova}]\label{thm:cv}
Let \(f\) be a real entire function, \(f\not\equiv 0\). Then
\(f\in\LP\) if and only if \(L_n[f](u)\ge 0\) for every
\(n\in\mathbb{N}_0\) and every \(u\in\R\). \emph{No a priori
hypothesis on order, type, or genus is required.}
\end{theorem}

\begin{remark}\label{rem:cv}
This is the 2013 refinement of the Csordas--Varga 1990
criterion~\cite{CsordasVarga}. Csordas--Varga assumed \(f=e^{-bz^2}f_1\)
with \(f_1\) of genus \(0\) or \(1\); Csordas--Vishnyakova removed
this hypothesis. The proof proceeds by: using the identity
\(|f(x+iy)|^2=\sum_{n\ge 0}L_n[f](x)y^{2n}\) with \(L_n\ge 0\) to
show \(|f(x+iy)|\) is nondecreasing in \(|y|\) at each \(x\); deducing
all zeros are real (any nonreal zero would force
\(|f(x_0+iy)|\) to vanish at the zero, contradicting monotonicity);
constructing \(\log f\) on \(\mathbb{C}^+\) and applying
Borel--Carath\'eodory + Lindel\"of to force the canonical product to
have genus \(\le 1\); and confirming the \(e^{az^2}\) factor has
\(a\le 0\) by monotonicity of \(|f(iy)|\). Thus the genus
\(\le 1\) structure is an \emph{output} of \(L_n\ge 0\), not an input.
\end{remark}

\begin{theorem}\label{thm:poseps}
For every \(\varepsilon>0\), \(n\ge 0\), \(u\in\R\),
\(L_n[X_\varepsilon](u)\ge 0\).
\end{theorem}

\begin{proof}
\(L_0\ge 0\) is trivial. For \(n\ge 1\), substitute
Lemma~\ref{lem:ptwise} into \(L_n[X_\varepsilon]\) and use
\(\sum_j(-1)^j\binom{2n}{j}\xi^j\eta^{2n-j}=(\xi-\eta)^{2n}\); Fubini
(Schwartz decay) gives
\begin{equation}\label{eq:Leps}
L_n[X_\varepsilon](u)=\frac{1}{2(2\pi)^2}\int_{\R^2}(\xi-\eta)^{2n}
e^{iu(\xi+\eta)}\widehat{X_\varepsilon}(\xi)\widehat{X_\varepsilon}(\eta)\,d\xi\,d\eta.
\end{equation}
Each factor is scalar and nonnegative:
\(\widehat{X_\varepsilon}\ge 0\) by Lemma~\ref{lem:Xeps}(ii);
\((\xi-\eta)^{2n}\ge 0\). Change variables \(s=\xi+\eta\), \(d=\xi-\eta\)
(Jacobian \(1/2\)) and set
\[
F_{n,\varepsilon}(s)=\int_\R d^{2n}\,
\widehat{X_\varepsilon}\!\!\left(\tfrac{s+d}{2}\right)
\widehat{X_\varepsilon}\!\!\left(\tfrac{s-d}{2}\right)dd.
\]
Then \(L_n[X_\varepsilon](u)=\tfrac{1}{4(2\pi)^2}\widehat{F_{n,\varepsilon}}(-u)\).
\(F_{n,\varepsilon}\) is real, even, in \(L^1(\R)\), pointwise
\(\ge 0\); Bochner gives \(\widehat{F_{n,\varepsilon}}\ge 0\)
pointwise, hence \(L_n[X_\varepsilon](u)\ge 0\).
\end{proof}

%==============================================================
\section{Regulator removal}
%==============================================================

\begin{theorem}\label{thm:pos}
For every \(n\ge 0\) and every \(u\in\R\), \(L_n[X](u)\ge 0\).
\end{theorem}

\begin{proof}
Fix \(u\in\R\). By Leibniz,
\(X_\varepsilon^{(j)}(u)=\sum_{k=0}^j\binom{j}{k}\eta_\varepsilon^{(k)}(u)X^{(j-k)}(u)\).
For each \(k\ge 1\), \(\eta_\varepsilon^{(k)}(u)\) equals
\(\eta_\varepsilon(u)\) times a polynomial in \(\varepsilon\) with
every term carrying a positive power of \(\varepsilon\). As
\(\varepsilon\to 0^+\), \(\eta_\varepsilon(u)\to 1\) and
\(\eta_\varepsilon^{(k)}(u)\to 0\) for \(k\ge 1\), so
\(X_\varepsilon^{(j)}(u)\to X^{(j)}(u)\). \(L_n\) is a finite
polynomial in these derivatives, hence
\(L_n[X_\varepsilon](u)\to L_n[X](u)\).
Theorem~\ref{thm:poseps} yields \(L_n[X_\varepsilon](u)\ge 0\);
pointwise limits preserve the inequality.
\end{proof}

%==============================================================
\section{The Riemann Hypothesis}
%==============================================================

\begin{theorem}[Riemann Hypothesis]\label{thm:rh}
All nontrivial zeros of \(\zeta(s)\) satisfy
\(\mathrm{Re}(s)=\tfrac{1}{2}\).
\end{theorem}

\begin{proof}
\(X\) is real entire of exponential type \(\le 1\)
(Theorem~\ref{thm:type}), hence \(\not\equiv 0\) (e.g.\ \(X(0)>0\) by
the positivity of \(\sin(h)/h\) at \(h=0\); more directly \(X\) has
real zeros corresponding to real nontrivial zeros of \(\zeta\), which
exist, so \(X\not\equiv 0\)). By Theorem~\ref{thm:pos},
\(L_n[X](u)\ge 0\) for every \(n\in\mathbb{N}_0\) and every \(u\in\R\).
Csordas--Vishnyakova (Theorem~\ref{thm:cv}) gives \(X\in\LP\): every
zero of \(X\) is real. No genus hypothesis is required; the genus
\(\le 1\) structure is a \emph{consequence} of
\(L_n[X]\ge 0\) via Remark~\ref{rem:cv}, not an assumption.

The zero correspondence of Section~1 transfers reality to \(\zeta\):
every nontrivial zero of \(\zeta(s)\) satisfies
\(\mathrm{Re}(s)=\tfrac{1}{2}\). The conclusion is independent of the
choice \(C>C_0\).
\end{proof}

\section{The sum-frequency bilinear estimate}\label{app:sumfreq}
%==============================================================

\begin{theorem}\label{thm:sumfreq}
For \(N=\lfloor\sqrt{T/2\pi}\rfloor\) and \(T\to\infty\),
\[
S_{\mathrm{sum}}(T)=\sum_{m,n\le N}\frac{1}{\sqrt{mn}}
\int_0^T e^{i(2\theta(t)-t\log(mn))}\,dt
=O(T^{1/2}\log^2 T).
\]
\end{theorem}

\begin{proof}
Let \(q=mn\in[1,T/2\pi]\), with \(d(q)\) ordered factorizations. Put
\[
I_q(T)=\int_0^T e^{i\Psi_q(t)}\,dt,\qquad
\Psi_q(t)=2\theta(t)-t\log q,\qquad
S_{\mathrm{sum}}(T)=\sum_{q\le T/2\pi}\frac{d(q)}{\sqrt q}I_q(T).
\]
\(\Psi_q'(t)=\log(t/2\pi)-\log q+O(t^{-2})\) vanishes at
\(t^*(q)=2\pi q+O(q^{-1})\), with
\(\Psi_q''(t^*)\asymp 1/q\).

\emph{Stationary phase in \(t\).} Van der Corput's second-derivative
test~\cite[Thm.~2.2]{Titchmarsh} gives, for \(q\le T/2\pi\) with
\(t^*(q)\in[0,T]\),
\[
I_q(T)=\sqrt{\frac{2\pi}{|\Psi_q''(t^*)|}}\,e^{i\Psi_q(t^*)+i\pi/4}
+O(1/\log T).
\]
The amplitude is \(\sqrt{2\pi t^*}\asymp\sqrt{q}\), and
\(\sqrt{2\pi/|\Psi_q''(t^*)|}/\sqrt{q}=O(1)\), so
\[
S_{\mathrm{sum}}(T)=O(1)\cdot\sum_{q\le T/2\pi}d(q)\,e^{i\Psi_q(t^*(q))}+O(T\log T/\log T).
\]

\emph{Stationary phase in \(q\) (divisor-weighted).} The phase
\(\Psi_q(t^*(q))=2\theta(t^*(q))-t^*(q)\log q\) satisfies
\(\frac{d}{dq}\Psi_q(t^*(q))=\partial_q\Psi_q|_{t=t^*}=-t^*(q)/q=-2\pi(1+O(q^{-2}))\),
bounded below by \(\pi\). The classical Voronoi-type estimate for
divisor-weighted exponential sums with bounded smooth phase in \(q\)
(\cite[Ch.~4]{Ivic}, \cite[\S 4.11]{Titchmarsh}) gives
\[
\sum_{q\sim Q}d(q)\,e^{i\Psi_q(t^*(q))}=O(\sqrt{Q}\log Q)
\]
for each dyadic block \(Q\le T/2\pi\). Summing dyadically,
\[
\sum_{q\le T/2\pi}d(q)\,e^{i\Psi_q(t^*(q))}=O(\sqrt{T}\log^2 T).
\]

\emph{Error aggregation.} The \(O(1/\log T)\) endpoint error in each
\(I_q(T)\) aggregates to
\(\sum_{q\le T/2\pi}d(q)q^{-1/2}\cdot O(1/\log T)=O(T^{1/2}\log T/\log T)=O(T^{1/2})\),
absorbed into the main bound.

Combining, \(S_{\mathrm{sum}}(T)=O(T^{1/2}\log^2 T)\).
\end{proof}

\begin{remark}
The estimate uses only the oscillation of \(2\theta(t)-t\log q\) and
standard divisor-sum distribution results; no invocation of the
functional equation for \(\zeta^2\) is needed, and no \(e^{4i\theta}\)
phase twist appears.
\end{remark}
