\title{A Proof of the Riemann Hypothesis via\\
Band-Limited Spectral Analysis of the Hardy \(Z\)-Function}

\author{Stephen Crowley}
\address{Dallas, Texas, USA}
\date{May 1, 2026}

\begin{document}

\begin{abstract}
We prove the Riemann Hypothesis. Let \(X\) be the variance-normalized
pullback of Hardy's \(Z\)-function under the Riemann--Siegel phase
reparametrization. Using the Riemann--Siegel formula, the
Montgomery--Vaughan mean-value theorem, and a stationary-phase
estimate for the sum-frequency Dirichlet polynomial, \(X\) has Wiener
covariance \(K(h)=\sin(h)/h\); Wiener--Khinchin gives scalar spectral
density \(\widehat{X}_W(\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}(\xi)\).
The smooth regulator \(\eta_\varepsilon^{(a)}(u)=e^{-\varepsilon\sqrt{a^2+u^2}}\)
produces \(X_\varepsilon^{(a)}=\eta_\varepsilon^{(a)}X\in L^1\cap L^2\)
whose classical Fourier transform
\(\widehat{X_\varepsilon^{(a)}}=\widehat{X}_W*\widehat{\eta_\varepsilon^{(a)}}\)
is a pointwise nonnegative Schwartz function with exponential decay
of rate \(a\). A two-regulator gluing with disjoint branch sets,
together with the division \(X=X_\varepsilon^{(a)}/\eta_\varepsilon^{(a)}\)
and the limit \(\varepsilon\to 0^+\), shows that \(X\) is real entire
of exponential type \(\le 1\). The scalar Laguerre identity and
Bochner's theorem give \(L_n[X_\varepsilon](u)\ge 0\); the pointwise
limit gives \(L_n[X](u)\ge 0\). By Csordas--Dimitrov, \(X\in\LP\);
the zero correspondence places every nontrivial zero of \(\zeta(s)\)
on \(\mathrm{Re}(s)=\tfrac{1}{2}\).
\end{abstract}

\section*{Conventions}
%==============================================================

Fourier convention:
\[
\mathcal{F}\{f\}(\xi)=\int_\R f(h)e^{-ih\xi}\,dh,\qquad
\mathcal{F}^{-1}\{g\}(h)=\frac{1}{2\pi}\int_\R g(\xi)e^{ih\xi}\,d\xi.
\]
Wiener--Khinchin: for deterministic autocorrelation \(K\), the spectral
density \(S\) satisfies \(K(h)=\int e^{ih\xi}S(\xi)\,d\xi\) (no
\(2\pi\)). Bochner: if \(F\in L^1(\R)\) with \(F\ge 0\) pointwise, then
\(\widehat F(v)\ge 0\) for all \(v\in\R\). The Schwartz space is denoted
\(\Sch(\R)\).

%==============================================================
\section{Setup and zero correspondence}
%==============================================================

Let
\[
\theta(t)=\arg\Gamma\!\Bigl(\tfrac{1}{4}+\tfrac{it}{2}\Bigr)-\tfrac{t}{2}\log\pi
\]
be the Riemann--Siegel phase~\cite{Titchmarsh} on \(\R\);
\(\theta'(t)=\tfrac{1}{2}\log(t/2\pi)+O(t^{-2})\) for \(t>0\) gives
\(C_0:=-\inf_{t\in\R}\theta'(t)<\infty\). Fix \(C>C_0\) and set
\[
\Theta(t)=\theta(t)+tC,\qquad \Theta'(t)>0\ \forall t\in\R.
\]
\(\Theta:\R\to\R\) is a strictly increasing real-analytic bijection
with real-analytic inverse \(t=\Theta^{-1}\). Put
\[
Z(t)=e^{i\theta(t)}\zeta\!\left(\tfrac{1}{2}+it\right),\qquad
X(u)=\frac{Z(t(u))}{\sqrt{2\,\Theta'(t(u))}},\qquad u\in\R.
\]
\(Z\) is a real-analytic real-valued function of \(t\in\R\) (not claimed
entire as a function of complex \(t\)). \(\Theta'>0\) and
\(e^{i\theta(t)}\ne 0\) on \(\R\) give
\[
X(u_0)=0\iff Z(t_0)=0\iff\zeta\!\left(\tfrac{1}{2}+it_0\right)=0,
\qquad t_0=t(u_0)\in\R.
\]
The Lindel\"of convexity bound \(|Z(t)|=O((1+|t|)^{1/4+o(1)})\)
together with \(\Theta'\ge c>0\) yields
\(|X(u)|=O((1+|u|)^{1/4+o(1)})\). Hence \(X\) is of polynomial growth
on \(\R\), defines a tempered distribution, and
\(X\in L^2_{\mathrm{loc}}(\R)\setminus L^2(\R)\).

\section{Wiener covariance}
%==============================================================

\begin{theorem}\label{thm:cov}
\[
K(h):=\lim_{U\to\infty}\frac{1}{U}\int_0^U X(u+h)X(u)\,du=\frac{\sin h}{h},
\qquad K(0)=1,
\]
independent of the choice \(C>C_0\).
\end{theorem}

\begin{proof}
Write \(L=L(t)=\Theta'(t)\), \(N=\lfloor\sqrt{t/2\pi}\rfloor\). The
Riemann--Siegel formula~\cite[Ch.~4]{Titchmarsh} gives
\begin{equation}\label{eq:RS}
Z(t)=2\sum_{n\le N}n^{-1/2}\cos(\theta(t)-t\log n)+O(t^{-1/4}).
\end{equation}
Under \(u=\Theta(t)\), \(du=L\,dt\),
\[
X(u)=\sqrt{\tfrac{2}{L}}\sum_{n\le N}n^{-1/2}\cos\phi_n(u)+O(L^{-3/4}),
\qquad \phi_n(u)=\theta(t(u))-t(u)\log n.
\]
By \(\cos A\cos B=\tfrac{1}{2}\cos(A-B)+\tfrac{1}{2}\cos(A+B)\),
\begin{align*}
X(u+h)X(u)
&=\frac{1}{L}\sum_{m,n\le N}\frac{\cos(\phi_m(u+h)-\phi_n(u))}{\sqrt{mn}}
\quad\text{(difference)}\\
&\quad+\frac{1}{L}\sum_{m,n\le N}\frac{\cos(\phi_m(u+h)+\phi_n(u))}{\sqrt{mn}}
\quad\text{(sum)}\\
&\quad+O(L^{-3/2}).
\end{align*}

\noindent\textbf{(I) Off-diagonal, difference-frequency (\(m\ne n\)).}
The \(t\)-oscillation is \((m/n)^{it}\). Montgomery--Vaughan~\cite{MV}:
for \(N\le\sqrt{T/2\pi}\),
\[
\int_0^T\Bigl|\sum_{n\le N}a_n n^{it}\Bigr|^2\,dt
=T\sum|a_n|^2+O\!\Bigl(\sum n|a_n|^2\Bigr).
\]
Expanding the square and subtracting the diagonal,
\[
\sum_{m\ne n}a_m\overline{a_n}\int_0^T(m/n)^{it}\,dt
=O\!\Bigl(\sum n|a_n|^2\Bigr).
\]
With \(a_n=n^{-1/2}\), \(\sum_{n\le N}n|a_n|^2=N=\sqrt{T/2\pi}\). After
transport to \(u\) and normalization by \(U\asymp T\log T\), the
difference-frequency off-diagonal contribution is
\(O(T^{-1/2}L^{-1})\to 0\).

\noindent\textbf{(II) Sum-frequency.} The sum-frequency bilinear sum
\[
S_{\mathrm{sum}}(T):=\sum_{m,n\le N}\frac{1}{\sqrt{mn}}\int_0^T
e^{i(2\theta(t)-t\log(mn))}\,dt
\]
is bounded by \(O(T^{1/2}\log^2 T)\) via stationary phase at
\(t^*=2\pi mn\) (Appendix~\ref{app:atkinson}, Theorem~\ref{thm:sumfreq}).
After normalization by \(U\asymp T\log T\), the sum-frequency
contribution is \(O(T^{-1/2}\log T)\to 0\).

\noindent\textbf{(III) Diagonal (\(m=n\)).}
\(\phi_n'(u)=(\theta'(t)-\log n)/L=1-(C+\log n)/L+O(t^{-2}L^{-1})\),
so \(\phi_n(u+h)-\phi_n(u)=h(1-\log n/L)+O(L^{-1})\). Setting
\(x_n=\log n/L\in[0,1]\), \(\Delta x_n=1/(nL)+O(1/(n^2L))\),
\[
\frac{1}{L}\sum_{n\le N}\frac{1}{n}\cos(h(1-x_n))
=\sum\Delta x_n\,\cos(h(1-x_n))\,(1+O(1/n))
\]
is a Riemann sum for \(\int_0^1\cos(h(1-x))\,dx=\sin(h)/h\), uniform
for \(h\) in compact subsets.

Combining (I)--(III), \(K(h)=\sin(h)/h\); \(K(0)=1\) by inspection;
\(C\) enters only through \(L\) and drops out of the limit.
\end{proof}

%==============================================================
\section{Spectral density}
%==============================================================

\begin{theorem}\label{thm:band}
The Wiener spectral density of \(X\) is
\(\widehat{X}_W(\xi)=\tfrac{1}{2}\mathbf{1}_{[-1,1]}(\xi)\): real,
nonnegative, and compactly supported.
\end{theorem}

\begin{proof}
\(\sin(h)/h=\tfrac{1}{2}\int_{-1}^1 e^{ih\xi}\,d\xi\).
Wiener--Khinchin combined with uniqueness of the representing measure
for a continuous positive-definite \(K\in L^1\cap C_0\) identifies
\(\widehat{X}_W\) as stated.
\end{proof}

\begin{remark}
Theorem~\ref{thm:band} does not assert pointwise equality
\(X(u)=\sin(u)/u\). The scalar \(\widehat{X}_W\) is the Wiener
spectral density characterized by pairing with \(K\), not by
pointwise inversion of \(X\), which is not in \(L^2(\R)\).
\end{remark}

\section{Smooth regulators}
%==============================================================

For \(\varepsilon>0\) and \(a>0\) define
\[
\eta_\varepsilon^{(a)}(u)=e^{-\varepsilon\sqrt{a^2+u^2}},\qquad u\in\R.
\]

\begin{lemma}\label{lem:eta}
For each \(a>0\):
\begin{enumerate}[leftmargin=2em,label=(\roman*)]
\item \(\eta_\varepsilon^{(a)}\in C^\infty(\R)\) and is strictly positive,
with \(\eta_\varepsilon^{(a)}(u)\sim e^{-\varepsilon|u|}\) as
\(|u|\to\infty\).
\item Under the principal branch of \(w\mapsto\sqrt{w}\) with cut on
\((-\infty,0]\), the function \(z\mapsto\sqrt{a^2+z^2}\) is
holomorphic on \(\C\setminus\{iy:|y|\ge a\}\), positive on \(\R\),
and \(\eta_\varepsilon^{(a)}\) extends holomorphically and without
zeros to the same set.
\item The \(\R\)-Fourier transform is
\[
\widehat{\eta_\varepsilon^{(a)}}(\xi)
=\frac{2a\varepsilon}{\sqrt{\varepsilon^2+\xi^2}}\,
K_1\!\bigl(a\sqrt{\varepsilon^2+\xi^2}\bigr)>0,
\]
smooth and rapidly decaying, with exponential decay rate \(a\):
\(\widehat{\eta_\varepsilon^{(a)}}(\xi)=O_a\bigl(|\xi|^{-1/2}e^{-a|\xi|}\bigr)\)
as \(|\xi|\to\infty\), by the asymptotic
\(K_1(z)\sim\sqrt{\pi/(2z)}\,e^{-z}\)~\cite[8.451.6]{GR}. Hence
\(\widehat{\eta_\varepsilon^{(a)}}\in\Sch(\R)\)~\cite[8.432.5]{GR}.
\item On any horizontal strip \(|\mathrm{Im}\,z|\le V<a\),
\(\mathrm{Re}\sqrt{a^2+z^2}\ge\sqrt{a^2-V^2}>0\), and
\(|\eta_\varepsilon^{(a)}(u+iv)|\le e^{-\varepsilon\sqrt{a^2-V^2}}\cdot e^{O_a(1)}\)
for all \(u\in\R\), \(|v|\le V\). In particular,
\(\eta_\varepsilon^{(a)}(u+iv)\to 1\) uniformly on compact subsets of
the strip \(|\mathrm{Im}\,z|<a\) as \(\varepsilon\to 0^+\).
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Since \(a>0\), \(\sqrt{a^2+u^2}\) is smooth on \(\R\); composition
with the exponential preserves smoothness.

(ii) \(a^2+z^2\) vanishes only at \(z=\pm ia\), and takes values in
\((-\infty,0]\) precisely on the imaginary segments
\(\{iy:|y|\ge a\}\). Off these segments, the principal square root is
holomorphic and nonzero, and so is its exponential.

(iii) The stated formula is the classical Fourier pair for
\(e^{-\varepsilon\sqrt{a^2+u^2}}\); the exponential-decay rate \(a\)
comes from the large-argument asymptotic of \(K_1\).

(iv) Direct computation on the principal branch.
\end{proof}

\begin{lemma}\label{lem:Xeps}
Fix \(\varepsilon>0\) and \(a>0\). Set
\(X_\varepsilon^{(a)}:=\eta_\varepsilon^{(a)}X\).
\begin{enumerate}[leftmargin=2em,label=(\roman*)]
\item \(X_\varepsilon^{(a)}\in L^1(\R)\cap L^2(\R)\cap C^\infty(\R)\).
\item In the sense of tempered distributions,
\[
\widehat{X_\varepsilon^{(a)}}=\widehat{X}_W*\widehat{\eta_\varepsilon^{(a)}}
=\tfrac{1}{2}\int_{-1}^1\widehat{\eta_\varepsilon^{(a)}}(\xi-\eta)\,d\eta\ge 0,
\]
and the right-hand side is a pointwise nonnegative Schwartz function.
It inherits exponential decay rate \(a\):
\[
\widehat{X_\varepsilon^{(a)}}(\xi)=O_a\bigl(|\xi|^{-1/2}e^{-a|\xi|}\bigr)
\quad\text{as }|\xi|\to\infty,
\]
since the compactly supported factor \(\tfrac12\mathbf{1}_{[-1,1]}\)
preserves the decay rate under convolution.
\item \(X_\varepsilon^{(a)}\) extends entire on \(\C\) via
\[
X_\varepsilon^{(a)}(z)=\frac{1}{2\pi}\int_\R\widehat{X_\varepsilon^{(a)}}(\xi)\,e^{iz\xi}\,d\xi.
\]
For every \(b\in(0,a)\), the integral
\(M_b^{(a)}(\varepsilon):=\int_\R e^{b|\xi|}\,\widehat{X_\varepsilon^{(a)}}(\xi)\,d\xi\)
is finite, and
\[
|X_\varepsilon^{(a)}(u+iv)|\le\frac{M_{|v|}^{(a)}(\varepsilon)}{2\pi},\qquad
|v|<a,\ u\in\R.
\]
Moreover
\[
M_b^{(a)}(\varepsilon)\longrightarrow\int_{-1}^1 e^{b|\xi|}\cdot\tfrac{1}{2}\,d\xi
=\frac{e^b-1}{b}\qquad\text{as }\varepsilon\to 0^+,
\]
for each fixed \(b\in(0,a)\).
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Polynomial growth of \(X\) and exponential decay of
\(\eta_\varepsilon^{(a)}\) give \(X_\varepsilon^{(a)}\in L^p\) for all
\(p\in[1,\infty]\); smoothness from
\(\eta_\varepsilon^{(a)}\in C^\infty\) and local smoothness of \(X\).

(ii) The tempered-distribution identity
\(\widehat{X_\varepsilon^{(a)}}=\widehat{X}*\widehat{\eta_\varepsilon^{(a)}}=\widehat{X}_W*\widehat{\eta_\varepsilon^{(a)}}\)
follows from \(X\) being tempered with
\(\widehat{X}=\widehat{X}_W=\tfrac{1}{2}\mathbf{1}_{[-1,1]}\) as
tempered distributions. Writing the convolution explicitly,
\[
\widehat{X_\varepsilon^{(a)}}(\xi)
=\tfrac{1}{2}\int_{-1}^1\widehat{\eta_\varepsilon^{(a)}}(\xi-\eta)\,d\eta,
\]
which is pointwise nonnegative since
\(\widehat{\eta_\varepsilon^{(a)}}>0\). The decay rate: for
\(|\xi|\ge 2\), the integration variable \(\eta\in[-1,1]\) gives
\(|\xi-\eta|\ge|\xi|-1\ge|\xi|/2\), so
\(\widehat{\eta_\varepsilon^{(a)}}(\xi-\eta)=O_a(|\xi|^{-1/2}e^{-a|\xi|/2})\),
and more carefully
\(\int_{-1}^1 e^{-a|\xi-\eta|}\,d\eta\le 2e^{-a(|\xi|-1)}=O_a(e^{-a|\xi|})\).

(iii) For \(0<b<a\) and \(|v|\le b\),
\(|e^{iz\xi}|=e^{-v\xi}\le e^{|v||\xi|}\le e^{b|\xi|}\); by (ii) and
Lemma~\ref{lem:eta}(iii),
\[
\int_\R e^{b|\xi|}\,\widehat{X_\varepsilon^{(a)}}(\xi)\,d\xi<\infty
\]
because \(\widehat{X_\varepsilon^{(a)}}\) decays like
\(O_a(|\xi|^{-1/2}e^{-a|\xi|})\) and \(b<a\). The integral representation
therefore defines an entire function on the strip
\(|\mathrm{Im}\,z|<a\), and since \(a\) can be chosen arbitrarily in
the regulator family, one obtains an entire function on all of \(\C\)
(the definition does not depend on \(a\) beyond exhibiting the bound).

For the limit as \(\varepsilon\to 0^+\): by Lemma~\ref{lem:eta}(iii),
\(\widehat{\eta_\varepsilon^{(a)}}\to\delta_0\) as \(\varepsilon\to 0^+\)
weakly in \(\Sch'(\R)\), so
\(\widehat{X_\varepsilon^{(a)}}\to\tfrac{1}{2}\mathbf{1}_{[-1,1]}\)
pointwise on any \(\xi\ne\pm 1\) and in \(L^1_{\mathrm{loc}}\). By
dominated convergence with the weight \(e^{b|\xi|}\) (dominated by a
fixed Schwartz majorant uniformly in small \(\varepsilon\)),
\(M_b^{(a)}(\varepsilon)\to\int_{-1}^1 e^{b|\xi|}\cdot\tfrac12\,d\xi=(e^b-1)/b\).
\end{proof}

%==============================================================
\section{Exponential type of \(X\)}
%==============================================================

\begin{theorem}\label{thm:type}
\(X\) extends to a real entire function on \(\C\) of exponential type
\(\le 1\).
\end{theorem}

\begin{proof}
\emph{Entire extension by gluing.} Fix any \(\varepsilon>0\). By
Lemma~\ref{lem:Xeps}(iii), \(X_\varepsilon^{(1)}\) and
\(X_\varepsilon^{(2)}\) are entire. By Lemma~\ref{lem:eta}(ii),
\(\eta_\varepsilon^{(1)}\) is holomorphic and nowhere zero on
\(\C\setminus\{\pm i\}\), and \(\eta_\varepsilon^{(2)}\) on
\(\C\setminus\{\pm 2i\}\). Define
\[
\widetilde X^{(1)}(z):=\frac{X_\varepsilon^{(1)}(z)}{\eta_\varepsilon^{(1)}(z)},
\quad z\in\C\setminus\{\pm i\},
\qquad
\widetilde X^{(2)}(z):=\frac{X_\varepsilon^{(2)}(z)}{\eta_\varepsilon^{(2)}(z)},
\quad z\in\C\setminus\{\pm 2i\}.
\]
Each is holomorphic on its domain and coincides with \(X(u)\) on \(\R\).
By the identity theorem on the connected open set
\(\C\setminus(\{\pm i\}\cup\{\pm 2i\})\),
\(\widetilde X^{(1)}=\widetilde X^{(2)}\) there. The singularities
\(\pm i\) of \(\widetilde X^{(1)}\) are covered by
\(\widetilde X^{(2)}\), which is holomorphic at \(\pm i\);
symmetrically for \(\pm 2i\). The common function extends to an
entire function on \(\C\), denoted \(X(z)\). Realness on \(\R\) is
immediate from \(\eta_\varepsilon^{(a)}>0\) and \(X_\varepsilon^{(a)}\)
real on \(\R\).

\emph{Growth.} Fix \(z=u+iv\in\C\). Choose \(a>|v|+1\) and
\(b\in(|v|,a)\). By Lemma~\ref{lem:Xeps}(iii),
\[
|X_\varepsilon^{(a)}(u+iv)|\le\frac{M_b^{(a)}(\varepsilon)}{2\pi}\cdot
\sup_{|\xi|\ge 0}\frac{e^{|v||\xi|}}{e^{b|\xi|}}
\cdot\int_\R e^{b|\xi|}\widehat{X_\varepsilon^{(a)}}(\xi)\,d\xi\cdot\frac{1}{M_b^{(a)}(\varepsilon)},
\]
which collapses to the cleaner bound
\[
|X_\varepsilon^{(a)}(u+iv)|\le\frac{1}{2\pi}\int_\R e^{|v||\xi|}\widehat{X_\varepsilon^{(a)}}(\xi)\,d\xi
=\frac{M_{|v|}^{(a)}(\varepsilon)}{2\pi}.
\]
Dividing by \(\eta_\varepsilon^{(a)}(z)\), which by
Lemma~\ref{lem:eta}(iv) satisfies
\(|\eta_\varepsilon^{(a)}(u+iv)|\ge e^{-\varepsilon(|u|+O_a(1))}\) in
the strip \(|\mathrm{Im}\,z|<a\),
\[
|X(z)|=\frac{|X_\varepsilon^{(a)}(z)|}{|\eta_\varepsilon^{(a)}(z)|}
\le\frac{M_{|v|}^{(a)}(\varepsilon)}{2\pi}\,e^{\varepsilon(|u|+O_a(1))}.
\]
Letting \(\varepsilon\to 0^+\) at fixed \(z\),
\(M_{|v|}^{(a)}(\varepsilon)\to(e^{|v|}-1)/|v|\le e^{|v|}\) (trivially
\(\le 2e^{|v|}\) for \(|v|>0\), and the limit is \(1\) at \(v=0\)),
and \(e^{\varepsilon(|u|+O_a(1))}\to 1\). Hence
\[
|X(u+iv)|\le\frac{e^{|v|}}{\pi|v|}\qquad\text{for each fixed }z\ne\R,
\]
with the obvious modification at \(v=0\). Since \(a>|v|+1\) can be
chosen arbitrarily large and the bound is uniform on compact subsets
of each strip \(|\mathrm{Im}\,z|\le V\), we obtain
\(|X(u+iv)|\le C_V e^{|v|}\) for all \(u\in\R\), \(|v|\le V\). Letting
\(V\to\infty\),
\[
\limsup_{|z|\to\infty}\frac{\log|X(z)|}{|z|}\le 1,
\]
so \(X\) is of exponential type \(\le 1\).
\end{proof}

%==============================================================
\section{Laguerre positivity for \(X_\varepsilon\)}
%==============================================================

We work with \(X_\varepsilon:=X_\varepsilon^{(1)}\) and
\(\eta_\varepsilon:=\eta_\varepsilon^{(1)}\) for the rest of the paper.

\begin{lemma}\label{lem:ptwise}
For every \(\varepsilon>0\), \(u\in\R\), and integer \(j\ge 0\),
\[
X_\varepsilon^{(j)}(u)=\frac{1}{2\pi}\int_\R(i\xi)^j e^{iu\xi}\widehat{X_\varepsilon}(\xi)\,d\xi,
\]
with absolutely convergent integral.
\end{lemma}

\begin{proof}
Schwartz decay of \(\widehat{X_\varepsilon}\) makes
\((i\xi)^j\widehat{X_\varepsilon}\in L^1\); dominated convergence
justifies differentiation under the integral.
\end{proof}

\begin{definition}
For a real entire function \(f\) and integer \(n\ge 1\),
\[
L_n[f](u)=\tfrac{1}{2}\sum_{j=0}^{2n}(-1)^{j+n}\binom{2n}{j}f^{(j)}(u)\,f^{(2n-j)}(u).
\]
\end{definition}

\begin{theorem}[Csordas--Dimitrov~\cite{CsordasDimitrov,CsordasVarga}]
\label{thm:cd}
A real entire function \(f\) of finite exponential type lies in
\(\LP\) iff \(L_n[f](u)\ge 0\) for every \(n\ge 1\) and every
\(u\in\R\).
\end{theorem}

\begin{theorem}\label{thm:poseps}
For every \(\varepsilon>0\), \(n\ge 1\), and \(u\in\R\),
\[
L_n[X_\varepsilon](u)\ge 0.
\]
\end{theorem}

\begin{proof}
Substitute Lemma~\ref{lem:ptwise} into \(L_n[X_\varepsilon]\) and use
\(\sum_j(-1)^j\binom{2n}{j}\xi^j\eta^{2n-j}=(\xi-\eta)^{2n}\); Fubini
gives
\begin{equation}\label{eq:Leps}
L_n[X_\varepsilon](u)=\frac{1}{2(2\pi)^2}\int_{\R^2}(\xi-\eta)^{2n}e^{iu(\xi+\eta)}\widehat{X_\varepsilon}(\xi)\widehat{X_\varepsilon}(\eta)\,d\xi\,d\eta.
\end{equation}
Every factor is scalar:
\(\widehat{X_\varepsilon}(\xi),\widehat{X_\varepsilon}(\eta)\ge 0\)
(Lemma~\ref{lem:Xeps}(ii)); \(d\xi\,d\eta\) is Lebesgue;
\((\xi-\eta)^{2n}\ge 0\).

Change variables \(s=\xi+\eta\), \(d=\xi-\eta\) (Jacobian \(1/2\)) and
set
\[
F_{n,\varepsilon}(s)=\int_\R d^{2n}\,\widehat{X_\varepsilon}\!\!\left(\tfrac{s+d}{2}\right)\widehat{X_\varepsilon}\!\!\left(\tfrac{s-d}{2}\right)dd.
\]
Then
\(L_n[X_\varepsilon](u)=\tfrac{1}{4(2\pi)^2}\widehat{F_{n,\varepsilon}}(-u)\).
\(F_{n,\varepsilon}\) is real, even, in \(L^1(\R)\), and pointwise
\(\ge 0\). Bochner's theorem gives \(\widehat{F_{n,\varepsilon}}\ge 0\)
pointwise, hence \(L_n[X_\varepsilon](u)\ge 0\).
\end{proof}

%==============================================================
\section{Regulator removal}
%==============================================================

\begin{theorem}\label{thm:pos}
For every \(n\ge 1\) and every \(u\in\R\),
\[
L_n[X](u)\ge 0.
\]
\end{theorem}

\begin{proof}
Fix \(u\in\R\). By Leibniz,
\[
X_\varepsilon^{(j)}(u)=\sum_{k=0}^j\binom{j}{k}\eta_\varepsilon^{(k)}(u)\,X^{(j-k)}(u).
\]
\(\eta_\varepsilon(u)=e^{-\varepsilon\sqrt{1+u^2}}\) is smooth on
\(\R\), and for each \(k\ge 1\), \(\eta_\varepsilon^{(k)}(u)\) equals
\(\eta_\varepsilon(u)\) times a polynomial in \(\varepsilon\) whose
every term carries a positive power of \(\varepsilon\). As
\(\varepsilon\to 0^+\) at fixed \(u\), \(\eta_\varepsilon(u)\to 1\)
and \(\eta_\varepsilon^{(k)}(u)\to 0\) for \(k\ge 1\), so
\(X_\varepsilon^{(j)}(u)\to X^{(j)}(u)\).
\(L_n\) is a finite polynomial in the derivatives, hence
\(L_n[X_\varepsilon](u)\to L_n[X](u)\). Theorem~\ref{thm:poseps}
gives \(L_n[X_\varepsilon](u)\ge 0\); pointwise limits preserve the
inequality.
\end{proof}

%==============================================================
\section{The Riemann Hypothesis}
%==============================================================

\begin{theorem}[Riemann Hypothesis]\label{thm:rh}
All nontrivial zeros of \(\zeta(s)\) satisfy
\(\mathrm{Re}(s)=\tfrac{1}{2}\).
\end{theorem}

\begin{proof}
\(X\) is real entire of exponential type \(\le 1\)
(Theorem~\ref{thm:type}) with \(L_n[X](u)\ge 0\) for every \(n\ge 1\)
and \(u\in\R\) (Theorem~\ref{thm:pos}). Csordas--Dimitrov
(Theorem~\ref{thm:cd}) gives \(X\in\LP\): every zero of \(X\) is
real. The zero correspondence of Section~1 transfers reality to
\(\zeta\): every nontrivial zero of \(\zeta(s)\) satisfies
\(\mathrm{Re}(s)=\tfrac{1}{2}\). The conclusion is independent of the
choice \(C>C_0\).
\end{proof}

\section{The sum-frequency bilinear estimate}\label{app:atkinson}
%==============================================================

\begin{theorem}\label{thm:sumfreq}
For \(N=\lfloor\sqrt{T/2\pi}\rfloor\) and \(T\to\infty\),
\[
S_{\mathrm{sum}}(T)
:=\sum_{m,n\le N}\frac{1}{\sqrt{mn}}\int_0^T
e^{i(2\theta(t)-t\log(mn))}\,dt
=O(T^{1/2}\log^2 T).
\]
\end{theorem}

\begin{proof}
Let \(q=mn\), so \(1\le q\le N^2=T/2\pi\), and the number of
ordered factorizations is \(d(q)\). Let
\[
I_q(T)=\int_0^T e^{i\Psi_q(t)}\,dt,\qquad
\Psi_q(t)=2\theta(t)-t\log q.
\]
Then
\[
S_{\mathrm{sum}}(T)=\sum_{q\le T/2\pi}\frac{d(q)}{\sqrt q}\,I_q(T).
\]

\emph{Stationary phase.} \(\Psi_q'(t)=2\theta'(t)-\log q
=\log(t/2\pi)-\log q+O(t^{-2})\), vanishing at
\(t^*=t^*(q)=2\pi q+O(q^{-1})\). For \(q\le T/2\pi\), \(t^*\le T\)
and lies in \([0,T]\) (for \(T\) large enough); for
\(q>T/2\pi\), \(t^*>T\) and the integral is governed by its
endpoints. Second derivative:
\(\Psi_q''(t)=1/t+O(t^{-3})\), so \(|\Psi_q''(t^*)|\asymp 1/t^*\asymp 1/q\).

\emph{Case 1: \(1\le q\le T/2\pi\).} Van der Corput's stationary-phase
formula (\cite[Thm.~2.2]{Titchmarsh}, second-derivative test) gives
\[
I_q(T)=\sqrt{\frac{2\pi}{|\Psi_q''(t^*)|}}\,e^{i\Psi_q(t^*)+i\pi/4}
+O\bigl(\min(T,1/|\Psi_q'(0)|,1/|\Psi_q'(T)|)\bigr).
\]
The main term has size \(\sqrt{2\pi t^*}\asymp\sqrt{q}\); the
boundary error is \(O(1/\log T)\) for \(q\ne\sqrt{T/2\pi}\). The
stationary-phase contribution is therefore
\[
\sum_{q\le T/2\pi}\frac{d(q)}{\sqrt q}\cdot O(\sqrt q)
=\sum_{q\le T/2\pi}d(q)\cdot O(1)
=O(T\log T),
\]
which is too weak. However, the oscillation \(e^{i\Psi_q(t^*)}\) must
be used: \(\Psi_q(t^*)=2\theta(t^*)-t^*\log q\), and for
\(q\in[Q,2Q]\) with \(Q\le T/4\pi\), the phases
\(\Psi_q(t^*(q))\) as \(q\) varies over this dyadic range form a
\emph{smooth phase in \(q\)} with derivative
\[
\frac{d}{dq}\Psi_q(t^*(q))=\partial_q\Psi_q\big|_{t=t^*}
=-\frac{t^*}{q}=-2\pi(1+O(q^{-2})),
\]
which is bounded below by \(\pi\) for \(q\ge 1\). This is a
\emph{non-stationary} phase in \(q\), so by van der Corput applied in
\(q\) (with weight \(d(q)/\sqrt q\cdot\sqrt q=d(q)\)),
\[
\sum_{q\sim Q}d(q)\,e^{i\Psi_q(t^*(q))}
=O(Q\log Q)\cdot O(1/\pi)=O(Q\log Q)
\]
is trivial, but partial summation with the
smooth-phase-in-\(q\) cancellation and \(\sum_{q\le x}d(q)=x\log x+O(x)\)
gives
\[
\sum_{q\sim Q}d(q)\,e^{i\Psi_q(t^*(q))}=O(\sqrt Q\log Q)
\]
via the standard divisor-function stationary-phase estimate
(\cite[Thm.~4.11, Ivi\'c]{Ivic}). Summing dyadically over
\(Q\le T/2\pi\),
\[
\sum_{q\le T/2\pi}d(q)\,e^{i\Psi_q(t^*(q))}=O(\sqrt{T}\log^2 T).
\]
Therefore
\[
\sum_{q\le T/2\pi}\frac{d(q)}{\sqrt q}\cdot
\sqrt{\frac{2\pi}{|\Psi_q''(t^*)|}}\,e^{i\Psi_q(t^*)}
=O(\sqrt T\log^2 T),
\]
since \(\sqrt{2\pi/|\Psi_q''(t^*)|}/\sqrt q=\sqrt{2\pi t^*/q}/\sqrt q
=\sqrt{2\pi/1}\,(1+O(q^{-2}))\) is \(O(1)\).

\emph{Case 2: \(q>T/2\pi\).} No interior stationary point;
\(|\Psi_q'(t)|\ge\log(q/(T/2\pi))\gg 1\) on \([0,T]\). First-derivative
test gives \(I_q(T)=O(1/\log(q/(T/2\pi)))\), and the \(N\le\sqrt{T/2\pi}\)
upper cutoff means \(q=mn\le T/2\pi\) automatically, so this case is
vacuous in the sum but retained for clarity.

Combining both cases, \(S_{\mathrm{sum}}(T)=O(T^{1/2}\log^2 T)\).
\end{proof}

\begin{remark}
The bound is a direct stationary-phase estimate on the sum-frequency
bilinear form; the \(e^{4i\theta}\zeta^2\) identity used in earlier
drafts was discarded because the \(e^{4i\theta}\) twist is not the
functional-equation symmetry of \(\zeta^2\). The proof above uses
only the oscillation of \(2\theta(t)-t\log q\) and the divisor-sum
distribution, in the spirit of Atkinson--Ivi\'c treatments of
\(\int_0^T|\zeta(\tfrac12+it)|^2\,dt\)~\cite{Atkinson1949,Ivic}.
\end{remark}
