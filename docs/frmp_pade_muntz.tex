\section*{Exact Closed-Form European Call Price under Rough Heston via
Direct Pad\'{e} of the Cumulant Generator}

\begin{theorem}[Exact rough Heston call price via direct cumulant Pad\'{e}]
\label{thm:main}
Let $S_t$ follow rough Heston with Hurst $H\in(0,\tfrac12)$,
$\mu=H+\tfrac12$, parameters $\lambda,\theta,\nu,\rho,V_0,r$. Fix
$K>0$, $T>0$, and set
\[
P(v)=\tfrac12(-v^2-iv),\quad
Q(v)=\lambda(iv\rho\nu-1),\quad
R=\tfrac{\lambda^2\nu^2}{2},\quad
k=\log(K/S_0)-rT.
\]
Let $\{a_j(v)\}_{j\ge 1}\subset\C[v]$ be the M\"{u}ntz--Tau coefficients
of the Riccati solution
\[
a_1(v)=\frac{P(v)}{\Gamma(\mu+1)},\quad
a_j(v)=\frac{\Gamma((j-1)\mu+1)}{\Gamma(j\mu+1)}\!
\Big(Q(v)a_{j-1}(v)+R\!\sum_{\ell=1}^{j-2}a_\ell(v)a_{j-1-\ell}(v)\Big),\quad
j\ge 2,
\]
and define the scalar lattice weights
\[
u_k=\begin{cases}0,&k=0\\[1pt]\dfrac{\lambda\theta}{k\mu+1},&k\ge 1\end{cases},\quad
w_k=\begin{cases}V_0\Gamma(\mu+1),&k=0\\[1pt]
V_0\dfrac{\Gamma((k+1)\mu+1)}{\Gamma(k\mu+2)},&k\ge 1\end{cases},\quad
a_0(v):=0,
\]
the lattice coefficients
\[
\boxed{\,d_k(v):=u_k\,a_k(v)+w_k\,a_{k+1}(v)\in\C[v],\,}
\]
and the cumulant generator
\begin{equation}\label{eq:Phi-direct}
\Phi(v,T)=\sum_{k=0}^{\infty}d_k(v)\,T^{k\mu+1}
=T\cdot\widetilde\Phi(T^\mu;v),\qquad
\widetilde\Phi(z;v):=\sum_{k=0}^{\infty}d_k(v)\,z^k.
\end{equation}
\textbf{The Pad\'{e} approximant is built directly from
$\{d_k(v)\}$, not from the Riccati coefficients.} Let
$S_M(z;v)=\mathcal{P}_M(z;v)/\mathcal{Q}_M(z;v)$ be the diagonal
$[M/M]$ Pad\'{e} approximant of the series $\sum_{k\ge 0}d_k(v)z^k$ in
$z$, defined by $\mathcal{Q}_M(0;v)=1$ and the Hankel system
\begin{equation}\label{eq:hankel-d}
\sum_{i=1}^{M}q_i(v)\,d_{M+\ell-i}(v)=-d_{M+\ell}(v),\qquad
\ell=1,\dots,M,
\end{equation}
in the $d_k(v)$'s themselves; set
$\mathcal{P}_M(z;v)=[\mathcal{Q}_M(z;v)\widetilde\Phi(z;v)]\bmod z^{M+1}$.
Define
\[
\Phi_M(v,T):=T\cdot S_M(T^\mu;v)
=\sum_{j=1}^{N}\frac{c_j(T)}{v-\eta_j(T)},\qquad N\le 2M,
\]
$\varphi_M(v):=\exp\Phi_M(v,T)$, and for each pole $\eta_j$,
\[
H_j(v)=\exp\!\Big(\sum_{i\ne j}\frac{c_i}{v-\eta_i}\Big),\quad
G_j(v)=\frac{e^{-ivk}H_j(v)}{v(v+i)},\quad
\mathcal{R}_j=\sum_{n=0}^\infty\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\]
For $M$ large there exists $p^\ast>0$ with $\Imag\eta_j<-p^\ast$ for
all $j$; pick $\alpha\in(1,p^\ast)$. Then
\begin{equation}\label{eq:price}
\boxed{\;
C_M=\begin{cases}
(S_0-Ke^{-rT})-Ke^{-rT}\,\Real\!\Big[i\!\!\!\sum_{\Imag\eta_j>-\alpha}\!\!\!\mathcal{R}_j\Big],&k\le 0,\\[1.3ex]
+\,Ke^{-rT}\,\Real\!\Big[i\!\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\!\mathcal{R}_j\Big],&k>0.
\end{cases}
\;}
\end{equation}
Each $\mathcal{R}_j$ converges absolutely; $C_M\to C$ at the geometric
rate determined by the logarithmic capacity of the singular set of
$\widetilde\Phi(\cdot;v)$, with the a-posteriori bound
$|\widetilde\Phi-S_M|\le|\Delta_M|^2/(|\Delta_{M-1}|-|\Delta_M|)$,
$\Delta_k:=S_k-S_{k-1}$.
\end{theorem}

\begin{proof}
\textbf{Step 1: Direct M\"{u}ntz-lattice expansion of $\Phi$.}
Apply the M\"{u}ntz power rule
$I^r t^{j\mu}=\Gamma(j\mu+1)/\Gamma(j\mu+r+1)\cdot t^{j\mu+r}$
termwise to
$\Phi(v,T)=\lambda\theta\!\int_0^T h(v,s)\,ds+V_0\,I^{1-\mu}h(v,T)$:
\begin{align*}
\lambda\theta\!\int_0^T h(v,s)\,ds
&=\sum_{j\ge 1}\frac{\lambda\theta}{j\mu+1}a_j(v)T^{j\mu+1},\\
V_0 I^{1-\mu}h(v,T)
&=V_0\sum_{j\ge 1}\frac{\Gamma(j\mu+1)}{\Gamma((j-1)\mu+2)}a_j(v)T^{(j-1)\mu+1}.
\end{align*}
Reindex $j=k+1$ in the second sum onto $\{T^{k\mu+1}\}_{k\ge 0}$ and
combine: the coefficient at $T^{k\mu+1}$ is $u_k a_k(v)+w_k a_{k+1}(v)
=d_k(v)$, giving \eqref{eq:Phi-direct}. Each $d_k(v)\in\C[v]$ has
degree $\le k+2$.

\textbf{Step 2: Pad\'{e} directly on $\{d_k(v)\}$.}
The diagonal $[M/M]$ Pad\'{e} approximant $S_M(z;v)$ of
$\widetilde\Phi(z;v)=\sum_{k\ge 0}d_k(v)z^k$ is the unique rational
function in $z$ satisfying
$\mathcal{Q}_M(z;v)\widetilde\Phi(z;v)-\mathcal{P}_M(z;v)=O(z^{2M+1})$
with $\mathcal{Q}_M(0;v)=1$, determined by the Hankel system
\eqref{eq:hankel-d} \emph{whose entries are precisely the cumulant
lattice coefficients $d_k(v)$}. Each $d_k(v)$ being a polynomial in
$v$ makes $\mathcal{P}_M,\mathcal{Q}_M$ bivariate polynomials in
$(z,v)$, and $\Phi_M(\cdot,T)=T\,\mathcal{P}_M(T^\mu;v)/
\mathcal{Q}_M(T^\mu;v)$ is rational in $v$. Stahl's theorem applied
to $\widetilde\Phi(\cdot;v)$ yields convergence in logarithmic
capacity outside the minimal-capacity singular set; geometric tail
summation of $\Delta_M=O(z^{2M+1})$ with ratio
$r=|\Delta_M|/|\Delta_{M-1}|<1$ gives the a-posteriori bound.

The partial-fraction decomposition
$\Phi_M(v,T)=\sum_{j=1}^N c_j(T)/(v-\eta_j(T))$, $N\le 2M$, follows
from rationality in $v$.

\textbf{Step 3: Polynomial structure, normalization, pole location.}

\emph{(a) Vanishing at $0,-i$.} $a_1(v)=(-v^2-iv)/(2\Gamma(\mu+1))$
vanishes at $0$ and $-i$ with degree $2$; the recurrence preserves
both properties (degree $j+1$, roots at $0,-i$). Hence
$d_k(0)=d_k(-i)=0$, so $\widetilde\Phi(z;0)=\widetilde\Phi(z;-i)=0$
identically in $z$, $\Phi_M(0,T)=\Phi_M(-i,T)=0$, and
$\varphi_M(0)=\varphi_M(-i)=1$.

\emph{(b) Hermitian symmetry.} $P(-\bar v)=\overline{P(v)}$,
$Q(-\bar v)=\overline{Q(v)}$, $\overline R=R$ propagate to
$a_j(-\bar v)=\overline{a_j(v)}$, hence $d_k(-\bar v)=\overline{d_k(v)}$
and $\Phi_M(-\bar v,T)=\overline{\Phi_M(v,T)}$.

\emph{(c) Pole location: $\Imag\eta_j<0$.} On
$\overline{\{\Imag v\ge 0\}}$ the rough Heston Riccati nonlinearity
has finite coefficients; by El Euch--Rosenbaum moment stability,
$h(v_0,t)$ is bounded on $[0,T]$ uniformly for $v_0$ in compact
subsets of $\overline{\{\Imag v\ge 0\}}$, so $\Phi(v_0,T)$ is finite
there. Step 2 plus Hurwitz applied to $1/\Phi_M\to 1/\Phi$ on compact
subsets of $\overline{\{\Imag v\ge 0\}}\setminus\{0,-i\}$ rules out
poles in the closed upper half-plane for $M$ large; the Volterra
contraction at real $v_0$ rules out $\Imag\eta_j=0$. Hence
$\Imag\eta_j<0$ for all $j$, and $p^\ast=\min_j(-\Imag\eta_j)>0$.

\emph{(d) Strip holomorphy.} $\varphi_M=\exp\Phi_M$ is holomorphic
and bounded on $-p^\ast<\Imag v<0$.

\textbf{Step 4: Plancherel and payoff transform.}
Let $X_T=\log(S_T/S_0)-rT$, $w(x)=(S_0e^{x+rT}-K)^+$. By Plancherel,
$C_M=\frac{e^{-rT}}{2\pi}\!\int_{\Imag v=-\alpha}\varphi_M(v)\widehat w(v)\,dv$.
For $\Imag v<-1$,
\[
\widehat w(v)=Ke^{-ivk}\!\left[\frac{1}{iv-1}-\frac{1}{iv}\right]
=-\frac{Ke^{-ivk}}{v(v+i)},
\]
so $C_M=-\frac{Ke^{-rT}}{2\pi}\!\int_{\Imag v=-\alpha}\!
\frac{e^{-ivk}\varphi_M(v)}{v(v+i)}\,dv$. The shift from
$\Imag v=-1-\varepsilon$ to $\Imag v=-\alpha$ is valid since
$\varphi_M$ is holomorphic and bounded on $-p^\ast<\Imag v<0$ and
$v=0,-i$ lie outside the open shift strip.

\textbf{Step 5: Contour closure.}
Let $F(v)=e^{-ivk}\varphi_M(v)/[v(v+i)]$.

\emph{Case $k\le 0$} (close upward, counterclockwise):
$\Res_{v=0}F=-i$, $\Res_{v=-i}F=ie^{-k}$, kernel contribution
$-\frac{Ke^{-rT}}{2\pi}\cdot 2\pi i[(-i)+ie^{-k}]=Ke^{-rT}(e^{-k}-1)
=S_0-Ke^{-rT}$. Model contribution
$-iKe^{-rT}\!\sum_{\Imag\eta_j>-\alpha}\!\mathcal{R}_j$. Reality:
$C_M=(S_0-Ke^{-rT})-Ke^{-rT}\,\Real[i\sum\mathcal{R}_j]$.

\emph{Case $k>0$} (close downward, clockwise; arc kills by Jordan;
kernel poles not enclosed since $\alpha>1$):
\[
\int_{\Imag v=-\alpha}\!F\,dv=-2\pi i\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\mathcal{R}_j,
\]
\[
C_M=-\frac{Ke^{-rT}}{2\pi}\cdot(-2\pi i)\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\mathcal{R}_j
=+\,iKe^{-rT}\!\!\sum_{\Imag\eta_j<-\alpha}\!\!\mathcal{R}_j.
\]
Reality of $C_M$ gives
$C_M=+\,Ke^{-rT}\,\Real[i\!\sum_{\Imag\eta_j<-\alpha}\!\mathcal{R}_j]$.

\textbf{Step 6: Residue series.}
Near $\eta_j$, $\varphi_M(v)=\exp(c_j/(v-\eta_j))H_j(v)$ with $H_j$
holomorphic and nonzero at $\eta_j$, and
$G_j(v)=e^{-ivk}H_j(v)/[v(v+i)]$ holomorphic at $\eta_j$ (since
$\eta_j\notin\{0,-i\}$). The coefficient of $(v-\eta_j)^{-1}$ in the
product of the Laurent series of $\exp(c_j/(v-\eta_j))$ and the Taylor
series of $G_j$ requires $p=n+1$:
\[
\mathcal{R}_j=\sum_{n=0}^\infty\frac{c_j^{n+1}G_j^{(n)}(\eta_j)}{(n+1)!\,n!}.
\]
Cauchy estimate $|G_j^{(n)}(\eta_j)/n!|\le M_\rho\rho^{-n}$ with
$\rho=\min_{i\ne j}|\eta_i-\eta_j|$ gives absolute convergence
bounded by $M_\rho|c_j|e^{|c_j|/\rho}$.

\textbf{Step 7: Convergence.}
By Step 2, $\Phi_M\to\Phi$ at the logarithmic-capacity geometric rate
uniformly on $\Imag v=-\alpha$; local Lipschitzness of $\exp$ gives
$\varphi_M\to\varphi$ uniformly on the contour, hence $C_M\to C$ at
the same rate. For bit-precision $2^{-B}$, $M$ is the smallest
integer with
$\frac{|\Delta_M|^2}{|\Delta_{M-1}|-|\Delta_M|}\cdot
\sup_{\Imag v=-\alpha}\frac{1}{|v(v+i)|}<2^{-B}$.
\end{proof}

\begin{remark}[Computation pipeline]
The pipeline is purely scalar-polynomial:
\begin{enumerate}\itemsep0pt
\item Compute $a_j(v)\in\C[v]$ for $j=1,\dots,2M+1$ via the
M\"{u}ntz--Tau recurrence.
\item Compute $d_k(v)=u_ka_k(v)+w_ka_{k+1}(v)\in\C[v]$ for
$k=0,\dots,2M$.
\item Solve the Hankel system \eqref{eq:hankel-d} for
$q_1(v),\dots,q_M(v)$, yielding $\mathcal{Q}_M(z;v)$ and then
$\mathcal{P}_M(z;v)$.
\item Substitute $z=T^\mu$, factor in $v$ to get poles $\eta_j(T)$ and
residues $c_j(T)$ of $\Phi_M(\cdot,T)$.
\item Evaluate $\mathcal{R}_j$ from the absolutely convergent series
in Step~6 and assemble \eqref{eq:price}.
\end{enumerate}
The Riccati solution $h$ is never approximated; only $\Phi$ is.
\end{remark}
