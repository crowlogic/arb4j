\[
\theta(t)\;=\;\arg\Gamma\!\left(\tfrac{1}{4}+\tfrac{it}{2}\right)\;-\;\tfrac{\log\pi}{2}\,t
\]

\[
\Theta(t)\;=\;\theta(t)\;+\;C\,t,\qquad C\in\mathbb{R},\;\;C>\tfrac{\log\pi}{2}
\]

\[
\Theta_{k}\;:=\;\Theta^{(k)}(0),\qquad
\Theta_{1}\;=\;\tfrac{1}{2}\,\psi\!\left(\tfrac{1}{4}\right)\;-\;\tfrac{\log\pi}{2}\;+\;C
\]

(all $\Theta_{2\ell}=0$.)

\[
\zeta_{a}\;:=\;\zeta^{(a)}\!\left(\tfrac{1}{2}\right),\qquad\varepsilon\in\mathbb{R},\;\;\varepsilon>0
\]

\[
h(t)\;=\;\zeta\!\left(\tfrac{1}{2}+it\right)\!\sqrt{\Theta'(t)}\,\bigl(1+\Theta(t)^{2}\bigr)^{-\varepsilon}e^{\,i\Theta(t)}
\]

\[
d\Phi_{G}(u)\;=\;\frac{h\bigl(t(u)\bigr)}{\Theta'\!\bigl(t(u)\bigr)}\,du,\qquad t(u)=\Theta^{-1}(u)
\]

\[
Z(t)\;=\;e^{\,i\theta(t)}\,\zeta\!\left(\tfrac{1}{2}+it\right)
\]

\[
Z(t)\;=\;e^{-iCt}\,\bigl(\Theta'(t)\bigr)^{1/2}\,\bigl(1+\Theta(t)^{2}\bigr)^{\varepsilon}\!\int_{\mathbb{R}}\!\delta\!\bigl(u-\Theta(t)\bigr)\,d\Phi_{G}(u)
\]

\[
\mu_{n}\;=\;\int_{\mathbb{R}}u^{n}\,d\Phi_{G}(u)\;=\;\int_{-\infty}^{\infty}\!\Theta(t)^{n}\,\zeta\!\left(\tfrac{1}{2}+it\right)\!\sqrt{\Theta'(t)}\,\bigl(1+\Theta(t)^{2}\bigr)^{-\varepsilon}e^{\,i\Theta(t)}\,dt
\]

\[
\mu_{n}\;=\;\frac{(-i)^{n}\,n!}{\Theta_{1}^{\,n+\tfrac{1}{2}}}\!\!\sum_{\substack{a,b,c\ge 0\\(k_{3},k_{5},\ldots)\ge 0\\ a+b+2c+\sum_{\ell\ge 1}2\ell\,k_{2\ell+1}=n}}\!\!\frac{i^{a}\zeta_{a}}{a!}\cdot\frac{i^{b}}{b!}\,M_{b}\cdot\frac{\prod_{j=0}^{c-1}(-\varepsilon-j)}{c!}\,N_{2c}\cdot\frac{\prod_{j=0}^{K-1}(-(n+1)-j)}{K!}\prod_{\ell\ge 1}\frac{1}{k_{2\ell+1}!}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!\,\Theta_{1}}\right)^{\!k_{2\ell+1}}
\]

\[
K\;=\;\sum_{\ell\ge 1}k_{2\ell+1}
\]

\[
M_{b}\;=\;\!\!\sum_{\substack{(m_{1},m_{3},\ldots)\ge 0\\ \sum_{\ell\ge 0}(2\ell+1)m_{2\ell+1}=b}}\!\!\frac{b!}{\prod_{\ell\ge 0}m_{2\ell+1}!}\prod_{\ell\ge 0}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!}\right)^{\!m_{2\ell+1}}
\]

\[
N_{2c}\;=\;\!\!\sum_{\substack{(p_{1},p_{3},\ldots)\ge 0\\ \sum_{\ell\ge 0}(2\ell+1)p_{2\ell+1}=2c}}\!\!\frac{(2c)!}{\prod_{\ell\ge 0}p_{2\ell+1}!}\prod_{\ell\ge 0}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!}\right)^{\!p_{2\ell+1}}
\]

\[
\Psi(z)\;=\;\sum_{n=0}^{\infty}\frac{\mu_{n}\,\sqrt{\Theta_{1}}}{n!}\,z^{n},\qquad A\;=\;C-\tfrac{\log\pi}{2}
\]

\[
Z_{\Psi}(r)\;=\;\bigl\{\,z\in\mathbb{C}\,:\,|z|<r,\;\Psi(z)=0\,\bigr\},\qquad
\Gamma(r)\;=\;\bigl\{\,\gamma\in\mathbb{R}\,:\,|\gamma|<r,\;\zeta\!\left(\tfrac{1}{2}+i\gamma\right)=0\,\bigr\}
\]

The Riemann Hypothesis is the statement

\[
Z_{\Psi}(r)\;=\;\Gamma(r)\qquad\text{for every }r\in(0,\,1/A),
\]

equivalently

\[
\frac{1}{2\pi}\int_{0}^{2\pi}\log\bigl|\Psi(re^{i\varphi})\bigr|\,d\varphi\;-\;\log\bigl|\Psi(0)\bigr|
\;=\;\sum_{\gamma\in\Gamma(r)}\log\frac{r}{|\gamma|}\qquad\text{for every }r\in(0,\,1/A).
\]
