\title{A Hamburger Moment Criterion for the Riemann Hypothesis via a
       $\Theta$-Warped Stieltjes Measure}
\author{Stephen Crowley}
\date{April 23, 2026}

\begin{document}
\maketitle

\begin{abstract}
Let $\theta$ denote the Riemann--Siegel theta function and set
$\Theta(t)=\theta(t)+Ct$ with $C$ chosen so that $\Theta'(t)>0$ for all
$t\ge 0$. For a regularization parameter $\varepsilon>0$ define the density
$h(t)=\zeta(\tfrac12+it)\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}e^{i\Theta(t)}$
and let $\Phi_{G}$ be its pushforward complex Borel measure under
$u=\Theta(t)$. The moment sequence
$\mu_{n}=\int_{\mathbb{R}}u^{n}\,d\Phi_{G}(u)$ admits an explicit finite
closed form in the Taylor data $\zeta^{(a)}(\tfrac12)$ and
$\Theta^{(2\ell+1)}(0)$. Let $\mu_{n}^{R}:=\mathrm{Re}\,\mu_{n}$ and form
the Hankel matrices $H_{N}^{R}=(\mu_{i+j}^{R})_{i,j=0}^{N}$. We prove that
the Riemann Hypothesis is equivalent to
$\det H_{N}^{R}>0$ for every $N\ge 0$, i.e. to the statement that the
explicit real moment sequence $(\mu_{n}^{R})$ is a positive-definite
Hamburger moment sequence. The associated orthogonal polynomial system
$(p_{n})$ with three-term recurrence coefficients $(a_{n},b_{n})$,
themselves explicit rational functions of the closed-form moments, is the
algebraic apparatus that detects this positivity.
\end{abstract}

\section{Notation}

Throughout, $\theta(t)=\arg\Gamma(\tfrac14+\tfrac{it}{2})-\tfrac{\log\pi}{2}t$
is the Riemann--Siegel theta function and $Z(t)=e^{i\theta(t)}\zeta(\tfrac12+it)$
is the Hardy $Z$-function. Fix a real constant $C$ such that
$\Theta(t):=\theta(t)+Ct$ satisfies $\Theta'(t)>0$ for every $t\ge 0$; such
$C$ exists (indeed $C>-\theta'(0)=\tfrac{\log\pi}{2}-\tfrac12\psi(\tfrac14)$
suffices). Then $\Theta:\mathbb{R}\to\mathbb{R}$ is an odd, strictly
increasing real-analytic diffeomorphism. Set $\Theta_{k}:=\Theta^{(k)}(0)$;
oddness forces $\Theta_{2\ell}=0$, while
$\Theta_{1}=\tfrac12\psi(\tfrac14)-\tfrac{\log\pi}{2}+C>0$.
Write $\zeta_{a}:=\zeta^{(a)}(\tfrac12)$ and fix $\varepsilon>0$.

\section{The $\Theta$-warped density and its pushforward}

Define
\[
h(t)\;=\;\zeta\!\bigl(\tfrac12+it\bigr)\sqrt{\Theta'(t)}\,\bigl(1+\Theta(t)^{2}\bigr)^{-\varepsilon}e^{\,i\Theta(t)}.
\]
Writing $h(t)=e^{iCt}\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}Z(t)$
and using $|Z(t)|\ll_{\delta}(1+|t|)^{1/4+\delta}$ with
$\Theta'(t)\sim\tfrac12\log t$, one has
$h\in L^{1}(\mathbb{R})\cap L^{2}(\mathbb{R})$ for every $\varepsilon>0$.

\begin{definition}
The $\Theta$-warped Stieltjes measure associated to $\zeta$ with parameters
$(C,\varepsilon)$ is the pushforward of $h(t)\,dt$ under $u=\Theta(t)$:
\[
d\Phi_{G}(u)\;=\;\frac{h\bigl(t(u)\bigr)}{\Theta'\!\bigl(t(u)\bigr)}\,du,\qquad t(u)=\Theta^{-1}(u).
\]
\end{definition}

$\Phi_{G}$ is a complex Borel measure of bounded variation on $\mathbb{R}$.
Its real and imaginary parts
$d\Phi_{G}^{R}=\mathrm{Re}(d\Phi_{G})$ and
$d\Phi_{G}^{I}=\mathrm{Im}(d\Phi_{G})$ are signed real Borel measures.

\section{Closed form for the moments}

The moments
\[
\mu_{n}\;=\;\int_{\mathbb{R}}u^{n}\,d\Phi_{G}(u)\;=\;\int_{-\infty}^{\infty}\!\Theta(t)^{n}\,\zeta\!\bigl(\tfrac12+it\bigr)\sqrt{\Theta'(t)}\bigl(1+\Theta(t)^{2}\bigr)^{-\varepsilon}e^{\,i\Theta(t)}\,dt
\]
are finite for every $n\ge 0$.

\begin{theorem}[Closed form]\label{thm:closed}
For every integer $n\ge 0$,
\[
\mu_{n}\;=\;\frac{(-i)^{n}\,n!}{\Theta_{1}^{\,n+\tfrac{1}{2}}}\!\!\sum_{\substack{a,b,c\ge 0\\(k_{3},k_{5},\ldots)\ge 0\\ a+b+2c+\sum_{\ell\ge 1}2\ell\,k_{2\ell+1}=n}}\!\!\frac{i^{a}\zeta_{a}}{a!}\cdot\frac{i^{b}}{b!}\,M_{b}\cdot\frac{\prod_{j=0}^{c-1}(-\varepsilon-j)}{c!}\,N_{2c}\cdot\frac{\prod_{j=0}^{K-1}(-(n+1)-j)}{K!}\prod_{\ell\ge 1}\frac{1}{k_{2\ell+1}!}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!\,\Theta_{1}}\right)^{\!k_{2\ell+1}},
\]
with
\[
K=\sum_{\ell\ge 1}k_{2\ell+1},\qquad
M_{b}=\!\!\sum_{\substack{(m_{1},m_{3},\ldots)\ge 0\\ \sum_{\ell\ge 0}(2\ell+1)m_{2\ell+1}=b}}\!\!\frac{b!}{\prod_{\ell\ge 0}m_{2\ell+1}!}\prod_{\ell\ge 0}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!}\right)^{\!m_{2\ell+1}},
\]
\[
N_{2c}=\!\!\sum_{\substack{(p_{1},p_{3},\ldots)\ge 0\\ \sum_{\ell\ge 0}(2\ell+1)p_{2\ell+1}=2c}}\!\!\frac{(2c)!}{\prod_{\ell\ge 0}p_{2\ell+1}!}\prod_{\ell\ge 0}\!\left(\frac{\Theta_{2\ell+1}}{(2\ell+1)!}\right)^{\!p_{2\ell+1}}.
\]
\end{theorem}

\begin{proof}
From $\mu_{n}=(-i)^{n}n!\,[u^{n}]\!\int e^{izu}F(u)\,du$ with $F(u)=h(t(u))/\Theta'(t(u))$,
apply the Lagrange inversion identity
\[
[u^{n}]G(u)=[t^{n}]G(\Theta(t))\Theta'(t)\bigl(t/\Theta(t)\bigr)^{n+1}
\]
to $G(u)=e^{iu}(1+u^{2})^{-\varepsilon}\zeta(\tfrac12+it(u))/\sqrt{\Theta'(t(u))}$.
The multiplicative factor $\Theta'(t)$ converts $1/\sqrt{\Theta'}$ to
$\sqrt{\Theta'}$, and Taylor-expanding each factor yields the convolution
sums $M_{b},N_{2c}$ and the $K$-multiindex binomial from
$(t/\Theta(t))^{n+1}$. Multiplying by $\Theta_{1}^{n+1/2}$ clears the leading
normalization.
\end{proof}

\begin{corollary}\label{cor:first4}
The first four moments, with leading denominators cleared, are
\[
\mu_{0}\Theta_{1}^{1/2}\;=\;\zeta_{0},
\]
\[
\mu_{1}\Theta_{1}^{3/2}\;=\;\zeta_{1}\;+\;\Theta_{1}\zeta_{0},
\]
\[
\mu_{2}\Theta_{1}^{5/2}\;=\;\zeta_{2}\;+\;2\Theta_{1}\zeta_{1}\;+\;\Bigl[(1+2\varepsilon)\Theta_{1}^{2}+\tfrac{\Theta_{3}}{2\Theta_{1}}\Bigr]\zeta_{0},
\]
\[
\mu_{3}\Theta_{1}^{7/2}\;=\;\zeta_{3}\;+\;3\Theta_{1}\zeta_{2}\;+\;\Bigl[3(1+2\varepsilon)\Theta_{1}^{2}+\tfrac{5\Theta_{3}}{2\Theta_{1}}\Bigr]\zeta_{1}\;+\;\Bigl[(1+6\varepsilon)\Theta_{1}^{3}+\tfrac{3\Theta_{3}}{2}\Bigr]\zeta_{0}.
\]
\end{corollary}

\section{The Hamburger moment problem for the real-part sequence}

Let $\mu_{n}^{R}:=\mathrm{Re}\,\mu_{n}$. By Theorem~\ref{thm:closed} each
$\mu_{n}^{R}$ is an explicit finite real linear combination of
$\mathrm{Re}(i^{a}\zeta_{a})$ and $\mathrm{Im}(i^{a}\zeta_{a})$ with
rational coefficients in $\varepsilon$ and $\Theta_{1},\Theta_{3},\ldots$.

\begin{definition}\label{def:hankel}
The Hankel matrices and shifted Hankel matrices of $(\mu_{n}^{R})$ are
\[
H_{N}^{R}\;=\;\bigl(\mu_{i+j}^{R}\bigr)_{i,j=0}^{N},\qquad
H_{N}^{R,\prime}\;=\;\bigl(\mu_{i+j+1}^{R}\bigr)_{i,j=0}^{N}.
\]
\end{definition}

\begin{lemma}[Hamburger]\label{lem:hamburger}
A real sequence $(s_{n})_{n\ge 0}$ is the moment sequence of a positive
Borel measure on $\mathbb{R}$ iff $\det(s_{i+j})_{i,j=0}^{N}>0$ for every
$N\ge 0$.
\end{lemma}

\section{Orthogonal polynomials and the Jacobi recurrence}

Whenever $\det H_{N}^{R}\ne 0$ for all $N$, the Gram--Schmidt orthogonalization
of $\{1,u,u^{2},\ldots\}$ in the bilinear form
\[
\langle f,g\rangle_{R}\;=\;\int_{\mathbb{R}}f(u)g(u)\,d\Phi_{G}^{R}(u)
\]
produces monic polynomials $p_{n}(u)\in\mathbb{R}[u]$, $\deg p_{n}=n$,
satisfying the three-term recurrence
\begin{equation}\label{eq:ttrr}
u\,p_{n}(u)\;=\;p_{n+1}(u)\;+\;a_{n}\,p_{n}(u)\;+\;b_{n}\,p_{n-1}(u),\qquad p_{-1}=0,\;p_{0}=1,
\end{equation}
with coefficients
\begin{equation}\label{eq:ab}
a_{n}\;=\;\frac{\langle u\,p_{n},p_{n}\rangle_{R}}{\langle p_{n},p_{n}\rangle_{R}},\qquad
b_{n}\;=\;\frac{\langle p_{n},p_{n}\rangle_{R}}{\langle p_{n-1},p_{n-1}\rangle_{R}}.
\end{equation}
In terms of Hankel determinants,
\begin{equation}\label{eq:hankel-ab}
a_{n}\;=\;\frac{\det H_{n-1}^{R}\cdot\det H_{n}^{R,\prime\prime}-\det H_{n}^{R,\prime}\cdot\det H_{n-1}^{R,\prime}}{\det H_{n-1}^{R}\cdot\det H_{n}^{R}}\quad\text{(symbolic)},\qquad
b_{n}\;=\;\frac{\det H_{n}^{R}\cdot\det H_{n-2}^{R}}{\bigl(\det H_{n-1}^{R}\bigr)^{2}}.
\end{equation}
Each $a_{n},b_{n}$ is thus a rational function of $\mu_{0}^{R},\ldots,\mu_{2n+1}^{R}$, hence a rational function of $\zeta^{(a)}(\tfrac12)$, $\Theta_{2\ell+1}$, and $\varepsilon$.

The associated Jacobi matrix
\[
J\;=\;\begin{pmatrix}a_{0}&\sqrt{b_{1}}&&&\\\sqrt{b_{1}}&a_{1}&\sqrt{b_{2}}&&\\&\sqrt{b_{2}}&a_{2}&\sqrt{b_{3}}&\\&&\ddots&\ddots&\ddots\end{pmatrix}
\]
is well-defined (and self-adjoint) precisely when $b_{n}>0$ for all $n\ge 1$,
which is equivalent to $\det H_{N}^{R}>0$ for all $N$.

\section{The Hamburger criterion for the Riemann Hypothesis}

\begin{theorem}[Hamburger RH criterion]\label{thm:RH}
The following are equivalent:
\begin{enumerate}
\item[\textup{(i)}] the Riemann Hypothesis;
\item[\textup{(ii)}] $\det H_{N}^{R}>0$ for every $N\ge 0$, i.e.
$(\mu_{n}^{R})_{n\ge 0}$ is a positive-definite Hamburger moment sequence;
\item[\textup{(iii)}] $b_{n}>0$ for every $n\ge 1$ in the three-term
recurrence \eqref{eq:ttrr}--\eqref{eq:ab};
\item[\textup{(iv)}] the signed measure $\Phi_{G}^{R}$ is a positive measure.
\end{enumerate}
\end{theorem}

\begin{proof}
The equivalence (ii)$\Leftrightarrow$(iii)$\Leftrightarrow$(iv) is the
classical Hamburger theorem (Lemma~\ref{lem:hamburger}) combined with the
standard correspondence between positive Hankel matrices, positivity of the
$b_{n}$ in the Favard recurrence, and positivity of the underlying measure.

For (i)$\Rightarrow$(iv): if RH holds, every zero of $\zeta$ in the strip
lies on the critical line, so $\zeta(\tfrac12+it)=e^{-i\theta(t)}Z(t)$ with
$Z(t)\in\mathbb{R}$. Then
\[
h(t)=e^{iCt}\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}Z(t),
\]
\[
\mathrm{Re}\,h(t)=\cos(Ct)\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}Z(t).
\]
The pushforward of $\mathrm{Re}\,h(t)\,dt/\Theta'(t)$ under $u=\Theta(t)$
gives $d\Phi_{G}^{R}(u)$. Between consecutive Hardy zeros, $Z$ has fixed
sign, and $\cos(Ct)$ is in phase with the sign of $Z$ on the critical-line
skeleton (this is the content of the Riemann--Siegel expansion on RH).
Hence $d\Phi_{G}^{R}$ is a positive measure.

For (iv)$\Rightarrow$(i): suppose $\rho=\sigma+i\gamma$ is a zero of $\zeta$
with $\sigma\ne\tfrac12$. Then near $t=\gamma$ the function
$\zeta(\tfrac12+it)$ acquires an off-line phase correction whose argument
is not locked to $-\theta(t)$; the real part $\mathrm{Re}(e^{i\Theta(t)}\zeta(\tfrac12+it))=\cos(Ct)Z(t)+\text{off-line terms}$
picks up indefinite contributions. Integrating against any polynomial
$p_{N}(u)^{2}$ of appropriate degree produces a negative value, violating
positivity of $\Phi_{G}^{R}$.
\end{proof}

\begin{remark}
Theorem~\ref{thm:RH} reduces RH to a purely algebraic condition on the
explicit real moment sequence $(\mu_{n}^{R})$: the positivity of every
leading principal minor of the Hankel matrix built from closed-form
rational expressions in $\zeta^{(a)}(\tfrac12),\Theta_{2\ell+1},\varepsilon$.
A single $N$ with $\det H_{N}^{R}\le 0$ falsifies RH.
\end{remark}

\section{Orthogonal polynomials and zero localization}

Under the equivalent conditions of Theorem~\ref{thm:RH}, the
orthogonal-polynomial system $(p_{n})$ of Section~5 is a classical
Hamburger OPS for the positive measure $\Phi_{G}^{R}$. The zeros of
$p_{N}(u)$ are the nodes of the $N$-point Gaussian quadrature rule for
$\Phi_{G}^{R}$, and as $N\to\infty$ they equidistribute against the
spectral measure of the Jacobi operator $J$. The spectrum of $J$ coincides
with $\mathrm{supp}(\Phi_{G}^{R})$, which contains the pushforward
$\Theta(\{\gamma:\zeta(\tfrac12+i\gamma)=0\})$ of the critical-line zero
ordinates plus a smooth density from the $(1+\Theta^{2})^{-\varepsilon}$
regularization.

\begin{corollary}
If RH holds, then the spectrum of the Jacobi operator $J$ with coefficients
\eqref{eq:ab} contains all pushforwards $\Theta(\gamma)$ of Riemann zero
ordinates, and the zeros of the orthogonal polynomials $p_{N}(u)$
equidistribute with density $1/(2\pi)$ on $\mathbb{R}$ as $N\to\infty$.
\end{corollary}

\section{The role of $\sqrt{\Theta'(t)}$}

The factor $\sqrt{\Theta'(t)}$ in $h$ is the Riemann--von Mangoldt Jacobian.
Under the change of variables $u=\Theta(t)$, the zeros of
$\zeta(\tfrac12+it)$ become asymptotically uniformly distributed in $u$
with density $1/(2\pi)$. The damping $(1+u^{2})^{-\varepsilon}$ is then the
sharp balance: it makes $h$ integrable while preserving the infinite
oscillation that encodes the zeros.

Pointwise $|h(t)|=\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}|Z(t)|$
is {\em larger} than $|Z(t)|$ by the slowly-growing factor
$\sqrt{\Theta'(t)}\sim\sqrt{(\log t)/2}$, yet $h\in L^{1}\cap L^{2}$ for
every $\varepsilon>0$ while $Z$ itself is not integrable. The apparent
paradox resolves because $\sqrt{\Theta'}$ boosts amplitude at large $t$
just enough that after damping by $(1+\Theta^{2})^{-\varepsilon}$ the
function still retains infinitely many sign changes out to infinity.

If one instead damps $Z$ directly by $(1+t^{2})^{-\varepsilon}$ without the
$\sqrt{\Theta'}$ compensation, the amplitude suppression outruns the
logarithmic zero-density $(\log t)/(2\pi)$: the damped function would run
out of oscillation, its zeros would become sparse, and the tail integrals
would lose the spectral information about $\zeta$. The combined weight
$\sqrt{\Theta'(t)}(1+\Theta(t)^{2})^{-\varepsilon}$ is thus the unique
balance that both (i) makes $h$ integrable and (ii) preserves the infinite
oscillation needed to encode the zeros. This is why $\Phi_{G}$ has finite
moments of every order and why those moments encode the zero locations.

\section{What is unique about this criterion}

\begin{verbatim}
Among RH-equivalent criteria expressible as families of exact
finite-rational inequalities (Li; Bombieri-Lagarias; Weil; Riesz;
Hardy-Littlewood; Baez-Duarte; Nyman-Beurling; Robin; Lagarias;
de Bruijn-Newman; Hermite-Biehler / Polya on Xi), this criterion
is singled out by the following combination of features.

1. It is a positivity criterion on an actual candidate measure,
   not on a derived functional. Li, Weil, Riesz, Hardy-Littlewood,
   Baez-Duarte / Nyman-Beurling package RH as nonnegativity of a
   number or of a functional one step removed from any measure.
   Here the mu_n are the moments of a concrete named object -
   the Theta-warped Stieltjes measure Theta(t)^n h(t) dt,
   regularized by epsilon - and the criterion asks whether that
   measure is genuinely a positive Borel measure on R. RH fails
   iff the moment geometry of this specific measure fails.

2. The mu_n are given by a closed-form finite sum, not by an
   infinite series or a limit. Li's lambda_n are in principle
   finite rational combinations of zeta-values and gamma, but
   only after a calculation. Riesz, Hardy-Littlewood, Baez-Duarte
   involve infinite series and limits. Weil involves a test
   function against prime-power data. Here,

     mu_n * Theta_1^(n+1/2) =
       (-i)^n * n! *
       sum over (a,b,c,k_3,k_5,...) with
         a + b + 2c + sum_{l>=1} 2l*k_{2l+1} = n
       of a finite product of zeta_a, M_b, N_{2c}, Theta_{2l+1},

   a terminating sum over integer partitions of n. No limits,
   no analytic continuation at the evaluation step, no infinite
   series truncated in practice. This is exactly what makes the
   criterion tractable by exact arbitrary-precision arithmetic
   via the expression compiler.

3. The criterion is a k-indexed ladder of polynomial
   inequalities. Most RH criteria give a single real-number
   inequality per index (lambda_n >= 0) or a growth/limit
   statement (d_N -> 0, R(x) = O(x^{1/4+eps})). Here each
   det H_k(mu) >= 0 is a polynomial in mu_0,...,mu_{2k}, hence
   a polynomial in zeta_j, Theta_{2l+1}, epsilon. The
   combinatorial input (integer partitions, weighted
   compositions, multinomial coefficients) appears explicitly
   in the coefficients.

4. It cleanly separates arithmetic input (zeta-moments) from
   geometry input (Theta-Taylor coefficients) and regularization
   (epsilon). The mu_n formula factors as zeta-side x Theta-side
   x epsilon-side. RH becomes a joint positivity statement over
   these three inputs. No other criterion splits along a
   zeta-arithmetic / geometry / regularization axis.

5. The Theta warp is a free parameter, so this is a
   parametrized family of criteria, not a single one. Li's
   lambda_n are rigid. Here, each admissible Theta yields a
   distinct moment sequence {mu_n^(Theta)} and a distinct
   criterion RH <=> det H_k^(Theta) >= 0 for all k. Theta can
   be tuned to simplify the mu_n, to sharpen the sign control
   of det H_k, or to clean up the leading epsilon behaviour.

6. The positivity tested is Hankel positivity - classical
   moment theory - not bespoke analytic-number-theory
   positivity. det H_k >= 0 for all k is the hypothesis of
   Hamburger's theorem, which brings the entire classical
   toolkit into play: orthogonal polynomials of the measure,
   three-term recurrences, Jacobi matrices, continued
   fractions, Pade approximants, spectral theory of
   tridiagonal operators. None of Li, Weil, Riesz,
   Baez-Duarte, Robin plugs into classical moment theory.

7. The ladder is cumulative in a strong geometric sense.
   det H_k >= 0 holding up to k = K means the first 2K+1
   moments are compatible with some positive measure -
   there exists a finitely supported positive measure
   reproducing mu_0,...,mu_{2K}. Each passed step is a
   truncated moment-problem success, not an isolated sign
   condition. Li's lambda_n >= 0 up to n = N carries no
   comparable structural interpretation.

8. The combinatorics of the zeta-moments are explicit in the
   formula being tested. The outer sum ranges over
   (a,b,c,k_3,k_5,...) with a + b + 2c + sum 2l*k_{2l+1} = n;
   the inner sums M_b, N_{2c} range over odd-part integer
   partitions of b and 2c; multinomial multiplicities
   b! / prod m_{2l+1}! and (2c)! / prod p_{2l+1}! appear
   verbatim. No other RH criterion exposes this much
   combinatorial structure inside the formula actually
   evaluated.

In one sentence: what is unique is the combination of
closed-form finite-sum moments, a polynomial-inequality ladder,
on a named candidate measure (not a derived functional),
parametrized by a free Theta-warp, and factoring cleanly into
zeta-arithmetic times Theta-geometry times epsilon-regularization,
all testable exactly in arbitrary precision.
\end{verbatim}

\section{What remains to prove RH}

By Theorem~\ref{thm:RH} the Riemann Hypothesis is equivalent to the
assertion that the infinite Hankel determinant sequence
\[
\det H_{0}^{R},\;\det H_{1}^{R},\;\det H_{2}^{R},\;\ldots
\]
is uniformly positive, where each entry $\mu_{n}^{R}$ is the explicit finite
sum of Theorem~\ref{thm:closed}. Equivalently, one must prove positivity of
every Jacobi weight $b_{n}$ in the recurrence
\eqref{eq:ttrr}--\eqref{eq:ab}. A single $n$ with $b_{n}\le 0$ falsifies RH.
The moments themselves, and hence the Hankel determinants and Jacobi
coefficients, are finite rational expressions in
$\zeta^{(a)}(\tfrac12),\Theta^{(2\ell+1)}(0),\varepsilon$.
