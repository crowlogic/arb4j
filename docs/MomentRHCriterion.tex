\section*{The closed-form moment}

\[
\mu_n \cdot \Theta_1^{\,n+\tfrac{1}{2}} \;=\; n!\,\sum_{\substack{p+2q+2Q+s=n\\ p,q,Q\ge 0,\ r_k\ge 0\\ \sum_{k\text{ odd}} k\,r_k = s\\ m := \sum_{k\text{ odd}} r_k}} \frac{(-1)^{q+Q+\sum_{k\ge 3,\,k\text{ odd}}\frac{k-1}{2}r_k}\,\dbinom{-\alpha}{q}\,J_Q}{p!\,\prod_{k\text{ odd}} r_k!}\;\zeta_m\;\Theta_1^{\,n-m}\,\prod_{k\text{ odd}}\bigl(\tau_k\,\Theta_1^{k}\bigr)^{r_k}
\]

with

\[
\tau_k \;=\; \Theta_1^{-k}\sum_{\substack{a_1,a_2,\ldots\ge 0\\ 2a_1+4a_2+6a_3+\cdots = k-1}}\frac{1}{k}\dbinom{-k}{a_1,a_2,\ldots,\textstyle\sum_i a_i}\prod_{i\ge 1}\!\left(\frac{\Theta_{2i+1}}{(2i+1)!\,\Theta_1}\right)^{a_i}
\]

\[
J_Q \;=\; \sum_{\substack{b_1,b_2,\ldots\ge 0\\ j:=\sum_i b_i}}\dbinom{1/2}{j}\frac{j!}{\prod_i b_i!}\prod_{i\ge 1}\!\left(\frac{\Theta_{2i+1}}{(2i)!\,\Theta_1}\right)^{b_i}\!\sum_{\substack{c_k\ge 0,\,k\text{ odd}\\ \sum c_k = 2\sum_i i\,b_i\\ \sum k\,c_k = 2Q}}\frac{(2\sum_i i\,b_i)!}{\prod c_k!}\prod_{k\text{ odd}}\tau_k^{\,c_k}
\]

\[
\zeta_m \;:=\; \zeta^{(m)}(\tfrac{1}{2}),\qquad \Theta_{2\ell+1} \;:=\; \Theta^{(2\ell+1)}(0).
\]

\section*{The criterion}

Let $\Omega_n := \mu_n \cdot \Theta_1^{\,n+\tfrac{1}{2}}$.

\[
\boxed{\;\mathrm{RH} \iff \limsup_{n\to\infty}\frac{|\Omega_n|^{1/n}}{n} \;\le\; \frac{\Theta_1}{A},\qquad A := \inf_{0<y<1/2}\left|\frac{\Theta(-iy)}{y}\right|.\;}
\]
